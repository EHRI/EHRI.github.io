57      private static class LruCache<A, B> extends LinkedHashMap<A, B> {
58          private final int maxEntries;
59  
60          public LruCache(final int maxEntries) {
61              super(maxEntries + 1, 1.0f, true);
62              this.maxEntries = maxEntries;
63          }
64  
65          @Override
66          protected boolean removeEldestEntry(final Map.Entry<A, B> eldest) {
67              return super.size() > maxEntries;
68          }
69      }

334     /**
335      * Convert a Frame into an EntityBundle that includes its @Fetch'd
336      * relations.
337      *
338      * @param item  The item vertex
339      * @param depth The maximum serialization depth
340      * @return A data bundle
341      * @throws SerializationError
342      */
343     private Bundle vertexToBundle(Vertex item, int depth, int maxDepth, boolean lite)
344             throws SerializationError {
345         try {
346             EntityClass type = EntityClass.withName((String) item
347                     .getProperty(EntityType.TYPE_KEY));
348             String id = item.getProperty(EntityType.ID_KEY);
349             logger.trace("Serializing {} ({}) at depth {}", id, type, depth);
350 
351             Class<? extends Frame> cls = type.getEntityClass();
352             Bundle.Builder builder = Bundle.Builder.withClass(type)
353                     .setId(id)
354                     .addData(getVertexData(item, type, lite))
355                     .addRelations(getRelationData(item,
356                             depth, maxDepth, lite, cls))
357                     .addMetaData(getVertexMeta(item, cls));
358             if (!lite) {
359                 builder.addMetaData(getVertexMeta(item, cls))
360                         .addMetaDataValue("gid", item.getId());
361             }
362             return builder.build();
363         } catch (IllegalArgumentException e) {
364             logger.error("Error serializing vertex with data: {}", getVertexData(item));
365             throw new SerializationError("Unable to serialize vertex: " + item, e);
366         }
367     }

369     private Bundle fetch(Frame frame, int depth, int maxDepth, boolean isLite) throws SerializationError {
370         if (cache != null) {
371             String key = frame.getId() + depth + isLite;
372             if (cache.containsKey(key))
373                 return cache.get(key);
374             Bundle bundle = vertexToBundle(frame.asVertex(), depth, maxDepth, isLite);
375             cache.put(key, bundle);
376             return bundle;
377         }
378         return vertexToBundle(frame.asVertex(), depth, maxDepth, isLite);
379     }

381     // TODO: Profiling shows that (unsurprisingly) this method is a
382     // performance hotspot. Rewrite it so that instead of using Frames
383     // method invocations to do the traversal, we use regular traversals
384     // whereever possible. Unfortunately the use of @JavaHandler Frame
385     // annotations will make this difficult.
386     private ListMultimap<String, Bundle> getRelationData(
387             Vertex item, int depth, int maxDepth, boolean lite, Class<?> cls) {
388         ListMultimap<String, Bundle> relations = ArrayListMultimap.create();
389         if (depth < maxDepth) {
390             Map<String, Method> fetchMethods = ClassUtils.getFetchMethods(cls);
391             logger.trace(" - Fetch methods: {}", fetchMethods);
392             for (Map.Entry<String, Method> entry : fetchMethods.entrySet()) {
393                 String relationName = entry.getKey();
394                 Method method = entry.getValue();
395 
396                 boolean isLite = liteMode || lite
397                         || shouldSerializeLite(method);
398 
399                 if (shouldTraverse(relationName, method, depth, isLite)) {
400                     int nextDepth = depth + 1;
401                     int nextMaxDepth = getNewMaxDepth(method, nextDepth, maxDepth);
402                     logger.trace("Fetching relation: {}, depth {}, {}",
403                             relationName, depth, method.getName());
404                     try {
405                         Object result = method.invoke(graph.frame(
406                                 item, cls));
407                         // The result of one of these fetchMethods should either
408                         // be a single Frame, or a Iterable<Frame>.
409                         if (result instanceof Iterable<?>) {
410                             for (Object d : (Iterable<?>) result) {
411                                 relations.put(relationName, fetch((Frame) d, nextDepth, nextMaxDepth, isLite));
412                             }
413                         } else {
414                             // This relationship could be NULL if, e.g. a
415                             // collection has no holder.
416                             if (result != null) {
417                                 relations.put(relationName, fetch((Frame) result, nextDepth, nextMaxDepth, isLite));
418                             }
419                         }
420                     } catch (Exception e) {
421                         e.printStackTrace();
422                         logger.error("Error serializing relationship for {} ({}): {}, depth {}, {}",
423                                 item, item.getProperty(EntityType.TYPE_KEY),
424                                 relationName, depth, method.getName());
425                         throw new RuntimeException(
426                                 "Unexpected error serializing Frame " + item, e);
427                     }
428                 }
429             }
430         }
431         return relations;
432     }

434     private int getNewMaxDepth(Method fetchMethod, int currentDepth, int currentMaxDepth) {
435         Fetch fetchProps = fetchMethod.getAnnotation(Fetch.class);
436         int max = fetchProps.numLevels();
437         int newMax = max == -1
438                 ? currentMaxDepth
439                 : Math.min(currentDepth + max, currentMaxDepth);
440         logger.trace("Current depth {}, fetch levels: {}, current max: {}, new max: {}, {}", currentDepth, max,
441                 currentMaxDepth, newMax, fetchMethod.getName());
442         return newMax;
443     }

445     private boolean shouldSerializeLite(Method method) {
446         Dependent dep = method.getAnnotation(Dependent.class);
447         Fetch fetch = method.getAnnotation(Fetch.class);
448         return dep == null && (fetch == null || !fetch.full());
449     }

451     private boolean shouldTraverse(String relationName, Method method, int level, boolean lite) {
452         // In order to avoid @Fetching the whole graph we track the
453         // depth parameter and increase it for every traversal.
454         // However the @Fetch annotation can also specify a maximum
455         // level of traversal beyond which we don't serialize.
456         Fetch fetchProps = method.getAnnotation(Fetch.class);
457         Dependent dep = method.getAnnotation(Dependent.class);
458 
459         if (fetchProps == null) {
460             return false;
461         }
462 
463         if (dependentOnly && dep == null) {
464             logger.trace(
465                     "Terminating fetch dependent only is specified: {}, ifBelowLevel {}, limit {}, {}",
466                     relationName, level, fetchProps.ifBelowLevel());
467             return false;
468         }
469 
470         if (lite && fetchProps.whenNotLite()) {
471             logger.trace(
472                     "Terminating fetch because it specifies whenNotLite: {}, ifBelowLevel {}, limit {}, {}",
473                     relationName, level, fetchProps.ifBelowLevel());
474             return false;
475         }
476 
477         if (level >= fetchProps.ifBelowLevel()) {
478             logger.trace(
479                     "Terminating fetch because level exceeded ifBelowLevel on fetch clause: {}, ifBelowLevel {}, " +
480                             "limit {}, {}",
481                     relationName, level, fetchProps.ifBelowLevel());
482             return false;
483         }
484 
485         // If the fetch should only be serialized at a certain ifBelowLevel and
486         // we've exceeded that, don't serialize.
487         if (fetchProps.ifLevel() != -1 && level > fetchProps.ifLevel()) {
488             logger.trace(
489                     "Terminating fetch because ifLevel clause found on {}, ifBelowLevel {}, {}",
490                     relationName, level);
491             return false;
492         }
493         return true;
494     }

496     /**
497      * Fetch a map of data from a vertex.
498      */
499     private Map<String, Object> getVertexData(Vertex item, EntityClass type, boolean lite) {
500         Map<String, Object> data = Maps.newHashMap();
501         Iterable<String> keys = lite
502                 ? getMandatoryOrSpecificProps(type)
503                 : item.getPropertyKeys();
504 
505         for (String key : keys) {
506             if (!(key.equals(EntityType.ID_KEY) || key
507                     .equals(EntityType.TYPE_KEY) || key.startsWith("_")))
508                 data.put(key, item.getProperty(key));
509         }
510         return data;
511     }

527     /**
528      * Fetch a map of metadata sourced from vertex properties.
529      * This is anything that begins with an underscore (but now
530      * two underscores)
531      */
532     private Map<String, Object> getVertexMeta(Vertex item, Class<?> cls) {
533         Map<String, Object> data = Maps.newHashMap();
534         for (String key : item.getPropertyKeys()) {
535             if (!key.startsWith("__") && key.startsWith("_")) {
536                 data.put(key.substring(1), item.getProperty(key));
537             }
538         }
539         Map<String, Method> metaMethods = ClassUtils.getMetaMethods(cls);
540         if (!metaMethods.isEmpty()) {
541             try {
542                 Object frame = graph.frame(item, cls);
543                 for (Map.Entry<String, Method> metaEntry : metaMethods.entrySet()) {
544                     Object value = metaEntry.getValue().invoke(frame);
545                     if (value != null) {
546                         data.put(metaEntry.getKey(), value);
547                     }
548                 }
549             } catch (Exception e) {
550                 throw new RuntimeException("Error fetching metadata", e);
551             }
552         }
553         return data;
554     }

564     /**
565      * Run a callback every time a node in a subtree is encountered, excepting
566      * the top-level node.
567      */
568     private <T extends Frame> void traverseSubtree(T item, int depth,
569             final TraversalCallback cb) {
570 
571         if (depth < maxTraversals) {
572             Class<?> cls = EntityClass.withName(
573                     (String) item.asVertex().getProperty(EntityType.TYPE_KEY))
574                     .getEntityClass();
575             Map<String, Method> fetchMethods = ClassUtils.getFetchMethods(cls);
576             for (Map.Entry<String, Method> entry : fetchMethods.entrySet()) {
577 
578                 String relationName = entry.getKey();
579                 Method method = entry.getValue();
580                 if (shouldTraverse(relationName, method, depth, false)) {
581                     try {
582                         Object result = method.invoke(graph.frame(
583                                 item.asVertex(), cls));
584                         if (result instanceof Iterable<?>) {
585                             int rnum = 0;
586                             for (Object d : (Iterable<?>) result) {
587                                 cb.process((Frame) d, depth,
588                                         entry.getKey(), rnum);
589                                 traverseSubtree((Frame) d, depth + 1, cb);
590                                 rnum++;
591                             }
592                         } else {
593                             if (result != null) {
594                                 cb.process((Frame) result, depth,
595                                         entry.getKey(), 0);
596                                 traverseSubtree((Frame) result,
597                                         depth + 1, cb);
598                             }
599                         }
600                     } catch (Exception e) {
601                         e.printStackTrace();
602                         throw new RuntimeException(
603                                 "Unexpected error serializing Frame", e);
604                     }
605                 }
606             }
607         }
