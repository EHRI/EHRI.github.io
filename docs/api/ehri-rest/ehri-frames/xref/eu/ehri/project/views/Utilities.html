34      /**
35       * Replace an existing property value with another value across an
36       * entire class of items.
37       *
38       * @param entityClass   the item class
39       * @param propertyName  the property name
40       * @param existingValue the existing value
41       * @param newValue      the new value
42       * @return the number of items changed
43       */
44      public long findReplacePropertyValue(EntityClass entityClass, String propertyName,
45                                           String existingValue, String newValue) {
46          checkLegalPropertyName(propertyName);
47          CloseableIterable<Vertex> vertices = manager.getVertices(entityClass);
48          try {
49              long changes = 0L;
50              for (Vertex v : vertices) {
51                  Object current = v.getProperty(propertyName);
52                  if (current instanceof List) {
53                      List<String> stringList = v.getProperty(propertyName);
54                      int valuesChanged = replaceInList(stringList, existingValue, newValue);
55                      if (valuesChanged > 0) {
56                          changes += valuesChanged;
57                          manager.setProperty(v, propertyName, stringList.toArray(new String[stringList.size()]));
58                      }
59                  } else if (current instanceof String) {
60                      if (existingValue.equals(current)) {
61                          changes++;
62                          manager.setProperty(v, propertyName, newValue);
63                      }
64                  }
65              }
66              return changes;
67          } finally {
68              vertices.close();
69          }
70      }

72      /**
73       * Replace a substring of a property value across an entire
74       * item class. Property values must either be strings of
75       * string arrays.
76       *
77       * @param entityClass  the item class
78       * @param propertyName the property name
79       * @param pattern      the existing value
80       * @param replacement  the new value
81       * @return the number of properties changed
82       */
83      public long findReplacePropertyValueRE(EntityClass entityClass, String propertyName,
84                                             Pattern pattern, String replacement) {
85          checkLegalPropertyName(propertyName);
86          CloseableIterable<Vertex> vertices = manager.getVertices(entityClass);
87          try {
88              long changes = 0L;
89              for (Vertex v : vertices) {
90                  Object current = v.getProperty(propertyName);
91                  if (current instanceof List) {
92                      List<String> stringList = v.getProperty(propertyName);
93                      int valuesChanged = regexReplaceInList(stringList, pattern, replacement);
94                      if (valuesChanged > 0) {
95                          changes += valuesChanged;
96                          manager.setProperty(v, propertyName, stringList.toArray(new String[stringList.size()]));
97                      }
98                  } else if (current instanceof String) {
99                      Matcher matcher = pattern.matcher((String) current);
100                     if (matcher.find()) {
101                         changes++;
102                         manager.setProperty(v, propertyName, matcher.replaceAll(replacement));
103                     }
104                 }
105             }
106             return changes;
107         } finally {
108             vertices.close();
109         }

112     /**
113      * Rename a property key across an entire class of items.
114      *
115      * @param entityClass     the item class
116      * @param propertyName    the existing property name
117      * @param newPropertyName the new property name
118      * @return the number of items changed
119      */
120     public long replacePropertyName(EntityClass entityClass, String propertyName, String newPropertyName) {
121         checkLegalPropertyName(propertyName);
122         CloseableIterable<Vertex> vertices = manager.getVertices(entityClass);
123         try {
124             long changes = 0L;
125             for (Vertex v : vertices) {
126                 Set<String> propertyKeys = v.getPropertyKeys();
127                 if (propertyKeys.contains(propertyName)) {
128                     Object current = v.getProperty(propertyName);
129                     manager.setProperty(v, newPropertyName, current);
130                     manager.setProperty(v, propertyName, null);
131                     changes++;
132                 }
133             }
134             return changes;
135         } finally {
136             vertices.close();
137         }
138     }

142     private void checkLegalPropertyName(String propertyName) {
143         if (propertyName == null
144                 || propertyName.trim().isEmpty()
145                 || propertyName.equals(EntityType.ID_KEY)
146                 || propertyName.equals(EntityType.TYPE_KEY)) {
147             throw new IllegalArgumentException("Invalid property name: " + propertyName);
148         }
149     }

151     /**
152      * Replaces items in the input list that equal the given existing value
153      * with the replacement string, returning the number of changes made.
154      *
155      * @param listInParam an input list that will be changed in place
156      * @param existing    an existing value
157      * @param replacement a replacement string
158      * @return the number of changes made in the list
159      */
160     private int replaceInList(List<String> listInParam, String existing, String replacement) {
161         int changes = 0;
162         for (int i = 0; i < listInParam.size(); i++) {
163             String itemAt = listInParam.get(i);
164             if (itemAt.equals(existing)) {
165                 changes++;
166                 listInParam.set(i, replacement);
167             }
168         }
169         return changes;
170     }

172     /**
173      * Replaces items in the input list that match the given pattern with the replacement string,
174      * returning the number of changes made.
175      *
176      * @param listInParam an input list that will be changed in place
177      * @param pattern     a regular expression pattern
178      * @param replacement a replacement string
179      * @return the number of changes made in the list
180      */
181     private int regexReplaceInList(List<String> listInParam, Pattern pattern, String replacement) {
182         int changes = 0;
183         for (int i = 0; i < listInParam.size(); i++) {
184             String itemAt = listInParam.get(i);
185             Matcher matcher = pattern.matcher(itemAt);
186             if (matcher.find()) {
187                 changes++;
188                 listInParam.set(i, matcher.replaceAll(replacement));
189             }
190         }
191         return changes;
192     }
