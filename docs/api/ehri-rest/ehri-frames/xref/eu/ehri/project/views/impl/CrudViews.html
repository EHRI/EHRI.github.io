33      /**
34       * Scoped Constructor.
35       *
36       * @param graph The graph
37       * @param cls   The entity class to return
38       * @param scope The permission scope
39       */
40      public CrudViews(FramedGraph<?> graph, Class<E> cls,
41              PermissionScope scope) {
42          this.graph = graph;
43          this.cls = cls;
44          this.scope = Optional.fromNullable(scope).or(SystemScope.getInstance());
45          helper = new ViewHelper(graph, this.scope);
46          acl = helper.getAclManager();
47          serializer = new Serializer(graph);
48          manager = GraphManagerFactory.getInstance(graph);
49      }

52       * Constructor.
53       *
54       * @param graph The graph
55       * @param cls   The entity class to return
56       */
57      public CrudViews(FramedGraph<?> graph, Class<E> cls) {
58          this(graph, cls, SystemScope.getInstance());

61      /**
62       * Fetch an item, as a user. This only provides access control.
63       *
64       * @param id The item id
65       * @param user The current user
66       * @return The given framed vertex
67       * @throws ItemNotFound
68       */
69      public E detail(String id, Accessor user) throws ItemNotFound {
70          E item = manager.getFrame(id, cls);
71          if (!acl.canAccess(item, user)) {
72              throw new ItemNotFound(id);
73          }
74          return item;
75      }

77      /**
78       * Update an object bundle, also updating dependent items.
79       *
80       * @param bundle     The item's data bundle
81       * @param user       The current user
82       * @return The updated framed vertex
83       * @throws PermissionDenied
84       * @throws ValidationError
85       * @throws ItemNotFound
86       */
87      public Mutation<E> update(Bundle bundle, Accessor user)
88              throws PermissionDenied, ValidationError, ItemNotFound {
89          E entity = graph.frame(manager.getVertex(bundle.getId()), cls);
90          helper.checkEntityPermission(entity, user, PermissionType.UPDATE);
91          return getPersister(scope).update(bundle, cls);
92      }

94      /**
95       * Create a new object of type `E` from the given data, within the scope of
96       * `scope`.
97       *
98       * @param bundle     The item's data bundle
99       * @param user       The current user
100      * @return The created framed vertex
101      * @throws PermissionDenied
102      * @throws ValidationError
103      */
104     public E create(Bundle bundle, Accessor user)
105             throws PermissionDenied, ValidationError {
106         helper.checkContentPermission(user, helper.getContentTypeEnum(cls),
107                 PermissionType.CREATE);
108         E item = getPersister(scope).create(bundle, cls);
109         // If a user creates an item, grant them OWNER perms on it.
110         // Owner permissions do not have a scope.
111         // FIXME: Currently a hack here so this doesn't apply to admin
112         // users - but it probably should...
113         if (!acl.belongsToAdmin(user)) {
114             acl.withScope(SystemScope.INSTANCE)
115                     .grantPermission(item, PermissionType.OWNER, user);
116         }
117         // If the scope is not the system, set the permission scope
118         // of the item too...
119         if (!scope.equals(SystemScope.getInstance())) {
120             item.setPermissionScope(scope);
121         }
122         return item;
123     }

125     /**
126      * Create or update a new object of type `E` from the given data, within the
127      * scope of `scope`.
128      *
129      * @param bundle     The item's data bundle
130      * @param user       The current user
131      * @return The created framed vertex
132      * @throws PermissionDenied
133      * @throws ValidationError
134      */
135     public Mutation<E> createOrUpdate(Bundle bundle, Accessor user)
136             throws PermissionDenied, ValidationError {
137         helper.checkContentPermission(user, helper.getContentTypeEnum(cls),
138                 PermissionType.CREATE);
139         helper.checkContentPermission(user, helper.getContentTypeEnum(cls),
140                 PermissionType.UPDATE);
141         return getPersister(scope).createOrUpdate(bundle, cls);
142     }

144     /**
145      * Delete an object bundle, following dependency cascades, within the scope
146      * of item `scope`.
147      *
148      * @param id The item ID
149      * @param user The current user
150      * @return The number of vertices deleted.
151      * @throws ItemNotFound
152      * @throws PermissionDenied
153      * @throws ValidationError
154      * @throws SerializationError
155      */
156     public Integer delete(String id, Accessor user) throws PermissionDenied,
157             ValidationError, SerializationError, ItemNotFound {
158         E item = manager.getFrame(id, cls);
159         helper.checkEntityPermission(item, user, PermissionType.DELETE);
160         return getPersister(scope).delete(serializer.vertexFrameToBundle(item));
161     }

163     /**
164      * Set the permission scope of the view.

169     public Crud<E> setScope(PermissionScope scope) {
170         return new CrudViews<E>(graph, cls,

172     }
