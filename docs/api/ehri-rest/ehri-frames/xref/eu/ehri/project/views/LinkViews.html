62      /**
63       * Create a link between two items.
64       *
65       * @param targetId1 the identifier of a AccessibleEntity target of this Annotation
66       * @param targetId2 the identifier of a Annotator source of this Annotation
67       * @param bundle    the annotation itself
68       * @param user the current user
69       * @return A new link
70       * @throws eu.ehri.project.exceptions.ItemNotFound
71       *
72       * @throws eu.ehri.project.exceptions.ValidationError
73       *
74       * @throws eu.ehri.project.exceptions.PermissionDenied
75       *
76       */
77      public Link createLink(String targetId1, String targetId2, List<String> bodies, Bundle bundle,
78              Accessor user) throws ItemNotFound, ValidationError, PermissionDenied {
79          LinkableEntity t1 = manager.getFrame(targetId1, LinkableEntity.class);
80          LinkableEntity t2 = manager.getFrame(targetId2, LinkableEntity.class);
81          helper.checkEntityPermission(t1, user, PermissionType.ANNOTATE);
82          // TODO: Should this require perms to link another item???
83          //helper.checkEntityPermission(t2, user, PermissionType.ANNOTATE);
84          Link link = new BundleDAO(graph).create(bundle, Link.class);
85          link.addLinkTarget(t1);
86          link.addLinkTarget(t2);
87          link.setLinker(user);
88          ActionManager.EventContext eventContext = new ActionManager(graph, t1).logEvent(
89                  graph.frame(user.asVertex(), Actioner.class),
90                  EventTypes.link, Optional.<String>absent());
91          eventContext.addSubjects(link).addSubjects(t2);
92          for (String body : bodies) {
93              AccessibleEntity item = manager.getFrame(body, AccessibleEntity.class);
94              link.addLinkBody(item);
95              eventContext.addSubjects(item);
96          }
97          return link;
98      }

100     /**
101      * Create a link between two items, along with an access point on the given description.
102      *
103      * @param targetId1       the identifier of a AccessibleEntity target of this Annotation
104      * @param targetId2 the identifier of a Annotator source of this Annotation
105      * @param bundle   the annotation itself
106      * @param user The current user
107      * @return A new link
108      * @throws ItemNotFound
109      * @throws ValidationError
110      * @throws PermissionDenied
111      */
112     public Link createAccessPointLink(String targetId1, String targetId2, String descriptionId, String bodyName,
113             String bodyType, Bundle bundle, Accessor user) throws ItemNotFound, ValidationError, PermissionDenied {
114         LinkableEntity t1 = manager.getFrame(targetId1, LinkableEntity.class);
115         LinkableEntity t2 = manager.getFrame(targetId2, LinkableEntity.class);
116         Description description = manager.getFrame(descriptionId, Description.class);
117         helper.checkEntityPermission(t1, user, PermissionType.ANNOTATE);
118         // TODO: Should this require perms to link another item???
119         //helper.checkEntityPermission(t2, user, PermissionType.ANNOTATE);
120         helper.checkEntityPermission(description.getEntity(), user, PermissionType.UPDATE);
121         Link link = new BundleDAO(graph).create(bundle, Link.class);
122         Bundle relBundle = new Bundle(EntityClass.UNDETERMINED_RELATIONSHIP)
123                 .withDataValue(Ontology.NAME_KEY, bodyName)
124                 .withDataValue(Ontology.UNDETERMINED_RELATIONSHIP_TYPE, bodyType)
125                 .withDataValue(Ontology.LINK_HAS_DESCRIPTION, link.getDescription());
126         UndeterminedRelationship rel = new BundleDAO(graph).create(relBundle, UndeterminedRelationship.class);
127         description.addUndeterminedRelationship(rel);
128         link.addLinkTarget(t1);
129         link.addLinkTarget(t2);
130         link.setLinker(user);
131         link.addLinkBody(rel);
132         ActionManager.EventContext eventContext = new ActionManager(graph).logEvent(
133                 t1, graph.frame(user.asVertex(), Actioner.class), EventTypes.link);
134         eventContext.addSubjects(link).addSubjects(t2).addSubjects(rel);
135         return link;
136     }
