49   * @author Mike Bryant (http://github.com/mikesname)
50   */
51  public class DescriptionViews <E extends DescribedEntity> {
52  
53      private final FramedGraph<?> graph;
54      private final GraphManager manager;
55      private final Crud<E> crud;
56      private final ViewHelper helper;
57      private final Serializer serializer;
58      private final ActionManager actionManager;
59  
60      public DescriptionViews(FramedGraph<?> graph, Class<E> parentClass) {
61          this.graph = graph;
62          this.manager = GraphManagerFactory.getInstance(graph);
63          this.crud = ViewFactory.getCrudNoLogging(graph, parentClass);
64          this.helper = new ViewHelper(graph);
65          this.serializer = new Serializer.Builder(graph).dependentOnly().build();
66          this.actionManager = new ActionManager(graph);
67      }
68  
69      public int delete(String parentId, String id, Accessor user, Optional<String> logMessage)
70              throws ItemNotFound, PermissionDenied, SerializationError {
71          E parent = crud.detail(parentId, user);
72          AccessibleEntity dependentItem = manager.getFrame(id, AccessibleEntity.class);
73          if (!itemsInSubtree(parent).contains(dependentItem)) {
74              throw new PermissionDenied("Given description does not belong to its parent item");
75          }
76          helper.checkEntityPermission(parent, user, PermissionType.UPDATE);
77          actionManager.newEventContext(parent, manager.cast(user, Actioner.class),
78                  EventTypes.deleteDependent, logMessage)
79                  .createVersion(dependentItem)
80                  .commit();
81          return getPersister(parent)
82                  .delete(serializer.vertexFrameToBundle(dependentItem));
83      }
84  
85      public <T extends AccessibleEntity> T create(String parentId, Bundle data,
86              Class<T> descriptionClass, Accessor user, Optional<String> logMessage)
87              throws ItemNotFound, PermissionDenied, ValidationError {
88          E parent = crud.detail(parentId, user);
89          helper.checkEntityPermission(parent, user, PermissionType.UPDATE);
90          T out = getPersister(parent).create(data, descriptionClass);
91          actionManager.newEventContext(parent, manager.cast(user, Actioner.class),
92                  EventTypes.createDependent, logMessage)
93                  .commit();
94          return out;
95      }
96  
97      public <T extends AccessibleEntity> Mutation<T> update(String parentId, Bundle data,
98                  Class<T> descriptionClass, Accessor user, Optional<String> logMessage)
99              throws ItemNotFound, PermissionDenied, ValidationError {
100         E parent = crud.detail(parentId, user);
101         helper.checkEntityPermission(parent, user, PermissionType.UPDATE);
102         Mutation<T> out = getPersister(parent).update(data, descriptionClass);
103         if (!out.unchanged()) {
104             actionManager
105                     .newEventContext(parent, manager.cast(user, Actioner.class),
106                             EventTypes.modifyDependent, logMessage)
107                     .createVersion(out.getNode(), out.getPrior().get())
108                     .commit();
109         }
110         return out;
111     }
112 
113     // Helpers
114     private BundleDAO getPersister(PermissionScope scope) {
115         return new BundleDAO(graph, scope.idPath());
116     }
117 
118     private Set<Frame> itemsInSubtree(Frame topLevel) {
119         final Set<Frame> items = Sets.newHashSet();
120         serializer.traverseSubtree(topLevel, new TraversalCallback() {
121             @Override
122             public void process(Frame frame, int depth, String relation, int relationIndex) {
123                 items.add(frame);
124             }
125         });
126         return items;
127     }
