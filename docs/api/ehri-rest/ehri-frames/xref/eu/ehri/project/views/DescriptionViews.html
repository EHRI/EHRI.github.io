28      public DescriptionViews(FramedGraph<?> graph, Class<E> parentClass) {
29          this.graph = graph;
30          this.manager = GraphManagerFactory.getInstance(graph);
31          this.crud = ViewFactory.getCrudNoLogging(graph, parentClass);
32          this.helper = new ViewHelper(graph);
33          this.serializer = new Serializer.Builder(graph).dependentOnly().build();
34          this.actionManager = new ActionManager(graph);
35      }

37      public Integer delete(String parentId, String id, Accessor user, Optional<String> logMessage)
38              throws ItemNotFound, PermissionDenied, SerializationError {
39          E parent = crud.detail(parentId, user);
40          Description desc = manager.getFrame(id, Description.class);
41          if (!parent.equals(desc.getDescribedEntity())) {
42              throw new PermissionDenied("Given description does not belong to its parent item");
43          }
44          helper.checkEntityPermission(parent, user, PermissionType.UPDATE);
45          actionManager.setScope(parent)
46                  .logEvent(parent, manager.cast(user, Actioner.class),
47                          EventTypes.deleteDependent, logMessage)
48                  .createVersion(desc);
49          return getPersister(parent).delete(serializer.vertexFrameToBundle(desc));
50      }

52      public <T extends Frame> T create(String parentId, Bundle data,
53              Class<T> descriptionClass, Accessor user, Optional<String> logMessage)
54              throws ItemNotFound, PermissionDenied, ValidationError {
55          E parent = crud.detail(parentId, user);
56          helper.checkEntityPermission(parent, user, PermissionType.UPDATE);
57          T out = getPersister(parent).create(data, descriptionClass);
58          actionManager.setScope(parent).logEvent(parent,
59                  manager.cast(user, Actioner.class),
60                  EventTypes.createDependent, logMessage);
61          return out;
62      }
63  
64      public <T extends Frame> Mutation<T> update(String parentId, Bundle data,
65                  Class<T> descriptionClass, Accessor user, Optional<String> logMessage)
66              throws ItemNotFound, PermissionDenied, ValidationError {
67          E parent = crud.detail(parentId, user);
68          helper.checkEntityPermission(parent, user, PermissionType.UPDATE);
69          Mutation<T> out = getPersister(parent).update(data, descriptionClass);
70          if (!out.unchanged()) {
71              actionManager.setScope(parent)
72                      .logEvent(parent, manager.cast(user, Actioner.class),
73                              EventTypes.modifyDependent, logMessage)
74                      .createVersion(out.getNode(), out.getPrior().get());
75          }
76          return out;
77      }
