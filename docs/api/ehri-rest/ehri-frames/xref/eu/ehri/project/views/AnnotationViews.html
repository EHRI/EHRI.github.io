56      /**
57       * Create an annotation for a dependent node of an item. The entity and the
58       * dependent item can be the same.
59       *
60       * @param id           the identifier of the AccessibleEntity this annotation is attached to, as a target
61       * @param did          the identifier of the dependent item
62       * @param bundle       the annotation itself
63       * @param user         the user creating the annotation
64       * @param accessibleTo Users or groups who can access this annotation
65       * @return the created annotation
66       * @throws PermissionDenied
67       * @throws AccessDenied
68       * @throws ValidationError
69       * @throws ItemNotFound
70       */
71      public Annotation createFor(String id, String did, Bundle bundle, Accessor user, Iterable<Accessor> accessibleTo)
72              throws PermissionDenied, AccessDenied, ValidationError, ItemNotFound {
73          AccessibleEntity entity = manager.getFrame(id, AccessibleEntity.class);
74          AnnotatableEntity dep = manager.getFrame(did, AnnotatableEntity.class);
75          helper.checkEntityPermission(entity, user, PermissionType.ANNOTATE);
76          helper.checkReadAccess(entity, user);
77  
78          if (!(entity.equals(dep) || isInSubtree(entity, dep))) {
79              // FIXME: Better error message here...
80              throw new PermissionDenied("Item is not covered by parent item's permissions");
81          }
82  
83          Annotation annotation = new BundleDAO(graph).create(bundle, Annotation.class);
84          entity.addAnnotation(annotation);
85          if (!entity.equals(dep)) {
86              dep.addAnnotationPart(annotation);
87          }
88          annotation.setAnnotator(graph.frame(user.asVertex(), Annotator.class));
89          acl.setAccessors(annotation, accessibleTo);
90          acl.withScope(SystemScope.INSTANCE)
91                  .grantPermission(annotation, PermissionType.OWNER, user);
92  
93          new ActionManager(graph, entity).logEvent(annotation,
94                  graph.frame(user.asVertex(), Actioner.class),
95                  EventTypes.annotation, Optional.<String>absent());
96          return annotation;
97      }
98  
99      private boolean isInSubtree(Frame parent, Frame child) {
100         // Check dependent is within item's subtree!
101         final Set<String> deps = Sets.newHashSet();
102         new Serializer(graph).traverseSubtree(parent, new TraversalCallback() {
103             @Override
104             public void process(Frame vertexFrame, int depth,
105                     String relation, int relationIndex) {
106                 deps.add(vertexFrame.getId());
107             }
108         });
109         return deps.contains(child.getId());
110     }
