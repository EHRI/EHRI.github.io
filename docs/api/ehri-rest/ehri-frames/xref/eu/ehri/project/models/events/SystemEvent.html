116     abstract class Impl implements JavaHandlerContext<Vertex>, SystemEvent {
117         public Iterable<AccessibleEntity> getSubjects() {
118             return frameVertices(gremlin().in(Ontology.ENTITY_HAS_EVENT)
119                     .as("n").in(Ontology.ENTITY_HAS_LIFECYCLE_EVENT)
120                     .loop("n", JavaHandlerUtils.noopLoopFunc, new PipeFunction<LoopPipe.LoopBundle<Vertex>, Boolean>() {
121                         @Override
122                         public Boolean compute(LoopPipe.LoopBundle<Vertex> vertexLoopBundle) {
123                             return isValidTarget(vertexLoopBundle.getObject());
124                         }
125                     }));
126         }

128         public AccessibleEntity getFirstSubject() {
129             // Ugh: horrible code duplication is horrible - unfortunately
130             // just calling getSubjects() fails for an obscure reason to do
131             // with Frames not being thinking it has an iterable???
132             GremlinPipeline<Vertex,Vertex> subjects = gremlin().in(Ontology.ENTITY_HAS_EVENT)
133                     .as("n").in(Ontology.ENTITY_HAS_LIFECYCLE_EVENT)
134                     .loop("n", JavaHandlerUtils.noopLoopFunc, new PipeFunction<LoopPipe.LoopBundle<Vertex>, Boolean>() {
135                         @Override
136                         public Boolean compute(LoopPipe.LoopBundle<Vertex> vertexLoopBundle) {
137                             return isValidTarget(vertexLoopBundle.getObject());
138                         }
139                     });
140             return (AccessibleEntity)(subjects.iterator().hasNext()
141                 ? frame(subjects.iterator().next())
142                 : null);
143         }
144 
145         public Actioner getActioner() {
146             GremlinPipeline<Vertex, Vertex> actioners = gremlin().in(Ontology.ENTITY_HAS_EVENT)
147                     .as("n").in(Ontology.ACTIONER_HAS_LIFECYCLE_ACTION)
148                     .loop("n", JavaHandlerUtils.noopLoopFunc, new PipeFunction<LoopPipe.LoopBundle<Vertex>, Boolean>() {
149                         @Override
150                         public Boolean compute(LoopPipe.LoopBundle<Vertex> vertexLoopBundle) {
151                             return isValidActioner(vertexLoopBundle.getObject());
152                         }
153                     });
154             return (Actioner)(actioners.iterator().hasNext()
155                     ? frame(actioners.iterator().next())
156                     : null);
157         }
158 
159         private boolean isValidActioner(Vertex vertex) {
160             return (!vertex.getVertices(Direction.IN,
161                     Ontology.ACTIONER_HAS_LIFECYCLE_ACTION).iterator().hasNext())
162                     && vertex.getProperty(EntityType.TYPE_KEY) != null;
163         }
164 
165         private boolean isValidTarget(Vertex vertex) {
166             return (!vertex.getVertices(Direction.IN,
167                     Ontology.ENTITY_HAS_LIFECYCLE_EVENT).iterator().hasNext())
168                     && vertex.getProperty(EntityType.TYPE_KEY) != null;
169         }
