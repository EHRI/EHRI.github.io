262     /**
263      * Return a Page instance containing a total of total items, and an iterable
264      * for the given page/count.
265      */
266     public <T extends Frame> Page<T> page(Iterable<T> vertices,
267             Accessor user, Class<T> cls) {
268         PipeFunction<Vertex, Boolean> aclFilterFunction = AclManager
269                 .getAclFilterFunction(user);
270 
271         GremlinPipeline<E, Vertex> pipeline = new GremlinPipeline<E, Vertex>(
272                 new FramedVertexIterableAdaptor<>(vertices))
273                 .filter(aclFilterFunction);
274 
275         if (stream) {
276             return new Page<>(graph.frameVertices(
277                     setPipelineRange(setOrder(applyFilters(pipeline))), cls), offset, limit, NO_COUNT);
278         } else {
279             // FIXME: We have to read the vertices into memory here since we
280             // can't re-use the iterator for counting and streaming.
281             ArrayList<Vertex> userVerts = Lists.newArrayList(applyFilters(pipeline).iterator());
282             Iterable<T> iterable = graph.frameVertices(
283                     setPipelineRange(setOrder(new GremlinPipeline<Vertex, Vertex>(
284                             userVerts))), cls);
285             return new Page<>(iterable, offset, limit, userVerts.size());
286         }
287     }

289     /**
290      * Return a Page instance containing a total of total items, and an iterable
291      * for the given page/count.
292      */
293     public Page<E> page(String key, String query, Accessor user) {
294         try (CloseableIterable<Vertex> countQ = manager.getVertices(key, query,
295                 ClassUtils.getEntityType(cls))) {
296             try (CloseableIterable<Vertex> indexQ = manager.getVertices(key,
297                     query, ClassUtils.getEntityType(cls))) {
298                 PipeFunction<Vertex, Boolean> aclFilterFunction = AclManager.getAclFilterFunction(user);
299                 long numItems = stream
300                         ? NO_COUNT
301                         : applyFilters(new GremlinPipeline<Vertex, Vertex>(countQ)
302                         .filter(aclFilterFunction)).count();
303 
304                 return new Page<>(
305                         graph.frameVertices(
306                                 setPipelineRange(setOrder(applyFilters(new GremlinPipeline<Vertex, Vertex>(
307                                         indexQ).filter(aclFilterFunction)))),
308                                 cls), offset, limit, numItems);
309             }
310         }
311     }

412     /**
413      * Add a set of string order clauses. Clauses must be of the form:
414      * <p>
415      * property__DIRECTION
416      *
417      * @param orderSpecs list of orderSpecs
418      */
419     public Query<E> orderBy(Iterable<String> orderSpecs) {
420         SortedMap<String, Sort> tmp = QueryUtils.parseOrderSpecs(orderSpecs);
421         Query<E> query = this;
422         for (Entry<String, Sort> entry : tmp.entrySet()) {
423             Optional<QueryUtils.TraversalPath> tp = QueryUtils.getTraversalPath(entry.getKey());
424             if (tp.isPresent()) {
425                 logger.debug("Adding traversal order: {}", entry.getKey());
426                 query = query.orderByTraversal(tp.get(), entry.getValue());
427             } else {
428                 logger.debug("Adding property order: {}", entry.getKey());
429                 query = query.orderBy(entry.getKey(), entry.getValue());
430             }
431         }
432         return query;
433     }

483      * Add a filter clause.
484      */
485     public Query<E> filter(String property, FilterPredicate predicate,
486             String value) {
487         ImmutableSortedMap.Builder<String, Pair<FilterPredicate, String>> builder
488                 = new ImmutableSortedMap.Builder<>(
489                 Ordering.natural());
490         for (Entry<String, Pair<FilterPredicate, String>> filter : filters.entrySet()) {
491             if (!filter.getKey().equals(property)) {
492                 builder.put(filter.getKey(), filter.getValue());
493             }
494         }
495         builder.put(property, new Pair<>(predicate, value));
496 
497         return new Query<>(graph, cls, scope, offset, limit, sort,
498                 traversalSort, defaultSort, builder.build(), depthFilters, traversalFilters, stream);
499     }

512     /**
513      * Add a set of unparsed filter clauses. Clauses must be of the format:
514      * property__PREDICATE:value
515      * <p>
516      * If PREDICATE is omitted, EQUALS is the default.
517      *
518      * @param filters list of filters
519      * @return a new query object
520      */
521     public Query<E> filter(Iterable<String> filters) {
522         // FIXME: This is really gross, but we want to allow the arguments
523         // to .filter() to be a mix of property filters and traversal path
524         // filters, because they'll usually just come straight from some
525         // query params.
526         SortedMap<String, Pair<FilterPredicate, String>> tmp = QueryUtils.parseFilters(filters);
527         Query<E> query = this;
528         for (Entry<String, Pair<FilterPredicate, String>> ft : tmp.entrySet()) {
529             Optional<QueryUtils.TraversalPath> tpath = QueryUtils.getTraversalPath(ft.getKey());
530             if (tpath.isPresent()) {
531                 logger.debug("Adding traversal filter: {}", tpath.get());
532                 query = query.filterTraversal(tpath.get(), ft.getValue().getA(), ft.getValue().getB());
533             } else {
534                 logger.debug("Adding property filter: {}", ft.getKey());
535                 query = query.filter(ft.getKey(), ft.getValue().getA(), ft.getValue().getB());
536             }
537         }
538         return query;
539     }

543     private <EE> GremlinPipeline<EE, Vertex> setPipelineRange(
544             GremlinPipeline<EE, Vertex> filter) {
545         int low = Math.max(0, offset);
546         if (limit < 0) {
547             // No way to skip a bunch of items in Gremlin without
548             // applying an end range... I guess this will break if
549             // we have more than 2147483647 items to traverse...
550             return filter.range(low, Integer.MAX_VALUE); // No filtering
551         } else if (limit == 0) {
552             return new GremlinPipeline<>(Lists.newArrayList());
553         } else {
554             // NB: The high range is inclusive, oddly.
555             return filter.range(low, low + (limit - 1));
556         }
557     }

559     private <EE> GremlinPipeline<EE, Vertex> setOrder(
560             GremlinPipeline<EE, Vertex> pipe) {
561         pipe = setTraversalOrdering(pipe);
562         if (sort.isEmpty()) {
563             if (defaultSort.isPresent()) {
564                 return pipe
565                         .order(getOrderFunction(new ImmutableSortedMap.Builder<String, Sort>(
566                                 Ordering.natural().nullsLast()).put(
567                                 defaultSort.get().getA(),
568                                 defaultSort.get().getB()).build()));
569             }
570             return pipe;
571         }
572         return pipe.order(getOrderFunction(sort));
573     }

578             pipe = pipe.order(getTraversalOrderFunction(entry.getKey(), entry.getValue()));

604     /**
605      * Get a PipeFunction which sorted vertices by the ordered list
606      * of sort parameters.
607      */
608     private PipeFunction<Pair<Vertex, Vertex>, Integer> getOrderFunction(
609             final SortedMap<String, Sort> sort) {
610         final Ordering<Comparable<?>> order = Ordering.natural().nullsLast();
611         return new PipeFunction<Pair<Vertex, Vertex>, Integer>() {
612             public Integer compute(Pair<Vertex, Vertex> pair) {
613                 ComparisonChain chain = ComparisonChain.start();
614                 for (Entry<String, Sort> entry : sort.entrySet()) {
615                     Vertex a = entry.getValue() == Sort.ASC ? pair.getA() : pair.getB();
616                     Vertex b = entry.getValue() == Sort.ASC ? pair.getB() : pair.getA();
617                     chain = chain.compare(
618                             (String) a.getProperty(entry.getKey()),
619                             (String) b.getProperty(entry.getKey()), order);
620                 }
621                 return chain.result();
622             }
623         };
624     }

626     /**
627      * Create a filter that limits the selected nodes to with a particular depth
628      * of a relationship chain. For example, if a set of nodes form a
629      * hierarchical relationship such as:
630      * <p>
631      * child -[childOf]-&gt; parent -[childOf]-&gt; grandparent
632      * <p>
633      * Then a depthFilter of childOf -&gt; 0 would filter out all except the
634      * grandparent node.
635      * <p>
636      * TODO: Figure out how to do this will Gremlin's loop() construct.
637      */
638     private PipeFunction<Vertex, Boolean> getDepthFilterFunction() {
639         return new PipeFunction<Vertex, Boolean>() {
640             public Boolean compute(Vertex vertex) {
641                 for (Entry<Pair<String, Direction>, Integer> entry : depthFilters.entrySet()) {
642                     int depthCount = 0;
643                     Vertex tmp = vertex;
644                     String label = entry.getKey().getA();
645                     Direction direction = entry.getKey().getB();
646                     while (tmp.getEdges(direction, label)
647                             .iterator().hasNext()) {
648                         tmp = tmp.getVertices(direction, label)
649                                 .iterator().next();
650                         depthCount++;
651                         if (depthCount > entry.getValue()) {
652                             return false;
653                         }
654                     }
655                 }
656                 return true;
657             }
658         };
659     }
660 
661     /**
662      * Create a function that filters nodes given a string and a predicate.
663      */
664     private PipeFunction<Vertex, Boolean> getFilterFunction() {
665         return new PipeFunction<Vertex, Boolean>() {
666             public Boolean compute(Vertex vertex) {
667                 for (Entry<String, Pair<FilterPredicate, String>> entry : filters
668                         .entrySet()) {
669                     String p = vertex.getProperty(entry.getKey());
670                     if (p == null || !matches(p, entry.getValue()
671                             .getB(), entry.getValue().getA())) {
672                         return false;
673                     }
674                 }
675                 return true;
676             }
677         };
678     }

695     private PipeFunction<Pair<Vertex, Vertex>, Integer> getTraversalOrderFunction(
696             QueryUtils.TraversalPath tp, final Sort sort) {
697         // FIXME: Make this less horribly inefficient!
698         final GremlinPipeline<Vertex, String> pipe = getOrderTraversalPipeline(tp);
699         final Map<Vertex, String> cache = Maps.newHashMap();
700         final Ordering<Comparable<?>> order = Ordering.natural().nullsLast();
701         return new PipeFunction<Pair<Vertex, Vertex>, Integer>() {
702             public Integer compute(Pair<Vertex, Vertex> pair) {
703                 String a = cache.get(pair.getA());
704                 String b = cache.get(pair.getB());
705                 if (!cache.containsKey(pair.getA())) {
706                     pipe.reset();
707                     if (pipe.start(pair.getA()).hasNext()) {
708                         a = pipe.next();
709                         cache.put(pair.getA(), a);
710                     }
711                 }
712                 if (!cache.containsKey(pair.getB())) {
713                     pipe.reset();
714                     if (pipe.start(pair.getB()).hasNext()) {
715                         b = pipe.next();
716                         cache.put(pair.getB(), b);
717                     }
718                 }
719                 return sort == Sort.ASC ? order.compare(a, b) : order.compare(b, a);
720             }
721         };

749     /**
750      * Add traversals to a pipeline given a set of String/Direction pairs.
751      */
752     private static <S> GremlinPipeline<S, Vertex> addTraversals(List<Pair<String, Direction>> paths,
753             Boolean reverse, GremlinPipeline<S, Vertex> pipe) {
754         List<Pair<String, Direction>> orderedPaths = reverse ? Lists.reverse(paths) : paths;
755         for (Pair<String, Direction> tp : orderedPaths) {
756             Direction direction = reverse ? tp.getB().opposite() : tp.getB();
757             String relation = tp.getA();
758             switch (direction) {
759                 case IN:
760                     logger.debug("Adding {} relation: {}", relation, direction);
761                     pipe = pipe.in(relation);
762                     break;
763                 case OUT:
764                     logger.debug("Adding {} relation: {}", relation, direction);
765                     pipe = pipe.out(relation);
766                     break;
767                 default:
768                     throw new IllegalArgumentException("Unexpected direction in traversal pipe function: " + tp.getB());
769             }
770         }
771         return pipe;
772     }

775     // FIXME: This has several limitations so far.
776     // - only handles properties cast as strings
777     // - doesn't do case-insensitive regexp matching.
778     private boolean matches(String a, String b, FilterPredicate predicate) {
779         switch (predicate) {
780             case EQUALS:
781                 return a.equals(b);
782             case IEQUALS:
783                 return a.equalsIgnoreCase(b);
784             case STARTSWITH:
785                 return a.startsWith(b);
786             case ENDSWITH:
787                 return a.endsWith(b);
788             case CONTAINS:
789                 return a.contains(b);
790             case ICONTAINS:
791                 return a.toLowerCase().contains(b.toLowerCase());
792             case MATCHES:
793                 return a.matches(b);
794             case GT:
795                 return a.compareTo(b) > 0;
796             case GTE:
797                 return a.compareTo(b) >= 0;
798             case LT:
799                 return a.compareTo(b) < 0;
800             case LTE:
801                 return a.compareTo(b) <= 0;
802             default:
803                 throw new RuntimeException("Unexpected filter predicate: "
804                         + predicate);
805         }
806     }
