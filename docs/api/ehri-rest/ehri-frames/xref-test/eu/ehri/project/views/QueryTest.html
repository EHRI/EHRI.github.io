159     @Test
160     public void testListWithPredicateFilter() throws IndexNotFoundException {
161         Query<DocumentaryUnit> query = new Query<DocumentaryUnit>(graph,
162                 DocumentaryUnit.class);
163 
164         // Query for document identifier c1.
165         List<DocumentaryUnit> list = toList(query.filter(
166                 Ontology.IDENTIFIER_KEY, Query.FilterPredicate.EQUALS,
167                 "c1").page(EntityClass.DOCUMENTARY_UNIT, validUser));
168         assertEquals(1, list.size());
169 
170         // Case-insensitive query
171         list = toList(query.filter(Ontology.IDENTIFIER_KEY,
172                 Query.FilterPredicate.IEQUALS, "C1").page(
173                 EntityClass.DOCUMENTARY_UNIT, validUser));
174         assertEquals(1, list.size());
175 
176         // Startswith...
177         list = toList(query.filter(Ontology.IDENTIFIER_KEY,
178                 Query.FilterPredicate.STARTSWITH, "c").page(
179                 EntityClass.DOCUMENTARY_UNIT, validUser));
180         assertEquals(4, list.size());
181 
182         // Endswith... should get one item (c1)
183         list = toList(query.filter(Ontology.IDENTIFIER_KEY,
184                 Query.FilterPredicate.ENDSWITH, "1").page(
185                 EntityClass.DOCUMENTARY_UNIT, validUser));
186         assertEquals(1, list.size());
187 
188         // Regexp... should get all doc units (c1-4)
189         list = toList(query.filter(Ontology.IDENTIFIER_KEY,
190                 Query.FilterPredicate.MATCHES, "^c\\d+$").page(
191                 EntityClass.DOCUMENTARY_UNIT, validUser));
192         assertEquals(4, list.size());
193 
194         // Less than... should get one item (c1)
195         list = toList(query.filter(Ontology.IDENTIFIER_KEY,
196                 Query.FilterPredicate.LT, "c2").page(
197                 EntityClass.DOCUMENTARY_UNIT, validUser));
198         assertEquals(1, list.size());
199         assertEquals("c1", list.get(0).getIdentifier());
200 
201         // Greater than... should get one item (c4)
202         list = toList(query.filter(Ontology.IDENTIFIER_KEY,
203                 Query.FilterPredicate.GT, "c3").page(
204                 EntityClass.DOCUMENTARY_UNIT, validUser));
205         assertEquals(1, list.size());
206         assertEquals("c4", list.get(0).getIdentifier());
207 
208         // Less than or equal... should get twos items (c1,c2)
209         list = toList(query.filter(Ontology.IDENTIFIER_KEY,
210                 Query.FilterPredicate.LTE, "c2")
211                 .orderBy(Ontology.IDENTIFIER_KEY, Query.Sort.ASC)
212                 .page(EntityClass.DOCUMENTARY_UNIT, validUser));
213         assertEquals(2, list.size());
214         assertEquals("c1", list.get(0).getIdentifier());
215         assertEquals("c2", list.get(1).getIdentifier());
216 
217         // Greater than or equal... should get two items (c3,c4)
218         Query<DocumentaryUnit> fQuery = query.filter(Ontology.IDENTIFIER_KEY,
219                 Query.FilterPredicate.GTE, "c3")
220                 .orderBy(Ontology.IDENTIFIER_KEY, Query.Sort.ASC);
221         list = toList(fQuery
222                 .page(EntityClass.DOCUMENTARY_UNIT, validUser));
223         assertEquals(2, list.size());
224         assertEquals("c3", list.get(0).getIdentifier());
225         assertEquals("c4", list.get(1).getIdentifier());
226         assertEquals(4, toList(fQuery.clearFilters()
227                 .page(EntityClass.DOCUMENTARY_UNIT, validUser)).size());
