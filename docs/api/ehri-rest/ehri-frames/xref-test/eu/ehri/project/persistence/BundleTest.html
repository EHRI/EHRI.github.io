100     @Test
101     public void testEqualsAndHashCode() throws Exception {
102         // A bundle with the same relationship data but
103         // in different orders should be equals()
104         Bundle bundle1 = Bundle.fromData(TestData.getTestDocBundle());
105         Bundle bundle2 = Bundle.fromData(TestData.getTestDocBundle());
106         assertEquals(bundle1, bundle2);
107         assertEquals(bundle1.hashCode(), bundle2.hashCode());
108 
109         // Add a new date period to both and ensure that have a
110         // different structural ordering
111         System.out.println(bundle1);
112         Bundle dp = new Bundle(EntityClass.DATE_PERIOD)
113                 .withDataValue(Ontology.DATE_PERIOD_START_DATE, "1900-01-01")
114                 .withDataValue(Ontology.DATE_PERIOD_END_DATE, "2000-01-01");
115         Bundle currentDp = BundleUtils.getBundle(bundle1, "describes[0]/hasDate[0]");
116         System.out.println("CURRENT: " + currentDp);
117         Bundle b1_2 = BundleUtils
118                 .setBundle(bundle1, "describes[0]/hasDate[-1]", currentDp);
119         Bundle b1_3 = BundleUtils.setBundle(b1_2, "describes[0]/hasDate[0]", dp);
120 
121         Bundle b2_2 = BundleUtils.setBundle(bundle2, "describes[0]/hasDate[-1]", dp);
122         assertEquals(BundleUtils.getBundle(b1_3, "describes[0]/hasDate[1]"),
123                 BundleUtils.getBundle(b2_2, "describes[0]/hasDate[0]"));
124         assertEquals(BundleUtils.getBundle(b1_3, "describes[0]/hasDate[0]"),
125                 BundleUtils.getBundle(b2_2, "describes[0]/hasDate[1]"));
126         assertEquals(b1_3, b2_2);
127     }

156         Bundle.Filter filter = new Bundle.Filter() {

159                 String lang = bundle.getDataValue(Ontology.LANGUAGE);

161                         && (lang != null
162                             && lang.equals("en"));
163             }
164         };
165         Bundle filtered = bundle.filterRelations(filter);
166         assertFalse((Ontology.DESCRIPTION_FOR_ENTITY).isEmpty());
167         assertTrue(filtered.getRelations(Ontology.DESCRIPTION_FOR_ENTITY).isEmpty());
