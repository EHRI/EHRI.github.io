57      @Test
58      public void testConceptHierarchy() throws Exception {
59          // Fruit, Apples and Bananas etc.
60  
61          Map<String, Object> data = new HashMap<String, Object>() {
62              {
63                  put(Ontology.IDENTIFIER_KEY, "fruit");
64              }
65          };
66          Vertex v_fruit = manager.createVertex("fruit_id",
67                  EntityClass.CVOC_CONCEPT, data);
68  
69          data = new HashMap<String, Object>() {
70              {
71                  put(Ontology.IDENTIFIER_KEY, "apples");
72              }
73          };
74          Vertex v_apples = manager.createVertex("applies_id",
75                  EntityClass.CVOC_CONCEPT, data);
76  
77          data = new HashMap<String, Object>() {
78              {
79                  put(Ontology.IDENTIFIER_KEY, "bananas");
80              }
81          };
82          Vertex v_bananas = manager.createVertex("bananas_id",
83                  EntityClass.CVOC_CONCEPT, data);
84  
85          data = new HashMap<String, Object>() {
86              {
87                  put(Ontology.IDENTIFIER_KEY, "trees");
88              }
89          };
90          Vertex v_trees = manager.createVertex("trees_id",
91                  EntityClass.CVOC_CONCEPT, data);
92  
93          // OK, so now we have fruit and more....
94          // See if we can frame them
95          Concept fruit = graph.frame(v_fruit, Concept.class);
96          Concept apples = graph.frame(v_apples, Concept.class);
97          Concept bananas = graph.frame(v_bananas, Concept.class);
98          Concept trees = graph.frame(v_trees, Concept.class);
99  
100         // OK, framed, now construct relations etc.
101         fruit.addNarrowerConcept(apples);
102         fruit.addNarrowerConcept(bananas);
103         graph.getBaseGraph().commit();
104 
105         // fruit should now be the broader concept
106         assertEquals(fruit.getId(), apples.getBroaderConcepts()
107                 .iterator().next().getId());
108         assertEquals(fruit.getId(), bananas.getBroaderConcepts()
109                 .iterator().next().getId());
110 
111         // make a relation to Trees concept
112         apples.addRelatedConcept(trees);
113         graph.getBaseGraph().commit();
114 
115         // is it symmetric?
116         assertEquals(apples.getId(), trees.getRelatedByConcepts()
117                 .iterator().next().getId());
118 
119         // TODO test removal of a relation
120     }
