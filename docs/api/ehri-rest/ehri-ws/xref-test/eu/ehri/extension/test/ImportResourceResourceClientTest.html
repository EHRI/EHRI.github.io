379     private boolean checkExists(String id, String userId) {
380         URI uri = ehriUriBuilder(GenericResource.ENDPOINT, id).build();
381         return callAs(userId, uri).get(ClientResponse.class).getStatus()
382                 == ClientResponse.Status.OK.getStatusCode();
383     }

385     private UriBuilder getImportUrl(String endPoint, String scopeId, String log, boolean tolerant) {
386         return ehriUriBuilder(ImportResource.ENDPOINT, endPoint)
387                 .queryParam(LOG_PARAM, log)
388                 .queryParam(SCOPE_PARAM, scopeId)
389                 .queryParam(TOLERANT_PARAM, String.valueOf(tolerant));
390     }

392     private InputStream getPayloadStream(String... resources)
393             throws URISyntaxException, UnsupportedEncodingException {
394         List<String> paths = Lists.newArrayList();
395         for (String resourceName : resources) {
396             URL resource = Resources.getResource(resourceName);
397             paths.add(new File(resource.toURI()).getAbsolutePath());
398         }
399         String payloadText = Joiner.on("\n").join(paths) + "\n";
400         return new ByteArrayInputStream(
401                 payloadText.getBytes("UTF-8"));
402     }
