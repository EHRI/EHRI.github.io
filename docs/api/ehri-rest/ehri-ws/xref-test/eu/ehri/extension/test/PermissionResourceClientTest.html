68      @Test
69      public void testSettingGlobalPermissionMatrix()
70              throws
71              UniformInterfaceException, IOException {
72  
73          ClientResponse response = jsonCallAs(getAdminUserProfileId(),
74                  ehriUri(ENDPOINT, LIMITED_USER_NAME))
75                  .get(ClientResponse.class);
76  
77          assertStatus(OK, response);
78          String data = response.getEntity(String.class);
79          System.out.println("DATA: " + data);
80  
81          List<Map<String, Map<String, List<String>>>> currentMatrix = getInheritedMatrix(data);
82          // Check we don't ALREADY have DocumentaryUnit -> create/delete perms
83          assertNull(currentMatrix.get(0).get(LIMITED_USER_NAME)
84                  .get(ContentTypes.DOCUMENTARY_UNIT.getName()));
85          assertNull(currentMatrix.get(0).get(LIMITED_USER_NAME)
86                  .get(ContentTypes.DOCUMENTARY_UNIT.getName()));
87  
88          // Set the permission
89          response = jsonCallAs(getAdminUserProfileId(),
90                  ehriUri(ENDPOINT, LIMITED_USER_NAME))
91                  .entity(jsonMapper.writeValueAsBytes(getTestMatrix()))
92                  .post(ClientResponse.class);
93  
94          assertStatus(OK, response);
95  
96          response = jsonCallAs(getAdminUserProfileId(),
97                  ehriUri(ENDPOINT, LIMITED_USER_NAME))
98                  .get(ClientResponse.class);
99  
100         assertStatus(OK, response);
101         data = response.getEntity(String.class);
102         List<Map<String, Map<String, List<String>>>> newMatrix = getInheritedMatrix(data);
103 
104         // Check we don't ALREADY have DocumentaryUnit -> create/delete perms
105         assertTrue(newMatrix.get(0).get(LIMITED_USER_NAME)
106                 .get(ContentTypes.DOCUMENTARY_UNIT.getName())
107                 .contains(PermissionType.CREATE.getName()));
108         assertTrue(newMatrix.get(0).get(LIMITED_USER_NAME)
109                 .get(ContentTypes.DOCUMENTARY_UNIT.getName())
110                 .contains(PermissionType.DELETE.getName()));
111     }

178     @Test
179     public void testSettingScopedPermissions() throws Exception {
180         // Grant permissions for a user to create items within this scope.
181 
182         String r2 = "r2";
183         String r3 = "r3";
184 
185         // The user shouldn't be able to create docs with r2
186         ClientResponse response = jsonCallAs(LIMITED_USER_NAME,
187                 getCreationUriFor(r2)).entity(jsonDocumentaryUnitTestStr)
188                 .post(ClientResponse.class);
189         assertStatus(UNAUTHORIZED, response);
190 
191         // Or r3...
192         response = jsonCallAs(LIMITED_USER_NAME, getCreationUriFor(r3))
193                 .entity(jsonDocumentaryUnitTestStr)
194                 .post(ClientResponse.class);
195         assertStatus(UNAUTHORIZED, response);
196 
197         // Now grant the user permissions to create just within
198         // the scope of r2
199         String permData = "{\"DocumentaryUnit\": [\"create\"]}";
200 
201         URI grantUri = ehriUri(ENDPOINT, LIMITED_USER_NAME, "scope", r2);
202 
203         response = jsonCallAs(getAdminUserProfileId(), grantUri)
204                 .entity(permData)
205                 .post(ClientResponse.class);
206         System.out.println(response.getEntity(String.class));
207         assertStatus(OK, response);
208 
209         // Now creation should succeed...
210         response = jsonCallAs(LIMITED_USER_NAME, getCreationUriFor(r2))
211                 .entity(jsonDocumentaryUnitTestStr)
212                 .post(ClientResponse.class);
213         assertStatus(CREATED, response);
214 
215         // But r3 should still fail...
216         // Or r3...
217         response = jsonCallAs(LIMITED_USER_NAME, getCreationUriFor(r3))
218                 .entity(jsonDocumentaryUnitTestStr)
219                 .post(ClientResponse.class);
220         assertStatus(UNAUTHORIZED, response);
221 
222         // And the user himself should not be able to grant
223         // others the ability to create within that scope.
224         String otherUserName = "linda";
225         String grantPermData = "{\"DocumentaryUnit\": [\"grant\"]}";
226         URI otherGrantUri = ehriUri(ENDPOINT, otherUserName, "scope", r2);
227 
228         response = jsonCallAs(LIMITED_USER_NAME, otherGrantUri).entity(grantPermData)
229                 .post(ClientResponse.class);
230 
231         assertStatus(UNAUTHORIZED, response);
232 
233     }

236     public void testSettingItemPermissions() throws Exception {
237 
238         // Fetch an existing item's data
239         String targetResourceId = "c4";
240         URI targetResourceUri = entityUri(Entities.DOCUMENTARY_UNIT, targetResourceId);
241 
242         ClientResponse response = jsonCallAs(LIMITED_USER_NAME, targetResourceUri)
243                 .get(ClientResponse.class);
244         assertStatus(OK, response);
245 
246         // First try and update the item
247         String testUpdateString = Bundle
248                 .fromString(response.getEntity(String.class))
249                 .withDataValue("testKey", "testValue").toJson();
250 
251         response = jsonCallAs(LIMITED_USER_NAME, targetResourceUri)
252                 .entity(testUpdateString)
253                 .put(ClientResponse.class);
254         assertStatus(UNAUTHORIZED, response);
255 
256         // Now grant the user permissions to update and delete just on this item
257         String permData = "[\"update\", \"delete\"]";
258 
259         // Set the permission
260         response = jsonCallAs(getAdminUserProfileId(),
261                 ehriUri(ENDPOINT,
262                         LIMITED_USER_NAME, "item", targetResourceId))
263                 .entity(permData)
264                 .post(ClientResponse.class);
265 
266         assertStatus(OK, response);
267 
268         // Retry the create action
269         response = jsonCallAs(LIMITED_USER_NAME, targetResourceUri).entity(testUpdateString)
270                 .put(ClientResponse.class);
271         // Should get UPDATED this time...
272         assertStatus(OK, response);
273 
274         // Finally, delete the item
275         response = jsonCallAs(LIMITED_USER_NAME, targetResourceUri)
276                 .delete(ClientResponse.class);
277 
278         // Should get NO_CONTENT this time...
279         assertStatus(NO_CONTENT, response);
280     }
