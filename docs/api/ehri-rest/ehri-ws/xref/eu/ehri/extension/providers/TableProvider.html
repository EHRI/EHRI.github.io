50  /**
51   * Provider for row/column data.
52   * <p>
53   * TODO: Handle CSV headers
54   */
55  @Provider
56  @Produces({MediaType.APPLICATION_JSON, "text/csv"})
57  @Consumes({MediaType.APPLICATION_JSON, "text/csv"})
58  public class TableProvider implements MessageBodyWriter<Table>, MessageBodyReader<Table>, JsonMessageBodyHandler {
59  
60      private static CsvMapper csvMapper = new CsvMapper()
61              .enable(CsvGenerator.Feature.STRICT_CHECK_FOR_QUOTING)
62              .enable(CsvParser.Feature.WRAP_AS_ARRAY);
63      private static CsvSchema csvSchema = csvMapper.schemaFor(List.class);
64  
65      private static TypeReference<List<List<String>>> typeRef = new TypeReference<List<List<String>>>() {
66      };
67  
68      @Override
69      public boolean isWriteable(Class<?> type, Type genericType, Annotation[] annotations, MediaType mediaType) {
70          return Table.class.equals(type);
71      }
72  
73      @Override
74      public long getSize(Table table, Class<?> aClass, Type type, Annotation[] annotations,
75              MediaType mediaType) {
76          return -1L;
77      }
78  
79      @Override
80      public void writeTo(Table table,
81              Class<?> aClass, Type type, Annotation[] annotations,
82              MediaType mediaType, MultivaluedMap<String, Object> headers,
83              OutputStream outputStream) throws IOException, WebApplicationException {
84          ObjectWriter writer = mediaType.isCompatible(MediaType.APPLICATION_JSON_TYPE)
85                  ? mapper.writer()
86                  : csvMapper.writer(csvSchema.withoutHeader());
87          writer.writeValue(outputStream, table.rows());
88      }
89  
90      @Override
91      public boolean isReadable(Class<?> type, Type genericType, Annotation[] annotations, MediaType mediaType) {
92          return Table.class.equals(type);
93      }
94  
95      @Override
96      public Table readFrom(Class<Table> type, Type genericType, Annotation[] annotations, MediaType mediaType, MultivaluedMap<String, String> httpHeaders, InputStream entityStream) throws IOException, WebApplicationException {
97          try {
98              ObjectReader reader = mediaType.isCompatible(MediaType.APPLICATION_JSON_TYPE)
99                      ? mapper.readerFor(typeRef)
100                     : csvMapper.readerFor(typeRef).with(csvSchema.withoutHeader());
101             return Table.of(reader.readValue(entityStream));
102         } catch (JsonMappingException | JsonParseException e) {
103             throw new WebDeserializationError(new DeserializationError(
104                     "Deserialization error for media type: " + mediaType, e));
105         }
106     }
