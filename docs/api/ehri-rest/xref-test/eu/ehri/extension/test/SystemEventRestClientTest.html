123     @Test
124     public void testPersonalisedEventList() throws Exception {
125 
126         // Create an event by updating an item...
127         ClientResponse response = jsonCallAs(getAdminUserProfileId(),
128                 ehriUri(Entities.REPOSITORY, "r1"))
129                 .entity(jsonAgentTestString)
130                 .header(AbstractRestResource.LOG_MESSAGE_HEADER_NAME, "Testing update")
131                 .put(ClientResponse.class);
132         assertStatus(OK, response);
133 
134         // At present, the personalised event stream for the validUser user should
135         // contain all the regular events.
136         String user = getRegularUserProfileId();
137         String personalisedEventUrl = "/" + Entities.SYSTEM_EVENT + "/aggregateForUser/" + user;
138         List<List<Map<String, Object>>> events = getItemListOfLists(personalisedEventUrl, user);
139         assertEquals(1, events.size());
140 
141         // Now only fetch events related to items we're watching - this list
142         // should currently be empty...
143         String personalisedEventUrlWatched = personalisedEventUrl + "?" + SystemEventResource.SHOW_PARAM + "=" + EventViews.ShowType.watched;
144         events = getItemListOfLists(personalisedEventUrlWatched, user);
145         assertEquals(0, events.size());
146 
147         // Now start watching item r1.
148         URI watchUrl = UriBuilder.fromPath(getExtensionEntryPointUri())
149                 .segment(Entities.USER_PROFILE)
150                 .segment(user)
151                 .segment(WATCHING)
152                 .queryParam(ID_PARAM, "r1").build();
153 
154         jsonCallAs(user, watchUrl).post(ClientResponse.class);
155 
156         // Now our event list should contain one item, the update
157         // we did initially.
158         events = getItemListOfLists(personalisedEventUrlWatched, user);
159         assertEquals(1, events.size());
160 
161         // Stop watching item r1, which should empty the list
162         jsonCallAs(user, watchUrl).delete(ClientResponse.class);
163 
164         events = getItemListOfLists(personalisedEventUrlWatched, user);
165         assertEquals(0, events.size());
166 
167         // Only get events for people we follow, excluding those
168         // for items we watch...
169         String personalisedEventUrlFollowed = personalisedEventUrl + "?" + SystemEventResource.SHOW_PARAM
170                 + "=" + EventViews.ShowType.followed;
171         events = getItemListOfLists(personalisedEventUrlFollowed, user);
172         assertEquals(0, events.size());
173 
174         // Now follow the other user...
175         URI followUrl = UriBuilder.fromPath(getExtensionEntryPointUri())
176                 .segment(Entities.USER_PROFILE)
177                 .segment(user)
178                 .segment(FOLLOWING)
179                 .queryParam(ID_PARAM, getAdminUserProfileId()).build();
180         jsonCallAs(user, followUrl).post(ClientResponse.class);
181 
182         // We should get the event again...
183         events = getItemListOfLists(personalisedEventUrlFollowed, user);
184         assertEquals(1, events.size());
185     }
