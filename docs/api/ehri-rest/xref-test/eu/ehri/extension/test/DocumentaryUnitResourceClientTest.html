100         // Create
101         ClientResponse response = jsonCallAs(getAdminUserProfileId(),
102                 entityUri(Entities.DOCUMENTARY_UNIT, "c1"))

115         assertThat(c4cc, not(containsString("no-store")));
116         assertThat(c4cc, containsString("max-age=" + AbstractResource.ITEM_CACHE_TIME));

136     public void testIntegrityError() throws Exception {
137         // Create
138         ClientResponse response = jsonCallAs(getAdminUserProfileId(), getCreationUri())
139                 .entity(jsonDocumentaryUnitTestStr).post(ClientResponse.class);
140 
141         assertStatus(CREATED, response);
142 
143         // Okay... now if we try and do the same things again we should
144         // get an integrity error because the identifiers are the same.
145         response = jsonCallAs(getAdminUserProfileId(), getCreationUri())
146                 .entity(jsonDocumentaryUnitTestStr).post(ClientResponse.class);
147         // Check the JSON gives use the correct error
148         String errString = response.getEntity(String.class);
149         assertStatus(BAD_REQUEST, response);
150 
151         JsonNode rootNode = jsonMapper.readTree(errString);
152         assertEquals(BAD_REQUEST.getReasonPhrase(), rootNode.path("error").asText());
153         JsonNode errValue = rootNode.path("details").path(ErrorSet.ERROR_KEY).path(
154                 Ontology.IDENTIFIER_KEY);
155         assertFalse(errValue.isMissingNode());
156     }

261         // -create data for testing, making this a child element of c1.
262         WebResource resource = client.resource(getCreationUri());
263         ClientResponse response = resource
264                 .accept(MediaType.APPLICATION_JSON)
265                 .type(MediaType.APPLICATION_JSON)

267                         getAdminUserProfileId())
268                 .entity(jsonDocumentaryUnitTestStr).post(ClientResponse.class);
269 
270         assertStatus(CREATED, response);
271         assertValidJsonData(response);
272         // response.getEntity(String.class)
273 
274         // Get created doc via the response location?
275         URI location = response.getLocation();
276 
277         resource = client.resource(location);
278         response = resource
279                 .accept(MediaType.APPLICATION_JSON)
280                 .header(AbstractResource.AUTH_HEADER_NAME,
281                         getAdminUserProfileId()).get(ClientResponse.class);
282         assertStatus(OK, response);
283 
284         // -get the data and change it
285         String json = response.getEntity(String.class);
286         String toUpdateJson = Bundle.fromString(json)
287                 .withDataValue("name", UPDATED_NAME).toJson();
288 
289         // -update
290         resource = client.resource(location);
291         response = resource
292                 .accept(MediaType.APPLICATION_JSON)
293                 .type(MediaType.APPLICATION_JSON)
294                 .header(AbstractResource.AUTH_HEADER_NAME,
295                         getAdminUserProfileId()).entity(toUpdateJson)
296                 .put(ClientResponse.class);
297         assertStatus(OK, response);
298 
299         // -get the data and convert to a bundle, is it changed?
300         resource = client.resource(location);
301         response = resource
302                 .accept(MediaType.APPLICATION_JSON)
303                 .header(AbstractResource.AUTH_HEADER_NAME,
304                         getAdminUserProfileId()).get(ClientResponse.class);
305         assertStatus(OK, response);
306 
307         // -get the data and convert to a bundle, is it OK?
308         String updatedJson = response.getEntity(String.class);
309         Bundle updatedEntityBundle = Bundle.fromString(updatedJson);
310         assertEquals(UPDATED_NAME, updatedEntityBundle.getDataValue("name"));
311     }

313     @Test
314     public void testPatchDocumentaryUnit() throws Exception {
315 
316         // -create data for testing, making this a child element of c1.
317         WebResource resource = client.resource(getCreationUri());
318         ClientResponse response = resource
319                 .accept(MediaType.APPLICATION_JSON)
320                 .type(MediaType.APPLICATION_JSON)
321                 .header(AbstractResource.AUTH_HEADER_NAME,
322                         getAdminUserProfileId())
323                 .entity(jsonDocumentaryUnitTestStr).post(ClientResponse.class);
324 
325         assertStatus(CREATED, response);
326 
327         // Get created doc via the response location?
328         URI location = response.getLocation();
329 
330         String toUpdateJson = partialJsonDocumentaryUnitTestStr;
331 
332         // - patch the data (using the Patch header)
333         resource = client.resource(location);
334         response = resource
335                 .accept(MediaType.APPLICATION_JSON)
336                 .type(MediaType.APPLICATION_JSON)
337                 .header(AbstractResource.AUTH_HEADER_NAME,
338                         getAdminUserProfileId())
339                 .header(AbstractResource.PATCH_HEADER_NAME, Boolean.TRUE.toString())
340                 .entity(toUpdateJson)
341                 .put(ClientResponse.class);
342         assertStatus(OK, response);
343 
344         // -get the data and convert to a bundle, is it patched?
345         resource = client.resource(location);
346         response = resource
347                 .accept(MediaType.APPLICATION_JSON)
348                 .header(AbstractResource.AUTH_HEADER_NAME,
349                         getAdminUserProfileId())
350                 .get(ClientResponse.class);
351         assertStatus(OK, response);
352 
353         // -get the data and convert to a bundle, is it OK?
354         String updatedJson = response.getEntity(String.class);
355         Bundle updatedEntityBundle = Bundle.fromString(updatedJson);
356         assertEquals(CREATED_ID, updatedEntityBundle.getDataValue(Ontology.IDENTIFIER_KEY));
357         assertEquals(PARTIAL_NAME, updatedEntityBundle.getDataValue(Ontology.NAME_KEY));
358     }
