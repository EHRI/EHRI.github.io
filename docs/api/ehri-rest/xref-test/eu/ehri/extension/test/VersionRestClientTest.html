30  import static com.sun.jersey.api.client.ClientResponse.Status.OK;
31  import static org.junit.Assert.assertFalse;
32  
33  public class VersionRestClientTest extends AbstractRestClientTest {
34  
35      @Test
36      public void testGetVersionsForItem() throws Exception {
37          // Create an item
38          String jsonAgentTestString = "{\"type\": \"Repository\", \"data\":{\"identifier\": \"jmp\"}}";
39          ClientResponse response = jsonCallAs(getAdminUserProfileId(),
40                  entityUri(Entities.REPOSITORY, "r1"))
41                  .entity(jsonAgentTestString)
42                  .put(ClientResponse.class);
43          assertStatus(OK, response);
44  
45          response = jsonCallAs(getAdminUserProfileId(),
46                  ehriUri(GenericResource.ENDPOINT, "r1", GenericResource.VERSIONS))
47                  .get(ClientResponse.class);
48  
49          String json = response.getEntity(String.class);
50          // Check the response contains a new version
51          JsonNode rootNode = jsonMapper.readTree(json);
52          assertFalse(rootNode.path(0).path(Bundle.DATA_KEY)
53                  .path(Ontology.VERSION_ENTITY_DATA).isMissingNode());
54          assertStatus(OK, response);
55      }
