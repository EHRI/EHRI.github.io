22  import eu.ehri.extension.test.helpers.ServerRunner;
23  import org.junit.After;
24  import org.junit.AfterClass;
25  import org.junit.Before;
26  import org.junit.BeforeClass;
27  
28  import java.io.IOException;

30  /**
31   * Test base class which starts a Neo4j server and loads the
32   * resource classes as an unmanaged extension.
33   */
34  public abstract class RunningServerTest {
35  
36      // Test server port - different from Neo4j default to prevent collisions.
37      final static private Integer testServerPort = 7575;
38  
39      // Mount point for EHRI resources
40      final static private String mountPoint = "ehri";
41  
42      private final static ServerRunner runner
43              = ServerRunner.getInstance(testServerPort, "eu.ehri.extension", mountPoint);
44  
45      @BeforeClass
46      public static void setUpBeforeClass() throws IOException {
47          runner.start();
48      }
49  
50      @AfterClass
51      public static void shutdownDatabase() {
52          runner.stop();
53      }
54  
55      String getExtensionEntryPointUri() {
56          return runner.getServer().baseUri() + mountPoint;
57      }
58  
59      @Before
60      public void setupDb() {
61          runner.setUpData();
62      }
63  
64      @After
65      public void resetDb() {
66          runner.tearDownData();
67      }
68  }
