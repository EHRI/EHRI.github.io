504     private UriBuilder getImportUrl(String endPoint, String scopeId, String log, boolean tolerant) {
505         return ehriUriBuilder(ImportResource.ENDPOINT, endPoint)
506                 .queryParam(LOG_PARAM, log)
507                 .queryParam(SCOPE_PARAM, scopeId)
508                 .queryParam(TOLERANT_PARAM, String.valueOf(tolerant));

511     private InputStream getPayloadStream(String... resources)
512             throws URISyntaxException, UnsupportedEncodingException {
513         List<String> paths = Lists.newArrayList();
514         for (String resourceName : resources) {
515             URL resource = Resources.getResource(resourceName);
516             paths.add(Paths.get(resource.toURI()).toAbsolutePath().toString());
517         }
518         String payloadText = Joiner.on("\n").join(paths) + "\n";
519         return new ByteArrayInputStream(
520                 payloadText.getBytes(Charsets.UTF_8));
521     }
