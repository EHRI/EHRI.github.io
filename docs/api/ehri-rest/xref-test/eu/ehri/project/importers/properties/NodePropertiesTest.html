30  public class NodePropertiesTest {
31      private static final org.slf4j.Logger logger = LoggerFactory.getLogger(NodePropertiesTest.class);
32      
33      NodeProperties pc;
34      @Before
35      public void init(){
36          pc  = new NodeProperties();
37          pc.setTitles("node,property,handlerTempName,required?,multivalued?");
38          pc.addRow("description,identifier,descriptionIdentifier,1,");
39          pc.addRow("authorityDescription,otherFormsOfName,,,1");
40          pc.addRow("description,languageCode,,1,");
41          pc.addRow("description,name,,1,");
42      } 
43      public NodePropertiesTest() {
44      }
45  
46      @Test
47      public void descriptionHasIdentifier() {
48          assertTrue(pc.hasProperty("description", "identifier"));
49          assertFalse(pc.hasProperty("description", "descriptionIdentifier"));
50      }
51      
52      @Test
53      public void descriptionMustHaveIdentifier() {
54          assertTrue(pc.isRequiredProperty("description", "identifier"));
55          assertFalse(pc.isRequiredProperty("description", "descriptionIdentifier"));
56          assertFalse(pc.isRequiredProperty("authorityDescription", "otherFormsOfName"));
57      }
58      
59      @Test
60      public void otherFormsOfNameIsMultivalued(){
61          assertTrue(pc.isMultivaluedProperty("authorityDescription", "otherFormsOfName"));
62          assertFalse(pc.isMultivaluedProperty("authorityDescription", "something"));
63      }
64      
65      @Test
66      public void descriptionIdentifierIsHandlerName(){
67          assertEquals("descriptionIdentifier", pc.getHandlerName("description", "identifier"));
68      }
69      
70      @Test
71      public void getAttributeWithValue(){
72           String propfile = "bundesarchive.properties";
73          logger.debug(propfile);
74          XmlImportProperties f = new XmlImportProperties(propfile);
75          
76          assertTrue(f.containsProperty("scopecontent/"));
77  //        phystech/p/=physicalCharacteristics
78          assertTrue(f.containsProperty("phystech/p/"));
79          assertEquals("physicalCharacteristics", f.getProperty("phystech/p/"));
80  
81          //did/unitid/@ehrilabel$ehri_main_identifier=objectIdentifier
82  
83          assertTrue(f.containsProperty("did/unitid/@encoding$Bestandssignatur"));
84          assertEquals("objectIdentifier", f.getProperty("did/unitid/@encoding$Bestandssignatur"));
85      }
86      
87      @Test
88      public void getPropertyWithoutSchema(){
89          String propfile = "dceuropeana.properties";
90          logger.debug(propfile);
91          XmlImportProperties f = new XmlImportProperties(propfile);
92          
93          assertTrue(f.containsProperty("identifier/"));
94          assertEquals("objectIdentifier", f.getProperty("identifier/"));
95          
96      }
