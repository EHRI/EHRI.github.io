41      private static final Logger logger = LoggerFactory.getLogger(PersonalitiesImporterTest.class);
42      protected final String SINGLE_EAD = "ukraine_small.csv";
43  
44      @Test
45      public void testImportItemsT() throws Exception {
46          Bundle ukrainianBundle = new Bundle(EntityClass.COUNTRY)
47                                  .withDataValue(Country.COUNTRY_CODE, "ua");
48          Bundle repoBundle = new Bundle(EntityClass.REPOSITORY).withDataValue(Ontology.IDENTIFIER_KEY, "ua-3311");
49          Bundle repoDescBundle = new Bundle(EntityClass.REPOSITORY_DESCRIPTION)
50                  .withDataValue(Ontology.NAME_KEY, "Центральний державний архів вищих органів влади і управління України")
51                  .withDataValue("otherFormsOfName", "Central State Archive of Supreme Bodies of Power and Government of Ukraine")
52                  .withDataValue(Ontology.LANGUAGE_OF_DESCRIPTION, "uk");
53          repoBundle = repoBundle.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, repoDescBundle);
54          repoBundle = repoBundle.withRelation(Ontology.REPOSITORY_HAS_COUNTRY, ukrainianBundle);
55          Repository repo = api(validUser).create(repoBundle, Repository.class);
56  
57          int count = getNodeCount(graph);
58          
59          final String logMessage = "Importing some Ukrainian records";
60          XmlImportProperties p = new XmlImportProperties("ukraine.properties");
61          assertTrue(p.containsProperty("identifier"));
62          assertTrue(p.containsProperty("level_of_description"));
63          assertTrue(p.containsProperty("project_judaica"));
64          
65          InputStream ios = ClassLoader.getSystemResourceAsStream(SINGLE_EAD);
66          ImportLog log = new CsvImportManager(graph, repo, validUser, false, false,
67                      UkrainianUnitImporter.class)
68                  .importInputStream(ios, logMessage);
69          assertTrue(log.hasDoneWork());
70  
71          /*
72           * 17 DocumentaryUnits
73           * 17 + 3 DocumentsDescription (there are 3 desc's with 2 languages)
74           * 17 + 3 DatePeriods
75           * 18 more import Event links (1 for every Unit, 1 for the User)
76           * 1 more import Event
77           */
78          assertEquals(count + 76, getNodeCount(graph));
79          printGraph(graph);
80      }
