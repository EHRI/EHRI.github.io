44  public class Eag2896Test extends AbstractImporterTest {
45  
46      private static final org.slf4j.Logger logger = LoggerFactory.getLogger(Eag2896Test.class);
47      protected final String SINGLE_UNIT = "eag-2896.xml";
48      // Depends on fixtures
49      protected final String TEST_COUNTRY = "r1";
50      // Depends on SINGLE_UNIT
51      protected final String IMPORTED_ITEM_ID = "NL-002896";
52      protected final String AGENT_DESC_ID = "NL-002896#desc";
53  
54      @Test
55      public void testImportItemsT() throws Exception {
56          Country country = manager.getFrame(TEST_COUNTRY, Country.class);
57          final String logMessage = "Importing a single EAG";
58  
59          int count = getNodeCount(graph);
60          logger.info("count of nodes before importing: " + count);
61  
62          InputStream ios = ClassLoader.getSystemResourceAsStream(SINGLE_UNIT);
63          SaxImportManager importManager = new SaxImportManager(graph, country, validUser, EagImporter.class, EagHandler.class);
64          ImportLog log = importManager.importFile(ios, logMessage);
65          //printGraph(graph);
66          // How many new nodes will have been created? We should have
67          // - 1 more Repository
68          // - 1 more RepositoryDescription
69          // - 1 more Address
70          // - 1 more UnknownProperty
71          // - 2 more MaintenanceEvent
72          // - 2 more linkEvents (1 for the Repository, 1 for the User)
73          // - 1 more SystemEvent
74  
75          int afterCount = count + 9;
76          assertEquals(afterCount, getNodeCount(graph));
77  
78          Iterable<Vertex> docs = graph.getVertices(Ontology.IDENTIFIER_KEY, IMPORTED_ITEM_ID);
79          assertTrue(docs.iterator().hasNext());
80          Repository unit = graph.frame(getVertexByIdentifier(graph, IMPORTED_ITEM_ID), Repository.class);
81          assertEquals(Entities.REPOSITORY, unit.getType());
82  
83          // check the child items
84          RepositoryDescription c1 = graph.frame(getVertexByIdentifier(graph, AGENT_DESC_ID), RepositoryDescription.class);
85          assertEquals(Entities.REPOSITORY_DESCRIPTION, c1.getType());
86          Object notes = c1.asVertex().getProperty(EagImporter.MAINTENANCE_NOTES);
87          if (notes instanceof String[]) {
88              fail("Maintenance notes property should not be an array");
89          } else {
90              assertTrue(notes instanceof String);
91          }
92  
93          // MB: Test priority hack - this should be pulled out of the
94          // maintenanceNotes field into its own int field
95          Object priority = unit.asVertex().getProperty(EagImporter.PRIORITY);
96          assertEquals(Integer.valueOf(5), priority);
97  
98          // Check scope
99          assertEquals(country, unit.getPermissionScope());
100 
101         //check whether the description has an Address attached to it
102         assertEquals(1, toList(c1.getAddresses()).size());
103 
104         assertEquals(2, toList(c1.getMaintenanceEvents()).size());
105         // Ensure that c1 is a description of the unit
106         for (Description d : unit.getDescriptions()) {
107             assertEquals(IMPORTED_ITEM_ID, d.getDescribedEntity().getIdentifier());
108         }
109 
110         // Check we've only got one action
111         assertEquals(1, log.getCreated());
112         SystemEvent ev = actionManager.getLatestGlobalEvent();
113         assertEquals(logMessage, ev.getLogMessage());
114 
115         // Ensure the import action has the right number of subjects.
116         List<AccessibleEntity> subjects = toList(ev.getSubjects());
117         assertEquals(1, subjects.size());
118         assertEquals(log.getChanged(), subjects.size());
119 
120         // Test idempotency
121         int edgeCount = getEdgeCount(graph);
122         ImportLog log2 = importManager.importFile(ClassLoader.getSystemResourceAsStream(SINGLE_UNIT), logMessage);
123         assertFalse(log2.hasDoneWork());
124         assertEquals(afterCount, getNodeCount(graph));
125         assertEquals(edgeCount, getEdgeCount(graph));
126     }
