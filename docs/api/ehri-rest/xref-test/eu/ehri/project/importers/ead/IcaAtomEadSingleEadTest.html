39  public class IcaAtomEadSingleEadTest extends AbstractImporterTest {
40      protected final String SINGLE_EAD = "single-ead.xml";
41  
42      // Depends on single-ead.xml
43      protected final String IMPORTED_ITEM_ID = "C00001";
44  
45      @Test
46      public void testImportItemsT() throws Exception {
47          Repository agent = manager.getEntity(TEST_REPO, Repository.class);
48          final String logMessage = "Importing a single EAD by IcaAtomEadSingleEad";
49  
50          int origCount = getNodeCount(graph);
51  
52          InputStream ios = ClassLoader.getSystemResourceAsStream(SINGLE_EAD);
53          ImportLog log = saxImportManager(IcaAtomEadImporter.class, IcaAtomEadHandler.class)
54                  .importFile(ios, logMessage);
55  
56          printGraph(graph);
57          // How many new nodes will have been created? We should have
58          // - 1 more DocumentaryUnit
59          // - 1 more DocumentDescription
60          // - 2 more DatePeriod
61          //TODO: test these UR's
62          // - 5 more UndeterminedRelationships
63          //TODO: test this UP
64          // - 1 more UnknownProperty
65          // - 2 more import Event links
66          // - 1 more import Event
67  
68          Iterable<Vertex> docs = graph.getVertices("identifier", IMPORTED_ITEM_ID);
69          assertTrue(docs.iterator().hasNext());
70          DocumentaryUnit unit = graph.frame(docs.iterator().next(), DocumentaryUnit.class);
71          for (Description d : unit.getDocumentDescriptions())
72              assertEquals("Test EAD Item", d.getName());
73  
74          // Test scope and content has correctly decoded data.
75          // This tests two bugs:
76          //  Space stripping: https://github.com/mikesname/ehri-rest/issues/12
77          //  Paragraph ordering: https://github.com/mikesname/ehri-rest/issues/13
78          Description firstDesc = unit.getDocumentDescriptions().iterator().next();
79          String scopeContent = firstDesc.getProperty("scopeAndContent");
80          String expected =
81                  "This is some test scope and content.\n\n" +
82                          "This contains Something & Something else.\n\n" +
83                          "This is another paragraph.";
84  
85          assertEquals(expected, scopeContent);
86  
87          // Check the right nodes get created.
88          int createCount = origCount + 13;
89  
90          // - 4 more UnderterminedRelationship nodes
91  
92          assertEquals(createCount, getNodeCount(graph));
93  
94          // Yet we've only created 1 *logical* item...
95          assertEquals(1, log.getChanged());
96  
97          List<SystemEvent> actions = toList(unit.getHistory());
98          // Check we've only got one action
99          assertEquals(1, actions.size());
100         assertEquals(logMessage, actions.get(0).getLogMessage());
101 
102         // Now re-import the same file
103         InputStream ios2 = ClassLoader.getSystemResourceAsStream(SINGLE_EAD);
104         ImportLog log2 = saxImportManager(IcaAtomEadImporter.class, IcaAtomEadHandler.class)
105                 .importFile(ios2, logMessage);
106 
107         // We should no new nodes (not even a SystemEvent)
108         assertEquals(createCount, getNodeCount(graph));
109         // And no logical item should've been updated
110         assertEquals(0, log2.getUpdated());
111 
112         // Check permission scopes
113         for (Accessible e : actionManager.getLatestGlobalEvent().getSubjects()) {
114             assertEquals(agent, e.getPermissionScope());
115         }
116     }
