142     @Test
143     public void testUpdate() throws Exception {
144         CrudViews<DocumentaryUnit> docViews = new CrudViews<DocumentaryUnit>(
145                 graph, DocumentaryUnit.class);
146         Bundle bundle = Bundle.fromData(TestData.getTestDocBundle());
147         DocumentaryUnit unit = docViews.create(bundle, validUser);
148         assertEquals(TestData.TEST_COLLECTION_NAME, unit.asVertex().getProperty("name"));
149 
150         String newName = TestData.TEST_COLLECTION_NAME + " with new stuff";
151         Bundle newBundle = bundle.withId(unit.getId()).withDataValue(
152                 "name", newName);
153 
154         DocumentaryUnit changedUnit = docViews.update(newBundle, validUser).getNode();
155         assertEquals(newName, changedUnit.asVertex().getProperty("name"));
156         DocumentDescription desc = graph.frame(
157                 changedUnit.getDescriptions().iterator().next().asVertex(),
158                 DocumentDescription.class);
159 
160         // Check the nested item was created correctly
161         DatePeriod datePeriod = desc.getDatePeriods().iterator().next();
162         assertTrue(datePeriod != null);
163         assertEquals(TestData.TEST_START_DATE, datePeriod.getStartDate());
164 
165         // And that the reverse relationship works.
166         assertEquals(desc.asVertex(), datePeriod.getEntity().asVertex());
167     }

215         CrudViews<DocumentaryUnit> docViews = new CrudViews<DocumentaryUnit>(
216                 graph, DocumentaryUnit.class);
217         Integer shouldDelete = 1;

219         // FIXME: Surely there's a better way of doing this???
220         Iterator<Description> descIter = item.getDescriptions().iterator();
221         for (; descIter.hasNext(); shouldDelete++) {
222             DocumentDescription d = graph.frame(descIter.next().asVertex(), DocumentDescription.class);
