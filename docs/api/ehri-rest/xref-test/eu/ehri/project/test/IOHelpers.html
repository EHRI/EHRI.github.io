18       * Create a zip file containing the named rsources
19       *
20       * @param file      a file object (typically a temp file)
21       * @param resources the resource names
22       * @throws URISyntaxException
23       * @throws IOException
24       */
25      public static void createZipFromResources(File file, String... resources)
26              throws URISyntaxException, IOException {
27          try (FileOutputStream fos = new FileOutputStream(file);
28               ZipOutputStream zos = new ZipOutputStream(fos)) {
29              for (String resource : resources) {
30                  URL url = Resources.getResource(resource);
31                  String name = new File(url.toURI()).getAbsolutePath();
32                  zos.putNextEntry(new ZipEntry(name));
33                  Resources.copy(url, zos);
34                  zos.closeEntry();
35              }
36          }
37      }
