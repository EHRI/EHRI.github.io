78      protected static class VertexProxy {
79          final String id;
80          final String type;
81          final Object gid;
82          final Map<String,Object> data;
83          public VertexProxy(Vertex v) {
84              this.id = v.getProperty(EntityType.ID_KEY);
85              this.type = v.getProperty(EntityType.TYPE_KEY);
86              this.gid = v.getId();
87              data = Maps.newHashMap();
88              for (String k : v.getPropertyKeys()) {
89                  if (!(k.equals(EntityType.ID_KEY) || k.equals(EntityType.TYPE_KEY))) {
90                      data.put(k, v.getProperty(k));
91                  }
92              }
93          }
94          public String toString() {
95              return "<" + id + " (" + type + ") [" + gid + "]>";
96          }
97  
98          public String toStringVerbose() {
99              StringBuilder builder = new StringBuilder(toString());
100             builder.append("\n");
101             for (Map.Entry<String,Object> entry : data.entrySet()) {
102                 builder.append(String.format("  %-15s : %s\n", entry.getKey(), entry.getValue()));
103             }
104             return builder.toString();
105         }

140         private Map<String, Integer> countDistinctTypes(Set<VertexProxy> vs){
141             Map<String, Integer> counts = new HashMap<String, Integer>();
142             for(VertexProxy v : vs){
143                 if(counts.containsKey(v.type))
144                     counts.put(v.type, counts.get(v.type) + 1);
145                 else
146                     counts.put(v.type, 1);
147             }
148             return counts;
149         }

151         public void printDebug(PrintStream printStream, boolean verbose) {
152             if (added.isEmpty() && removed.isEmpty()) {
153                 printStream.println("GraphDiff - No nodes added or removed");
154             } else {
155                 printStream.println("GraphDiff - Nodes added: " + added.size());
156                 for (VertexProxy proxy : added) {
157                     printStream.println(verbose ? proxy.toStringVerbose() : proxy.toString());
158                 }
159                 Map<String, Integer> counts = countDistinctTypes(added);
160                 for(String key: counts.keySet()){
161                     printStream.println(key + ": " + counts.get(key));
162                 }
163                 printStream.println("GraphDiff - Nodes removed: " + removed.size());
164                 for (VertexProxy proxy : removed) {
165                     printStream.println(verbose ? proxy.toStringVerbose() : proxy.toString());
166                 }
167                 counts = countDistinctTypes(removed);
168                 for(String key: counts.keySet()){
169                     printStream.println(key + ": " + counts.get(key));
170                 }
171             }
172         }
