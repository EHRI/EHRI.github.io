100     private String testExport(DocumentaryUnit unit, String lang) throws Exception {
101         Ead2002Exporter exporter = new Ead2002Exporter(graph);
102         ByteArrayOutputStream baos = new ByteArrayOutputStream();
103         exporter.export(unit, baos, lang);
104         String xml = baos.toString("UTF-8");
105         isValidEad(xml);
106         return xml;
107     }

109     private String testImportExport(Repository repository, String resourceName,
110             String topLevelIdentifier, String lang) throws Exception {
111         InputStream ios = ClassLoader.getSystemResourceAsStream(resourceName);
112         ImportManager importManager = new SaxImportManager(graph, repository, validUser,
113                 IcaAtomEadImporter.class, IcaAtomEadHandler.class);
114         importManager.importFile(ios, "Testing import/export");
115 
116         DocumentaryUnit fonds = graph.frame(
117                 getVertexByIdentifier(graph, topLevelIdentifier), DocumentaryUnit.class);
118         Ead2002Exporter exporter = new Ead2002Exporter(graph);
119         ByteArrayOutputStream baos = new ByteArrayOutputStream();
120         exporter.export(fonds, baos, lang);
121         String xml = baos.toString("UTF-8");
122         isValidEad(xml);
123         return xml;
124     }
