50      @Test
51      public void testWithInitializationError() throws Exception {
52          try {
53              assertEquals(0, getNodeCount(testGraph));
54              new GraphInitializer(testGraph).initialize();
55              FixtureLoader loader = new YamlFixtureLoader(testGraph);
56              loader.loadTestData();
57              fail("Loading fixtures on an initialized graph should have" +
58                      " thrown an integrity error");
59          } catch (Exception e) {
60              assertThat(e.getMessage(), containsString("Integrity error"));
61          }
62      }

64      @Test
65      public void testWithInitialization() throws Exception {
66          // NB: An initialized Neo4j 1.9 graph has a root node
67          assertEquals(0, getNodeCount(testGraph));
68          new GraphInitializer(testGraph).initialize();
69          FixtureLoader loader = new YamlFixtureLoader(testGraph)
70              .setInitializing(false);
71          loader.loadTestData();
72          assertTrue(getNodeCount(testGraph) > 1);
73      }
74
