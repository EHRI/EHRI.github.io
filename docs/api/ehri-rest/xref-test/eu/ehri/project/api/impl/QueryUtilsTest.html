33  public class QueryUtilsTest {
34      @Test
35      public void testParseFilters() throws Exception {
36          SortedMap<String, Pair<QueryApi.FilterPredicate, Object>> filters
37                  = QueryUtils.parseFilters(Lists.newArrayList("identifier__GT:c1"));
38          assertTrue(filters.containsKey("identifier"));
39          Pair<QueryApi.FilterPredicate, Object> pair = filters.get("identifier");
40          assertEquals(QueryApi.FilterPredicate.GT, pair.getA());
41          assertEquals("c1", pair.getB());
42      }
43  
44      @Test
45      public void testParseOrderSpecs() throws Exception {
46          SortedMap<String, QueryApi.Sort> orders = QueryUtils
47                  .parseOrderSpecs(Lists.newArrayList("identifier__DESC", "name"));
48          assertTrue(orders.containsKey("identifier"));
49          QueryApi.Sort sort = orders.get("identifier");
50          assertEquals(QueryApi.Sort.DESC, sort);
51          QueryApi.Sort nameSort = orders.get("name");
52          assertEquals(QueryApi.Sort.ASC, nameSort);
53      }
