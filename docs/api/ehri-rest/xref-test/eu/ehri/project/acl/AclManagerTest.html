366     @Test
367     public void testCreateOwnerScenario() throws Exception {
368         loader.loadTestData("archivist-permissions.yaml");
369         UserProfile headUser = manager.getEntity("hauser", UserProfile.class);
370         UserProfile user1 = manager.getEntity("auser1", UserProfile.class);
371         UserProfile user2 = manager.getEntity("auser2", UserProfile.class);
372         Group headArchivists = manager.getEntity("head-archivists", Group.class);
373         Group archivists = manager.getEntity("archivists", Group.class);
374         Repository repo = manager.getEntity("repo", Repository.class);
375 
376         AclManager acl = new AclManager(graph, repo);
377         // The grants in the permissions should give these particular
378         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, CREATE, headArchivists));
379         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, UPDATE, headArchivists));
380         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, DELETE, headArchivists));
381 
382         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, CREATE, archivists));
383         assertFalse(acl.hasPermission(DOCUMENTARY_UNIT, UPDATE, archivists));
384         assertFalse(acl.hasPermission(DOCUMENTARY_UNIT, DELETE, archivists));
385 
386         // Check these grants are correctly inherited by the users.
387         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, CREATE, headUser));
388         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, UPDATE, headUser));
389         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, DELETE, headUser));
390 
391         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, CREATE, user1));
392         assertFalse(acl.hasPermission(DOCUMENTARY_UNIT, UPDATE, user1));
393         assertFalse(acl.hasPermission(DOCUMENTARY_UNIT, DELETE, user1));
394 
395         // New create some stuff
396         LoggingCrudViews<DocumentaryUnit> views
397                 = new LoggingCrudViews<DocumentaryUnit>(graph, DocumentaryUnit.class, repo);
398 
399         DocumentaryUnit headdoc1 = views.create(
400                 Bundle.fromData(TestData.getTestDocBundle())
401                         .withDataValue(Ontology.IDENTIFIER_KEY, "head-doc"), headUser);
402         DocumentaryUnit userdoc1 = views.create(
403                 Bundle.fromData(TestData.getTestDocBundle())
404                         .withDataValue(Ontology.IDENTIFIER_KEY, "user-doc-1"), user1);
405         DocumentaryUnit userdoc2 = views.create(
406                 Bundle.fromData(TestData.getTestDocBundle())
407                         .withDataValue(Ontology.IDENTIFIER_KEY, "user-doc-2"), user2);
408 
409         // Ensure Head Archivist can update/delete user1's doc
410         assertTrue(acl.hasPermission(userdoc1, UPDATE, headArchivists));
411         assertTrue(acl.hasPermission(userdoc1, DELETE, headArchivists));
412         // But doesn't own it
413         assertFalse(acl.hasPermission(userdoc1, OWNER, headArchivists));
414 
415         // Check the calculated permission sets.
416         assertTrue(acl.getInheritedItemPermissions(userdoc1, user1).has(OWNER));
417         assertFalse(acl.getInheritedItemPermissions(userdoc2, user1).has(OWNER));
418         assertTrue(acl.getInheritedItemPermissions(userdoc2, user2).has(OWNER));
419         assertFalse(acl.getInheritedItemPermissions(userdoc1, user2).has(OWNER));
420 
421         // Ensure user1 can update/delete his own doc
422         assertTrue(acl.hasPermission(userdoc1, OWNER, user1));
423         assertTrue(acl.hasPermission(userdoc1, UPDATE, user1));
424         assertTrue(acl.hasPermission(userdoc1, DELETE, user1));
425 
426         // Ensure neither user1 or user2 can update/delete head's docs
427         assertFalse(acl.hasPermission(headdoc1, OWNER, user1));
428         assertFalse(acl.hasPermission(headdoc1, UPDATE, user1));
429         assertFalse(acl.hasPermission(headdoc1, DELETE, user1));
430         assertFalse(acl.hasPermission(headdoc1, OWNER, user2));
431         assertFalse(acl.hasPermission(headdoc1, UPDATE, user2));
432         assertFalse(acl.hasPermission(headdoc1, DELETE, user2));
433 
434         // Ensure neither user1 or user2 can update/delete each other's docs
435         assertFalse(acl.hasPermission(userdoc1, OWNER, user2));
436         assertFalse(acl.hasPermission(userdoc1, UPDATE, user2));
437         assertFalse(acl.hasPermission(userdoc1, DELETE, user2));
438         assertFalse(acl.hasPermission(userdoc2, OWNER, user1));
439         assertFalse(acl.hasPermission(userdoc2, UPDATE, user1));
440         assertFalse(acl.hasPermission(userdoc2, DELETE, user1));
441     }
