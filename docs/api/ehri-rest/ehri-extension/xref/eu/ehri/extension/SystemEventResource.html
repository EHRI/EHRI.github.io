69       * List global events. Standard list parameters for paging apply.
70       *
71       * @param eventTypes Filter events by type
72       * @param itemTypes  Filter events based on the item type of their subjects
73       * @param itemIds    Filter events pertaining to specific item IDs
74       * @param users      Filter events based on the user IDs they involve
75       * @param from       Exclude events prior to this data
76       * @param to         Exclude events after this data
77       * @throws BadRequester
78       */
79      @GET
80      @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
81      @Path("/list")
82      public Response listEvents(
83              final @QueryParam(EVENT_TYPE_PARAM) List<EventTypes> eventTypes,
84              final @QueryParam(ITEM_TYPE_PARAM) List<String> itemTypes,
85              final @QueryParam(ITEM_ID_PARAM) List<String> itemIds,
86              final @QueryParam(USER_PARAM) List<String> users,
87              final @QueryParam(FROM_PARAM) String from,
88              final @QueryParam(TO_PARAM) String to)
89              throws BadRequester {
90  
91          Accessor user = getRequesterUserProfile();
92          EventViews eventViews = new EventViews(graph)
93                  .withEventTypes(eventTypes.toArray(new EventTypes[eventTypes.size()]))
94                  .withEntityTypes(itemTypes.toArray(new String[itemTypes.size()]))
95                  .withIds(itemIds.toArray(new String[itemIds.size()]))
96                  .withUsers(users.toArray(new String[users.size()]))
97                  .from(from)
98                  .to(to);
99  
100         return streamingList(eventViews.list(getQuery(cls), user));
101     }
