161     @DELETE
162     @Path("/{id:.+}/all")
163     public Response deleteAllVocabularyConcepts(@PathParam("id") String id)
164             throws ItemNotFound, AccessDenied, PermissionDenied {
165         try (final Tx tx = graph.getBaseGraph().beginTx()) {
166             UserProfile user = getCurrentUser();
167             Vocabulary vocabulary = views.detail(id, user);
168             CrudViews<Concept> conceptViews = new CrudViews<>(
169                     graph, Concept.class, vocabulary);
170             ActionManager actionManager = new ActionManager(graph, vocabulary);
171             Iterable<Concept> concepts = vocabulary.getConcepts();
172             if (concepts.iterator().hasNext()) {
173                 ActionManager.EventContext context = actionManager
174                         .newEventContext(user, EventTypes.deletion, getLogMessage());
175                 for (Concept concept : concepts) {
176                     context.addSubjects(concept);
177                     conceptViews.delete(concept.getId(), user);
178                 }
179             }
180             tx.success();
181             return Response.status(Status.OK).build();
182         } catch (SerializationError | ValidationError e) {
183             throw new RuntimeException(e);
184         }
185     }

210     public final static String TURTLE_MIMETYPE = "text/turtle";

212     public final static String N3_MIMETYPE = "application/n-triples";
213     public final BiMap<String, String> RDF_MIMETYPE_FORMATS = ImmutableBiMap.of(

215             TURTLE_MIMETYPE, "TTL",
216             RDF_XML_MIMETYPE, "RDF/XML"

219     /**
220      * Export the given vocabulary as SKOS.
221      *
222      * @param id      the vocabulary id
223      * @param format  the RDF format. Can be one of: RDF/XML, N3, TTL
224      * @param baseUri the base URI for exported items
225      * @return a SKOS vocabulary
226      * @throws IOException
227      * @throws ItemNotFound
228      */
229     @GET
230     @Path("{id:.+}/export")
231     @Produces({TURTLE_MIMETYPE, RDF_XML_MIMETYPE, N3_MIMETYPE})
232     public Response exportSkos(@PathParam("id") String id,
233             final @QueryParam("format") String format,
234             final @QueryParam("baseUri") String baseUri)
235             throws IOException, ItemNotFound {
236         final String rdfFormat = getRdfFormat(format, "TTL");
237         final String base = baseUri == null ? SkosRDFVocabulary.DEFAULT_BASE_URI : baseUri;
238         final MediaType mediaType = MediaType.valueOf(RDF_MIMETYPE_FORMATS
239                 .inverse().get(rdfFormat));
240         final Tx tx = graph.getBaseGraph().beginTx();
241         try {
242             final Accessor user = getRequesterUserProfile();
243             final Vocabulary vocabulary = views.detail(id, user);
244             final SkosExporter skosImporter = new JenaSkosExporter(graph, vocabulary)
245                     .setFormat(format);
246             return Response.ok(new StreamingOutput() {
247                 @Override
248                 public void write(OutputStream outputStream) throws IOException, WebApplicationException {
249                     skosImporter.export(outputStream, base);
250                 }
251             }).type(mediaType + "; charset=utf-8").build();
252         } catch (Exception e) {
253             tx.close();
254             throw e;
255         }
256     }

258     private String getRdfFormat(String format, String defaultFormat) {
259         if (format == null) {
260             for (String mimeValue : RDF_MIMETYPE_FORMATS.keySet()) {
261                 MediaType mime = MediaType.valueOf(mimeValue);
262                 if (requestHeaders.getAcceptableMediaTypes().contains(mime)) {
263                     return RDF_MIMETYPE_FORMATS.get(mimeValue);
264                 }
265             }
266             return defaultFormat;
267         } else {
268             return RDF_MIMETYPE_FORMATS.containsValue(format) ? format : defaultFormat;
269         }
270     }
