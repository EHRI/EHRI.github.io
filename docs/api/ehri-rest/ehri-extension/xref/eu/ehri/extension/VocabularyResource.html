152     @DELETE
153     @Path("/{id:.+}/all")
154     public Response deleteAllVocabularyConcepts(@PathParam("id") String id)
155             throws ItemNotFound, BadRequester, AccessDenied, PermissionDenied {
156         try {
157             UserProfile user = getCurrentUser();
158             Vocabulary vocabulary = views.detail(id, user);
159             CrudViews<Concept> conceptViews = new CrudViews<Concept>(
160                     graph, Concept.class, vocabulary);
161             ActionManager actionManager = new ActionManager(graph, vocabulary);
162             Iterable<Concept> concepts = vocabulary.getConcepts();
163             if (concepts.iterator().hasNext()) {
164                 ActionManager.EventContext context = actionManager
165                         .logEvent(user, EventTypes.deletion, getLogMessage());
166                 for (Concept concept : concepts) {
167                     context.addSubjects(concept);
168                     conceptViews.delete(concept.getId(), user);
169                 }
170             }
171             graph.getBaseGraph().commit();
172             return Response.status(Status.OK).build();
173         } catch (SerializationError e) {
174             graph.getBaseGraph().rollback();
175             throw new RuntimeException(e);
176         } catch (ValidationError e) {
177             graph.getBaseGraph().rollback();
178             throw new RuntimeException(e);
179         } finally {
180             cleanupTransaction();
181         }
182     }
