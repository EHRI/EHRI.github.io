68      /**
69       * Update the childCount property on hierarchical items with the number of
70       * children they contain.
71       *
72       * @throws Exception
73       */
74      @POST
75      @Produces(MediaType.APPLICATION_JSON)
76      @Path("/_rebuildChildCache")
77      public Response rebuildChildCache() throws Exception {
78          graph.getBaseGraph().checkNotInTransaction();
79          try {
80              for (DocumentaryUnit unit : manager.getFrames(EntityClass.DOCUMENTARY_UNIT, DocumentaryUnit.class)) {
81                  unit.updateChildCountCache();
82              }
83              for (Repository repository : manager.getFrames(EntityClass.REPOSITORY, Repository.class)) {
84                  repository.updateChildCountCache();
85              }
86              for (Country country : manager.getFrames(EntityClass.COUNTRY, Country.class)) {
87                  country.updateChildCountCache();
88              }
89              for (Group group : manager.getFrames(EntityClass.GROUP, Group.class)) {
90                  group.updateChildCountCache();
91              }
92              for (Concept concept : manager.getFrames(EntityClass.CVOC_CONCEPT, Concept.class)) {
93                  concept.updateChildCountCache();
94              }
95              for (Vocabulary vocabulary : manager.getFrames(EntityClass.CVOC_VOCABULARY, Vocabulary.class)) {
96                  vocabulary.updateChildCountCache();
97              }
98              for (AuthoritativeSet set : manager.getFrames(EntityClass.AUTHORITATIVE_SET, AuthoritativeSet.class)) {
99                  set.updateChildCountCache();
100             }
101             for (VirtualUnit vu : manager.getFrames(EntityClass.VIRTUAL_UNIT, VirtualUnit.class)) {
102                 vu.updateChildCountCache();
103             }
104 
105             graph.getBaseGraph().commit();
106             return Response.status(Status.OK).build();
107         } finally {
108             cleanupTransaction();
109         }
110     }

229     /**
230      * Create a new user with a default name and identifier.
231      *
232      * @param jsonData Additional key/value data for the created object
233      * @param groups   IDs for groups to which the user should belong
234      * @return A new user
235      * @throws Exception
236      */
237     @POST
238     @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
239     @Consumes(MediaType.APPLICATION_JSON)
240     @Path("/createDefaultUserProfile")
241     public Response createDefaultUserProfile(String jsonData,
242                                              @QueryParam(GROUP_PARAM) List<String> groups) throws Exception {
243         graph.getBaseGraph().checkNotInTransaction();
244         try {
245             String ident = getNextDefaultUserId();
246             Bundle bundle = Bundle.Builder.withClass(EntityClass.USER_PROFILE)
247                     .addDataValue(Ontology.IDENTIFIER_KEY, ident)
248                     .addDataValue(Ontology.NAME_KEY, ident)
249                     .addData(parseUserData(jsonData))
250                     .build();
251 
252             // NB: This assumes that admin's ID is the same as its identifier.
253             Accessor accessor = manager.getFrame(Group.ADMIN_GROUP_IDENTIFIER,
254                     Accessor.class);
255             Crud<UserProfile> view = ViewFactory.getCrudWithLogging(graph, UserProfile.class);
256             UserProfile user = view.create(bundle, accessor);
257 
258             // add to the groups
259             for (String groupId : groups) {
260                 Group group = manager.getFrame(groupId, EntityClass.GROUP, Group.class);
261                 group.addMember(user);
262             }
263 
264             // Grant them owner permissions on their own account.
265             new AclManager(graph).grantPermission(user, PermissionType.OWNER, user
266             );
267             graph.getBaseGraph().commit();
268             return creationResponse(user);
269         } finally {
270             cleanupTransaction();
271         }
272     }

276     private String getNextDefaultUserId() {
277         CloseableIterable<Vertex> query = manager.getVertices(EntityClass.USER_PROFILE);
278         try {
279             long start = Iterables.size(query) + 1;
280             while (manager.exists(String.format(DEFAULT_USER_ID_FORMAT,
281                     DEFAULT_USER_ID_PREFIX, start))) start++;
282             return String.format(DEFAULT_USER_ID_FORMAT, DEFAULT_USER_ID_PREFIX, start);
283         } finally {
284             query.close();
285         }
286     }
