64      /**
65       * Update the childCount property on hierarchical items with the number of
66       * children they contain.
67       *
68       * @throws Exception
69       */
70      @POST
71      @Produces(MediaType.APPLICATION_JSON)
72      @Path("/_rebuildChildCache")
73      public Response rebuildChildCache() throws Exception {
74          graph.getBaseGraph().checkNotInTransaction();
75          try {
76              for (DocumentaryUnit unit : manager.getFrames(EntityClass.DOCUMENTARY_UNIT, DocumentaryUnit.class)) {
77                  unit.updateChildCountCache();
78              }
79              for (Repository repository : manager.getFrames(EntityClass.REPOSITORY, Repository.class)) {
80                  repository.updateChildCountCache();
81              }
82              for (Country country : manager.getFrames(EntityClass.COUNTRY, Country.class)) {
83                  country.updateChildCountCache();
84              }
85              for (Group group : manager.getFrames(EntityClass.GROUP, Group.class)) {
86                  group.updateChildCountCache();
87              }
88              for (Concept concept : manager.getFrames(EntityClass.CVOC_CONCEPT, Concept.class)) {
89                  concept.updateChildCountCache();
90              }
91              for (Vocabulary vocabulary : manager.getFrames(EntityClass.CVOC_VOCABULARY, Vocabulary.class)) {
92                  vocabulary.updateChildCountCache();
93              }
94              for (AuthoritativeSet set : manager.getFrames(EntityClass.AUTHORITATIVE_SET, AuthoritativeSet.class)) {
95                  set.updateChildCountCache();
96              }
97              for (VirtualUnit vu : manager.getFrames(EntityClass.VIRTUAL_UNIT, VirtualUnit.class)) {
98                  vu.updateChildCountCache();
99              }
100 
101             graph.getBaseGraph().commit();
102             return Response.status(Status.OK).build();
103         } finally {
104             cleanupTransaction();
105         }
106     }

126     /**
127      * Create a new user with a default name and identifier.
128      *
129      * @param jsonData Additional key/value data for the created object
130      * @param groups   IDs for groups to which the user should belong
131      * @return A new user
132      * @throws Exception
133      */
134     @POST
135     @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
136     @Consumes(MediaType.APPLICATION_JSON)
137     @Path("/createDefaultUserProfile")
138     public Response createDefaultUserProfile(String jsonData,
139                                              @QueryParam(GROUP_PARAM) List<String> groups) throws Exception {
140         graph.getBaseGraph().checkNotInTransaction();
141         try {
142             String ident = getNextDefaultUserId();
143             Bundle bundle = Bundle.Builder.withClass(EntityClass.USER_PROFILE)
144                     .addDataValue(Ontology.IDENTIFIER_KEY, ident)
145                     .addDataValue(Ontology.NAME_KEY, ident)
146                     .addData(parseUserData(jsonData))
147                     .build();
148 
149             // NB: This assumes that admin's ID is the same as its identifier.
150             Accessor accessor = manager.getFrame(Group.ADMIN_GROUP_IDENTIFIER,
151                     Accessor.class);
152             Crud<UserProfile> view = ViewFactory.getCrudWithLogging(graph, UserProfile.class);
153             UserProfile user = view.create(bundle, accessor);
154 
155             // add to the groups
156             for (String groupId : groups) {
157                 Group group = manager.getFrame(groupId, EntityClass.GROUP, Group.class);
158                 group.addMember(user);
159             }
160 
161             // Grant them owner permissions on their own account.
162             new AclManager(graph).grantPermission(user, PermissionType.OWNER, user
163             );
164             graph.getBaseGraph().commit();
165             return creationResponse(user);
166         } finally {
167             cleanupTransaction();
168         }
169     }

173     private String getNextDefaultUserId() {
174         CloseableIterable<Vertex> query = manager.getVertices(EntityClass.USER_PROFILE);
175         try {
176             long start = Iterables.size(query) + 1;
177             while (manager.exists(String.format(DEFAULT_USER_ID_FORMAT,
178                     DEFAULT_USER_ID_PREFIX, start))) start++;
179             return String.format(DEFAULT_USER_ID_FORMAT, DEFAULT_USER_ID_PREFIX, start);
180         } finally {
181             query.close();
182         }
183     }

192             };
193             return jsonMapper.readValue(json, typeRef);
194         }
