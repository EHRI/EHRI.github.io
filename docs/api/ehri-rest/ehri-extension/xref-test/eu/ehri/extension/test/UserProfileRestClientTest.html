173         assertTrue(groupIds.contains(GROUP_ID1));
174         assertTrue(groupIds.contains(GROUP_ID2));

177     private Set<String> getGroupIdsFromEntityJson(String jsonString) throws JSONException {
178         JSONObject obj = new JSONObject(jsonString);
179         Set<String> groupIds = new HashSet<String>();
180         JSONArray jsonArray = ((JSONObject) obj.get("relationships")).getJSONArray("belongsTo");
181         for (int i = 0; i < jsonArray.length(); i++) {
182             JSONObject item = jsonArray.getJSONObject(i);
183             groupIds.add(item.getString("id"));
184         }
185         return groupIds;
186     }

207     @Test
208     public void testFollowAndUnfollow() throws Exception {
209         String user1 = "reto";
210         String user2 = "mike";
211         String user3 = "linda";
212 
213         String followingUrl = "/" + Entities.USER_PROFILE + "/" + user1 + "/" + FOLLOWING;
214         List<Map<String, Object>> followers = getItemList(followingUrl, user1);
215         assertTrue(followers.isEmpty());
216 
217         URI followUrl1 = UriBuilder.fromPath(getExtensionEntryPointUri())
218                 .segment(Entities.USER_PROFILE)
219                 .segment(user1)
220                 .segment(FOLLOWING)
221                 .queryParam(ID_PARAM, user2).build();
222         ClientResponse response = jsonCallAs(user1, followUrl1).post(ClientResponse.class);
223         assertStatus(OK, response);
224         followers = getItemList(followingUrl, user1);
225         assertEquals(1, followers.size());
226 
227         URI followUrl2 = UriBuilder.fromPath(getExtensionEntryPointUri())
228                 .segment(Entities.USER_PROFILE)
229                 .segment(user1)
230                 .segment(FOLLOWING)
231                 .queryParam(ID_PARAM, user3).build();
232         response = jsonCallAs(user1, followUrl2).post(ClientResponse.class);
233         assertStatus(OK, response);
234         followers = getItemList(followingUrl, user1);
235         assertEquals(2, followers.size());
236 
237         // Hitting the same URL as a GET should give us a boolean...
238         URI isFollowingUrl = ehriUri(Entities.USER_PROFILE, user1, IS_FOLLOWING, user2);
239         response = jsonCallAs(user1, isFollowingUrl).get(ClientResponse.class);
240         assertStatus(OK, response);
241         assertEquals("true", response.getEntity(String.class));
242 
243         URI hasFollowerUrl = ehriUri(Entities.USER_PROFILE, user2, IS_FOLLOWER, user1);
244         response = jsonCallAs(user2, hasFollowerUrl).get(ClientResponse.class);
245         assertStatus(OK, response);
246         assertEquals("true", response.getEntity(String.class));
247 
248         response = jsonCallAs(user1, followUrl1).delete(ClientResponse.class);
249         assertStatus(OK, response);
250         followers = getItemList(followingUrl, user1);
251         assertEquals(1, followers.size());
252     }
