98          String appleIdStr = appleLocationStr.substring(appleLocationStr
99                  .lastIndexOf('/') + 1);
100 
101         // make apple narrower of fruit
102         WebResource resource = client.resource(fruitLocation + "/narrower/"
103                 + appleIdStr);
104         response = resource
105                 .accept(MediaType.APPLICATION_JSON)
106                 .type(MediaType.APPLICATION_JSON)
107                 .header(AbstractRestResource.AUTH_HEADER_NAME,
108                         getAdminUserProfileId()).post(ClientResponse.class);
109         // Hmm, it's a post request, but we don't create a vertex (but we do an
110         // edge...)
111         assertStatus(OK, response);
112 
113         // get fruit's narrower concepts
114         response = testGet(fruitLocation + "/list");
115         // check if apple is in there
116         assertTrue(containsIdentifier(response, "apple"));
117 
118         // check if apple's broader is fruit
119         // get apple's broader concepts
120         response = testGet(ehriUri("cvocConcept", appleIdStr, "broader"));
121         // check if fruit is in there
122         assertTrue(containsIdentifier(response, "fruit"));
123 
124         // Test removal of one narrower concept
125         resource = client.resource(fruitLocation + "/narrower/" + appleIdStr);
126         response = resource
127                 .accept(MediaType.APPLICATION_JSON)
128                 .type(MediaType.APPLICATION_JSON)
129                 .header(AbstractRestResource.AUTH_HEADER_NAME,
130                         getAdminUserProfileId()).delete(ClientResponse.class);
131         assertStatus(OK, response);
132 
133         // apple should still exist
134         response = testGet(ehriUri("cvocConcept", appleIdStr));
135         assertStatus(OK, response);
136 
137         // but not as a narrower of fruit!
138         response = testGet(fruitLocation + "/list");
139         // check if apple is NOT in there

238     public boolean containsIdentifier(final ClientResponse response,
239             final String idStr) throws IOException {
240         String json = response.getEntity(String.class);
241         JsonNode rootNode = jsonMapper.readValue(json, JsonNode.class);
242         JsonNode idPath = rootNode.path(0).path("data").path("identifier");
243         return idPath.isTextual() && idPath.asText().equals(idStr);
244 
245     }
