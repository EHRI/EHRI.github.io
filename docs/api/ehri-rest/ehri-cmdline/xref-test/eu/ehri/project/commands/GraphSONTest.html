42      @Test
43      public void testSaveDumpAndRead() throws Exception {
44  
45          // 1. Load the (YAML) fixtures into a graph.
46          // 2. Dump it as JSON.
47          // 3. Load the JSON into another graph.
48          // 4. Compare the before and after graphs and check
49          //     nothing's changed.
50  
51          // Setup will create the database but not load the fixtures...
52          FramedGraph<? extends TransactionalGraph> graph1 = getFramedGraph();
53          helper = FixtureLoaderFactory.getInstance(graph1);
54          helper.loadTestData();
55          graph1.getBaseGraph().commit();
56  
57          List<VertexProxy> graphState1 = getGraphState(graph1);
58  
59          File temp = File.createTempFile("temp-file-name", ".tmp");
60          temp.deleteOnExit();
61          assertEquals(0L, temp.length());
62  
63          String filePath = temp.getAbsolutePath();
64          String[] outArgs = new String[]{"-d", "out", filePath};
65  
66          GraphSON graphSON = new GraphSON();
67          CommandLine outCmdLine = graphSON.getCmdLine(outArgs);
68  
69          assertEquals(0, graphSON.execWithOptions(graph1, outCmdLine));
70          graph1.shutdown();
71  
72          assertTrue(temp.exists());
73          assertTrue(temp.length() > 0L);
74  
75          FramedGraph<? extends TransactionalGraph> graph2 = getFramedGraph();
76  
77          String[] inArgs = new String[]{"-d", "in", filePath};
78          CommandLine inCmdLine = graphSON.getCmdLine(inArgs);
79          assertEquals(0, graphSON.execWithOptions(graph2, inCmdLine));
80  
81          List<VertexProxy> graphState2 = getGraphState(graph2);
82  
83          GraphDiff graphDiff = diffGraph(graphState1, graphState2);
84          assertEquals(0, graphDiff.added.size());
85          assertEquals(0, graphDiff.removed.size());
86      }
