27          // 2. Dump it as JSON.
28          // 3. Load the JSON into another graph.
29          // 4. Compare the before and after graphs and check
30          //     nothing's changed.
31  
32          // Setup will create the database but not load the fixtures...
33          FramedGraph<? extends TransactionalGraph> graph1 = getTestGraph();
34          helper = FixtureLoaderFactory.getInstance(graph1);
35          helper.loadTestData();
36  
37          List<VertexProxy> graphState1 = getGraphState(graph1);
38  
39          File temp = File.createTempFile("temp-file-name", ".tmp");
40          temp.deleteOnExit();
41          assertEquals(0L, temp.length());
42  
43          String filePath = temp.getAbsolutePath();
44          String[] outArgs = new String[]{"-d", "out", filePath};
45  
46          GraphSON graphSON = new GraphSON();
47          CommandLine outCmdLine = graphSON.getCmdLine(outArgs);
48  
49          assertEquals(0, graphSON.execWithOptions(graph1, outCmdLine));
50          graph1.shutdown();
51  
52          assertTrue(temp.exists());
53          assertTrue(temp.length() > 0L);
54  
55          FramedGraph<? extends TransactionalGraph> graph2 = getTestGraph();
56  
57          String[] inArgs = new String[]{"-d", "in", filePath};
58          CommandLine inCmdLine = graphSON.getCmdLine(inArgs);
59          assertEquals(0, graphSON.execWithOptions(graph2, inCmdLine));
60  
61          List<VertexProxy> graphState2 = getGraphState(graph2);
62  
63          GraphDiff graphDiff = diffGraph(graphState1, graphState2);
64          assertEquals(0, graphDiff.added.size());
65          assertEquals(0, graphDiff.removed.size());
66      }

68      private FramedGraph<? extends TransactionalGraph> getTestGraph() {

71          // we don't need - when loading and dumping into a new graph it
