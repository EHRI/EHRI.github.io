35          options.addOption(new Option("user", true,
36                  "Identifier of user to import as"));
37          options.addOption(new Option("tolerant", false,
38                  "Don't fail on individual row errors."));
39          options.addOption(new Option("log", true,
40                  "Log message for import action."));

43       @Override
44      public int execWithOptions(final FramedGraph<? extends TransactionalGraph> graph,
45              CommandLine cmdLine) throws Exception {
46  
47          GraphManager manager = GraphManagerFactory.getInstance(graph);
48          String logMessage = "Imported from command-line";
49          if (cmdLine.hasOption("log")) {
50              logMessage = cmdLine.getOptionValue("log");
51          }
52  
53          boolean tolerant = cmdLine.hasOption("tolerant");
54  
55          if (cmdLine.getArgList().size() < 1)
56              throw new RuntimeException(getHelp());
57  
58          List<String> filePaths = new LinkedList<String>();
59          for (int i = 0; i < cmdLine.getArgList().size(); i++) {
60              filePaths.add((String) cmdLine.getArgList().get(i));
61          }
62  
63          try {
64              // Find the agent
65              PermissionScope scope = SystemScope.getInstance();
66              if (cmdLine.hasOption("scope")) {
67                  scope = manager.getFrame(cmdLine.getOptionValue("scope"), PermissionScope.class);
68              }
69  
70              // Find the user
71              UserProfile user = manager.getFrame(cmdLine.getOptionValue("user"),
72                      UserProfile.class);
73  
74              ImportLog log = new CsvImportManager(graph, scope, user, importer)
75                      .setTolerant(tolerant).importFiles(filePaths, logMessage);
76              
77              System.out.println("Import completed. Created: " + log.getCreated()
78                      + ", Updated: " + log.getUpdated());
79              if (log.getErrored() > 0) {
80                  System.out.println("Errors:");
81                  for (Map.Entry<String, String> entry : log.getErrors().entrySet()) {
82                      System.out.printf(" - %-20s : %s%n", entry.getKey(),
83                              entry.getValue());
84                  }
85              }
86          } catch (Exception e) {
87              e.printStackTrace();
88              return 1;
89          }
90          return 0;
91      }
