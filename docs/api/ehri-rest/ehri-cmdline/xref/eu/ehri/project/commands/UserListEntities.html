20  package eu.ehri.project.commands;

22  import com.tinkerpop.frames.FramedGraph;
23  import eu.ehri.project.core.GraphManager;
24  import eu.ehri.project.core.GraphManagerFactory;
25  import eu.ehri.project.models.EntityClass;
26  import eu.ehri.project.models.UserProfile;
27  import eu.ehri.project.models.base.AccessibleEntity;
28  import eu.ehri.project.views.Query;
29  import org.apache.commons.cli.CommandLine;
30  import org.apache.commons.cli.Option;
31  import org.apache.commons.cli.Options;

33  /**
34   * List items of a specific type as a given user, respecting access controls.
35   *
36   * @author Mike Bryant (https://github.com/mikesname)
37   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
38   */
39  public class UserListEntities extends BaseCommand {
40  
41      final static String NAME = "user-list";
42  
43      public UserListEntities() {
44      }
45  
46      @Override
47      public String getHelp() {
48          return "Usage: user-list [OPTIONS] <type>";
49      }
50  
51      @Override
52      public String getUsage() {
53          return "List entities of a given type as a given user.";
54      }
55      
56      @Override
57      public void setCustomOptions(Options options) {
58          options.addOption(new Option("user", true,
59                  "Identifier of user to list items as"));
60      }
61  
62      @Override
63      public int execWithOptions(FramedGraph<?> graph,
64              CommandLine cmdLine) throws Exception {
65  
66          if (cmdLine.getArgList().size() < 1)
67              throw new RuntimeException(getHelp());
68  
69          GraphManager manager = GraphManagerFactory.getInstance(graph);
70          EntityClass type = EntityClass.withName(cmdLine.getArgs()[0]);
71          Class<?> cls = type.getJavaClass();
72  
73          if (!AccessibleEntity.class.isAssignableFrom(cls))
74              throw new RuntimeException("Unknown accessible entity: " + type);
75  
76          UserProfile user = manager.getFrame(
77                  cmdLine.getOptionValue("user"), UserProfile.class);
78  
79          @SuppressWarnings("unchecked")
80          Query<AccessibleEntity> query = new Query<>(graph,
81                  (Class<AccessibleEntity>) cls);
82          for (AccessibleEntity acc : query.page(user)) {
83              System.out.println(acc.getId());
84          }
85  
86          return 0;
87      }
88  }
