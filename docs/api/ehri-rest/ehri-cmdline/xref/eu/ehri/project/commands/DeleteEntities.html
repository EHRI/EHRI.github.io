41  /**
42   * Delete an entire class of entities via the command line.
43   * 
44   * @author Mike Bryant (https://github.com/mikesname)
45   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
46   */
47  public class DeleteEntities extends BaseCommand {
48  
49      final static String NAME = "delete-all";
50  
51      public DeleteEntities() {
52      }
53  
54      @Override
55      protected void setCustomOptions(Options options) {
56          options.addOption(new Option("user", true,
57             "Identifier of user to import as"));
58          options.addOption(new Option("log", true,
59                  "Log message for import action."));
60      }
61  
62      @Override
63      public String getHelp() {
64          return String.format("Usage: %s [OPTIONS] <type>", NAME);
65      }
66  
67      @Override
68      public String getUsage() {
69          return "Delete ALL entities of a given type.";
70      }
71  
72      @Override
73      public int execWithOptions(FramedGraph<?> graph,
74              CommandLine cmdLine) throws Exception {
75  
76          // the first argument is the entity type, and that must be specified
77          if (cmdLine.getArgList().size() < 1)
78              throw new RuntimeException(getHelp());
79          EntityClass type = EntityClass.withName(cmdLine.getArgs()[0]);
80          Class<?> cls = type.getEntityClass();
81  
82          String logMessage = "Deleting items of type " + type + " via the command-line";
83          if (cmdLine.hasOption("log")) {
84              logMessage = cmdLine.getOptionValue("log");
85          }
86  
87          if (!AccessibleEntity.class.isAssignableFrom(cls))
88              throw new RuntimeException("Unknown accessible entity: " + type);
89  
90          GraphManager manager = GraphManagerFactory.getInstance(graph);
91  
92          // Find the user
93          UserProfile user = manager.getFrame(cmdLine.getOptionValue("user"),
94                  UserProfile.class);
95  
96          new ActionManager(graph).newEventContext(user,
97                  EventTypes.deletion,
98                  getLogMessage(logMessage))
99                  .commit();
100         deleteIds(graph, manager, type, user);
101 
102         return 0;
103     }
104 
105     private void deleteIds(FramedGraph<?> graph, GraphManager manager, EntityClass type, UserProfile user)
106             throws SerializationError, ValidationError, ItemNotFound, PermissionDenied {
107         CrudViews<AccessibleEntity> views = new CrudViews<>(graph, AccessibleEntity.class);
108         try (CloseableIterable<AccessibleEntity> items = manager.getFrames(type, AccessibleEntity.class)) {
109             for (AccessibleEntity acc : items) {
110                 System.out.println(acc.getId());
111                 views.delete(acc.getId(), graph.frame(user.asVertex(), Accessor.class));
112             }
113         }
114     }
