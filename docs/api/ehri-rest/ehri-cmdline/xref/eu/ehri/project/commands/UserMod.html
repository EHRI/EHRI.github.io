41  /**
42   * Modify an existing user.
43   *
44   * @author Mike Bryant (https://github.com/mikesname)
45   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
46   */
47  public class UserMod extends BaseCommand implements Command {
48  
49      final static String NAME = "usermod";
50  
51      public UserMod() {
52      }
53  
54      @Override
55      protected void setCustomOptions(Options options) {
56          options.addOption(new Option("group", true,
57                  "A group to add the user to"));
58          options.addOption(new Option("c", "comment", false,
59                  "Log message for create action action."));
60      }
61  
62      @Override
63      public String getHelp() {
64          return "Usage: usermod [OPTIONS] <user-identifier>";
65      }
66  
67      @Override
68      public String getUsage() {
69          return "Add an existing user to a group";
70      }
71  
72      @Override
73      public int execWithOptions(final FramedGraph<? extends TransactionalGraph> graph,
74              CommandLine cmdLine) throws ItemNotFound, ValidationError,
75              PermissionDenied, DeserializationError {
76  
77          GraphManager manager = GraphManagerFactory.getInstance(graph);
78          final String logMessage = cmdLine.getOptionValue("c",
79                  "Adding user to groups");
80  
81          if (cmdLine.getArgList().size() < 1)
82              throw new RuntimeException(getHelp());
83  
84          // Fetch the admin accessor, who's going to do the work.
85          Actioner admin = manager.getFrame(Group.ADMIN_GROUP_IDENTIFIER,
86                  Actioner.class);
87  
88          String userId = (String) cmdLine.getArgList().get(0);
89  
90          String[] groups = {};
91          if (cmdLine.hasOption("group")) {
92              groups = cmdLine.getOptionValues("group");
93          }
94  
95          try {
96              UserProfile user = manager.getFrame(userId,
97                      EntityClass.USER_PROFILE, UserProfile.class);
98  
99              EventContext actionCtx = new ActionManager(graph).newEventContext(
100                     user, admin, EventTypes.modification,
101                     getLogMessage(logMessage));
102 
103             for (String groupId : groups) {
104                 Group group = manager.getFrame(groupId, EntityClass.GROUP,
105                         Group.class);
106                 group.addMember(user);
107                 actionCtx.addSubjects(group);
108             }
109             actionCtx.commit();
110             graph.getBaseGraph().commit();
111         } catch (Exception e) {
112             graph.getBaseGraph().rollback();
113             throw new RuntimeException(e);
114         }
115 
116         return 0;
117     }
