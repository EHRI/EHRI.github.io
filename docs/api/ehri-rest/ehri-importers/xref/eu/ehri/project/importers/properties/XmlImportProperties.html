64      /**
65       * Get a set of property keys that have a given value
66       * @param value the value that requested properties must have
67       * @return a set of key names that have the given value
68       */
69      public Set<String> getPropertiesWithValue(String value) {
70          Set<String> ps = new HashSet<String>();
71          Properties p = getProperties();
72          for (Entry<Object, Object> property : p.entrySet()) {
73              if (value.equals(property.getValue().toString()))
74                  ps.add(property.getKey().toString());
75          }
76          return ps;
77      }
78  
79      public String getFirstPropertyWithValue(String value) {
80          Properties p = getProperties();
81          for (Entry<Object, Object> property : p.entrySet()) {
82              if (value.equals(property.getValue().toString()))
83                  return property.getKey().toString();
84          }
85          return null;
86      }

88      public Set<String> getAllNonAttributeProperties() {
89          Set<String> ps = new HashSet<String>();
90          Properties p = getProperties();
91          for (Object key : p.keySet()) {
92              if (!key.toString().startsWith("@"))
93                  ps.add(key.toString());
94          }
95          return ps;
96      }

147     /**
148      * used to try to find the xml attribute value 'ehri_main_identifier' in
149      * properties like 'did/unitid/@ehrilabel$ehri_main_identifier=objectIdentifier'
150      *
151      * @param key       did/unitid/
152      * @param attribute @ehrilabel
153      * @return ehri_main_identifier or null if no attribute value could be found in this property file
154      */
155     public String getValueForAttributeProperty(String key, String attribute) {
156         Properties p = getProperties();
157         for (Object k_object : p.keySet()) {
158             String k = k_object.toString();
159             if (k.startsWith(key + attribute) && k.contains("$")) {
160                 return k.substring(1 + k.indexOf("$"));
161             }
162         }
163         return null;
164     }

167 abstract class PropertyLoader {
168 
169     private static final Logger logger = LoggerFactory.getLogger(PropertyLoader.class);
170 
171     /**
172      * Looks up a resource named 'name' in the classpath. The resource must map to a file with .properties extention.
173      * The name is assumed to be absolute and can use either "/" or "." for package segment separation with an optional
174      * leading "/" and optional ".properties" suffix. Thus, the following names refer to the same resource:
175      * <p/>
176      * <pre>
177      * some.pkg.Resource
178      * some.pkg.Resource.properties
179      * some/pkg/Resource
180      * some/pkg/Resource.properties
181      * /some/pkg/Resource
182      * /some/pkg/Resource.properties
183      * </pre>
184      *
185      * @param name   classpath resource name [may not be null]
186      * @param loader classloader through which to load the resource [null is equivalent to the application loader]
187      * @return resource converted to java.util.Properties [may be null if the resource was not found and
188      *         THROW_ON_LOAD_FAILURE is false]
189      * @throws IllegalArgumentException if the resource was not found and THROW_ON_LOAD_FAILURE is true
190      */
191     public static Properties loadProperties(String name, ClassLoader loader) {
192         logger.debug("loading " + name + " ...");
193         if (name == null) {
194             throw new IllegalArgumentException("null input: name");
195         }
196 
197         if (name.endsWith(SUFFIX)) {
198             name = name.substring(0, name.length() - SUFFIX.length());
199         }
200 
201         Properties result = null;
202 
203         InputStream in = null;
204         try {
205             if (loader == null) {
206                 loader = ClassLoader.getSystemClassLoader();
207             }
208 
209             if (LOAD_AS_RESOURCE_BUNDLE) {
210                 name = name.replace('/', '.');
211                 // Throws MissingResourceException on lookup failures:
212                 final ResourceBundle rb = ResourceBundle.getBundle(name, Locale
213                         .getDefault(), loader);
214 
215                 result = new Properties();
216                 for (Enumeration<String> keys = rb.getKeys(); keys
217                         .hasMoreElements(); ) {
218                     final String key = keys.nextElement();
219                     final String value = rb.getString(key);
220 
221                     result.put(key, value);
222                 }
223             } else {
224 
225                 name = name.replace('.', '/');
226 
227                 if (!name.endsWith(SUFFIX)) {
228                     name = name.concat(SUFFIX);
229                 }
230 
231                 // Returns null on lookup failures:
232                 in = loader.getResourceAsStream(name);
233                 if (in != null) {
234                     result = new Properties();
235                     result.load(in); // Can throw IOException
236                 } else {
237                     logger.error(name + " is not a Resource");
238                     FileInputStream ios = new FileInputStream(name);
239                     logger.info("trying to read " + name + " as fileinputstream");
240 
241                     in = ios;
242                     result = new Properties();
243                     result.load(in); // Can throw IOException
244                 }
245             }
246         } catch (Exception e) {
247             logger.error(e.getMessage());
248             result = null;
249         } finally {
250             if (in != null) {
251                 try {
252                     in.close();
253                 } catch (Throwable ignore) {
254                 }
255             }
256         }
257 
258         if (THROW_ON_LOAD_FAILURE && (result == null)) {
259             throw new IllegalArgumentException("could not load ["
260                     + name
261                     + "]"
262                     + " as "
263                     + (LOAD_AS_RESOURCE_BUNDLE ? "a resource bundle"
264                     : "a classloader resource"));
265         }
266 
267         return result;
268     }
269 
270     /**
271      * A convenience overload of {@link #loadProperties(String, ClassLoader)} that uses the current thread's context
272      * classloader.
273      */
274     public static Properties loadProperties(final String name) {
275         return loadProperties(name, Thread.currentThread()
276                 .getContextClassLoader());
277     }
278 
279     private static final boolean THROW_ON_LOAD_FAILURE = true;
280     private static final boolean LOAD_AS_RESOURCE_BUNDLE = false;
281     private static final String SUFFIX = ".properties";
282 }
