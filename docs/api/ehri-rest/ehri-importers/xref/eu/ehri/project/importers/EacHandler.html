38  /**
39   * Handler of EAC-CPF files. Creates a {@link Map} for the {@link EacImporter} to
40   * import.
41   * Makes use of eac.properties with format: part/of/path/=attribute
42   *
43   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
44   */
45  public class EacHandler extends EaHandler {
46  
47      private final ImmutableMap<String, Class<? extends Frame>> possibleSubnodes
48              = ImmutableMap.<String, Class<? extends Frame>>builder()
49              .put("maintenanceEvent", MaintenanceEvent.class)
50              .put("relation", Annotation.class)
51              .put("book", Annotation.class)
52              .put("bookentry", Annotation.class)
53              .put("accessPoint", Annotation.class)
54              .put("name", UnknownProperty.class).build();
55  
56      private static final Logger logger = LoggerFactory.getLogger(EacHandler.class);
57  
58      public EacHandler(AbstractImporter<Map<String, Object>> importer) {
59          super(importer, new XmlImportProperties("eac.properties"));
60      }
61  
62      public EacHandler(AbstractImporter<Map<String, Object>> importer, XmlImportProperties properties) {
63          super(importer, properties);
64      }
65  
66      @Override
67      protected boolean needToCreateSubNode(String key) {
68          return possibleSubnodes.containsKey(getImportantPath(currentPath));
69      }
70  
71      @Override
72      public void endElement(String uri, String localName, String qName) throws SAXException {
73          //if a subnode is ended, add it to the super-supergraph
74          super.endElement(uri, localName, qName);
75          if (needToCreateSubNode(getImportantPath(currentPath))) {
76              Map<String, Object> currentGraph = currentGraphPath.pop();
77              putSubGraphInCurrentGraph(getImportantPath(currentPath), currentGraph);
78              depth--;
79          }
80  
81          currentPath.pop();
82  
83          //an EAC file consists of only 1 element, so if we're back at the root, we're done
84          if (currentPath.isEmpty()) {
85              try {
86                  logger.debug("depth close " + depth + " " + qName);
87                  //TODO: add any mandatory fields not yet there:
88                  if (!currentGraphPath.peek().containsKey("objectIdentifier")) {
89                      putPropertyInCurrentGraph("objectIdentifier", "id");
90                  }
91  
92                  //TODO: name can have only 1 value, others are otherFormsOfName
93                  if (currentGraphPath.peek().containsKey(Ontology.NAME_KEY)) {
94                      String name = chooseName(currentGraphPath.peek().get(Ontology.NAME_KEY));
95                      overwritePropertyInCurrentGraph(Ontology.NAME_KEY, name);
96                  }
97                  if (!currentGraphPath.peek().containsKey(Ontology.LANGUAGE_OF_DESCRIPTION)) {
98                      logger.debug("no " + Ontology.LANGUAGE_OF_DESCRIPTION + " found");
99                      putPropertyInCurrentGraph(Ontology.LANGUAGE_OF_DESCRIPTION, "en");
100                 }
101 
102                 importer.importItem(currentGraphPath.pop(), Lists.<String>newArrayList());
103 
104             } catch (ValidationError ex) {
105                 logger.error(ex.getMessage());
106             }
107         }
108     }
