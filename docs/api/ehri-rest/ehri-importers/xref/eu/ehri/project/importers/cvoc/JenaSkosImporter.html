158     /**
159      * Import an input stream.
160      *
161      * @param ios        The SKOS file input stream
162      * @param logMessage A log message
163      * @return A log of imported nodes
164      * @throws IOException
165      * @throws ValidationError
166      */
167     @Override
168     public ImportLog importFile(InputStream ios, String logMessage)
169             throws IOException, ValidationError {
170 
171         // Create a new action for this import
172         ActionManager.EventContext eventContext = new ActionManager(framedGraph, vocabulary).newEventContext(
173                 actioner, EventTypes.ingest, getLogMessage(logMessage));
174         // Create a manifest to store the results of the import.
175         ImportLog log = new ImportLog(eventContext);
176 
177         OntModel model = ModelFactory.createOntologyModel();
178         model.read(ios, null, format);
179         OntClass conceptClass = model.getOntClass(SkosRDFVocabulary.CONCEPT.getURI().toString());
180         logger.debug("in import file: " + SkosRDFVocabulary.CONCEPT.getURI());
181         ExtendedIterator<? extends OntResource> extendedIterator = conceptClass.listInstances();
182         Map<Resource, Concept> imported = Maps.newHashMap();
183 
184         while (extendedIterator.hasNext()) {
185             Resource item = extendedIterator.next();
186 
187             try {
188                 Mutation<Concept> graphConcept = importConcept(item);
189                 imported.put(item, graphConcept.getNode());
190 
191                 switch (graphConcept.getState()) {
192                     case UNCHANGED:
193                         log.addUnchanged();
194                         break;
195                     case CREATED:
196                         log.addCreated();
197                         eventContext.addSubjects(graphConcept.getNode());
198                         break;
199                     case UPDATED:
200                         log.addUpdated();
201                         eventContext.addSubjects(graphConcept.getNode());
202                         break;
203                 }
204             } catch (ValidationError validationError) {
205                 if (tolerant) {
206                     logger.error(validationError.getMessage());
207                     log.setErrored(item.toString(), validationError.getMessage());
208                 } else {
209                     throw validationError;
210                 }
211             }
212         }
213 
214         for (Map.Entry<Resource, Concept> pair : imported.entrySet()) {
215             hookupRelationships(pair.getKey(), pair.getValue(), imported);
216         }
217 
218         for (Concept concept : imported.values()) {
219             vocabulary.addItem(concept);
220             concept.setPermissionScope(vocabulary);
221         }
222 
223         if (log.hasDoneWork()) {
224             eventContext.commit();
225         }
226 
227         return log;
228     }

230     private Mutation<Concept> importConcept(Resource item) throws ValidationError {
231         logger.debug("Importing: {}", item);
232         Bundle.Builder builder = Bundle.Builder.withClass(EntityClass.CVOC_CONCEPT)
233                 .addDataValue(Ontology.IDENTIFIER_KEY, getId(URI.create(item.getURI())));
234 
235         Map<AuthoritativeItem, String> linkedConcepts = Maps.newHashMap();
236 
237         List<Bundle> undetermined = getUndeterminedRelations(item, linkedConcepts);
238         for (Bundle description : getDescriptions(item)) {
239             Bundle withRels = description
240                     .withRelations(Ontology.HAS_ACCESS_POINT, undetermined);
241             builder.addRelation(Ontology.DESCRIPTION_FOR_ENTITY, withRels);
242         }
243 
244         Mutation<Concept> mut = dao.createOrUpdate(builder.build(), Concept.class);
245         solveUndeterminedRelationships(mut.getNode(), linkedConcepts);
246         return mut;
247     }

249     private void solveUndeterminedRelationships(Concept unit, Map<AuthoritativeItem, String> linkedConcepts) {
250         GraphManager manager = GraphManagerFactory.getInstance(framedGraph);
251         for (AuthoritativeItem concept : linkedConcepts.keySet()) {
252             try {
253                 String reltype = linkedConcepts.get(concept);
254                 Bundle linkBundle = new Bundle(EntityClass.LINK)
255                         .withDataValue(Ontology.LINK_HAS_TYPE, "associate")
256                         .withDataValue(reltype.substring(0, reltype.indexOf(":")), reltype.substring(reltype.indexOf(":") + 1))
257                         .withDataValue(Ontology.LINK_HAS_DESCRIPTION, EaImporter.RESOLVED_LINK_DESC);
258                 UserProfile user = manager.getFrame(actioner.getId(), UserProfile.class);
259                 Link link;
260                 link = new CrudViews<>(framedGraph, Link.class).create(linkBundle, user);
261                 unit.addLink(link);
262                 concept.addLink(link);
263             } catch (ItemNotFound | ValidationError | PermissionDenied ex) {
264                 logger.error(ex.getMessage());
265             }
266         }
267     }

269     private List<Bundle> getUndeterminedRelations(Resource item, Map<AuthoritativeItem, String> linkedItems) {
270         List<Bundle> undetermined = Lists.newArrayList();
271 
272         for (Map.Entry<String, URI> rel : SkosRDFVocabulary.RELATION_PROPS.entrySet()) {
273             for (RDFNode annotation : getObjectWithPredicate(item, rel.getValue())) {
274                 if (annotation.isLiteral()) {
275                     undetermined.add(Bundle.Builder.withClass(EntityClass.ACCESS_POINT)
276                             .addDataValue(Ontology.ANNOTATION_TYPE, rel.getKey())
277                             .addDataValue(Ontology.NAME_KEY, annotation.toString())
278                             .build());
279                 } else {
280                     if (rel.getKey().startsWith("skos:") || rel.getKey().startsWith("sem:")) {
281                         String prefix = rel.getKey().startsWith("skos:") ? "skos" : "sem";
282                         Optional<AuthoritativeItem> found = findRelatedConcept(annotation.toString());
283                         if (found.isPresent()) {
284                             linkedItems.put(found.get(), rel.getKey());
285                         } else {
286                             undetermined.add(Bundle.Builder.withClass(EntityClass.ACCESS_POINT)
287                                     .addDataValue(Ontology.ANNOTATION_TYPE, "associate")
288                                     .addDataValue(prefix, rel.getKey().substring(rel.getKey().indexOf(":") + 1))
289                                     .addDataValue(Ontology.NAME_KEY, annotation.toString())
290                                     .build());
291                         }
292                     }
293                 }
294             }
295         }
296         return undetermined;
297     }

299     private Optional<AuthoritativeItem> findRelatedConcept(String name) {
300         if (name != null) {
301             String[] domains = name.split("/");
302             if (domains.length > 2) {
303                 String cvocId = domains[domains.length - 2];
304                 String conceptId = domains[domains.length - 1];
305                 AuthoritativeSet referredSet;
306                 try {
307                     GraphManager manager = GraphManagerFactory.getInstance(framedGraph);
308                     referredSet = manager.getFrame(cvocId, AuthoritativeSet.class);
309                     for (AuthoritativeItem authItem : referredSet.getAuthoritativeItems()) {
310                         if (authItem.getIdentifier().equals(conceptId)) {
311                             return Optional.of(authItem);
312                         }
313                     }
314                 } catch (ItemNotFound ex) {
315                     logger.error("AuthoritativeSet with id " + cvocId + " not found. " + ex.getMessage());
316                 }
317             }
318         }
319         return Optional.absent();
320     }

327     private List<RDFNode> getObjectWithPredicate(Resource item, final URI propUri) {
328         // NB: this should be possible with simply item.listProperties(propUri)
329         // but for some reason that doesn't work... I can't grok why.
330         return item.listProperties().filterKeep(new Filter<Statement>() {
331             @Override
332             public boolean accept(Statement statement) {
333                 return statement.getPredicate().hasURI(propUri.toString());
334             }
335         }).mapWith(new Map1<Statement, RDFNode>() {
336             @Override
337             public RDFNode map1(Statement statement) {
338                 return statement.getObject();
339             }
340         }).toList();
341     }

343     private void connectRelation(Concept current, Resource item, Map<Resource, Concept> others,
344             URI propUri, ConnectFunc connectFunc) {
345         for (RDFNode other : getObjectWithPredicate(item, propUri)) {
346             if (other.isResource()) {
347                 Concept related = others.get(other.asResource());
348                 if (related != null) {
349                     connectFunc.connect(current, related);
350                 }
351             }
352         }
353     }

355     private void hookupRelationships(Resource item, Concept current,
356             Map<Resource, Concept> conceptMap) {
357 
358         connectRelation(current, item, conceptMap, SkosRDFVocabulary.BROADER.getURI(), new ConnectFunc() {
359             @Override
360             public void connect(Concept current, Concept related) {
361                 related.addNarrowerConcept(current);
362             }
363         });
364 
365         connectRelation(current, item, conceptMap, SkosRDFVocabulary.NARROWER.getURI(), new ConnectFunc() {
366             @Override
367             public void connect(Concept current, Concept related) {
368                 current.addNarrowerConcept(related);
369             }
370         });
371 
372         connectRelation(current, item, conceptMap, SkosRDFVocabulary.RELATED.getURI(), new ConnectFunc() {
373             @Override
374             public void connect(Concept current, Concept related) {
375                 current.addRelatedConcept(related);
376             }
377         });
378     }

380     private List<Bundle> getDescriptions(Resource item) {
381         List<Bundle> descriptions = Lists.newArrayList();
382 
383         for (RDFNode property : getObjectWithPredicate(item, SkosRDFVocabulary.PREF_LABEL.getURI())) {
384             if (!property.isLiteral()) {
385                 continue;
386             }
387 
388             Bundle.Builder builder = Bundle.Builder.withClass(EntityClass.CVOC_CONCEPT_DESCRIPTION);
389 
390             Literal literalPrefName = property.asLiteral();
391             String languageCode = isValidLanguageCode(literalPrefName.getLanguage())
392                     ? Helpers.iso639DashTwoCode(literalPrefName.getLanguage())
393                     : defaultLang;
394 
395             builder.addDataValue(Ontology.NAME_KEY, literalPrefName.getString())
396                     .addDataValue(Ontology.LANGUAGE, languageCode);
397 
398             for (Map.Entry<String, URI> prop : SkosRDFVocabulary.GENERAL_PROPS.entrySet()) {
399                 for (RDFNode target : getObjectWithPredicate(item, prop.getValue())) {
400                     if (target.isLiteral()) {
401                         if (prop.getKey().equals("latitude/longitude")) {
402                             String[] latLon = target.asLiteral().getString().split(",");
403                             if (latLon.length > 1) {
404                                 builder.addDataValue("latitude", latLon[0]);
405                                 builder.addDataValue("longitude", latLon[1]);
406                             }
407                         } else {
408                             builder.addDataValue(prop.getKey(), target.asLiteral().getString());
409                         }
410                     } else {
411                         builder.addDataValue(prop.getKey(), target.toString());
412                     }
413                 }
414             }
415 
416             for (Map.Entry<String, List<URI>> prop : SkosRDFVocabulary.LANGUAGE_PROPS.entrySet()) {
417                 List<String> values = Lists.newArrayList();
418                 for (URI uri : prop.getValue()) {
419                     for (RDFNode target : getObjectWithPredicate(item, uri)) {
420                         if (target.isLiteral()) {
421                             Literal literal = target.asLiteral();
422                             String propLanguageCode = isValidLanguageCode(literal.getLanguage())
423                                     ? Helpers.iso639DashTwoCode(literal.getLanguage())
424                                     : defaultLang;
425                             if (propLanguageCode.equals(languageCode)) {
426                                 values.add(literal.getString());
427                             }
428                         }
429                     }
430                 }
431                 if (!values.isEmpty()) {
432                     // Sorting the related literal values in
433                     // natural order gives consistency between
434                     // import/export round-trips. The data is
435                     // otherwise unsorted.
436                     Collections.sort(values);
437                     builder.addDataValue(prop.getKey(), values);
438                 }
439             }
440             Bundle bundle = builder.build();
441             logger.trace(bundle.toJson());
442             descriptions.add(bundle);
443         }
444 
445         return descriptions;
446     }

448     private String getId(URI uri) {
449         return uri.getPath().substring(uri.getPath().lastIndexOf("/") + 1)
450                 + (uri.getQuery() != null ? uri.getQuery() : "")
451                 + (uri.getFragment() != null ? uri.getFragment() : "");
452     }
