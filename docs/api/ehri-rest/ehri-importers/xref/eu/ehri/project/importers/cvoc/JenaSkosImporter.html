168      * Import an input stream.
169      *
170      * @param ios The SKOS file input stream
171      * @param logMessage A log message
172      * @return A log of imported nodes
173      * @throws IOException
174      * @throws ValidationError
175      */
176     @Override
177     public ImportLog importFile(InputStream ios, String logMessage)
178             throws IOException, ValidationError {
179         try {
180             // Create a new action for this import
181             final ActionManager.EventContext eventContext = new ActionManager(framedGraph, vocabulary).logEvent(
182                     actioner, EventTypes.ingest, getLogMessage(logMessage));
183             // Create a manifest to store the results of the import.
184             final ImportLog log = new ImportLog(eventContext);
185 
186             OntModel model = ModelFactory.createOntologyModel();
187             model.read(ios, null, format);
188             OntClass conceptClass = model.getOntClass(SkosRDFVocabulary.CONCEPT.getURI().toString());
189             logger.debug("in import file: " + SkosRDFVocabulary.CONCEPT.getURI().toString());
190             ExtendedIterator<? extends OntResource> extendedIterator = conceptClass.listInstances();
191             Map<Resource, Concept> imported = Maps.newHashMap();
192 
193             while (extendedIterator.hasNext()) {
194                 Resource item = extendedIterator.next();
195 
196                 try {
197                     Mutation<Concept> graphConcept = importConcept(item);
198                     imported.put(item, graphConcept.getNode());
199 
200                     switch (graphConcept.getState()) {
201                         case UNCHANGED:
202                             log.addUnchanged();
203                             break;
204                         case CREATED:
205                             log.addCreated();
206                             eventContext.addSubjects(graphConcept.getNode());
207                             break;
208                         case UPDATED:
209                             log.addUpdated();
210                             eventContext.addSubjects(graphConcept.getNode());
211                             break;
212                     }
213                 } catch (ValidationError validationError) {
214                     if (tolerant) {
215                         logger.error(validationError.getMessage());
216                         log.setErrored(item.toString(), validationError.getMessage());
217                     } else {
218                         throw validationError;
219                     }
220                 }
221             }
222 
223             for (Map.Entry<Resource, Concept> pair : imported.entrySet()) {
224                 hookupRelationships(pair.getKey(), pair.getValue(), imported);
225             }
226 
227             for (Concept concept : imported.values()) {
228                 vocabulary.addItem(concept);
229                 concept.setPermissionScope(vocabulary);
230             }
231 
232             if (log.hasDoneWork()) {
233                 framedGraph.getBaseGraph().commit();
234             } else {
235                 framedGraph.getBaseGraph().rollback();
236             }
237 
238             return log;
239         } catch (ValidationError error) {
240             framedGraph.getBaseGraph().rollback();
241             throw error;
242         }
243     }

245     private Mutation<Concept> importConcept(Resource item) throws ValidationError {
246         logger.debug("Importing: {}", item.toString());
247         Bundle.Builder builder = Bundle.Builder.withClass(EntityClass.CVOC_CONCEPT)
248                 .addDataValue(Ontology.IDENTIFIER_KEY, getId(URI.create(item.getURI())));
249 
250         Map<Concept, String> linkedConcepts = new HashMap<Concept, String>();
251 
252         List<Bundle> undetermined = getUndeterminedRelations(item, linkedConcepts);
253         for (Bundle description : getDescriptions(item)) {
254             Bundle withRels = description
255                     .withRelations(Ontology.HAS_ACCESS_POINT, undetermined);
256             builder.addRelation(Ontology.DESCRIPTION_FOR_ENTITY, withRels);
257         }
258 
259         Mutation<Concept> mut = dao.createOrUpdate(builder.build(), Concept.class);
260         solveUndeterminedRelationships(mut.getNode(), linkedConcepts);
261         return mut;
262     }

264     private void solveUndeterminedRelationships(Concept unit, Map<Concept, String> linkedConcepts) {
265         GraphManager manager = GraphManagerFactory.getInstance(framedGraph);
266         for (Concept concept : linkedConcepts.keySet()) {
267             try {
268                 String reltype = linkedConcepts.get(concept);
269                 Bundle linkBundle = new Bundle(EntityClass.LINK)
270                         .withDataValue(Ontology.LINK_HAS_TYPE, "associate")
271                         .withDataValue(reltype.substring(0, reltype.indexOf(":")), reltype.substring(reltype.indexOf(":")+1))
272                         .withDataValue(Ontology.LINK_HAS_DESCRIPTION, EaImporter.RESOLVED_LINK_DESC);
273                 UserProfile user = manager.getFrame(actioner.getId(), UserProfile.class);
274                 Link link;
275                 link = new CrudViews<Link>(framedGraph, Link.class).create(linkBundle, user);
276                 unit.addLink(link);
277                 concept.addLink(link);
278             } catch (ItemNotFound ex) {
279                 logger.error(ex.getMessage());
280             } catch (PermissionDenied ex) {
281                 logger.error(ex.getMessage());
282             } catch (ValidationError ex) {
283                 java.util.logging.Logger.getLogger(JenaSkosImporter.class.getName()).log(Level.SEVERE, null, ex);
284             }
285 
286         }
287     }

289     private List<Bundle> getUndeterminedRelations(Resource item, Map<Concept, String> linkedConcepts) {
290         List<Bundle> undetermined = Lists.newArrayList();
291 
292         for (Map.Entry<String, URI> rel : RELATION_PROPS.entrySet()) {
293             for (RDFNode annotation : getObjectWithPredicate(item, rel.getValue())) {
294                 if (annotation.isLiteral()) {
295                     undetermined.add(new Bundle(EntityClass.UNDETERMINED_RELATIONSHIP)
296                             .withDataValue(Ontology.ANNOTATION_TYPE, rel.getKey())
297                             .withDataValue(Ontology.NAME_KEY, annotation.toString()));
298                 } else {
299                     if (rel.getKey().startsWith("skos:") || rel.getKey().startsWith("sem:") ) {
300                         String prefix = rel.getKey().startsWith("skos:") ? "skos" : "sem";
301                         Concept found = findRelatedConcept(annotation.toString());
302                         if (found != null) {
303                             linkedConcepts.put(found, rel.getKey());
304                         } else {
305                             undetermined.add(new Bundle(EntityClass.UNDETERMINED_RELATIONSHIP)
306                                     .withDataValue(Ontology.ANNOTATION_TYPE, "associate")
307                                     .withDataValue(prefix, rel.getKey().substring(rel.getKey().indexOf(":")+1))
308                                     .withDataValue(Ontology.NAME_KEY, annotation.toString()));
309                         }
310                     }
311                 }
312             }
313         }
314         return undetermined;
315     }

317     private Concept findRelatedConcept(String name) {
318         if (name != null) {
319             String[] domains = name.split("/");
320             if (domains.length > 2) {
321                 String cvoc_id = domains[domains.length - 2];
322                 String concept_id = domains[domains.length - 1];
323                 Vocabulary referredvocabulary;
324                 try {
325                     GraphManager manager = GraphManagerFactory.getInstance(framedGraph);
326                     referredvocabulary = manager.getFrame(cvoc_id, Vocabulary.class);
327                     for (Concept concept : referredvocabulary.getConcepts()) {
328                         if (concept.getIdentifier().equals(concept_id)) {
329                             return concept;
330                         }
331                     }
332                 } catch (ItemNotFound ex) {
333                     logger.error("Vocabulary with id " + cvoc_id + " not found. " + ex.getMessage());
334                 }
335             }
336         }
337         return null;
338     }

345     private List<RDFNode> getObjectWithPredicate(final Resource item, final URI propUri) {
346         // NB: this should be possible with simply item.listProperties(propUri)
347         // but for some reason that doesn't work... I can't grok why.
348         return item.listProperties().filterKeep(new Filter<Statement>() {
349             @Override
350             public boolean accept(Statement statement) {
351                 return statement.getPredicate().hasURI(propUri.toString());
352             }
353         }).mapWith(new Map1<Statement, RDFNode>() {
354             @Override
355             public RDFNode map1(Statement statement) {
356                 return statement.getObject();
357             }
358         }).toList();
359     }

361     private void connectRelation(Concept current, Resource item, Map<Resource, Concept> others,
362             URI propUri, ConnectFunc connectFunc) {
363         for (RDFNode other : getObjectWithPredicate(item, propUri)) {
364             if (other.isResource()) {
365                 Concept related = others.get(other.asResource());
366                 if (related != null) {
367                     connectFunc.connect(current, related);
368                 }
369             }
370         }

373     private void hookupRelationships(Resource item, Concept current,
374             Map<Resource, Concept> conceptMap) {
375 
376         connectRelation(current, item, conceptMap, SkosRDFVocabulary.BROADER.getURI(), new ConnectFunc() {
377             @Override
378             public void connect(Concept current, Concept related) {
379                 related.addNarrowerConcept(current);
380             }
381         });
382 
383         connectRelation(current, item, conceptMap, SkosRDFVocabulary.NARROWER.getURI(), new ConnectFunc() {
384             @Override
385             public void connect(Concept current, Concept related) {
386                 current.addNarrowerConcept(related);
387             }
388         });
389 
390         connectRelation(current, item, conceptMap, SkosRDFVocabulary.RELATED.getURI(), new ConnectFunc() {
391             @Override
392             public void connect(Concept current, Concept related) {
393                 current.addRelatedConcept(related);
394             }
395         });
396     }

398     private List<Bundle> getDescriptions(Resource item) {
399         List<Bundle> descriptions = Lists.newArrayList();
400 
401         for (RDFNode property : getObjectWithPredicate(item, SkosRDFVocabulary.PREF_LABEL.getURI())) {
402             if (!property.isLiteral()) {
403                 continue;
404             }
405 
406             Bundle.Builder builder = Bundle.Builder.withClass(EntityClass.CVOC_CONCEPT_DESCRIPTION);
407 
408             Literal literalPrefName = property.asLiteral();
409             String languageCode = isValidLanguageCode(literalPrefName.getLanguage())
410                     ? Helpers.iso639DashTwoCode(literalPrefName.getLanguage())
411                     : defaultLang;
412 
413             builder.addDataValue(Ontology.NAME_KEY, literalPrefName.getString())
414                     .addDataValue(Ontology.LANGUAGE, languageCode);
415 
416             for (Map.Entry<String, URI> prop : GENERAL_PROPS.entrySet()) {
417                 for (RDFNode target : getObjectWithPredicate(item, prop.getValue())) {
418                     if (target.isLiteral()) {
419                         if (prop.getKey().equals("latitude/longitude")) {
420                             String[] latlong = target.asLiteral().getString().split(",");
421                             if (latlong.length > 1) {
422                                 builder.addDataValue("latitude", latlong[0]);
423                                 builder.addDataValue("longitude", latlong[1]);
424                             }
425                         } else {
426                             builder.addDataValue(prop.getKey(), target.asLiteral().getString());
427                         }
428                     } else {
429                         builder.addDataValue(prop.getKey(), target.toString());
430                     }
431                 }
432             }
433 
434             for (Map.Entry<String, URI> prop : LANGUAGE_PROPS.entrySet()) {
435                 List<String> values = Lists.newArrayList();
436 
437                 for (RDFNode target : getObjectWithPredicate(item, prop.getValue())) {
438                     if (target.isLiteral()) {
439                         Literal literal = target.asLiteral();
440                         String propLanguageCode = isValidLanguageCode(literal.getLanguage())
441                                 ? Helpers.iso639DashTwoCode(literal.getLanguage())
442                                 : defaultLang;
443                         if (propLanguageCode.equals(languageCode)) {
444                             values.add(literal.getString());
445                         }
446                     }
447                 }
448                 if (!values.isEmpty()) {
449                     builder.addDataValue(prop.getKey(), values);
450                 }
451             }
452             Bundle bundle = builder.build();
453             logger.trace(bundle.toJson());
454             descriptions.add(bundle);
455         }
456 
457         return descriptions;
458     }

460     private String getId(URI uri) {
461         return uri.getPath().substring(uri.getPath().lastIndexOf("/") + 1)
462                 + (uri.getQuery() != null ? uri.getQuery() : "")
463                 + (uri.getFragment() != null ? uri.getFragment() : "");
