137     public ImportLog importFile(InputStream ios, String logMessage)
138             throws IOException, ValidationError, InputParseError {
139         try {
140             // Create a new action for this import
141             final EventContext eventContext = new ActionManager(framedGraph, vocabulary).logEvent(
142                     actioner, EventTypes.ingest, getLogMessage(logMessage));
143             // Create a manifest to store the results of the import.
144             final ImportLog log = new ImportLog(eventContext);
145 
146             // Do the import...
147             importFile(ios, eventContext, log);
148             // If nothing was imported, remove the action...
149             commitOrRollback(log.hasDoneWork());
150             return log;
151         } catch (ValidationError e) {
152             commitOrRollback(false);
153             throw e;
154         } catch (Exception e) {
155             commitOrRollback(false);
156             throw new RuntimeException(e);
157         }
158     }

160     private void importFile(InputStream ios, final ActionManager.EventContext eventContext,
161             final ImportLog log) throws IOException, ValidationError,
162             InputParseError, InvalidXmlDocument, InvalidInputFormatError, IntegrityError {
163 
164         // XML parsing boilerplate...
165         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
166         factory.setNamespaceAware(true);
167         factory.setValidating(!tolerant);
168 
169         try {
170             DocumentBuilder builder = factory.newDocumentBuilder();
171             if (tolerant)
172                 builder.setEntityResolver(new DummyEntityResolver());
173             Document doc = builder.parse(ios);
174             logger.debug("xml encoding: " + doc.getXmlEncoding());
175             importDocWithinAction(doc, eventContext, log);
176         } catch (ParserConfigurationException e) {
177             logger.error(e.getMessage());
178             throw new RuntimeException(e);
179         } catch (SAXException e) {
180             logger.error(e.getMessage());
181             throw new InputParseError(e);
182         }
183 
184     }

205     /**
206      * Do the Concept data extraction and create all the Concepts
207      *
208      * @param doc
209      * @param action
210      * @param manifest
211      * @throws ValidationError
212      * @throws InvalidInputFormatError
213      * @throws IntegrityError
214      */
215     private void createConcepts(Document doc, final ActionManager.EventContext action,
216             final ImportLog manifest) throws ValidationError,
217             InvalidInputFormatError, IntegrityError {
218 
219         // the Concept is extracted from the XML
220         // but how to use the namespacing properly?
221         // for now suppose the root node is rdf:RDF
222         if (doc.getDocumentElement().getNodeName().equals("rdf:RDF")) {
223             Element rdfElement = doc.getDocumentElement();
224             // get all concepts,  without XPath
225             // NOTE the concept can be in a skos:Concept or an rdf:Description
226             // for now assume we have RDF !
227             NodeList nodeList = rdfElement.getElementsByTagName("rdf:Description");
228             for (int i = 0; i < nodeList.getLength(); i++) {
229                 Node childNode = nodeList.item(i);
230                 Element element = (Element) childNode; // it should be!
231 
232                 if (isConceptElement(element)) {
233                     try {
234                         Bundle unit = constructBundleForConcept(element);
235 
236                         BundleDAO persister = new BundleDAO(framedGraph, vocabulary.idPath());
237                         Mutation<Concept> mutation = persister.createOrUpdate(unit,
238                                 Concept.class);
239                         Concept frame = mutation.getNode();
240 
241 
242                         // Set the vocabulary/concept relationship
243                         handleCallbacks(mutation, manifest);
244 
245                         if (mutation.created() || mutation.unchanged()) {
246                             // when concept was successfully persisted!
247                             action.addSubjects(frame);
248                         }
249 
250                         // FIXME: Handle case where relationships have changed on update???
251                         if (mutation.created()) {
252                             frame.setVocabulary(vocabulary);
253                             frame.setPermissionScope(vocabulary);
254 
255                             // Create and add a ConceptPlaceholder
256                             // for making the vocabulary (relation) structure in the next step
257                             List<String> broaderIds = getBroaderConceptIds(element);
258                             logger.debug("Concept has " + broaderIds.size()
259                                     + " broader ids: " + broaderIds.toString());
260                             List<String> relatedIds = getRelatedConceptIds(element);
261                             logger.debug("Concept has " + relatedIds.size()
262                                     + " related ids: " + relatedIds.toString());
263 
264                             String storeId = unit.getId();//id;
265                             String skosId = (String) unit.getDataValue(CONCEPT_URL);
266                             // referal
267                             logger.debug("Concept store id = " + storeId + ", skos id = " + skosId);
268                             conceptLookup.put(skosId, new ConceptPlaceholder(storeId, broaderIds, relatedIds, frame));
269                         }
270                     } catch (ValidationError validationError) {
271                         if (tolerant) {
272                             logger.error(validationError.getMessage());
273                         } else {
274                             throw validationError;
275                         }
276                     }
277                 }
278             }
279         }
280     }

282     /**
283      * Extract data and construct the bundle for a new Concept
284      *
285      * @param element
286      * @throws ValidationError
287      */
288     private Bundle constructBundleForConcept(Element element) throws ValidationError {
289         Bundle unit = new Bundle(EntityClass.CVOC_CONCEPT, extractCvocConcept(element));
290 
291         // add the description data to the concept as relationship
292         Map<String, Object> descriptions = extractCvocConceptDescriptions(element);
293 
294         for (String key : descriptions.keySet()) {
295             logger.debug("description for: " + key);
296             Map<String, Object> d = (Map<String, Object>) descriptions.get(key);
297             logger.debug("languageCode = " + d.get(Ontology.LANGUAGE_OF_DESCRIPTION));
298 
299             Bundle descBundle = new Bundle(EntityClass.CVOC_CONCEPT_DESCRIPTION, d);
300             Map<String, Object> rel = extractRelations(element, "owl:sameAs");
301             if (!rel.isEmpty()) {
302                 descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT,
303                         new Bundle(EntityClass.UNDETERMINED_RELATIONSHIP, rel));
304             }
305 
306             // NOTE maybe test if prefLabel is there?
307             unit = unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
308         }
309 
310         return unit;
311     }

374     /**
375      * Create the broader/narrower relations for all the concepts
376      *
377      * @param doc
378      * @param eventContext
379      * @param manifest
380      * @throws ValidationError
381      * @throws InvalidInputFormatError
382      * @throws IntegrityError
383      */
384     private void createBroaderNarrowerRelations(Document doc, final EventContext eventContext,
385             final ImportLog manifest) throws ValidationError,
386             InvalidInputFormatError, IntegrityError {
387 
388         // check the lookup and start making the BT relations (and narrower implicit)
389         // visit all concepts an see if they have broader concepts
390         for (String skosId : conceptLookup.keySet()) {
391             ConceptPlaceholder conceptPlaceholder = conceptLookup.get(skosId);
392             if (!conceptPlaceholder.broaderIds.isEmpty()) {
393                 logger.debug("Concept with skos id [" + skosId
394                         + "] has broader concepts of " + conceptPlaceholder.broaderIds.size());
395                 // find them in the lookup
396                 for (String bsId : conceptPlaceholder.broaderIds) {
397                     logger.debug(bsId);
398                     if (conceptLookup.containsKey(bsId)) {
399                         // found
400                         logger.debug("Found mapping from: " + bsId
401                                 + " to: " + conceptLookup.get(bsId).storeId);
402 
403                         createBroaderNarrowerRelation(conceptLookup.get(bsId), conceptPlaceholder);
404                     } else {
405                         // not found
406                         logger.debug("Found NO mapping for: " + bsId);
407                         // NOTE What does this mean; refers to an External resource, not in this file?
408                     }
409                 }
410             }
411         }
412     }

414     /**
415      * Create the Broader to Narrower Concept relation
416      * Note that we cannot use the storeId's and retrieve them from the database.
417      * we need to use the Concept objects from the placeholders in the lookup.
418      *
419      * @param bcp
420      * @param ncp
421      */
422     private void createBroaderNarrowerRelation(ConceptPlaceholder bcp, ConceptPlaceholder ncp) {
423         logger.debug("Creating Broader: " + bcp.storeId + " to Narrower: " + ncp.storeId);
424 
425         // An item CANNOT be a narrower version of itself!
426         if (bcp.concept.equals(ncp.concept)) {
427             logger.error("Ignoring cyclic narrower relationship on {}", bcp.concept.getId());
428         } else {
429             bcp.concept.addNarrowerConcept(ncp.concept);
430         }
431     }

437      * @param eventContext

443     private void createNonspecificRelations(Document doc, final EventContext eventContext,

446 
447         // check the lookup and start making the relations
448         // visit all concepts an see if they have a related concept
449         for (String skosId : conceptLookup.keySet()) {

451             if (!conceptPlaceholder.relatedIds.isEmpty()) {
452                 logger.debug("Concept with skos id [" + skosId
453                         + "] has related concepts");
454                 // find them in the lookup
455                 for (String toCId : conceptPlaceholder.relatedIds) {
456                     if (conceptLookup.containsKey(toCId)) {
457                         // found
458                         logger.debug("Found mapping from: " + toCId
459                                 + " to: " + conceptLookup.get(toCId).storeId);
460 
461                         createNonspecificRelation(conceptPlaceholder, conceptLookup.get(toCId));
462                     } else {
463                         // not found
464                         logger.debug("Found NO mapping for: " + toCId);
465                         // NOTE What does this mean; refers to an External resource, not in this file?
466                     }
467                 }
468             }
469         }
470     }

472     /**
473      * Create a non specific relation; this is the 'skos:related' on a concept
474      *
475      * @param from
476      * @param to
477      */
478     private void createNonspecificRelation(ConceptPlaceholder from, ConceptPlaceholder to) {
479 
480         // Prevent creating the relation both ways
481         // NOTE, optimization possible if we could prevent for checking everything twice!
482         if (!reverseRelationExists(from, to)) {
483             logger.debug("Creating relation from: " + from.storeId + " to: " + to.storeId);
484             from.concept.addRelatedConcept(to.concept);
485         }
486     }

488     /**
489      * Check if the 'reverse' relation is exists already, via relatedBy
490      *
491      * @param from
492      * @param to
493      */
494     private boolean reverseRelationExists(ConceptPlaceholder from, ConceptPlaceholder to) {
495         boolean result = false;
496         for (Concept relatedBy : from.concept.getRelatedByConcepts()) {
497             //logger.debug("Related By: " + relatedBy.asVertex().getProperty(EntityType.ID_KEY));
498             String relatedByStoreId = (String) relatedBy.asVertex().getProperty(EntityType.ID_KEY);
499             if (relatedByStoreId == to.storeId) {
500                 result = true;
501                 break; // found
502             }
503         }
504         logger.debug("Relation exists: " + result + " for: " + to.storeId);
505         return result;
506     }

508     /**
509      * Check if the element represents a Concept
510      *
511      * @param element
512      */
513     private boolean isConceptElement(Element element) {
514         boolean result = false;
515 
516         // For now check if it has an rdf:type element
517         // with the attribute rdf:resource="http://www.w3.org/2004/02/skos/core#Concept"
518 
519         NodeList nodeList = element.getElementsByTagName("rdf:type");
520         for (int i = 0; i < nodeList.getLength(); i++) {
521             Node typeNode = nodeList.item(i);
522             Node namedItem = typeNode.getAttributes().getNamedItem("rdf:resource");
523             if (namedItem.getNodeValue().contentEquals("http://www.w3.org/2004/02/skos/core#Concept")) {
524                 result = true;
525             }
526         }
527 
528         return result;
529     }

534      * Extract the Concept information (but not the descriptions)
535      *
536      * @param conceptNode
537      * @throws ValidationError
538      */
539     Map<String, Object> extractCvocConcept(Node conceptNode)
540             throws ValidationError {
541         Map<String, Object> dataMap = new HashMap<String, Object>();
542 
543         // are we using the rdf:about attribute as 'identifier'
544         Node namedItem = conceptNode.getAttributes().getNamedItem("rdf:about");
545         String value = namedItem.getNodeValue();
546         // Hack! Use everything after the last '/'
547         String idvalue = value.substring(value.lastIndexOf('/') + 1);
548         dataMap.put(Ontology.IDENTIFIER_KEY, idvalue);
549         dataMap.put(CONCEPT_URL, value);
550 
551         logger.debug("Extracting Concept id: " + dataMap.get(Ontology.IDENTIFIER_KEY));
552 
553         return dataMap;
554     }

556     private Map<String, Object> extractRelations(Element element, String skosName) {
557         Map<String, Object> relationNode = new HashMap<String, Object>();
558 
559         NodeList textNodeList = element.getElementsByTagName(skosName);
560         for (int i = 0; i < textNodeList.getLength(); i++) {
561             Node textNode = textNodeList.item(i);
562             // get value
563             String text = textNode.getTextContent();
564             logger.debug("text: \"" + text + "\", skos name: " + skosName);
565 
566             // add to all descriptionData maps
567             relationNode.put(Ontology.ANNOTATION_TYPE, skosName);
568             relationNode.put(Ontology.NAME_KEY, text);
569         }
570         return relationNode;
571     }

573     /**
574      * Extract the Descriptions information for a concept
575      *
576      * @param conceptElement
577      */
578     Map<String, Object> extractCvocConceptDescriptions(Element conceptElement) {
579         // extract and process the textual items (with a language)
580         // one description for each language, so the languageCode serve as a key into the map
581         Map<String, Object> descriptionData = new HashMap<String, Object>();
582 
583         // one and only one
584         extractAndAddToLanguageMapSingleValuedTextToDescriptionData(descriptionData, Ontology.NAME_KEY, "skos:prefLabel", conceptElement);
585         // multiple alternatives is logical
586         extractAndAddMultiValuedTextToDescriptionData(descriptionData,
587                 Ontology.CONCEPT_ALTLABEL, "skos:altLabel", conceptElement);
588         // just allow multiple, its not forbidden by Skos
589         extractAndAddMultiValuedTextToDescriptionData(descriptionData,
590                 Ontology.CONCEPT_SCOPENOTE, "skos:scopeNote", conceptElement);
591         // just allow multiple, its not forbidden by Skos
592         extractAndAddMultiValuedTextToDescriptionData(descriptionData,
593                 Ontology.CONCEPT_DEFINITION, "skos:definition", conceptElement);
594         //<geo:lat>52.43333333333333</geo:lat>
595         extractAndAddToAllMapsSingleValuedTextToDescriptionData(descriptionData, "latitude", "geo:lat", conceptElement);
596         //<geo:long>20.716666666666665</geo:long>
597         extractAndAddToAllMapsSingleValuedTextToDescriptionData(descriptionData, "longitude", "geo:long", conceptElement);
598 
599         //<owl:sameAs>http://www.yadvashem.org/yv/he/research/ghettos_encyclopedia/ghetto_details.asp?cid=1</owl:sameAs>
600         //TODO: must be an UndeterminedRelation, which can then later be resolved
601 
602         // NOTE we could try to also add everything else, using the skos tagname as a key?
603 
604         return descriptionData;
605     }

607     private void extractAndAddToAllMapsSingleValuedTextToDescriptionData(Map<String, Object> descriptionData,
608             String textName, String skosName, Element conceptElement) {
609         NodeList textNodeList = conceptElement.getElementsByTagName(skosName);
610         for (int i = 0; i < textNodeList.getLength(); i++) {
611             Node textNode = textNodeList.item(i);
612             // get value
613             String text = textNode.getTextContent();
614             logger.debug("text: \"" + text + "\", skos name: " + skosName + ", property: " + textName);
615 
616             // add to all descriptionData maps
617             for (String key : descriptionData.keySet()) {
618                 Object map = descriptionData.get(key);
619                 if (map instanceof Map) {
620                     ((Map<String, Object>) map).put(textName, text);
621                 } else {
622                     logger.warn(key + " no description map found");
623                 }
624             }
625         }
626     }

628     /**
629      * Extract a 'single valued' textual description property and add it to the data
630      *
631      * @param descriptionData
632      * @param textName
633      * @param skosName
634      * @param conceptElement
635      */
636     private void extractAndAddToLanguageMapSingleValuedTextToDescriptionData(Map<String, Object> descriptionData,
637             String textName, String skosName, Element conceptElement) {
638 
639         NodeList textNodeList = conceptElement.getElementsByTagName(skosName);
640         for (int i = 0; i < textNodeList.getLength(); i++) {
641             Node textNode = textNodeList.item(i);
642             // get lang attribute, we must have that!
643             Node langItem = textNode.getAttributes().getNamedItem("xml:lang");
644             String lang = langItem.getNodeValue();
645             // get value
646             String text = textNode.getTextContent();
647             logger.debug("text: \"" + text + "\" lang: \"" + lang + "\"" + ", skos name: " + skosName);

665

667         for (int i = 0; i < textNodeList.getLength(); i++) {
668             Node textNode = textNodeList.item(i);
669             // get lang attribute, we must have that!
670             Node langItem = textNode.getAttributes().getNamedItem("xml:lang");
671             String lang = langItem.getNodeValue();
672             // get value
673             String text = textNode.getTextContent();
674             logger.debug("text: \"" + text + "\" lang: \"" + lang + "\"" + ", skos name: " + skosName);
675             // add to descriptionData
676             Map<String, Object> d = getOrCreateDescriptionForLanguage(descriptionData, lang);
677             // get the array if it is there, otherwise create it first
678             if (d.containsKey(textName)) {
679                 // should be a list, add it
680                 ((List<String>) d.get(textName)).add(text);
681             } else {
682                 // create a list first
683                 List<String> textList = new ArrayList<String>();
684                 textList.add(text);

686             }

761     private void commitOrRollback(boolean okay) {
762         TransactionalGraph baseGraph = framedGraph.getBaseGraph();
763         if (baseGraph instanceof TxCheckedNeo4jGraph) {
764             TxCheckedNeo4jGraph graph = (TxCheckedNeo4jGraph) baseGraph;
765             if (!okay && graph.isInTransaction()) {
766                 graph.rollback();
