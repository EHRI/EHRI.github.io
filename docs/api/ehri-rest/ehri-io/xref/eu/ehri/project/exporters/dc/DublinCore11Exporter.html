72      @Override
73      public Document export(Described item, String langCode)
74              throws IOException {
75  
76          // Root
77          Document doc = documentBuilder.newDocument();
78  
79          Element rootElem = doc.createElementNS(OAI_NS, "oai_dc:dc");
80          rootElem.setAttribute("xmlns:oai_dc", OAI_NS);
81          rootElem.setAttribute("xmlns:dc", DC_NS);
82          rootElem.setAttribute("xmlns:xsi", "http://www.w3.org/2001/XMLSchema-instance");
83          rootElem.setAttributeNS("http://www.w3.org/2001/XMLSchema-instance",
84                  "xs:schemaLocation", OAI_NS + " http://www.openarchives.org/OAI/2.0/oai_dc.xsd");
85          doc.appendChild(rootElem);
86  
87          addElement(doc, rootElem, "identifier", item.getIdentifier());
88  
89          Optional<Description> descOpt = LanguageHelpers.getBestDescription(
90                  item, Optional.<Description>absent(), langCode);
91  
92          for (Description desc : descOpt.asSet()) {
93              addElement(doc, rootElem, "title", desc.getName());
94  
95              for (String attr : propertyMappings.keySet()) {
96                  Collection<String> mappedKeys = propertyMappings.get(attr);
97                  for (String key : mappedKeys) {
98                      Object value = desc.getProperty(key);
99                      if (value != null) {
100                         if (value instanceof String) {
101                             addElement(doc, rootElem, attr, value);
102                         } else if (value instanceof List) {
103                             for (Object v : (List) value) {
104                                 addElement(doc, rootElem, attr, v);
105                             }
106                         }
107                         break;
108                     }
109                 }
110             }
111 
112             for (AccessPoint accessPoint : desc.getAccessPoints()) {
113                 AccessPointType type = accessPoint.getRelationshipType();
114                 switch (type) {
115                     case creator:
116                     case subject:
117                         addElement(doc, rootElem, type.name(), accessPoint.getName());
118                         break;
119                     case person:
120                     case corporateBody:
121                     case family:
122                         addElement(doc, rootElem, "relation", accessPoint.getName());
123                         break;
124                     case place:
125                         addElement(doc, rootElem, "coverage", accessPoint.getName());
126                         break;
127                     default:
128                 }
129             }
130         }
131 
132         return doc;
133     }
134 
135     private void addElement(Document doc, Element parent, String tag, Object value) {
136         Element elem = doc.createElementNS(DC_NS, "dc:" + tag);
137         String text = valueTransformers.containsKey(tag)
138                 ? valueTransformers.get(tag).apply(value)
139                 : value.toString();
140         elem.setTextContent(text);
141         parent.appendChild(elem);
142     }
