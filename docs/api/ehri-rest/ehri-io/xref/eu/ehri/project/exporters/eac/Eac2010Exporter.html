107     @Override
108     public Document export(final HistoricalAgent agent, String langCode) throws IOException {
109         Document doc = documentBuilder.newDocument();
110         Comment boilerplateComment = doc.createComment(Resources.toString(
111                 Resources.getResource("export-boilerplate.txt"), StandardCharsets.UTF_8));
112         doc.appendChild(boilerplateComment);
113 
114         Element rootElem = doc.createElement("eac-cpf");
115         rootElem.setAttribute("xmlns", "urn:isbn:1-931666-33-4");
116         rootElem.setAttribute("xmlns:xlink", "http://www.w3.org/1999/xlink");
117         rootElem.setAttributeNS("http://www.w3.org/2001/XMLSchema-instance",
118                 "xs:schemaLocation",
119                 "urn:isbn:1-931666-33-4 http://eac.staatsbibliothek-berlin.de/schema/cpf.xsd");
120         doc.appendChild(rootElem);
121 
122         for (Description desc : Helpers.getBestDescription(
123                 agent, Optional.<Description>absent(), langCode).asSet()) {
124 
125             addControlSection(agent, doc, rootElem, desc);
126 
127             Element cpfDescElem = doc.createElement("cpfDescription");
128             rootElem.appendChild(cpfDescElem);
129 
130             addIdentitySection(agent, desc, doc, cpfDescElem);
131 
132             Element descElem = doc.createElement("description");
133             cpfDescElem.appendChild(descElem);
134 
135             addDatesOfExistence(doc, descElem, desc);
136 
137             for (Map.Entry<String, String> entry : descriptiveTextMappings.entrySet()) {
138                 addTextElements(doc, descElem, desc, entry.getKey(), entry.getValue());
139             }
140             for (Map.Entry<String, String> entry : pureTextMappings.entrySet()) {
141                 addPureTextElements(doc, descElem, desc, entry.getKey(), entry.getValue());
142             }
143 
144             addRelations(doc, cpfDescElem, agent, desc, langCode);
145         }
146 
147         return doc;
148     }

150     private void addRelations(Document doc, Element cpfDesc, HistoricalAgent agent,
151             Description desc, String langCode) {
152 
153         List<Link> linkRels = Lists.newArrayList(agent.getLinks());
154         if (!linkRels.isEmpty()) {
155             Element relationsElem = doc.createElement("relations");
156             cpfDesc.appendChild(relationsElem);
157 
158             for (Link link : linkRels) {
159 
160                 Element cpfRelElem = doc.createElement("cpfRelation");
161                 // FIXME: Harmonise this attribute
162                 cpfRelElem.setAttribute("cpfRelationType", "associative");
163                 relationsElem.appendChild(cpfRelElem);
164 
165                 // Look for a body which is an access point
166                 for (String id : getLinkEntityId(agent, link).asSet()) {
167                     cpfRelElem.setAttribute("xlink:href", id);
168                 }
169                 for (String name : getLinkName(agent, desc, link, langCode).asSet()) {
170                     Element cpfRelEntry = doc.createElement("relationEntry");
171                     cpfRelEntry.setTextContent(name);
172                     cpfRelElem.appendChild(cpfRelEntry);
173                 }
174                 for (String name : getLinkDescription(link).asSet()) {
175                     Element descNote = doc.createElement("descriptiveNote");
176                     createCDataElement(doc, descNote, "p", name);
177                     cpfRelElem.appendChild(descNote);
178                 }
179             }
180         }

183     private void addDatesOfExistence(Document doc, Element descElem, Description desc) {
184         List<DatePeriod> allDates = Lists.newArrayList(desc.as(HistoricalAgentDescription.class)
185                 .getDatePeriods());
186         List<DatePeriod> existence = Lists.newArrayList();
187         for (DatePeriod datePeriod : allDates) {
188             if (DatePeriod.DatePeriodType.existence.equals(datePeriod.getDateType())) {
189                 existence.add(datePeriod);
190             }
191         }
192         if (existence.isEmpty() && !allDates.isEmpty()) {
193             existence.add(allDates.get(0));
194         }
195 
196         String datesOfExistence = desc.getProperty("datesOfExistence");
197         if (!existence.isEmpty() || datesOfExistence != null) {
198             Element existDates = doc.createElement("existDates");
199             descElem.appendChild(existDates);
200 
201             for (DatePeriod datePeriod : existence) {
202                 Element dpElem = doc.createElement("dateRange");
203                 existDates.appendChild(dpElem);
204 
205                 String startDate = datePeriod.getStartDate();
206                 String endDate = datePeriod.getEndDate();
207 
208                 if (startDate != null) {
209                     Element start = doc.createElement("fromDate");
210                     String startYear = String.valueOf(new DateTime(startDate).year().get());
211                     start.setAttribute("standardDate", startYear);
212                     start.setTextContent(startYear);
213                     dpElem.appendChild(start);
214                 }
215 
216                 if (endDate != null) {
217                     Element end = doc.createElement("toDate");
218                     String endYear = String.valueOf(new DateTime(endDate).year().get());
219                     end.setAttribute("standardDate", endYear);
220                     end.setTextContent(endYear);
221                     dpElem.appendChild(end);
222                 }
223             }
224 
225             if (existence.isEmpty() && datesOfExistence != null) {
226                 Element date = doc.createElement("date");
227                 existDates.appendChild(date);
228                 date.setTextContent(datesOfExistence);
229             } else if (datesOfExistence != null) {
230                 Element note = doc.createElement("descriptiveNote");
231                 existDates.appendChild(note);
232                 createCDataElement(doc, note, "p", datesOfExistence);
233             }
234         }

495     protected Optional<String> getLinkDescription(Link link) {
496         String desc = link.getDescription();
497         if (desc == null) {
498             for (Accessible other : link.getLinkBodies()) {
499                 if (other.getType().equals(Entities.ACCESS_POINT)) {
500                     AccessPoint ap = other.as(AccessPoint.class);
501                     desc = ap.getProperty("description");
502                 }
503             }
504         }
505         if (desc != null && !desc.trim().isEmpty()) {
506             return Optional.of(desc);
507         }
508         return Optional.absent();
509     }

511     protected Optional<String> getLinkName(Described entity,
512             Description description, Link link, String lang) {
513         for (Accessible other : link.getLinkBodies()) {
514             // We only use an access point body for the name of this link
515             // if the access point is on the current entity (otherwise the
516             // link will have the same name as our current item.)
517             if (other.getType().equals(Entities.ACCESS_POINT)) {
518                 AccessPoint ap = other.as(AccessPoint.class);
519                 for (AccessPoint outAp : description.getAccessPoints()) {
520                     if (outAp.equals(ap)) {
521                         return Optional.of(ap.getName());
522                     }
523                 }
524             }
525         }
526         for (Accessible other : link.getLinkTargets()) {
527             if (!other.equals(entity)) {
528                 return Optional.of(getEntityName(other.as(Described.class), lang));
529             }
530         }
531 
532         return Optional.absent();
533     }

535     protected Optional<String> getLinkEntityId(Described entity, Link link) {
536         for (Accessible other : link.getLinkTargets()) {
537             if (!other.equals(entity)) {
538                 return Optional.of(other.getId());
539             }
540         }
541         return Optional.absent();
542     }

553     private void addTextElements(Document doc, Element parent, Description desc, String key, String path)
554             throws IOException {
555         for (Object prop : Optional.fromNullable(desc.getProperty(key)).asSet()) {
556             String elemName = path;
557             Element top = parent;
558             if (elemName.contains("/")) {
559                 List<String> strings = Splitter.on("/").splitToList(elemName);
560                 for (int i = 0; i < strings.size() - 1; i++) {
561                     Element next = doc.createElement(strings.get(i));
562                     top.appendChild(next);
563                     top = next;
564                 }
565                 elemName = strings.get(strings.size() - 1);
566             }
567             List values = prop instanceof List
568                     ? (List) prop
569                     : Lists.newArrayList(prop);
570             for (Object value : values) {
571                 Element propElem = (Element) top.appendChild(doc.createElement(elemName));
572                 propElem.setTextContent(value.toString());
573             }
574         }
575     }

581             parent.appendChild(propElem);
