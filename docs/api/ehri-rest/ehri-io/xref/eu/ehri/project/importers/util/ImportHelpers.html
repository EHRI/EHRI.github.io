111     /**
112      * Extract properties from the itemData Map that are marked as unknown, and return them in a new Map.
113      *
114      * @param itemData a Map containing raw properties of a unit
115      * @return returns a Map with all keys from itemData that start with SaxXmlHandler.UNKNOWN
116      * @throws ValidationError never
117      */
118     public static Map<String, Object> extractUnknownProperties(Map<String, Object> itemData) throws ValidationError {
119         Map<String, Object> unknowns = Maps.newHashMap();
120         for (Map.Entry<String, Object> key : itemData.entrySet()) {
121             if (key.getKey().startsWith(UNKNOWN_PREFIX)) {
122                 unknowns.put(key.getKey().substring(UNKNOWN_PREFIX.length()), key.getValue());
123             }
124         }
125         return unknowns;
126     }

129     /**
130      * only properties that have the multivalued-status can actually be multivalued. all other properties will be
131      * flattened by this method.
132      *
133      * @param key    a property key
134      * @param value  a property value
135      * @param entity the EntityClass with which this frameMap must comply
136      */
137     public static Object flattenNonMultivaluedProperties(String key, Object value, EntityClass entity) {
138         if (value instanceof List
139                 && !(nodeProperties.hasProperty(entity.getName(), key)
140                 && nodeProperties.isMultivaluedProperty(entity.getName(), key))) {
141             logger.trace("Flattening array property value: {}: {}", key, value);
142             return stringJoiner.join((List) value);
143         } else {
144             return value;
145         }
146     }

168     /**
169      * Extract a Map containing the properties of a documentary unit's description.
170      * Excludes unknown properties, object identifier(s), maintenance events, relations,
171      * addresses and *Access relations.
172      *
173      * @param itemData a Map containing raw properties of a unit
174      * @param entity   an EntityClass to get the multi-valuedness of properties for
175      * @return a Map representation of a DocumentDescription
176      */
177     public static Map<String, Object> extractUnitDescription(Map<String, Object> itemData, EntityClass entity) {
178         Map<String, Object> description = Maps.newHashMap();
179 
180         description.put(Ontology.CREATION_PROCESS, Description.CreationProcess.IMPORT.toString());
181 
182         for (Map.Entry<String, Object> itemProperty : itemData.entrySet()) {
183             if (itemProperty.getKey().equals("descriptionIdentifier")) {
184                 description.put(Ontology.IDENTIFIER_KEY, itemProperty.getValue());
185             } else if (!itemProperty.getKey().startsWith(UNKNOWN_PREFIX)
186                     && !itemProperty.getKey().equals(OBJECT_IDENTIFIER)
187                     && !itemProperty.getKey().equals(Ontology.IDENTIFIER_KEY)
188                     && !itemProperty.getKey().equals(Ontology.OTHER_IDENTIFIERS)
189                     && !itemProperty.getKey().startsWith(Entities.MAINTENANCE_EVENT)
190                     && !itemProperty.getKey().startsWith(Entities.ACCESS_POINT)
191                     && !itemProperty.getKey().startsWith("IGNORE")
192                     && !itemProperty.getKey().startsWith("address/")
193                     && !itemProperty.getKey().endsWith("AccessPoint")) {
194                 description.put(itemProperty.getKey(), flattenNonMultivaluedProperties(
195                         itemProperty.getKey(), itemProperty.getValue(), entity));
196             }
197         }
198 
199         return description;
200     }

218     /**
219      * Extract datevalues from datamap
220      *
221      * @param data the data map
222      * @return returns a List with the separated datevalues
223      */
224     static Map<String, String> returnDatesAsString(Map<String, Object> data) {
225         Map<String, String> datesAsString = Maps.newHashMap();
226         Object value;
227         for (Map.Entry<String, Object> property : data.entrySet()) {
228             if (dates.containsProperty(property.getKey()) && (value = property.getValue()) != null) {
229                 if (property.getValue() instanceof String) {
230                     String dateValue = (String) value;
231                     for (String d : dateValue.split(",")) {
232                         datesAsString.put(d, property.getKey());
233                     }
234                 } else if (property.getValue() instanceof List) {
235                     for (String s : (List<String>) value) {
236                         datesAsString.put(s, property.getKey());
237                     }
238                 }
239             }
240         }
241         return datesAsString;
242     }

244     /**
245      * Extract a list of entity bundles for DatePeriods from the data,
246      * attempting to parse the unitdate attribute.
247      *
248      * @param data the data map
249      */
250     public static List<Map<String, Object>> extractDates(Map<String, Object> data) {
251         List<Map<String, Object>> extractedDates = Lists.newArrayList();
252 
253         for (String key : data.keySet()) {
254             if (key.equals(Entities.DATE_PERIOD) && data.get(key) instanceof List) {
255                 for (Map<String, Object> event : (List<Map<String, Object>>) data.get(key)) {
256                     extractedDates.add(getSubNode(event));
257                 }
258             }
259         }
260 
261         Map<String, String> dateValues = returnDatesAsString(data);
262         for (String s : dateValues.keySet()) {
263             extractDate(s).ifPresent(extractedDates::add);
264         }
265         return extractedDates;
266     }

268     /**
269      * The dates that have been extracted to the extractedDates will be removed from the data map
270      *
271      * @param data           the data map
272      * @param extractedDates the set of extracted dates
273      */
274     public static void replaceDates(Map<String, Object> data, List<Map<String, Object>> extractedDates) {
275         Map<String, String> dateValues = returnDatesAsString(data);
276         Map<String, String> dateTypes = Maps.newHashMap();
277         for (String dateValue : dateValues.keySet()) {
278             dateTypes.put(dateValues.get(dateValue), null);
279         }
280         for (Map<String, Object> dateMap : extractedDates) {
281             dateValues.remove(dateMap.get(Ontology.DATE_HAS_DESCRIPTION));
282         }
283         //replace dates in data map
284         for (String datevalue : dateValues.keySet()) {
285             String dateType = dateValues.get(datevalue);
286             logger.debug("{} -- {}", datevalue, dateType);
287             if (dateTypes.containsKey(dateType) && dateTypes.get(dateType) != null) {
288                 dateTypes.put(dateType, dateTypes.get(dateType) + ", " + datevalue.trim());
289             } else {
290                 dateTypes.put(dateType, datevalue.trim());
291             }
292         }
293         for (String dateType : dateTypes.keySet()) {
294             logger.debug("datetype {} -- {}", dateType, dateTypes.get(dateType));
295             if (dateTypes.get(dateType) == null) {
296                 data.remove(dateType);
297             } else {
298                 data.put(dateType, dateTypes.get(dateType));
299             }
300             logger.debug("" + data.get(dateType));
301         }
302     }

305      * Extract the data from a sub-node.
306      *
307      * @param event a Map of event properties
308      * @return a data map
309      */
310     public static Map<String, Object> getSubNode(Map<String, Object> event) {
311         Map<String, Object> me = Maps.newHashMap();
312         for (Map.Entry<String, Object> eventEntry : event.entrySet()) {
313             // Hack for EAG 1 and 2012 compatibility - maps maintenance event
314             // types from old to new values
315             if (eventEntry.getKey().equals(Ontology.MAINTENANCE_EVENT_TYPE)) {
316                 me.put(Ontology.MAINTENANCE_EVENT_TYPE, MaintenanceEventType
317                         .withName((String) eventEntry.getValue()).toString());
318             } else {
319                 me.put(eventEntry.getKey(), eventEntry.getValue());
320             }
321         }
322         if (!me.containsKey(Ontology.MAINTENANCE_EVENT_TYPE)) {
323             me.put(Ontology.MAINTENANCE_EVENT_TYPE, MaintenanceEventType.updated.name());
324         }
325         return me;
326     }

332     /**
333      * Normalise a date in a string.
334      *
335      * @param date       a String date that needs formatting
336      * @param endOfPeriod a string signifying whether this date is the begin of
337      *                   a period or the end of a period
338      * @return a String containing the formatted date.
339      */
340     static String normaliseDate(String date, boolean endOfPeriod) {
341         String returnDate = isoDateTimeFormat.print(DateTime.parse(date));
342         if (returnDate.startsWith("00")) {
343             returnDate = "19" + returnDate.substring(2);
344             date = "19" + date;
345         }
346         if (endOfPeriod) {
347             if (!date.equals(returnDate)) {
348                 ParsePosition p = new ParsePosition(0);
349                 yearMonthDateFormat.parse(date, p);
350                 if (p.getIndex() > 0) {
351                     returnDate = isoDateTimeFormat.print(DateTime.parse(date).plusMonths(1).minusDays(1));
352                 } else {
353                     p = new ParsePosition(0);
354                     yearDateFormat.parse(date, p);
355                     if (p.getIndex() > 0) {
356                         returnDate = isoDateTimeFormat.print(DateTime.parse(date).plusYears(1).minusDays(1));
357                     }
358                 }
359             }
360         }
361         return returnDate;
362     }

372     /**
373      * Stores this property value pair in the given graph node representation.
374      * If the value is effectively empty, nothing happens.
375      * If the property already exists, it is added to the value list.
376      *
377      * @param c        a Map representation of a graph node
378      * @param property the key to store the value for
379      * @param value    the value to store
380      */
381     public static void putPropertyInGraph(Map<String, Object> c, String property, String value) {
382         String normValue = normaliseValue(property, value);
383         if (normValue == null || normValue.isEmpty()) {
384             return;
385         }
386         logger.debug("putProp: {} -> {}", property, normValue);
387         if (c.containsKey(property)) {
388             Object currentValue = c.get(property);
389             if (currentValue instanceof List) {
390                 ((List) currentValue).add(normValue);
391             } else {
392                 c.put(property, Lists.newArrayList(currentValue, normValue));
393             }
394         } else {
395             c.put(property, normValue);
396         }
397     }

400      * Attempt to extract some date periods. This does not currently put the dates into ISO form.
401      *
402      * @param date the data map
403      * @return returns a Map with DatePeriod.DATE_PERIOD_START_DATE and DatePeriod.DATE_PERIOD_END_DATE values

405     private static Optional<Map<String, Object>> extractDate(String date) {
406         Map<String, Object> data = matchDate(date);
407         return data.isEmpty() ? Optional.empty() : Optional.of(data);
408     }

410     private static Map<String, Object> matchDate(String date) {
411         Map<String, Object> data = Maps.newHashMap();
412         for (Pattern re : datePatterns) {
413             Matcher matcher = re.matcher(date);
414             if (matcher.matches()) {
415                 logger.debug("matched {}", date);
416                 data.put(Ontology.DATE_PERIOD_START_DATE, normaliseDate(matcher.group(1)));
417                 data.put(Ontology.DATE_PERIOD_END_DATE, normaliseDate(matcher.group(matcher
418                         .groupCount() > 1 ? 2 : 1), true));
419                 data.put(Ontology.DATE_HAS_DESCRIPTION, date);
420                 break;
421             }
422         }
423         return data;
424     }

426     private static String normaliseValue(String property, String value) {
427         String trimmedValue = StringUtils.normalizeSpace(value);
428         // Language codes are converted to their 3-letter alternates
429         return property.startsWith(LANGUAGE_KEY_PREFIX)
430                 ? LanguageHelpers.iso639DashTwoCode(trimmedValue)
431                 : trimmedValue;
432     }

450     private static NodeProperties loadNodeProperties() {
451         try (InputStream fis = ImportHelpers.class.getClassLoader().getResourceAsStream(NODE_PROPERTIES);
452              BufferedReader br = new BufferedReader(new InputStreamReader(fis, Charsets.UTF_8))) {
453             NodeProperties nodeProperties = new NodeProperties();
454             String headers = br.readLine();
455             nodeProperties.setTitles(headers);
456 
457             String line;
458             while ((line = br.readLine()) != null) {
459                 nodeProperties.addRow(line);
460             }
461             return nodeProperties;
462         } catch (IOException ex) {
463             throw new RuntimeException(ex);
464         } catch (NullPointerException npe) {
465             throw new RuntimeException("Missing or empty properties file: " + NODE_PROPERTIES);
466         }
