63  /**
64   * Import EAD for a given repository into the database. Due to the laxness of the EAD standard this is a fairly complex
65   * procedure. An EAD a single entity at the highest level of description or multiple top-level entities, with or without
66   * a hierarchical structure describing their child items. This means that we need to recursively descend through the
67   * archdesc and c,c01-12 levels.
68   * <p/>
69   * TODO: Extensive cleanups, optimisation, and rationalisation.
70   */

91      /**
92       * Import a single archdesc or c01-12 item, keeping a reference to the hierarchical depth.
93       *
94       * @param itemData The raw data map
95       * @param idPath   The identifiers of parent documents,
96       *                 not including those of the overall permission scope
97       * @throws ValidationError when the itemData does not contain an identifier for the unit or...
98       */
99      @Override
100     public AbstractUnit importItem(Map<String, Object> itemData, List<String> idPath)
101             throws ValidationError {
102 
103         BundleManager persister = getPersister(idPath);
104 
105         List<Map<String, Object>> extractedDates = extractDates(itemData);
106         replaceDates(itemData, extractedDates);
107 
108         Bundle descBundle = new Bundle(EntityClass.DOCUMENTARY_UNIT_DESCRIPTION, extractUnitDescription(itemData, EntityClass.DOCUMENTARY_UNIT_DESCRIPTION));
109         // Add dates and descriptions to the bundle since they're @Dependent
110         // relations.
111         for (Map<String, Object> dpb : extractedDates) {
112             descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
113         }
114         for (Map<String, Object> rel : extractRelations(itemData)) {//, (String) unit.getErrors().get(Identifiable.IDENTIFIER_KEY)
115             logger.debug("relation found: {}", rel.get(Ontology.NAME_KEY));
116             for (String s : rel.keySet()) {
117                 logger.debug(s);
118             }
119             descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT, new Bundle(EntityClass.ACCESS_POINT, rel));
120         }
121         Map<String, Object> unknowns = extractUnknownProperties(itemData);
122         if (!unknowns.isEmpty()) {
123             StringBuilder unknownProperties = new StringBuilder();
124             for (String u : unknowns.keySet()) {
125                 unknownProperties.append(u);
126             }
127             logger.debug("Unknown Properties found: {}", unknownProperties);
128             descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, new Bundle(EntityClass.UNKNOWN_PROPERTY, unknowns));
129         }
130 
131         // extractDocumentaryUnit does not throw ValidationError on missing ID
132         Bundle unit = new Bundle(unitEntity, extractDocumentaryUnit(itemData));
133 
134         // Check for missing identifier, throw an exception when there is no ID.
135         if (unit.getDataValue(Ontology.IDENTIFIER_KEY) == null) {
136             throw new ValidationError(unit, Ontology.IDENTIFIER_KEY,
137                     "Missing identifier " + Ontology.IDENTIFIER_KEY);
138         }
139         logger.debug("Imported item: " + itemData.get("name"));
140 
141         Mutation<DocumentaryUnit> mutation =
142                 persister.createOrUpdate(mergeWithPreviousAndSave(unit, descBundle, idPath), DocumentaryUnit.class);
143         DocumentaryUnit frame = mutation.getNode();
144 
145         // Set the repository/item relationship
146         if (idPath.isEmpty() && mutation.created()) {
147             EntityClass scopeType = manager.getEntityClass(permissionScope);
148             if (scopeType.equals(EntityClass.REPOSITORY)) {
149                 Repository repository = framedGraph.frame(permissionScope.asVertex(), Repository.class);
150                 frame.setRepository(repository);
151                 frame.setPermissionScope(repository);
152             } else if (scopeType.equals(unitEntity)) {
153                 DocumentaryUnit parent = framedGraph.frame(permissionScope.asVertex(), DocumentaryUnit.class);
154                 parent.addChild(frame);
155                 frame.setPermissionScope(parent);
156             } else {
157                 logger.error("Unknown scope type for documentary unit: {}", scopeType);
158             }
159         }
160         handleCallbacks(mutation);
161         logger.debug("============== " + frame.getIdentifier() + " created:" + mutation.created());
162         if (mutation.created()) {
163             solveUndeterminedRelationships(frame, descBundle);
164         }
165         return frame;
166

170     /**
171      * Finds any bundle in the graph with the same ObjectIdentifier.
172      * If there is no bundle with this identifier, it is created.
173      * If it exists and a Description in the given language exists from the same source file,
174      * the description is replaced. If the description is from another source, it is added to the
175      * bundle's descriptions.
176      *
177      * @param unit       the DocumentaryUnit to be saved
178      * @param descBundle the documentsDescription to replace any previous ones with this language
179      * @param idPath     the ID path of this bundle (will be relative to the ID path of the permission scope)
180      * @return A bundle with description relationships merged.
181      * @throws ValidationError
182      */
183     protected Bundle mergeWithPreviousAndSave(Bundle unit, Bundle descBundle, List<String> idPath) throws ValidationError {
184         final String languageOfDesc = descBundle.getDataValue(Ontology.LANGUAGE_OF_DESCRIPTION);
185         final String thisSourceFileId = descBundle.getDataValue(Ontology.SOURCEFILE_KEY);
186 
187         logger.debug("merging: descBundle's language = {}, sourceFileId = {}",
188                 languageOfDesc, thisSourceFileId);
189         /*
190          * for some reason, the idpath from the permissionscope does not contain the parent documentary unit.
191          * TODO: so for now, it is added manually
192          */
193         List<String> itemIdPath = Lists.newArrayList(getPermissionScope().idPath());
194         for (String p : idPath) {
195             itemIdPath.add(p);
196         }
197         Bundle unitWithIds = unit.generateIds(itemIdPath);
198         logger.debug("merging: docUnit's graph id = {}", unitWithIds.getId());
199         // If the bundle exists, we merge
200         if (manager.exists(unitWithIds.getId())) {
201             try {
202                 // read the current itemâ€™s bundle
203                 Bundle oldBundle = mergeSerializer
204                         .vertexToBundle(manager.getVertex(unitWithIds.getId()));
205 
206                 // filter out dependents that a) are descriptions, b) have the same language/code
207                 Bundle.Filter filter = new Bundle.Filter() {
208                     @Override
209                     public boolean remove(String relationLabel, Bundle bundle) {
210                         String lang = bundle.getDataValue(Ontology.LANGUAGE);
211                         String oldSourceFileId = bundle.getDataValue(Ontology.SOURCEFILE_KEY);
212                         return bundle.getType().equals(EntityClass.DOCUMENTARY_UNIT_DESCRIPTION)
213                                 && (lang != null
214                                 && lang.equals(languageOfDesc)
215                                 && (oldSourceFileId != null && oldSourceFileId.equals(thisSourceFileId))
216                         );
217                     }
218                 };
219                 Bundle filtered = oldBundle.filterRelations(filter);
220 
221                 // if this desc-id already exists, but with a different sourceFileId, 
222                 // change the desc-id
223                 String defaultDescIdentifier = DescriptionIdGenerator.descriptionJoiner.join(
224                         unitWithIds.getId(), languageOfDesc.toLowerCase());
225                 logger.debug("merging: defaultDescIdentifier = {}", defaultDescIdentifier);
226                 String newDescIdentifier = defaultDescIdentifier + HIERARCHY_SEPARATOR + Slugify.slugify(thisSourceFileId,
227                         SLUG_REPLACE);
228                 logger.debug("merging: newDescIdentifier = {}", newDescIdentifier);
229                 String generatedId = DescriptionIdGenerator.INSTANCE.generateId(itemIdPath, filtered);
230                 logger.debug("merging: generated ID = {}", generatedId);
231                 // First see whether this has been done before and a desc with the new id exists
232                 if (manager.exists(newDescIdentifier)) {
233                     descBundle = descBundle.withDataValue(Ontology.IDENTIFIER_KEY, newDescIdentifier);
234                     String anotherGeneratedId = DescriptionIdGenerator.INSTANCE.generateId(itemIdPath, descBundle);
235                     logger.debug("merging: new desc ID exists, new generated ID = {}", anotherGeneratedId);
236                 } else if (manager.exists(defaultDescIdentifier)) {
237                     Bundle oldDescBundle = mergeSerializer
238                             .vertexToBundle(manager.getVertex(defaultDescIdentifier));
239                     //if the previous had NO sourcefile_key OR it was different:
240                     if (oldDescBundle.getDataValue(Ontology.SOURCEFILE_KEY) == null
241                             || !thisSourceFileId.equals(oldDescBundle.getDataValue(Ontology.SOURCEFILE_KEY).toString())) {
242                         descBundle = descBundle.withDataValue(Ontology.IDENTIFIER_KEY, thisSourceFileId);
243                         logger.info("other description found ({}), creating new description id: {}",
244                                 defaultDescIdentifier, descBundle.getDataValue(Ontology.IDENTIFIER_KEY));
245                         String anotherGeneratedId = DescriptionIdGenerator.INSTANCE.generateId(itemIdPath, descBundle);
246                         logger.debug("merging: def desc ID exists, source file differs, another generated ID = {}", anotherGeneratedId);
247                     }
248                 }
249 
250                 return unitWithIds.withRelations(filtered.getRelations())
251                         .withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
252 
253             } catch (SerializationError ex) {
254                 throw new ValidationError(unit, "serialization error", ex.getMessage());
255             } catch (ItemNotFound ex) {
256                 throw new ValidationError(unit, "item not found exception", ex.getMessage());
257             }
258         } else { // else we create a new bundle.
259             return unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
260         }

263     /**
264      * subclasses can override this method to cater to their special needs for UndeterminedRelationships
265      * by default, it expects something like this in the original EAD:
266      * <p/>
267      * <pre>
268      * {@code
269      * <persname source="terezin-victims" authfilenumber="PERSON.ITI.1514982">Kien,
270      * Leonhard (* 11.5.1886)</persname>
271      * }
272      * </pre>
273      * <p/>
274      * it works in unison with the extractRelations() method.
275      *
276      * @param unit       the current unit
277      * @param descBundle - not used
278      * @throws ValidationError
279      */
280     protected void solveUndeterminedRelationships(DocumentaryUnit unit, Bundle descBundle) throws ValidationError {
281         //Try to resolve the undetermined relationships
282         //we can only create the annotations after the DocumentaryUnit and its Description have been added to the graph,
283         //so they have id's. 
284         for (Description unitdesc : unit.getDescriptions()) {
285             // Put the set of relationships into a HashSet to remove duplicates.
286             for (AccessPoint rel : Sets.newHashSet(unitdesc.getAccessPoints())) {
287                 /*
288                  * the wp2 undetermined relationship that can be resolved have a 'cvoc' and a 'concept' attribute.
289                  * they need to be found in the vocabularies that are in the graph
290                  */
291                 if (rel.getPropertyKeys().contains("cvoc")) {
292                     String cvoc_id = rel.getProperty("cvoc");
293                     String concept_id = rel.getProperty("concept");
294                     if (concept_id == null) {
295                         concept_id = rel.getProperty("target");
296                     }
297                     logger.debug("cvoc:" + cvoc_id + "  concept:" + concept_id);
298                     Vocabulary vocabulary;
299                     try {
300                         vocabulary = manager.getEntity(cvoc_id, Vocabulary.class);
301                         for (Concept concept : vocabulary.getConcepts()) {
302                             logger.debug("*********************" + concept.getId() + " " + concept.getIdentifier());
303                             if (concept.getIdentifier().equalsIgnoreCase(concept_id)) {
304                                 try {
305                                     Bundle linkBundle = new Bundle(EntityClass.LINK)
306                                             .withDataValue(Ontology.LINK_HAS_TYPE, rel.getProperty("type").toString())
307                                             .withDataValue(Ontology.LINK_HAS_DESCRIPTION, RESOLVED_LINK_DESC);
308                                     UserProfile user = manager.getEntity(this.log.getActioner().getId(), UserProfile.class);
309                                     Link link = new CrudViews<>(framedGraph, Link.class).create(linkBundle, user);
310                                     unit.addLink(link);
311                                     concept.addLink(link);
312                                     link.addLinkBody(rel);
313                                     logger.debug("link created between {} and {}", concept_id, concept.getId());
314                                 } catch (PermissionDenied ex) {
315                                     logger.error(ex.getMessage());
316                                 }
317 
318                             }
319 
320                         }
321                     } catch (ItemNotFound ex) {
322                         logger.error("Vocabulary with id " + cvoc_id + " not found. " + ex.getMessage());
323                     }
324 
325                 } else {
326                     logger.debug("no cvoc found");
327                 }
328 
329             }
330         }
331     }

333     @SuppressWarnings("unchecked")
334     @Override
335     protected Iterable<Map<String, Object>> extractRelations(Map<String, Object> data) {
336         String REL = "relation";
337         List<Map<String, Object>> list = Lists.newArrayList();
338         for (String key : data.keySet()) {
339             if (key.equals(REL)) {
340                 //name identifier
341                 for (Map<String, Object> origRelation : (List<Map<String, Object>>) data.get(key)) {
342                     Map<String, Object> relationNode = Maps.newHashMap();
343                     if (origRelation.containsKey("type")) {
344                         //try to find the original identifier
345                         relationNode.put(LINK_TARGET, origRelation.get("concept"));
346                         //try to find the original name
347                         relationNode.put(Ontology.NAME_KEY, origRelation.get("name"));
348                         relationNode.put("cvoc", origRelation.get("cvoc"));
349                         relationNode.put(Ontology.ACCESS_POINT_TYPE, origRelation.get("type"));
350                     } else {
351                         relationNode.put(Ontology.NAME_KEY, origRelation.get(REL));
352                     }
353                     if (!relationNode.containsKey(Ontology.ACCESS_POINT_TYPE)) {
354                         logger.debug("relationNode without type: " + relationNode.get(Ontology.NAME_KEY));
355                         relationNode.put(Ontology.ACCESS_POINT_TYPE, "corporateBodyAccess");
356                     }
357                     list.add(relationNode);
358                 }
359             } else if (key.endsWith(ACCESS_POINT)) {
360 
361                 logger.debug(key + data.get(key).getClass());
362                 if (data.get(key) instanceof List) {
363                     for (Object o : (List) data.get(key)) {
364                         logger.debug("" + o.getClass());
365                     }
366                     //type, targetUrl, targetName, notes
367                     for (Map<String, Object> origRelation : (List<Map<String, Object>>) data.get(key)) {
368                         if (origRelation.isEmpty()) {
369                             break;
370                         }
371                         Map<String, Object> relationNode = Maps.newHashMap();
372                         for (String eventkey : origRelation.keySet()) {
373                             if (eventkey.endsWith(ACCESS_POINT)) {
374                                 relationNode.put(Ontology.ACCESS_POINT_TYPE, eventkey.substring(0, eventkey.indexOf("Point")));
375                                 relationNode.put(Ontology.NAME_KEY, origRelation.get(eventkey));
376 //logger.debug("------------------" + eventkey.substring(0, eventkey.indexOf("Point")) + ": "+ origRelation.get(eventkey));                            
377                             } else {
378                                 relationNode.put(eventkey, origRelation.get(eventkey));
379                             }
380                         }
381                         if (!relationNode.containsKey(Ontology.ACCESS_POINT_TYPE)) {
382                             relationNode.put(Ontology.ACCESS_POINT_TYPE, "corporateBodyAccess");
383                         }
384                         //if no name is given, it was apparently an empty <controlaccess> tag?
385                         if (relationNode.containsKey(Ontology.NAME_KEY)) {
386                             list.add(relationNode);
387                         }
388                     }
389                 } else {
390                     Map<String, Object> relationNode = Maps.newHashMap();
391                     relationNode.put(Ontology.ACCESS_POINT_TYPE, key.substring(0, key.indexOf("Point")));
392                     relationNode.put(Ontology.NAME_KEY, data.get(key));
393                     list.add(relationNode);
394 
395                 }
396             }
397         }
398         return list;
399     }

443     /**
444      * Creates a Map containing properties of a Documentary Unit description.
445      * These properties are the unit description's properties: all except the doc unit identifiers and unknown properties.
446      *
447      * @param itemData Map of all extracted information
448      * @param depth    depth of node in the tree
449      * @return a Map representing a Documentary Unit Description node
450      * @throws ValidationError
451      */
452     protected Map<String, Object> extractDocumentDescription(Map<String, Object> itemData, int depth) throws ValidationError {
453 
454         Map<String, Object> unit = Maps.newHashMap();
455         for (String key : itemData.keySet()) {
456             if (!(key.equals(OBJECT_ID)
457                     || key.equals(Ontology.OTHER_IDENTIFIERS)
458                     || key.startsWith(SaxXmlHandler.UNKNOWN))) {
459                 unit.put(key, itemData.get(key));
460             }
461         }
462         return unit;
463     }
