46  /**
47   * Handler of EAC-CPF files. Creates a {@link Map} for the {@link EacImporter} to
48   * import.
49   * Makes use of eac.properties with format: part/of/path/=attribute
50   */
51  public class EacHandler extends SaxXmlHandler {
52  
53      // Artificial subnodes
54      private static final String NAME_ENTRY = "NameEntry";
55  
56      private final ImmutableMap<String, Class<? extends Entity>> possibleSubNodes = ImmutableMap
57              .<String, Class<? extends Entity>>builder()
58              .put(Entities.MAINTENANCE_EVENT, MaintenanceEvent.class)
59              .put(Entities.ACCESS_POINT, AccessPoint.class)
60              .put(Entities.DATE_PERIOD, DatePeriod.class)
61              .put(NAME_ENTRY, UnknownProperty.class)
62              .build();
63  
64      private static final Logger logger = LoggerFactory.getLogger(EacHandler.class);
65  
66      public EacHandler(ItemImporter<Map<String, Object>, ?> importer) {
67          super(importer, new XmlImportProperties("eac.properties"));
68      }
69  
70      public EacHandler(ItemImporter<Map<String, Object>, ?> importer, XmlImportProperties properties) {
71          super(importer, properties);
72      }
73  
74      @Override
75      protected boolean needToCreateSubNode(String key) {
76          return possibleSubNodes.containsKey(getMappedProperty(currentPath));
77      }
78  
79      @Override
80      public void endElement(String uri, String localName, String qName) throws SAXException {
81          //if a subnode is ended, add it to the super-supergraph
82          super.endElement(uri, localName, qName);
83          if (needToCreateSubNode(getMappedProperty(currentPath))) {
84              Map<String, Object> currentGraph = currentGraphPath.pop();
85              putSubGraphInCurrentGraph(getMappedProperty(currentPath), currentGraph);
86              depth--;
87          }
88  
89          currentPath.pop();
90  
91          //an EAC file consists of only 1 element, so if we're back at the root, we're done
92          if (currentPath.isEmpty()) {
93              try {
94                  logger.debug("depth close {} {}", depth, qName);
95                  //TODO: add any mandatory fields not yet there:
96                  if (!currentGraphPath.peek().containsKey(ImportHelpers.OBJECT_IDENTIFIER)) {
97                      putPropertyInCurrentGraph(ImportHelpers.OBJECT_IDENTIFIER, "id");
98                  }
99  
100                 //TODO: name can have only 1 value, others are otherFormsOfName
101                 if (currentGraphPath.peek().containsKey(NAME_ENTRY)) {
102                     String name = chooseName(currentGraphPath.peek().get(NAME_ENTRY));
103                     overwritePropertyInCurrentGraph(Ontology.NAME_KEY, name);
104                 }
105                 if (!currentGraphPath.peek().containsKey(Ontology.LANGUAGE_OF_DESCRIPTION)) {
106                     logger.debug("no {} found", Ontology.LANGUAGE_OF_DESCRIPTION);
107                     putPropertyInCurrentGraph(Ontology.LANGUAGE_OF_DESCRIPTION, "eng");
108                 }
109 
110                 importer.importItem(currentGraphPath.pop(), Lists.<String>newArrayList());
111 
112             } catch (ValidationError ex) {
113                 logger.error(ex.getMessage());
114             }
115         }
116     }
117 
118     private String chooseName(Object names) {
119         String nameValue;
120         if (names instanceof String) {
121             nameValue = names.toString();
122         } else if (names instanceof List) {
123             Object firstName = ((List) names).get(0);
124             if (firstName instanceof String) {
125                 nameValue = firstName.toString();
126             } else {
127                 Map nameMap = (Map) firstName;
128                 if (nameMap.get("namePart") instanceof String) {
129                     nameValue = nameMap.get("namePart").toString();
130                 } else if (nameMap.get("namePart") instanceof List) {
131                     nameValue = "";
132                     for (Object p : (List) nameMap.get("namePart")) {
133                         nameValue += p + " ";
134                     }
135 
136                 } else {
137                     nameValue = nameMap.get("namePart").toString();
138                 }
139             }
140 
141             Set<String> otherNames = Sets.newHashSet();
142             for (int i = 1; i < ((List) names).size(); i++) {
143                 Map m = (Map) ((List) names).get(i);
144                 Object namePart = m.get("namePart");
145                 if (namePart != null && !namePart.toString().trim().equalsIgnoreCase(nameValue)) {
146                     otherNames.add(namePart.toString().trim());
147                     logger.debug("other name: {}", namePart);
148                 }
149             }
150             otherNames.forEach(n -> putPropertyInCurrentGraph("otherFormsOfName", n));
151         } else {
152             logger.warn("no {} found", Ontology.NAME_KEY);
153             nameValue = "no title";
154         }
155         return nameValue.trim();
156     }
