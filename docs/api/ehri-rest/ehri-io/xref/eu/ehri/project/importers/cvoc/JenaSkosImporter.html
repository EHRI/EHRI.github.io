168     /**
169      * Import an input stream.
170      *
171      * @param ios        The SKOS file input stream
172      * @param logMessage A log message
173      * @return A log of imported nodes
174      */
175     @Override
176     public ImportLog importFile(InputStream ios, String logMessage)
177             throws IOException, ValidationError {
178 
179         // Create a new action for this import
180         Optional<String> logMsg = getLogMessage(logMessage);
181         ActionManager.EventContext eventContext = new ActionManager(framedGraph, vocabulary).newEventContext(
182                 actioner, EventTypes.ingest, logMsg);
183         // Create a manifest to store the results of the import.
184         ImportLog log = new ImportLog(logMsg);
185 
186         OntModel model = ModelFactory.createOntologyModel();
187         model.read(ios, null, format);
188         OntClass conceptClass = model.getOntClass(SkosRDFVocabulary.CONCEPT.getURI().toString());
189         logger.debug("in import file: {}", SkosRDFVocabulary.CONCEPT.getURI());
190         ExtendedIterator<? extends OntResource> extendedIterator = conceptClass.listInstances();
191         Map<Resource, Concept> imported = Maps.newHashMap();
192 
193         while (extendedIterator.hasNext()) {
194             Resource item = extendedIterator.next();
195 
196             try {
197                 Mutation<Concept> graphConcept = importConcept(item);
198                 imported.put(item, graphConcept.getNode());
199 
200                 switch (graphConcept.getState()) {
201                     case UNCHANGED:
202                         log.addUnchanged();
203                         break;
204                     case CREATED:
205                         log.addCreated();
206                         eventContext.addSubjects(graphConcept.getNode());
207                         break;
208                     case UPDATED:
209                         log.addUpdated();
210                         eventContext.addSubjects(graphConcept.getNode());
211                         break;
212                 }
213             } catch (ValidationError validationError) {
214                 if (tolerant) {
215                     logger.error(validationError.getMessage());
216                     log.addError(item.toString(), validationError.getMessage());
217                 } else {
218                     throw validationError;
219                 }
220             }
221         }
222 
223         for (Map.Entry<Resource, Concept> pair : imported.entrySet()) {
224             hookupRelationships(pair.getKey(), pair.getValue(), imported);
225         }
226 
227         for (Concept concept : imported.values()) {
228             vocabulary.addItem(concept);
229             concept.setPermissionScope(vocabulary);
230         }
231 
232         if (log.hasDoneWork()) {
233             eventContext.commit();
234         }
235 
236         return log;
237     }

239     private Mutation<Concept> importConcept(Resource item) throws ValidationError {
240         logger.debug("Importing: {}", item);
241         Bundle.Builder builder = Bundle.Builder.withClass(EntityClass.CVOC_CONCEPT)
242                 .addDataValue(Ontology.IDENTIFIER_KEY, getId(URI.create(item.getURI())));
243 
244         Map<AuthoritativeItem, String> linkedConcepts = Maps.newHashMap();
245 
246         List<Bundle> unknown = getAdditionalRelations(item, linkedConcepts);
247         for (Bundle description : getDescriptions(item)) {
248             Bundle withRels = description
249                     .withRelations(Ontology.HAS_UNKNOWN_PROPERTY, unknown);
250             builder.addRelation(Ontology.DESCRIPTION_FOR_ENTITY, withRels);
251         }
252 
253         Mutation<Concept> mut = dao.createOrUpdate(builder.build(), Concept.class);
254         solveUndeterminedRelationships(mut.getNode(), linkedConcepts);
255         return mut;
256     }

258     private void solveUndeterminedRelationships(Concept unit, Map<AuthoritativeItem, String> linkedConcepts) {
259         Api api = ApiFactory.noLogging(framedGraph, actioner.as(UserProfile.class));
260 
261         for (AuthoritativeItem concept : linkedConcepts.keySet()) {
262             try {
263                 String reltype = linkedConcepts.get(concept);
264                 Bundle linkBundle = new Bundle(EntityClass.LINK)
265                         .withDataValue(Ontology.LINK_HAS_TYPE, "associate")
266                         .withDataValue(reltype.substring(0, reltype.indexOf(":")), reltype.substring(reltype.indexOf(":") + 1))
267                         .withDataValue(Ontology.LINK_HAS_DESCRIPTION, SaxXmlImporter.RESOLVED_LINK_DESC);
268                 Link link = api.create(linkBundle, Link.class);
269                 unit.addLink(link);
270                 concept.addLink(link);
271             } catch (ValidationError | PermissionDenied | DeserializationError ex) {
272                 logger.error(ex.getMessage());
273             }
274         }
275     }

277     private List<Bundle> getAdditionalRelations(Resource item, Map<AuthoritativeItem, String> linkedItems) {
278         List<Bundle> unknown = Lists.newArrayList();
279 
280         for (Map.Entry<String, URI> rel : SkosRDFVocabulary.RELATION_PROPS.entrySet()) {
281             for (RDFNode annotation : getObjectWithPredicate(item, rel.getValue())) {
282                 if (annotation.isLiteral()) {
283                     unknown.add(Bundle.Builder.withClass(EntityClass.UNKNOWN_PROPERTY)
284                             .addDataValue(rel.getKey(), annotation.toString())
285                             .build());
286                 } else {
287                     if (rel.getKey().startsWith("skos:") || rel.getKey().startsWith("sem:")) {
288                         Optional<AuthoritativeItem> found = findRelatedConcept(annotation.toString());
289                         if (found.isPresent()) {
290                             linkedItems.put(found.get(), rel.getKey());
291                         } else {
292                             unknown.add(Bundle.Builder.withClass(EntityClass.UNKNOWN_PROPERTY)
293                                     .addDataValue(rel.getKey(), annotation.toString())
294                                     .build());
295                         }
296                     }
297                 }
298             }
299         }
300         return unknown;
301     }

303     private Optional<AuthoritativeItem> findRelatedConcept(String name) {
304         if (name != null) {
305             String[] domains = name.split("/");
306             if (domains.length > 2) {
307                 String cvocId = domains[domains.length - 2];
308                 String conceptId = domains[domains.length - 1];
309                 AuthoritativeSet referredSet;
310                 try {
311                     GraphManager manager = GraphManagerFactory.getInstance(framedGraph);
312                     referredSet = manager.getEntity(cvocId, AuthoritativeSet.class);
313                     for (AuthoritativeItem authItem : referredSet.getAuthoritativeItems()) {
314                         if (authItem.getIdentifier().equals(conceptId)) {
315                             return Optional.of(authItem);
316                         }
317                     }
318                 } catch (ItemNotFound ex) {
319                     logger.error("AuthoritativeSet with id {} not found: {}", cvocId, ex.getMessage());
320                 }
321             }
322         }
323         return Optional.empty();
324     }

331     private List<RDFNode> getObjectWithPredicate(Resource item, final URI propUri) {
332         // NB: this should be possible with simply item.listProperties(propUri)
333         // but for some reason that doesn't work... I can't grok why.
334         return item.listProperties().filterKeep(statement ->
335                 statement.getPredicate()
336                         .hasURI(propUri.toString()))
337                 .mapWith(Statement::getObject).toList();
338     }

340     private void connectRelation(Concept current, Resource item, Map<Resource, Concept> others,
341             URI propUri, ConnectFunc connectFunc) {
342         for (RDFNode other : getObjectWithPredicate(item, propUri)) {
343             if (other.isResource()) {
344                 Concept related = others.get(other.asResource());
345                 if (related != null) {
346                     connectFunc.connect(current, related);
347                 }
348             }
349         }
350     }

352     private void hookupRelationships(Resource item, Concept current,
353             Map<Resource, Concept> conceptMap) {
354 
355         connectRelation(current, item, conceptMap, SkosRDFVocabulary.BROADER.getURI(),
356                 (it, other) -> other.addNarrowerConcept(it));
357 
358         connectRelation(current, item, conceptMap, SkosRDFVocabulary.NARROWER.getURI(),
359                 (it, other) -> it.addNarrowerConcept(other));
360 
361         connectRelation(current, item, conceptMap, SkosRDFVocabulary.RELATED.getURI(),
362                 (it, other) -> it.addRelatedConcept(other));
363     }

365     private List<Bundle> getDescriptions(Resource item) {
366         List<Bundle> descriptions = Lists.newArrayList();
367 
368         for (RDFNode property : getObjectWithPredicate(item, SkosRDFVocabulary.PREF_LABEL.getURI())) {
369             if (!property.isLiteral()) {
370                 continue;
371             }
372 
373             Bundle.Builder builder = Bundle.Builder.withClass(EntityClass.CVOC_CONCEPT_DESCRIPTION);
374 
375             Literal literalPrefName = property.asLiteral();
376             String langCode2Letter = literalPrefName.getLanguage();
377             String langCode3Letter = getLanguageCode(langCode2Letter, defaultLang);
378             Optional<String> descCode = getScriptCode(langCode2Letter);
379 
380             builder.addDataValue(Ontology.NAME_KEY, literalPrefName.getString())
381                     .addDataValue(Ontology.LANGUAGE, langCode3Letter);
382             descCode.ifPresent(code -> builder.addDataValue(Ontology.IDENTIFIER_KEY, code));
383 
384             for (Map.Entry<String, URI> prop : SkosRDFVocabulary.GENERAL_PROPS.entrySet()) {
385                 for (RDFNode target : getObjectWithPredicate(item, prop.getValue())) {
386                     if (target.isLiteral()) {
387                         if (prop.getKey().equals("latitude/longitude")) {
388                             String[] latLon = target.asLiteral().getString().split(",");
389                             if (latLon.length > 1) {
390                                 builder.addDataValue("latitude", latLon[0]);
391                                 builder.addDataValue("longitude", latLon[1]);
392                             }
393                         } else {
394                             builder.addDataValue(prop.getKey(), target.asLiteral().getString());
395                         }
396                     } else {
397                         builder.addDataValue(prop.getKey(), target.toString());
398                     }
399                 }
400             }
401 
402             for (Map.Entry<String, List<URI>> prop : SkosRDFVocabulary.LANGUAGE_PROPS.entrySet()) {
403                 List<String> values = Lists.newArrayList();
404                 for (URI uri : prop.getValue()) {
405                     for (RDFNode target : getObjectWithPredicate(item, uri)) {
406                         if (target.isLiteral()) {
407                             Literal literal = target.asLiteral();
408                             String propLang2Letter = literal.getLanguage();
409                             String propLanguageCode = getLanguageCode(propLang2Letter, defaultLang);
410                             Optional<String> propDescCode = getScriptCode(propLang2Letter);
411                             if (propLanguageCode.equals(langCode3Letter) && propDescCode.equals(descCode)) {
412                                 values.add(literal.getString());
413                             }
414                         }
415                     }
416                 }
417                 if (!values.isEmpty()) {
418                     // Sorting the related literal values in
419                     // natural order gives consistency between
420                     // import/export round-trips. The data is
421                     // otherwise unsorted.
422                     Collections.sort(values);
423                     builder.addDataValue(prop.getKey(), values);
424                 }
425             }
426             Bundle bundle = builder.build();
427             logger.trace(bundle.toJson());
428             descriptions.add(bundle);
429         }
430 
431         return descriptions;
432     }

434     private static String getLanguageCode(String langCode2Letter, String defaultLang) {
435         if (langCode2Letter == null || langCode2Letter.trim().isEmpty()) {
436             return defaultLang;
437         }
438         List<String> parts = codeSplitter.splitToList(langCode2Letter);
439         if (parts.isEmpty()) {
440             return defaultLang;
441         } else {
442             return LanguageHelpers.iso639DashTwoCode(parts.get(0));
443         }
444     }

446     private static Optional<String> getScriptCode(String langCode) {
447         List<String> parts = codeSplitter.splitToList(langCode);
448         return parts.size() > 1 ? Optional.of(parts.get(1)) : Optional.empty();
449     }

451     private String getId(URI uri) {
452         if (baseURI != null && uri.toString().startsWith(baseURI)) {
453             return uri.toString().substring(baseURI.length());
454         } else if (uri.getFragment() != null) {
455             return uri.getFragment();
456         } else {
457             return uri.getPath().substring(uri.getPath().lastIndexOf("/") + 1)
458                     + (uri.getQuery() != null ? uri.getQuery() : "")
459                     + (uri.getFragment() != null ? uri.getFragment() : "");
460         }
461     }
