45  /**
46   * Import EAD describing a Virtual Collection. some rules governing virtual collections:
47   * <p>
48   * the archdesc should describe the purpose of this vc. it can not in itself refer to a DU.
49   * <p>
50   * every c level is either 1) a virtual level (=VirtualLevel), or 2) it points to an existing DocumentaryUnit
51   * (=VirtualReferrer) (and consequently to the entire subtree beneath it) 1) there is no repository-tag with a
52   * ehri-label
53   * <p>
54   * 2) there is exactly one repository-tag with an ehri-label &lt;repository
55   * label="ehri_repository_vc"&gt;il-002777&lt;/repository&gt; (this will not be shown in the portal) and exactly one unitid with
56   * a ehri-main-identifier label, that is identical to the existing unitid within the graph for this repository
57   * <p>
58   * all other tags will be ignored, since the DocumentsDescription of the referred DocumentaryUnit will be shown. there
59   * should not be any c-levels beneath such a c-level
60   */
61  public class VirtualEadImporter extends EaImporter {

81      /**
82       * Import a single archdesc or c01-12 item, keeping a reference to the hierarchical depth. this will import the
83       * structure as VirtualUnits, which either have a DocDescription (VirtualLevel, like series) or they point to an
84       * existing DocDesc from an existing DocumentaryUnit (VirtualReferrer).
85       *
86       * @param itemData The data map
87       * @param idPath   The identifiers of parent documents, not including those of the overall permission scope
88       * @throws ValidationError when the itemData does not contain an identifier for the unit or...
89       */
90      @Override
91      public AbstractUnit importItem(Map<String, Object> itemData, List<String> idPath)
92              throws ValidationError {
93  
94          BundleDAO persister = getPersister(idPath);
95  
96          boolean isVirtualLevel = isVirtualLevel(itemData);
97          Bundle unit = new Bundle(unitEntity, extractVirtualUnit(itemData));
98  
99          if (isVirtualLevel) { //VirtualLevel
100             // Check for missing identifier, throw an exception when there is no ID.
101             if (unit.getDataValue(Ontology.IDENTIFIER_KEY) == null) {
102                 throw new ValidationError(unit, Ontology.IDENTIFIER_KEY,
103                         "Missing identifier " + Ontology.IDENTIFIER_KEY);
104             }
105             logger.debug("Imported item: " + itemData.get("name"));
106             Bundle descBundle = new Bundle(EntityClass.DOCUMENTARY_UNIT_DESCRIPTION, extractUnitDescription(itemData, EntityClass.DOCUMENTARY_UNIT_DESCRIPTION));
107             // Add dates and descriptions to the bundle since they're @Dependent
108             // relations.
109             for (Map<String, Object> dpb : extractDates(itemData)) {
110                 descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
111             }
112             for (Map<String, Object> rel : extractRelations(itemData)) {//, (String) unit.getErrors().get(Identifiable.IDENTIFIER_KEY)
113                 logger.debug("relation found: " + rel.get(Ontology.NAME_KEY));
114                 descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT, new Bundle(EntityClass.ACCESS_POINT, rel));
115             }
116             Map<String, Object> unknowns = extractUnknownProperties(itemData);
117             if (!unknowns.isEmpty()) {
118                 logger.debug("Unknown Properties found");
119                 descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, new Bundle(EntityClass.UNKNOWN_PROPERTY, unknowns));
120             }
121             unit = unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
122             Mutation<VirtualUnit> mutation =
123                     persister.createOrUpdate(unit, VirtualUnit.class);
124             VirtualUnit frame = mutation.getNode();
125             // Set the repository/item relationship
126             //TODO: figure out another way to determine we're at the root, so we can get rid of the depth param
127             if (idPath.isEmpty() && mutation.created()) {
128                 EntityClass scopeType = manager.getEntityClass(permissionScope);
129                 if (scopeType.equals(EntityClass.USER_PROFILE)) {
130                     UserProfile responsibleUser = framedGraph.frame(permissionScope.asVertex(), UserProfile.class);
131                     frame.setAuthor(responsibleUser);
132                     //the top Virtual Unit does not have a permissionScope. 
133                 } else if (scopeType.equals(unitEntity)) {
134                     VirtualUnit parent = framedGraph.frame(permissionScope.asVertex(), VirtualUnit.class);
135                     parent.addChild(frame);
136                     frame.setPermissionScope(parent);
137                 } else {
138                     logger.error("Unknown scope type for virtual unit: {}", scopeType);
139                 }
140             }
141             handleCallbacks(mutation);
142             return frame;
143         } else { //find the referred Documentary Unit and RETURN it
144             try {
145                 //find the DocumentaryUnit using the repository_id/unit_id combo
146                 return findReferredToDocumentaryUnit(itemData);
147             } catch (ItemNotFound ex) {
148                 throw new ValidationError(unit, ex.getKey(), ex.getMessage());
149             }
150         }
151     }

153     @SuppressWarnings("unchecked")
154     @Override
155     protected Iterable<Map<String, Object>> extractRelations(Map<String, Object> data) {
156         String REL = "AccessPoint";
157         List<Map<String, Object>> list = Lists.newArrayList();
158         for (String key : data.keySet()) {
159             if (key.endsWith(REL)) {
160                 logger.debug(key + " found in data");
161                 //type, targetUrl, targetName, notes
162                 for (Map<String, Object> origRelation : (List<Map<String, Object>>) data.get(key)) {
163                     Map<String, Object> relationNode = Maps.newHashMap();
164                     for (String eventkey : origRelation.keySet()) {
165                         logger.debug(eventkey);
166                         if (eventkey.endsWith(REL)) {
167                             relationNode.put(Ontology.ACCESS_POINT_TYPE, eventkey);
168                             relationNode.put(Ontology.NAME_KEY, origRelation.get(eventkey));
169                         } else {
170                             relationNode.put(eventkey, origRelation.get(eventkey));
171                         }
172                     }
173                     if (!relationNode.containsKey(Ontology.ACCESS_POINT_TYPE)) {
174                         relationNode.put(Ontology.ACCESS_POINT_TYPE, "corporateBodyAccessPoint");
175                     }
176                     list.add(relationNode);
177                 }
178             }
179         }
180         return list;
181     }

183     /**
184      * Creates a Map containing properties of a Virtual Unit.
185      * <p>
186      * These properties are the unit's identifiers.
187      *
188      * @param itemData Map of all extracted information
189      * @return a Map representing a Documentary Unit node
190      * @throws ValidationError
191      */
192     protected Map<String, Object> extractVirtualUnit(Map<String, Object> itemData) throws ValidationError {
193         Map<String, Object> unit = Maps.newHashMap();
194         if (itemData.get(OBJECT_ID) != null) {
195             unit.put(Ontology.IDENTIFIER_KEY, itemData.get(OBJECT_ID));
196         }
197         if (itemData.get(Ontology.OTHER_IDENTIFIERS) != null) {
198             logger.debug("otherIdentifiers is not null");
199             unit.put(Ontology.OTHER_IDENTIFIERS, itemData.get(Ontology.OTHER_IDENTIFIERS));
200         }
201         return unit;
202     }

209     /**
210      * Creates a Map containing properties of a Documentary Unit description. These properties are the unit
211      * description's properties: all except the doc unit identifiers and unknown properties.
212      *
213      * @param itemData Map of all extracted information
214      * @param depth    depth of node in the tree
215      * @return a Map representing a Documentary Unit Description node
216      * @throws ValidationError
217      */
218     protected Map<String, Object> extractDocumentDescription(Map<String, Object> itemData, int depth) throws ValidationError {
219 
220         Map<String, Object> unit = Maps.newHashMap();
221         for (String key : itemData.keySet()) {
222             if (!(key.equals(OBJECT_ID)
223                     || key.equals(Ontology.OTHER_IDENTIFIERS)
224                     || key.startsWith(SaxXmlHandler.UNKNOWN))) {
225                 unit.put(key, itemData.get(key));
226             }
227         }
228         return unit;
229     }

240     private boolean isVirtualLevel(Map<String, Object> itemData) {
241         return !(itemData.containsKey(REPOID) && itemData.containsKey(UNITID));
242     }
243 
244     private DocumentaryUnit findReferredToDocumentaryUnit(Map<String, Object> itemData) throws ItemNotFound {
245         if (itemData.containsKey(REPOID) && itemData.containsKey(UNITID)) {
246             String repositoryid = itemData.get(REPOID).toString();
247             String unitid = itemData.get(UNITID).toString();
248             Repository repository = manager.getEntity(repositoryid, Repository.class);
249             for (DocumentaryUnit unit : repository.getAllCollections()) {
250                 logger.debug(unit.getIdentifier() + " " + unit.getId() + " " + unitid);
251                 if (unit.getIdentifier().equals(unitid)) {
252                     return unit;
253                 }
254             }
255             throw new ItemNotFound(String.format("No item %s found in repo %s", unitid, repositoryid));
256         }
257         throw new ItemNotFound("Apparently no repositoryid/unitid combo given");
258 
259     }
