331     public static Optional<String> countryCodeToContinent(String countryCode) {
332         String continentCode = countryCodesToContinents.get(countryCode.toUpperCase());

339     /**
340      * Get the best description for a given language code.
341      *
342      * @param item         a described item
343      * @param priorDescOpt if the object is hierarchical, the parent-level
344      *                     description
345      * @param langCode     a 3-letter language code.
346      * @return the best matching description found
347      */
348     public static Optional<Description> getBestDescription(Described item, Optional<Description> priorDescOpt, String langCode) {
349         List<Description> descriptions = Lists.newArrayList(item.getDescriptions());
350         Collections.sort(descriptions, (d1, d2) -> d1.getId().compareTo(d2.getId()));
351         Description fallBack = null;
352         for (Description description : descriptions) {
353             if (fallBack == null) {
354                 fallBack = description;
355             }
356             // First of all, check the description code (usually set to the
357             // EAD file ID.) If this is the same as the parent, return the
358             // current description.
359             for (Description parent : priorDescOpt.map(Collections::singleton).orElse(Collections.emptySet())) {
360                 for (String code : Optional.ofNullable(parent.getDescriptionCode())
361                         .map(Collections::singleton).orElse(Collections.emptySet())) {
362                     if (code.equals(description.getDescriptionCode())) {
363                         return Optional.of(description);
364                     }
365                 }
366             }
367 
368             // Otherwise, fall back to the first one with the same language
369             if (description.getLanguageOfDescription().equalsIgnoreCase(langCode)) {
370                 return Optional.of(description);
371             }
372         }
373         return Optional.ofNullable(fallBack);
374     }

376     public static Optional<Description> getBestDescription(Described item, String langCode) {
377         return getBestDescription(item, Optional.empty(), langCode);

380     /**
381      * Take an ISO-639-1 code or a language name and try and map to a valid ISO639-2 code.
382      *
383      * @param nameOrCode a language code or name to convert
384      * @return the ISO 639-2 language code for that code or name, or the input string if
385      * no conversion was found
386      */
387     public static String iso639DashTwoCode(String nameOrCode) {
388         if (nameOrCode.length() == 2 && locale2Map.containsKey(nameOrCode)) {
389             return locale2Map.get(nameOrCode).getISO3Language();
390         } else if (nameOrCode.length() == 3 && iso639BibTermLookup.containsKey(nameOrCode)) {
391             return iso639BibTermLookup.get(nameOrCode);
392         } else if (nameOrCode.length() > 3 && localeNameMap.containsKey(nameOrCode.toLowerCase())) {
393             return localeNameMap.get(nameOrCode.toLowerCase()).getISO3Language();
394             /* FIXME the localeNameMap depends on locale and translating an
395              * English name to a code fails when executed on
396              * e.g. a server with non-English locale
397              */
398         }
399         return nameOrCode;
400     }

402     /**
403      * Take an ISO-639-2 code or a language name and try and map to a valid ISO639-1 code.
404      *
405      * @param nameOrCode a language code or name to convert
406      * @return the ISO 639-1 language code for that code or name, or the input string if
407      * no conversion was found
408      */
409     public static String iso639DashOneCode(String nameOrCode) {
410         if (nameOrCode.length() == 3 && locale3Map.containsKey(nameOrCode)) {
411             return locale3Map.get(nameOrCode);
412         } else if (nameOrCode.length() == 3 && iso639BibTermLookup.containsKey(nameOrCode)) {
413             return locale3Map.get(iso639BibTermLookup.get(nameOrCode));
414         } else if (nameOrCode.length() > 3 && localeNameMap.containsKey(nameOrCode.toLowerCase())) {
415             return localeNameMap.get(nameOrCode.toLowerCase()).getLanguage();
416         } else if (nameOrCode.length() > 2 && nameOrCode.contains("-")) {
417             // Attempt to handle codes like 'heb-Hebr' and 'eng-Latn'
418             List<String> parts = Lists.newArrayList(codeSplitter.split(nameOrCode));
419             if (parts.size() == 1) {
420                 return iso639DashOneCode(parts.get(0));
421             } else if (parts.size() == 2) {
422                 return iso639DashOneCode(parts.get(0)) + "-" + parts.get(1);
423             }
424         }
425         return nameOrCode;
426     }

428     /**
429      * Convert a 2- or 3-letter language code to its (English) name.
430      * <p>
431      * If no name can be found the input code will be returned.
432      *
433      * @param code a 2- or 3-letter ISO639 code
434      * @return a language name, with the original code as a fall back
435      */
436     public static String codeToName(String code) {
437         if (code.length() == 2 && locale2Map.containsKey(code)) {
438             return locale2Map.get(code).getDisplayLanguage(Locale.ENGLISH);
439         } else if (code.length() == 3) {
440             String termCode = iso639BibTermLookup.containsKey(code)
441                     ? iso639BibTermLookup.get(code)
442                     : code;
443             String twoCode = locale3Map.get(termCode);
444             if (locale2Map.containsKey(twoCode)) {
445                 return locale2Map.get(twoCode).getDisplayLanguage(Locale.ENGLISH);
446             }
447         }
448         return code;
449     }
