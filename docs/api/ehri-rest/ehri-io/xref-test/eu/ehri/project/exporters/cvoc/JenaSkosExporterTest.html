41      @Test
42      public void testImportExportRoundtrip() throws Exception {
43          Map<String, String> files = ImmutableMap.of(
44                  FILE1, "RDF/XML",
45                  FILE2, "N3",
46                  FILE3, "RDF/XML",
47                  FILE4, "RDF/XML",
48                  FILE5, "RDF/XML"
49          );
50          for (Map.Entry<String, String> entry : files.entrySet()) {
51              SkosImporter importer = new JenaSkosImporter(graph, actioner, vocabulary);
52              importer.setFormat(entry.getValue().equalsIgnoreCase("") ? null : entry.getValue())
53                      .importFile(ClassLoader.getSystemResourceAsStream(entry.getKey()), "test");
54  
55              List<VertexProxy> before = getGraphState(graph);
56              int edgeCountBefore = getEdgeCount(graph);
57              SkosExporter exporter = new JenaSkosExporter(graph, vocabulary)
58                      .setFormat(entry.getValue().equalsIgnoreCase("") ? null : entry.getValue());
59              OutputStream outputStream = new ByteArrayOutputStream();
60              exporter.export(outputStream, null);
61              String skos = outputStream.toString();
62              ImportLog log = importer.setFormat(entry.getValue())
63                      .importFile(new ByteArrayInputStream(skos.getBytes()), "test");
64              log.printReport();
65              List<VertexProxy> after = getGraphState(graph);
66              assertTrue(log.getUnchanged() > 0);
67              assertEquals(0, log.getChanged());
68              assertEquals(0, log.getCreated());
69              assertEquals(0, log.getErrored());
70              GraphDiff graphDiff = diffGraph(before, after);
71              assertTrue(graphDiff.added.isEmpty());
72              assertTrue(graphDiff.removed.isEmpty());
73              int edgeCountAfter = getEdgeCount(graph);
74              assertEquals(edgeCountBefore, edgeCountAfter);
75          }
76      }
