50      @Test
51      public void testCorrectEventNodesAreCreated() throws DeserializationError, ValidationError {
52          ActionManager am = new ActionManager(graph);
53          // Create a user and log it
54          Bundle userBundle = Bundle.fromData(TestData.getTestUserBundle());
55          UserProfile user = new BundleManager(graph).create(userBundle, UserProfile.class);
56          ActionManager.EventContext ctx1 = am.newEventContext(user,
57                  graph.frame(validUser.asVertex(), Actioner.class),
58                  EventTypes.creation);
59          SystemEvent first = ctx1.commit();
60  
61          // Create a repository and log that too...
62          Bundle repoBundle = Bundle.fromData(TestData.getTestAgentBundle());
63          Repository repository = new BundleManager(graph).create(repoBundle, Repository.class);
64  
65          ActionManager.EventContext ctx2 = am.newEventContext(repository,
66                  graph.frame(validUser.asVertex(), Actioner.class),
67                  EventTypes.creation);
68          assertEquals(EventTypes.creation, ctx2.getEventType());
69          SystemEvent second = ctx2.commit();
70  
71          // Check exactly one Event was created
72          assertEquals(1, Iterables.count(second.getSubjects()));
73          // Check item cache is correct...
74          assertEquals(1L, second.subjectCount());
75          assertNotNull(second.getActioner());
76  
77          // Check the user is correctly linked
78          assertEquals(validUser, second.getActioner());
79  
80          assertEquals(1, Iterables.count(repository.getHistory()));
81          assertNotNull(repository.getLatestEvent());
82  
83          // Check the latest event in the list is the one we want...
84          SystemEvent top = am.getLatestGlobalEvents().iterator().next();
85          assertEquals(top, second);
86  
87          // Check the full list of system events contains both items
88          List<SystemEvent> events = toList(am.getLatestGlobalEvents());
89          assertEquals(2, events.size());
90          assertEquals(second, events.get(0));
91          assertEquals(first, events.get(1));
92      }

131     @Test
132     public void testCreatingVersions() throws Exception {
133         Repository r1 = manager.getEntity("r1", Repository.class);
134         ActionManager am = new ActionManager(graph, r1);
135 
136         Bundle docBundle = Bundle.fromData(TestData.getTestDocBundle());
137         BundleManager dao = new BundleManager(graph);
138         DocumentaryUnit doc = dao.create(docBundle, DocumentaryUnit.class);
139         ActionManager.EventContext ctx1 = am.newEventContext(doc,
140                 graph.frame(validUser.asVertex(), Actioner.class),
141                 EventTypes.creation);
142         ctx1.commit();
143         assertNull(doc.getPriorVersion());
144         Mutation<DocumentaryUnit> update = dao.update(docBundle
145                 .withId(doc.getId())
146                 .withDataValue("identifier", "changed"), DocumentaryUnit.class);
147         ActionManager.EventContext ctx2 = am.newEventContext(doc,
148                 graph.frame(validUser.asVertex(), Actioner.class),
149                 EventTypes.modification).createVersion(doc, docBundle);
150         SystemEvent event = ctx2.commit();
151         assertTrue(update.updated());
152         assertTrue(event.getPriorVersions().iterator().hasNext());
153         assertEquals(1, Iterables.count(event.getPriorVersions()));
154         Version version = event.getPriorVersions().iterator().next();
155         assertNotNull(doc.getPriorVersion());
156         assertEquals(version, doc.getPriorVersion());
157 
158         // Create another event and ensure versions are ordered correctly
159         dao.update(docBundle
160                 .withId(doc.getId())
161                 .withDataValue("identifier", "changed-again"), DocumentaryUnit.class);
162         SystemEvent event2 = ctx2.commit();
163         assertEquals(1, Iterables.count(event2.getPriorVersions()));
164         assertEquals(2, Iterables.count(doc.getAllPriorVersions()));
165     }
