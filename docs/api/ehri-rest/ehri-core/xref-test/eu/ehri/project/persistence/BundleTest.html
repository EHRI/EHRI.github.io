130     @Test
131     public void testEqualsAndHashCode() throws Exception {
132         // A bundle with the same relationship data but
133         // in different orders should be equals()
134         Bundle bundle1 = Bundle.fromData(TestData.getTestDocBundle());
135         Bundle bundle2 = Bundle.fromData(TestData.getTestDocBundle());
136         assertEquals(bundle1, bundle2);
137         assertEquals(bundle1.hashCode(), bundle2.hashCode());
138 
139         // Add a new date period to both and ensure that have a
140         // different structural ordering
141         System.out.println(bundle1);
142         Bundle dp = new Bundle(EntityClass.DATE_PERIOD)
143                 .withDataValue(Ontology.DATE_PERIOD_START_DATE, "1900-01-01")
144                 .withDataValue(Ontology.DATE_PERIOD_END_DATE, "2000-01-01");
145         Bundle currentDp = BundleUtils.getBundle(bundle1, "describes[0]/hasDate[0]");
146         Bundle b1_2 = BundleUtils
147                 .setBundle(bundle1, "describes[0]/hasDate[-1]", currentDp);
148         Bundle b1_3 = BundleUtils.setBundle(b1_2, "describes[0]/hasDate[0]", dp);
149 
150         Bundle b2_2 = BundleUtils.setBundle(bundle2, "describes[0]/hasDate[-1]", dp);
151         assertEquals(BundleUtils.getBundle(b1_3, "describes[0]/hasDate[1]"),
152                 BundleUtils.getBundle(b2_2, "describes[0]/hasDate[0]"));
153         assertEquals(BundleUtils.getBundle(b1_3, "describes[0]/hasDate[0]"),
154                 BundleUtils.getBundle(b2_2, "describes[0]/hasDate[1]"));
155         assertEquals(b1_3, b2_2);
156     }

195     @Test
196     public void testMergeDataWithForTree() throws Exception {
197         Bundle nested = new Bundle(EntityClass.DATE_PERIOD)
198                 .withDataValue(Ontology.DATE_PERIOD_START_DATE, "2001-01-01");
199         // Add a nested node to the bundle we're patching and generate IDs
200         Bundle start = bundle
201                 .withRelation(Ontology.DESCRIPTION_FOR_ENTITY,
202                         new Bundle(EntityClass.MAINTENANCE_EVENT)
203                                 .withDataValue(Ontology.MAINTENANCE_EVENT_TYPE,
204                                         MaintenanceEventType.created.toString()));
205 
206         Bundle withIds = start.generateIds(Sets.<String>newHashSet());
207         Bundle toMerge = BundleUtils.setBundle(withIds, "describes[0]/hasDate[-1]", nested)
208                 .generateIds(Sets.<String>newHashSet());
209         Bundle patch = BundleUtils.set(withIds, "describes[0]/name", "Foobar 2");
210         Bundle merged = toMerge.mergeDataWith(patch);
211         assertNotSame(merged, bundle);
212         assertEquals("Foobar 2",
213                 BundleUtils.get(merged, "describes[0]/name"));
214         assertEquals("created",
215                 BundleUtils.get(merged, "describes[1]/eventType"));
216         assertEquals("2001-01-01",
217                 BundleUtils.get(merged, "describes[0]/hasDate[0]/startDate"));
218     }
