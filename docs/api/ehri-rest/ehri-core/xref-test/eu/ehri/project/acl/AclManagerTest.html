368     @Test
369     public void testCreateOwnerScenario() throws Exception {
370         loader.loadTestData("archivist-permissions.yaml");
371         UserProfile headUser = manager.getEntity("hauser", UserProfile.class);
372         UserProfile user1 = manager.getEntity("auser1", UserProfile.class);
373         UserProfile user2 = manager.getEntity("auser2", UserProfile.class);
374         Group headArchivists = manager.getEntity("head-archivists", Group.class);
375         Group archivists = manager.getEntity("archivists", Group.class);
376         Repository repo = manager.getEntity("repo", Repository.class);
377 
378         AclManager acl = new AclManager(graph, repo);
379         // The grants in the permissions should give these particular
380         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, CREATE, headArchivists));
381         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, UPDATE, headArchivists));
382         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, DELETE, headArchivists));
383 
384         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, CREATE, archivists));
385         assertFalse(acl.hasPermission(DOCUMENTARY_UNIT, UPDATE, archivists));
386         assertFalse(acl.hasPermission(DOCUMENTARY_UNIT, DELETE, archivists));
387 
388         // Check these grants are correctly inherited by the users.
389         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, CREATE, headUser));
390         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, UPDATE, headUser));
391         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, DELETE, headUser));
392 
393         assertTrue(acl.hasPermission(DOCUMENTARY_UNIT, CREATE, user1));
394         assertFalse(acl.hasPermission(DOCUMENTARY_UNIT, UPDATE, user1));
395         assertFalse(acl.hasPermission(DOCUMENTARY_UNIT, DELETE, user1));
396 
397         // New create some stuff
398         LoggingCrudViews<DocumentaryUnit> views
399                 = new LoggingCrudViews<DocumentaryUnit>(graph, DocumentaryUnit.class, repo);
400 
401         DocumentaryUnit headdoc1 = views.create(
402                 Bundle.fromData(TestData.getTestDocBundle())
403                         .withDataValue(Ontology.IDENTIFIER_KEY, "head-doc"), headUser);
404         DocumentaryUnit userdoc1 = views.create(
405                 Bundle.fromData(TestData.getTestDocBundle())
406                         .withDataValue(Ontology.IDENTIFIER_KEY, "user-doc-1"), user1);
407         DocumentaryUnit userdoc2 = views.create(
408                 Bundle.fromData(TestData.getTestDocBundle())
409                         .withDataValue(Ontology.IDENTIFIER_KEY, "user-doc-2"), user2);
410 
411         // Ensure Head Archivist can update/delete user1's doc
412         assertTrue(acl.hasPermission(userdoc1, UPDATE, headArchivists));
413         assertTrue(acl.hasPermission(userdoc1, DELETE, headArchivists));
414         // But doesn't own it
415         assertFalse(acl.hasPermission(userdoc1, OWNER, headArchivists));
416 
417         // Check the calculated permission sets.
418         assertTrue(acl.getInheritedItemPermissions(userdoc1, user1).has(OWNER));
419         assertFalse(acl.getInheritedItemPermissions(userdoc2, user1).has(OWNER));
420         assertTrue(acl.getInheritedItemPermissions(userdoc2, user2).has(OWNER));
421         assertFalse(acl.getInheritedItemPermissions(userdoc1, user2).has(OWNER));
422 
423         // Ensure user1 can update/delete his own doc
424         assertTrue(acl.hasPermission(userdoc1, OWNER, user1));
425         assertTrue(acl.hasPermission(userdoc1, UPDATE, user1));
426         assertTrue(acl.hasPermission(userdoc1, DELETE, user1));
427 
428         // Ensure neither user1 or user2 can update/delete head's docs
429         assertFalse(acl.hasPermission(headdoc1, OWNER, user1));
430         assertFalse(acl.hasPermission(headdoc1, UPDATE, user1));
431         assertFalse(acl.hasPermission(headdoc1, DELETE, user1));
432         assertFalse(acl.hasPermission(headdoc1, OWNER, user2));
433         assertFalse(acl.hasPermission(headdoc1, UPDATE, user2));
434         assertFalse(acl.hasPermission(headdoc1, DELETE, user2));
435 
436         // Ensure neither user1 or user2 can update/delete each other's docs
437         assertFalse(acl.hasPermission(userdoc1, OWNER, user2));
438         assertFalse(acl.hasPermission(userdoc1, UPDATE, user2));
439         assertFalse(acl.hasPermission(userdoc1, DELETE, user2));
440         assertFalse(acl.hasPermission(userdoc2, OWNER, user1));
441         assertFalse(acl.hasPermission(userdoc2, UPDATE, user1));
442         assertFalse(acl.hasPermission(userdoc2, DELETE, user1));
443     }
