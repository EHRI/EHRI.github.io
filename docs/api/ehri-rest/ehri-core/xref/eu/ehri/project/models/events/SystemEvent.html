45  /**
46   * Frame class representing an event that happened in the
47   * graph.
48   */
49  @EntityType(EntityClass.SYSTEM_EVENT)
50  public interface SystemEvent extends Accessible {
51  
52      @Meta(ItemHolder.CHILD_COUNT)
53      @JavaHandler
54      long subjectCount();
55  
56      /**
57       * Fetch the time stamp of this event.
58       *
59       * @return a UTF timestamp string
60       */
61      @Mandatory
62      @Property(Ontology.EVENT_TIMESTAMP)
63      String getTimestamp();
64  
65      /**
66       * Fetch the event type of this event.
67       *
68       * @return an {@link EventTypes} value
69       */
70      @Mandatory
71      @Property(Ontology.EVENT_TYPE)
72      EventTypes getEventType();
73  
74      /**
75       * Fetch the log message associated with this event.
76       *
77       * @return A string
78       */
79      @Property(Ontology.EVENT_LOG_MESSAGE)
80      String getLogMessage();
81  
82      /**
83       * Fetch the actioner who triggered this event.
84       *
85       * @return A user profile instance
86       */
87      @Fetch(value = Ontology.EVENT_HAS_ACTIONER, numLevels = 0)
88      @JavaHandler
89      Actioner getActioner();
90  
91      /**
92       * Fetch the subject items to whom this event pertains.
93       *
94       * @return an iterable of frame items
95       */
96      @JavaHandler
97      Iterable<Accessible> getSubjects();
98  
99      /**
100      * If new versions have been created, fetch the prior versions
101      * of the subjects that were affected by this event.
102      *
103      * @return an iterable of version nodes
104      */
105     @Fetch(value = Ontology.VERSION_HAS_EVENT, ifLevel = 0)
106     @Adjacency(label = Ontology.VERSION_HAS_EVENT, direction = Direction.IN)
107     Iterable<Version> getPriorVersions();
108 
109     /**
110      * Fetch the first subject of this event. This is a shortcut
111      * method to avoid having to fetch many items.
112      *
113      * @return an item frame
114      */
115     @Fetch(value = Ontology.EVENT_HAS_FIRST_SUBJECT, ifLevel = 0)
116     @JavaHandler
117     Accessible getFirstSubject();
118 
119     /**
120      * Fetch the event chronologically prior to this one.
121      *
122      * @return an event item
123      */
124     @Adjacency(label = Ontology.ACTIONER_HAS_LIFECYCLE_ACTION)
125     SystemEvent getPriorEvent();
126 
127     /**
128      * Fetch the "scope" of this event, or the context in which a
129      * given creation/modification/deletion event is happening.
130      *
131      * @return the event scope
132      */
133     @Fetch(value = Ontology.EVENT_HAS_SCOPE, ifLevel = 0)
134     @Adjacency(label = Ontology.EVENT_HAS_SCOPE, direction = Direction.OUT)
135     Entity getEventScope();
136 
137     /**
138      * Set the scope of this event.
139      *
140      * @param frame a scope item

142     @Adjacency(label = Ontology.EVENT_HAS_SCOPE, direction = Direction.OUT)
143     void setEventScope(Entity frame);

145     /**
146      * Implementation of complex methods.
147      */
148     abstract class Impl implements JavaHandlerContext<Vertex>, SystemEvent {
149 
150         @Override
151         public long subjectCount() {
152             return gremlin().inE(Ontology.ENTITY_HAS_EVENT).count();
153         }
154 
155         @Override
156         public Iterable<Accessible> getSubjects() {
157             return frameVertices(gremlin().in(Ontology.ENTITY_HAS_EVENT)
158                     .as("n").in(Ontology.ENTITY_HAS_LIFECYCLE_EVENT)
159                     .loop("n", JavaHandlerUtils.noopLoopFunc, new PipeFunction<LoopPipe.LoopBundle<Vertex>, Boolean>() {
160                         @Override
161                         public Boolean compute(LoopPipe.LoopBundle<Vertex> vertexLoopBundle) {
162                             return isValidEndpoint(vertexLoopBundle.getObject(),
163                                     Ontology.ENTITY_HAS_LIFECYCLE_EVENT);
164                         }
165                     }));
166         }
167 
168         @Override
169         public Accessible getFirstSubject() {
170             // Ugh: horrible code duplication is horrible - unfortunately
171             // just calling getSubjects() fails for an obscure reason to do
172             // with Frames not being thinking it has an iterable???
173             GremlinPipeline<Vertex, Vertex> subjects = gremlin().in(Ontology.ENTITY_HAS_EVENT)
174                     .as("n").in(Ontology.ENTITY_HAS_LIFECYCLE_EVENT)
175                     .loop("n", JavaHandlerUtils.noopLoopFunc, new PipeFunction<LoopPipe.LoopBundle<Vertex>, Boolean>() {
176                         @Override
177                         public Boolean compute(LoopPipe.LoopBundle<Vertex> vertexLoopBundle) {
178                             return isValidEndpoint(vertexLoopBundle.getObject(),
179                                     Ontology.ENTITY_HAS_LIFECYCLE_EVENT);
180                         }
181                     });
182             return (Accessible) (subjects.iterator().hasNext()
183                     ? frame(subjects.iterator().next())
184                     : null);
185         }
186 
187         @Override
188         public Actioner getActioner() {
189             GremlinPipeline<Vertex, Vertex> actioners = gremlin().in(Ontology.ACTION_HAS_EVENT)
190                     .as("n").in(Ontology.ACTIONER_HAS_LIFECYCLE_ACTION)
191                     .loop("n", JavaHandlerUtils.noopLoopFunc, new PipeFunction<LoopPipe.LoopBundle<Vertex>, Boolean>() {
192                         @Override
193                         public Boolean compute(LoopPipe.LoopBundle<Vertex> vertexLoopBundle) {
194                             return isValidEndpoint(vertexLoopBundle.getObject(),
195                                     Ontology.ACTIONER_HAS_LIFECYCLE_ACTION);
196                         }
197                     });
198             return (Actioner) (actioners.iterator().hasNext()
199                     ? frame(actioners.iterator().next())
200                     : null);
201         }
202 
203         private boolean isValidEndpoint(Vertex vertex, String linkRel) {
204             // A node at the end of an event link chain will have
205             //  - a) no in-coming event link relations
206             //  - b) be of a different type than an event link
207             if (vertex.getEdges(Direction.IN, linkRel).iterator().hasNext()) {
208                 return false;
209             } else {
210                 String type = vertex.getProperty(EntityType.TYPE_KEY);
211                 return type != null && !type.equals(Entities.EVENT_LINK);
212             }
213         }
214     }
