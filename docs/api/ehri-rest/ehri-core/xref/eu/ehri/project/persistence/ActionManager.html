53  /**
54   * Class for dealing with actions.
55   * <p>
56   * Events are captured as a linked list with new events placed
57   * at the head. The head event is connected to a node called the
58   * {@link eu.ehri.project.models.events.SystemEventQueue}. Events
59   * can have subjects (the thing the event is happening to) and
60   * actioners (the person initiating the event.) A subject's events
61   * and an actioner's actions likewise for a linked list so it is
62   * possible to fetch new events easily and prevent having to sort
63   * by timestamp, etc. Schematically, the graph thus formed looks
64   * something like:
65   * <p>
66   * <pre>
67   * <code>
68   * Actioner              SystemEventQueue             Subject
69   * \/                        \/                      \/
70   * [lifecycleAction]     [lifecycleActionStream]     [lifecycleEvent]
71   * |                         |                       |
72   * e3--[actionHasEvent]-&gt;-- Event 3 ---[hasEvent]--&lt;--e3
73   * \/                        \/                      \/
74   * [lifecycleAction]         [lifecycleAction]       [lifecycleEvent]
75   * |                         |                       |
76   * e2--[actionHasEvent]-&gt;-- Event 2 ---[hasEvent]--&lt;--e2
77   * \/                        \/                      \/
78   * [lifecycleAction]         [lifecycleAction]       [lifecycleEvent]
79   * |                         |                       |
80   * e1--[actionHasEvent]-&gt;-- Event 1 ---[hasEvent]--&lt;--e1
81   * </code>
82   * </pre>
83   */
84  public final class ActionManager {

215                     .addDataValue(Ontology.VERSION_ENTITY_CLASS, frame.getType())

218             EventContext ctx = new EventContext(actionManager, actioner,

251         /**
252          * Flush this event log to the graph.
253          */
254         public SystemEvent commit() {
255             Vertex vertex = getLinkNode(Ontology.ACTIONER_HAS_LIFECYCLE_ACTION);
256             replaceAtHead(actioner.asVertex(), vertex,
257                     Ontology.ACTIONER_HAS_LIFECYCLE_ACTION,
258                     Ontology.ACTIONER_HAS_LIFECYCLE_ACTION, Direction.OUT);
259             SystemEvent systemEvent = createGlobalEvent(timestamp, actionType, logMessage);
260             addActionerLink(systemEvent.asVertex(), vertex);
261 
262             for (Entity entity : subjects) {
263                 Vertex subjectVertex = getLinkNode(
264                         Ontology.ENTITY_HAS_LIFECYCLE_EVENT);
265                 replaceAtHead(entity.asVertex(), subjectVertex,
266                         Ontology.ENTITY_HAS_LIFECYCLE_EVENT,
267                         Ontology.ENTITY_HAS_LIFECYCLE_EVENT, Direction.OUT);
268                 addSubjectLink(systemEvent.asVertex(), subjectVertex);
269             }
270 
271             // Create the version.
272             if (versionBundle.isPresent() && versionFrame.isPresent()) {
273                 try {
274                     Entity subject = versionFrame.get();
275                     Bundle version = versionBundle.get();
276                     Version ev = dao.create(version, Version.class);
277                     replaceAtHead(subject.asVertex(), ev.asVertex(),
278                             Ontology.ENTITY_HAS_PRIOR_VERSION,
279                             Ontology.ENTITY_HAS_PRIOR_VERSION, Direction.OUT);
280                     graph.addEdge(null, ev.asVertex(),
281                             systemEvent.asVertex(), Ontology.VERSION_HAS_EVENT);
282                 } catch (ValidationError validationError) {
283                     throw new RuntimeException(validationError);
284                 }
285             }
286 
287             return systemEvent;
288         }

388     /**
389      * Determine if two events are the same according to the following
390      * definition:
391      * <p>
392      * <ol>
393      *     <li>They have the same scope and subject</li>
394      *     <li>They have the same actioner</li>
395      *     <li>They are both of the same type</li>
396      *     <li>They have the same log message, if any</li>
397      * </ol>
398      * <p>
399      * This function allows filtering an event stream for duplicates,
400      * like someone repeatedly updating the same item.
401      *
402      * @param event1 the first event
403      * @param event2 the second event
404      * @param timeDiffInSeconds the elapsed time between the events
405      *
406      * @return whether or not the events are effectively the same.
407      */
408     public static boolean canAggregate(SystemEvent event1, SystemEvent event2, int timeDiffInSeconds) {
409         // NB: Fetching all these props and relations is potentially quite
410         // costly, so we want to short-circuit and return early is possible,
411         // starting with the least-costly to fetch attributes.
412         EventTypes eventType1 = event1.getEventType();
413         EventTypes eventType2 = event2.getEventType();
414         if (eventType1 != null && eventType2 != null && !eventType1.equals(eventType2)) {
415             return false;
416         }
417 
418         String logMessage1 = event1.getLogMessage();
419         String logMessage2 = event2.getLogMessage();
420         if (logMessage1 != null && logMessage2 != null && !logMessage1.equals(logMessage2)) {
421             return false;
422         }
423 
424         if (timeDiffInSeconds > -1) {
425             DateTime event1Time = DateTime.parse(event1.getTimestamp());
426             DateTime event2Time = DateTime.parse(event2.getTimestamp());
427             int timeDiff = Seconds.secondsBetween(event1Time, event2Time).getSeconds();
428             if (timeDiff >= timeDiffInSeconds) {
429                 return false;
430             }
431         }
432 
433         Entity eventScope1 = event1.getEventScope();
434         Entity eventScope2 = event2.getEventScope();
435         if (eventScope1 != null && eventScope2 != null && !eventScope1.equals(eventScope2)) {
436             return false;
437         }
438 
439         Accessible entity1 = event1.getFirstSubject();
440         Accessible entity2 = event2.getFirstSubject();
441         if (entity1 != null && entity2 != null && !entity1.equals(entity2)) {
442             return false;
443         }
444 
445         Actioner actioner1 = event1.getActioner();
446         Actioner actioner2 = event2.getActioner();
447         if (actioner1 != null && actioner2 != null && !actioner1.equals(actioner2)) {
448             return false;
449         }
450 
451         // Okay, fall through...
452         return true;
453     }

455     /**
456      * Test if two events are sequential with the same actioner. This is used
457      * for aggregating repeated events.
458      *
459      * @param first the first temporal event
460      * @param second the second temporal event
461      * @return whether events can be aggregated by user
462      */
463     public static boolean sequentialWithSameAccessor(SystemEvent first, SystemEvent second) {
464         Vertex firstVertex = first.asVertex();
465         Vertex secondVertex = second.asVertex();
466 
467         for (Vertex flink : firstVertex.getVertices(Direction.IN, Ontology.ACTION_HAS_EVENT)) {
468             for (Vertex slink : secondVertex.getVertices(Direction.IN, Ontology.ACTION_HAS_EVENT)) {
469                 for (Edge chain : slink.getEdges(Direction.OUT, Ontology.ACTIONER_HAS_LIFECYCLE_ACTION)) {
470                     return chain.getVertex(Direction.IN).equals(flink);
471                 }
472             }
473         }
474         return false;
475     }

477     public static boolean sameAs(SystemEvent event1, SystemEvent event2) {
478         return canAggregate(event1, event2, -1);
479     }

483     private SystemEvent createGlobalEvent(String timestamp, EventTypes type, Optional<String> logMessage) {
484         // Create a global event and insert it at the head of the system queue. The
485         // relationship from the *system* node to the new latest action is
486         // *type* Stream.
487         try {
488             logger.trace("Creating global event root");
489             Vertex system = manager.getVertex(GLOBAL_EVENT_ROOT, EntityClass.SYSTEM);
490             Bundle ge = Bundle.Builder.withClass(EntityClass.SYSTEM_EVENT)
491                     .addDataValue(Ontology.EVENT_TYPE, type.toString())
492                     .addDataValue(Ontology.EVENT_TIMESTAMP, timestamp)
493                     .addDataValue(Ontology.EVENT_LOG_MESSAGE, logMessage.or(""))
494                     .build();
495             SystemEvent ev = dao.create(ge, SystemEvent.class);
496             if (!scope.equals(SystemScope.getInstance())) {
497                 ev.setEventScope(scope);
498             }
499             replaceAtHead(system, ev.asVertex(), Ontology.ACTIONER_HAS_LIFECYCLE_ACTION + "Stream", Ontology.ACTIONER_HAS_LIFECYCLE_ACTION, Direction.OUT);
500             return ev;
501         } catch (ItemNotFound e) {
502             e.printStackTrace();
503             throw new RuntimeException("Fatal error: system node (id: 'system') was not found. " +
504                     "Perhaps the graph was incorrectly initialised?");
505         } catch (ValidationError e) {
506             e.printStackTrace();
507             throw new RuntimeException(
508                     "Unexpected validation error creating action", e);
509         }
510     }

541     /**
542      * Given a vertex <em>head</em> that forms that start of a chain <em>relation</em> with
543      * direction <em>direction</em>, insert vertex <em>insert</em> <strong>after</strong>
544      * the head of the chain.
545      *
546      * @param head      The current vertex queue head node
547      * @param newHead   The replacement head node
548      * @param relation  The relationship between them
549      * @param direction The direction of the relationship
550      */
551     private void replaceAtHead(Vertex head, Vertex newHead, String headRelation,
552                                String relation, Direction direction) {
553         Iterator<Vertex> iter = head.getVertices(direction, headRelation).iterator();
554         if (iter.hasNext()) {
555             Vertex current = iter.next();
556             for (Edge e : head.getEdges(direction, headRelation)) {
557                 graph.removeEdge(e);
558             }
559             graph.addEdge(null, newHead, current, relation);
560 
561         }
562         graph.addEdge(null, head, newHead, headRelation);
563     }
