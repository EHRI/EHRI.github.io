78      // Aggregator function that aggregates adjacent events by 'strict' similarity,
79      // which in practice means:
80      // - same event type
81      // - same subject(s)
82      // - same actioner
83      private static final AggregatorPipe.AggregatorFunction<SystemEvent> strictAggregator = new AggregatorPipe
84              .AggregatorFunction<SystemEvent>() {
85          @Override
86          public boolean aggregate(SystemEvent a, SystemEvent b, int count) {
87              return ActionManager.sameAs(a, b);
88          }
89      };
90  
91      // Aggregator function that aggregates adjacent events by actioner
92      private static final AggregatorPipe.AggregatorFunction<SystemEvent> userAggregator = new AggregatorPipe
93              .AggregatorFunction<SystemEvent>() {
94          @Override
95          public boolean aggregate(SystemEvent a, SystemEvent b, int count) {
96              return count < MAX_AGGREGATION && ActionManager.sequentialWithSameAccessor(b, a);
97          }
98      };

342     private Iterable<SystemEvent> initStream() {
343         // If we're filtering the list for specific user's actions
344         // it's much more efficient to aggregate the user(s) event
345         // streams directly via than scanning the global one, even
346         // though we then have to sort the events to combine them
347         // into a newest-first stream.
348         if (users.isEmpty() && ids.isEmpty()) {
349             // No item/user filter: scan the global queue...
350             System.out.println("GLOBAL QUEUE");
351             return actionManager.getLatestGlobalEvents();
352         } else  {
353             List<Actioner> actioners = getItems(users, Actioner.class);
354             List<Accessible> entities = getItems(ids, Accessible.class);
355             if (actioners.size() == 1) {
356                 // Single user: return user's action queue
357                 return actioners.get(0).getActions();
358             } else if (entities.size() == 1) {
359                 // Single item: return item's history queue
360                 return entities.get(0).getHistory();
361             } else if (actioners.size() > 1) {
362                 // Merge multiple user action queues
363                 List<Iterable<SystemEvent>> actions = Lists.newArrayList();
364                 for (Actioner actioner : actioners) {
365                     actions.add(actioner.getActions());
366                 }
367                 return mergeEventQueues(actions);
368             } else {
369                 // Merge multiple item history queues
370                 List<Iterable<SystemEvent>> histories = Lists.newArrayList();
371                 for (Accessible entity : entities) {
372                     histories.add(entity.getHistory());
373                 }
374                 return mergeEventQueues(histories);
375             }
376         }
377     }

379     private Iterable<SystemEvent> mergeEventQueues(List<Iterable<SystemEvent>> queues) {
380         return Iterables.mergeSorted(queues, new Comparator<SystemEvent>() {
381             @Override
382             public int compare(SystemEvent event1, SystemEvent event2) {
383                 return event2.getTimestamp().compareTo(event1.getTimestamp());
384             }
385         });

388     private GremlinPipeline<SystemEvent, SystemEvent> applyAclFilter(GremlinPipeline<SystemEvent, SystemEvent> pipe,
389             Accessor asUser) {
390         final PipeFunction<Vertex, Boolean> aclFilterTest = AclManager.getAclFilterFunction(asUser);
391 
392         // Filter items accessible to this asUser... hide the
393         // event if any subjects or the scope are inaccessible
394         // to the asUser.
395         return pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
396             @Override
397             public Boolean compute(SystemEvent event) {
398                 Entity eventScope = event.getEventScope();
399                 if (eventScope != null && !aclFilterTest.compute(eventScope.asVertex())) {
400                     return false;
401                 }
402                 for (Accessible e : event.getSubjects()) {
403                     if (!aclFilterTest.compute(e.asVertex())) {
404                         return false;
405                     }
406                 }
407                 return true;
408             }
409         });
410     }

412     private GremlinPipeline<SystemEvent, SystemEvent> filterEvents(
413             GremlinPipeline<SystemEvent, SystemEvent> pipe) {
414 
415         if (!eventTypes.isEmpty()) {
416             pipe = pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
417                 @Override
418                 public Boolean compute(SystemEvent event) {
419                     return eventTypes.contains(event.getEventType());
420                 }
421             });
422         }
423 
424         if (!ids.isEmpty()) {
425             pipe = pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
426                 @Override
427                 public Boolean compute(SystemEvent event) {
428                     for (Accessible e : event.getSubjects()) {
429                         if (ids.contains(e.getId())) {
430                             return true;
431                         }
432                     }
433                     return false;
434                 }
435             });
436         }
437 
438         if (!entityTypes.isEmpty()) {
439             pipe = pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
440                 @Override
441                 public Boolean compute(SystemEvent event) {
442                     for (Accessible e : event.getSubjects()) {
443                         if (entityTypes.contains(manager.getEntityClass(e))) {
444                             return true;
445                         }
446                     }
447                     return false;
448                 }
449             });
450         }
451 
452         if (!users.isEmpty()) {
453             pipe = pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
454                 @Override
455                 public Boolean compute(SystemEvent event) {
456                     Actioner actioner = event.getActioner();
457                     return actioner != null && users.contains(actioner.getId());
458                 }
459             });
460         }
461 
462         // Add from/to filters (depends on timestamp strings comparing the right way!
463         if (from.isPresent()) {
464             pipe = pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
465                 @Override
466                 public Boolean compute(SystemEvent event) {
467                     String timestamp = event.getTimestamp();
468                     return from.get().compareTo(timestamp) >= 0;
469                 }
470             });
471         }
472 
473         if (to.isPresent()) {
474             pipe = pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
475                 @Override
476                 public Boolean compute(SystemEvent event) {
477                     String timestamp = event.getTimestamp();
478                     return to.get().compareTo(timestamp) <= 0;
479                 }
480             });
481         }
482 
483         return pipe;
484     }

486     private GremlinPipeline<SystemEvent, SystemEvent> getPersonalisedEvents(UserProfile asUser, Accessor accessor) {
487         // Add optional filters for event type, item type, and asUser...
488         GremlinPipeline<SystemEvent, SystemEvent> pipe = new GremlinPipeline<>(
489                 actionManager.getLatestGlobalEvents());
490 
491         // Add additional generic filters
492         pipe = filterEvents(pipe);
493 
494         // Filter out those we're not watching, or are actioned
495         // by users we're not following...
496 
497         if (showType.contains(ShowType.watched)) {
498             // Set IDs to items this asUser is watching...
499             final List<String> watching = Lists.newArrayList();
500             for (Watchable item : asUser.getWatching()) {
501                 watching.add(item.getId());
502             }
503 
504             pipe = pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
505                 @Override
506                 public Boolean compute(SystemEvent event) {
507                     for (Accessible e : event.getSubjects()) {
508                         if (watching.contains(e.getId())) {
509                             return true;
510                         }
511                     }
512                     return false;
513                 }
514             });
515         }
516 
517         if (showType.contains(ShowType.followed)) {
518             final List<String> following = Lists.newArrayList();
519             for (UserProfile other : asUser.getFollowing()) {
520                 following.add(other.getId());
521             }
522 
523             pipe = pipe.filter(new PipeFunction<SystemEvent, Boolean>() {
524                 @Override
525                 public Boolean compute(SystemEvent event) {
526                     Actioner actioner = event.getActioner();
527                     return actioner != null && following.contains(actioner.getId());
528                 }
529             });
530         }
531 
532         return applyAclFilter(pipe, accessor);
533     }

662     private GremlinPipeline<SystemEvent, List<SystemEvent>> aggregateFilter(GremlinPipeline<SystemEvent, SystemEvent> pipeline) {
663         switch (aggregation) {
664             case strict:
665                 return pipeline.add(new AggregatorPipe<>(strictAggregator));
666             case user:
667                 return pipeline.add(new AggregatorPipe<>(userAggregator));
668             default:
669                 // Default case: no aggregation, so simply wrap each event in a list
670                 return pipeline.transform(new PipeFunction<SystemEvent, List<SystemEvent>>() {
671                     @Override
672                     public List<SystemEvent> compute(SystemEvent event) {
673                         return Lists.newArrayList(event);
674                     }
675                 });
676         }
677     }

679     private <E, S> GremlinPipeline<E, S> setPipelineRange(
680             GremlinPipeline<E, S> filter) {
681         int low = Math.max(0, offset);
682         if (limit < 0) {
683             // No way to skip a bunch of items in Gremlin without
684             // applying an end range... I guess this will break if
685             // we have more than 2147483647 items to traverse...
686             return filter.range(low, Integer.MAX_VALUE); // No filtering
687         } else if (limit == 0) {
688             return new GremlinPipeline<>(Lists.newArrayList());
689         } else {
690             // NB: The high range is inclusive, oddly.
691             return filter.range(low, low + (limit - 1));
692         }
693     }

695     private <T> List<T> getItems(Collection<String> itemIds, Class<T> cls) {
696         List<T> items = Lists.newArrayList();
697         for (String itemId : itemIds) {
698             try {
699                 items.add(manager.getEntity(itemId, cls));
700             } catch (ItemNotFound itemNotFound) {
701                 logger.warn("Invalid event filter item: " + itemId);
702             }
703         }
704         return items;
