41  /**
42   * Delegation functions for ID generation.
43   */
44  public class IdGeneratorUtils {
45      /**
46       * Separator for ID components.
47       */
48      public static final String HIERARCHY_SEPARATOR = "-";
49      public static final String SLUG_REPLACE = "_";
50  
51      private static final Joiner hierarchyJoiner = Joiner.on(HIERARCHY_SEPARATOR);
52  
53      protected final static Logger logger = LoggerFactory.getLogger(IdGeneratorUtils.class);
54  
55      public static ListMultimap<String, String> handleIdCollision(Collection<String> scopeIds,
56              String dataKey, String ident) {
57  
58          logger.error("ID Generation error: {}={} (scope: {})", dataKey, ident, Lists.newArrayList(scopeIds));
59          ListMultimap<String, String> errors = ArrayListMultimap.create();
60          errors.put(dataKey, MessageFormat.format(
61                  Messages.getString("BundleDAO.uniquenessError"), ident));
62          return errors;
63      }
64  
65  
66      /**
67       * Uses the items identifier and its entity type to generate a (supposedly)
68       * unique ID.
69       *
70       * @param scope  the permission scope
71       * @param bundle the item's data bundle
72       * @param ident  the item's identifier
73       */
74      public static String generateId(PermissionScope scope, Bundle bundle, String ident) {
75          LinkedList<String> scopeIds = Lists.newLinkedList();
76          if (scope != null && !scope.equals(SystemScope.getInstance())) {
77              for (PermissionScope s : scope.getPermissionScopes())
78                  scopeIds.addFirst(s.getIdentifier());
79              scopeIds.add(scope.getIdentifier());
80          }
81          return generateId(scopeIds, bundle, ident);
82      }
83  
84      /**
85       * Use an array of scope IDs and the bundle data to generate a unique
86       * id within a given scope.
87       *
88       * @param scopeIds An array of scope ids
89       * @param bundle   The input bundle
90       * @param ident    the item's identifier
91       * @return The complete id string
92       */
93      public static String generateId(Collection<String> scopeIds, Bundle bundle, String ident) {
94  
95          // Validation should have ensured that ident exists...
96          if (ident == null || ident.trim().isEmpty()) {
97              throw new RuntimeException("Invalid null identifier for "
98                      + bundle.getType().getName() + ": " + bundle.getData());
99          }
100         List<String> newIds = Lists.newArrayList(scopeIds);
101         newIds.add(ident);
102         return joinPath(newIds);
103     }
104 
105     /**
106      * Join an identifier path to form a full ID.
107      * Duplicate parts of the part are removed.
108      * This check is case sensitive.
109      *
110      * @param path A non-empty list of identifier strings.
111      * @return The resultant path ID.
112      */
113     public static String joinPath(Collection<String> path) {
114         // If the last part of the path is included in the
115         // current part, remove that chunk, providing that
116         // there is something left over.
117         List<String> newPaths = Lists.newArrayList();
118         String last = null;
119         for (String ident : path) {
120             if (last == null) {
121                 newPaths.add(ident);
122             } else {
123                 if (ident.startsWith(last) && !ident.equals(last)) {
124                     newPaths.add(ident.substring(last.length()));
125                 } else {
126                     newPaths.add(ident);
127                 }
128             }
129             last = ident;
130         }
131 
132         // Slugify the path sections...
133         List<String> slugged = Lists.newArrayList();
134         for (String p : newPaths) {
135             slugged.add(Slugify.slugify(p, SLUG_REPLACE));
136         }
137 
138         return hierarchyJoiner.join(slugged);
139     }
