45  /**
46   * Util class for re-generating the IDs for a given
47   * set of items. Hierarchical IDs are used to enforce uniqueness
48   * of identifiers within a particular parent-child scope. In
49   * theory, since identifiers are generally very stable this works
50   * fine, but occasionally they change, invalidating the identifiers
51   * of their child items.
52   * <p/>
53   * We also sometimes tweak the ID generation algorithm itself, which
54   * necessitates bulk ID re-generation.
55   *
56   * @author Mike Bryant (http://github.com/mikesname)
57   */
58  public class IdRegenerator {
59      private final FramedGraph<?> graph;
60      private final GraphManager manager;
61      private final Serializer depSerializer;
62      private final boolean dryrun;
63      private final boolean skipCollisions;
64      private final boolean collisionMode;
65  
66      public static class IdCollisionError extends Exception {
67          public IdCollisionError(String from, String to) {
68              super(String.format("Renaming %s to %s collide with existing item",
69                      from, to));
70          }
71      }
72  
73      public IdRegenerator(FramedGraph<?> graph) {
74          this(graph, true, false, false);
75      }
76  
77      private IdRegenerator(FramedGraph<?> graph, boolean dryrun, boolean skipCollisions,
78              boolean collisionMode) {
79          this.graph = graph;
80          this.manager = GraphManagerFactory.getInstance(graph);
81          this.depSerializer = new Serializer.Builder(graph).dependentOnly().build();
82          this.dryrun = dryrun;
83          this.skipCollisions = skipCollisions;
84          this.collisionMode = collisionMode;
85      }
86  
87      public List<List<String>> reGenerateIds(PermissionScope scope, Iterable<? extends Frame> items) throws
88              IdCollisionError {
89          List<List<String>> remaps = Lists.newArrayList();
90          for (Frame item: items) {
91              Optional<List<String>> optionalRemap = reGenerateId(scope, item);
92              if (optionalRemap.isPresent()) {
93                  remaps.add(optionalRemap.get());
94              }
95          }
96          return remaps;
97      }

99      public List<List<String>> reGenerateIds(Iterable<? extends AccessibleEntity> items) throws IdCollisionError {
100         List<List<String>> remaps = Lists.newArrayList();
101         for (AccessibleEntity item: items) {
102             Optional<List<String>> optionalRemap = reGenerateId(item);
103             if (optionalRemap.isPresent()) {
104                 remaps.add(optionalRemap.get());
105             }
106         }
107         return remaps;
108     }

110     public Optional<List<String>> reGenerateId(AccessibleEntity item) throws IdCollisionError {
111         return reGenerateId(item.getPermissionScope(), item);
112     }

114     public Optional<List<String>> reGenerateId(PermissionScope permissionScope, Frame item)
115             throws IdCollisionError {
116         String currentId = item.getId();
117         Collection<String> idChain = Lists.newArrayList();
118         if (permissionScope != null && !permissionScope.equals(SystemScope.getInstance())) {
119             idChain.addAll(permissionScope.idPath());
120         }
121 
122         EntityClass entityClass = manager.getEntityClass(item);
123         try {
124             IdGenerator idgen = entityClass.getIdgen();
125             Bundle itemBundle = depSerializer.vertexFrameToBundle(item);
126             String newId = idgen.generateId(idChain, itemBundle);
127             if (collisionMode) {
128                 if (!newId.equals(currentId) && manager.exists(newId)) {
129                     List<String> collision = Lists.newArrayList(currentId, newId);
130                     return Optional.of(collision);
131                 } else {
132                     return Optional.absent();
133                 }
134             } else {
135                 if (!newId.equals(currentId)) {
136                     if (manager.exists(newId)) {
137                         if (!skipCollisions) {
138                             throw new IdCollisionError(currentId, newId);
139                         } else {
140                             return Optional.absent();
141                         }
142                     } else {
143                         if (!dryrun) {
144                             manager.renameVertex(item.asVertex(), currentId, newId);
145 
146                             // Rename all the descriptions
147                             String idBase = idgen.getIdBase(itemBundle);
148                             Collection<String> descIdChain = Lists.newArrayList(idChain);
149                             descIdChain.add(idBase);
150                             for (Description d : manager.cast(item, DescribedEntity.class).getDescriptions()) {
151                                 Bundle desc = depSerializer.vertexFrameToBundle(d);
152                                 String newDescriptionId = desc.getType().getIdgen().generateId(descIdChain, desc);
153                                 manager.renameVertex(d.asVertex(), d.getId(), newDescriptionId);
154                             }
155 
156                             // Change the ID on any versions...
157                             for (Version v : manager.cast(item, VersionedEntity.class).getAllPriorVersions()) {
158                                 manager.setProperty(v.asVertex(), Ontology.VERSION_ENTITY_ID, newId);
159                             }
160                         }
161                         List<String> remap = Lists.newArrayList(currentId, newId);
162                         return Optional.of(remap);
163                     }
164                 } else {
165                     return Optional.absent();
166                 }
167             }
168         } catch (SerializationError e) {
169             throw new RuntimeException(e);
170         }
171     }

173     /**
174      * Obtain a re-generator that will actually perform the rename
175      * step.
176      *
177      * @param doIt whether to actually commit changes
178      *
179      * @return a new, more dangerous, re-generator
180      */
181     public IdRegenerator withActualRename(boolean doIt) {
182         return new IdRegenerator(graph, !doIt, skipCollisions, collisionMode);
183     }
184 
185     /**
186      * Obtain a re-generator that will skip items that would collide
187      * with existing items.
188      *
189      * @param skipCollisions whether or not to error on an ID collision
190      *
191      * @return a new, more tolerant, re-generator
192      */
193     public IdRegenerator skippingCollisions(boolean skipCollisions) {
194         return new IdRegenerator(graph, dryrun, skipCollisions, collisionMode);
195     }
196 
197     /**
198      * Obtain a re-generator that only outputs items with IDs that if renamed
199      * would collide with another item's ID.
200      *
201      * @param collisionMode only output items that would collide if renamed
202      *
203      * @return a new, more tolerant, re-generator
204      */
205     public IdRegenerator collisionMode(boolean collisionMode) {
206         return new IdRegenerator(graph, dryrun, skipCollisions, collisionMode);
207     }
