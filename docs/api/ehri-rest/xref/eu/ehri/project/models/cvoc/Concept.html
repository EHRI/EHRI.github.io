38  /**
39   * This models the thesaurus terms or keywords in a way that is better managing multi-linguality. 
40   * The terms are then labels of concepts in a specific language and concepts can have many labels.  
41   * This is following the SKOS-Core concept, but not fully so we don't use 'SKOS' in the class name. 
42   * For SKOS core see: http://www.w3.org/2009/08/skos-reference/skos.html
43   * 
44   * Also note that the labels and textual information of a single language 
45   * are all placed in a ConceptDescription, following the pattern 
46   * that the textual details of an entity are described by a Description entity.  
47   * 
48   * A nice glossary of terms used with controlled vocabularies can be found here: 
49   * http://www.willpowerinfo.co.uk/glossary.htm
50   */
51  @EntityType(EntityClass.CVOC_CONCEPT)
52  public interface Concept extends Described, AuthoritativeItem, ItemHolder {
53  
54      // NB: As an AuthoritativeItem the set will be @Fetched automatically
55      @Mandatory
56      @Adjacency(label = Ontology.ITEM_IN_AUTHORITATIVE_SET)
57      Vocabulary getVocabulary();
58  
59      @Adjacency(label = Ontology.ITEM_IN_AUTHORITATIVE_SET)
60      void setVocabulary(Vocabulary vocabulary);
61  
62      @Property(CHILD_COUNT)
63      long getChildCount();
64  
65      // relations to other concepts
66      
67      // Note that multiple broader concepts are possible
68      @Fetch(Ontology.CONCEPT_HAS_BROADER)
69      @Adjacency(label = Ontology.CONCEPT_HAS_NARROWER, direction=Direction.IN)
70      Iterable<Concept> getBroaderConcepts();
71  
72      // NOTE: don't put a Fetch on it, because it can be a large tree of concepts
73      @Adjacency(label = Ontology.CONCEPT_HAS_NARROWER, direction = Direction.OUT)
74      @InverseOf(Ontology.CONCEPT_HAS_BROADER)
75      Iterable<Concept> getNarrowerConcepts();
76  
77      @JavaHandler
78      void addNarrowerConcept(Concept concept);
79  
80      @JavaHandler
81      void removeNarrowerConcept(Concept concept);
82  
83      
84      // Related concepts, should be like a symmetric associative link... 
85      @Adjacency(label = Ontology.CONCEPT_HAS_RELATED)
86      Iterable<Concept> getRelatedConcepts();
87  
88      @JavaHandler
89      void addRelatedConcept(Concept concept);
90  
91      @Adjacency(label = Ontology.CONCEPT_HAS_RELATED)
92      void removeRelatedConcept(Concept concept);
93      
94      // Hmm, does not 'feel' symmetric
95      @Adjacency(label = Ontology.CONCEPT_HAS_RELATED, direction=Direction.IN)
96      Iterable<Concept> getRelatedByConcepts();
97  
98      /**
99       * Implementation of complex methods.
100      */
101     abstract class Impl  implements JavaHandlerContext<Vertex>, Concept {
102 
103         public long getChildCount() {
104             return gremlin().outE(Ontology.CONCEPT_HAS_NARROWER).count();
105         }
106 
107         public void addRelatedConcept(Concept related) {
108             JavaHandlerUtils.addUniqueRelationship(it(),
109                     related.asVertex(), Ontology.CONCEPT_HAS_RELATED);
110         }
111 
112         public void addNarrowerConcept(Concept concept) {
113             if (!concept.asVertex().equals(it())) {
114                 JavaHandlerUtils.addUniqueRelationship(it(),
115                         concept.asVertex(), Ontology.CONCEPT_HAS_NARROWER);
116             }
117         }
118 
119         public void removeNarrowerConcept(Concept concept) {
120             JavaHandlerUtils.removeAllRelationships(it(), concept.asVertex(),
121                     Ontology.CONCEPT_HAS_NARROWER);
122         }
123     }
124 }
