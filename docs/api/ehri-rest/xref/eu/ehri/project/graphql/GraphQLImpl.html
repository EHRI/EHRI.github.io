139     private static Map<String, Object> mapOf(Object... items) {
140         Preconditions.checkArgument(items.length % 2 == 0, "Items must be pairs of key/value");
141         Map<String, Object> map = Maps.newHashMap();
142         for (int i = 0; i < items.length; i += 2) {
143             map.put(((String) items[i]), items[i + 1]);
144         }
145         return map;
146     }

148     private static int decodeCursor(String cursor, int defaultVal) {
149         try {
150             return cursor != null
151                     ? Math.max(-1, Integer.parseInt(Base64.fromBase64(cursor)))
152                     : defaultVal;
153         } catch (NumberFormatException e) {
154             return defaultVal;
155         }
156     }

158     private static int getLimit(Integer limitArg, boolean stream) {
159         if (limitArg == null) {
160             return stream ? -1 : DEFAULT_LIST_LIMIT;
161         } else if (limitArg < 0) {
162             return stream ? limitArg : MAX_LIST_LIMIT;
163         } else {
164             return stream ? limitArg : Math.min(MAX_LIST_LIMIT, limitArg);
165         }
166     }
167 
168     private static int getOffset(String afterCursor, String fromCursor) {
169         if (afterCursor != null) {
170             return decodeCursor(afterCursor, -1) + 1;
171         } else if (fromCursor != null) {
172             return decodeCursor(fromCursor, 0);
173         } else {
174             return 0;
175         }
176     }

197     private DataFetcher topLevelDocDataFetcher() {
198         Iterable<Country> countries = api().query()
199                 .setStream(true).setLimit(-1).page(EntityClass.COUNTRY, Country.class);
200         Iterable<Iterable<Entity>> docs = Iterables.transform(countries, c ->
201                 Iterables.transform(c.getTopLevelDocumentaryUnits(), d -> d.as(Entity.class)));
202         return connectionDataFetcher(() -> Iterables.concat(docs));
203     }

205     private DataFetcher entityTypeConnectionDataFetcher(EntityClass type) {
206         // FIXME: The only way to get a list of all items of a given
207         // type via the API alone is to run a query as a stream w/ no limit.
208         // However, this means that ACL filtering will be applied twice,
209         // once here, and once by the connection data fetcher, which also
210         // applies pagination. This is a bit gross but the speed difference
211         // appears to be negligible.
212         return connectionDataFetcher(() -> api().query()
213                 .setStream(true).setLimit(-1).page(type, Entity.class));
214     }

216     private DataFetcher hierarchicalOneToManyRelationshipConnectionFetcher(
217             Function<Entity, Iterable<? extends Entity>> top, Function<Entity, Iterable<? extends Entity>> all) {
218         // Depending on the value of the "all" argument, return either just
219         // the top level items or everything in the tree.
220         return env -> {
221             boolean allOrTop = (Boolean) Optional.ofNullable(env.getArgument(ALL_PARAM)).orElse(false);
222             Function<Entity, Iterable<? extends Entity>> func = allOrTop ? all : top;
223             return connectionDataFetcher(() -> func.apply(((Entity) env.getSource()))).get(env);
224         };
225     }

231     private DataFetcher connectionDataFetcher(Supplier<Iterable<? extends Entity>> iter) {
232         // NB: The data fetcher takes a supplier here so lazily generated
233         // streams can be invoked more than one (if, e.g. both the items array
234         // and the edges array is needed.) Otherwise we would have to somehow
235         // reset the Iterable.
236         return env -> {
237             int limit = getLimit(env.getArgument(FIRST_PARAM), stream);
238             int offset = getOffset(env.getArgument(AFTER_PARAM), env.getArgument(FROM_PARAM));
239             return stream && limit < 0
240                     ? lazyConnectionDataFetcher(iter, limit, offset)
241                     : strictConnectionDataFetcher(iter, limit, offset);
242         };
243     }

245     private Map<String, Object> connectionData(Iterable<Entity> items,
246             Iterable<Map<String, Object>> edges, String nextCursor, String prevCursor) {
247         return mapOf(
248                 ITEMS, items,
249                 EDGES, edges,
250                 PAGE_INFO, mapOf(
251                         HAS_NEXT_PAGE, nextCursor != null,
252                         NEXT_PAGE, nextCursor,
253                         HAS_PREVIOUS_PAGE, prevCursor != null,
254                         PREVIOUS_PAGE, prevCursor
255                 )
256         );
257     }
258 
259     private Map<String, Object> strictConnectionDataFetcher(Supplier<Iterable<? extends Entity>> iter, int limit, int offset) {
260         QueryApi query = api().query().setLimit(limit).setOffset(offset);
261         QueryApi.Page<Entity> page = query.page(iter.get(), Entity.class);
262         List<Entity> items = Lists.newArrayList(page);
263 
264         // Create a list of edges, with the cursor taking into
265         // account each item's offset
266         List<Map<String, Object>> edges = Lists.newArrayListWithExpectedSize(items.size());
267         for (int i = 0; i < items.size(); i++) {
268             edges.add(mapOf(
269                     CURSOR, Base64.toBase64(String.valueOf(offset + i)),
270                     NODE, items.get(i)
271             ));
272         }
273 
274         boolean hasNext = page.getOffset() + items.size() < page.getTotal();
275         boolean hasPrev = page.getOffset() > 0;
276         String nextCursor = Base64.toBase64(String.valueOf(offset + limit));
277         String prevCursor = Base64.toBase64(String.valueOf(offset - limit));
278         return connectionData(items, edges, hasNext ? nextCursor : null, hasPrev ? prevCursor : null);
279     }

281     private Map<String, Object> lazyConnectionDataFetcher(Supplier<Iterable<? extends Entity>> iter, int limit, int offset) {
282         QueryApi query = api().query().setLimit(limit).setOffset(offset);
283         QueryApi.Page<Entity> items = query.page(iter.get(), Entity.class);
284         boolean hasPrev = items.getOffset() > 0;
285 
286         // Create a list of edges, with the cursor taking into
287         // account each item's offset
288         final AtomicInteger index = new AtomicInteger();
289         Iterable<Map<String, Object>> edges = Iterables.transform(
290                 query.page(iter.get(), Entity.class), item -> mapOf(
291                         CURSOR, Base64.toBase64(String.valueOf(offset + index.getAndIncrement())),
292                         NODE, item
293                 ));
294 
295         String prevCursor = Base64.toBase64(String.valueOf(offset - limit));
296         return connectionData(items, edges, null, hasPrev ? prevCursor : null);
297     }

317     private static DataFetcher listDataFetcher(DataFetcher fetcher) {
318         return environment -> {
319             Object obj = fetcher.get(environment);
320             if (obj == null) {
321                 return Collections.emptyList();
322             } else if (obj instanceof List) {
323                 return obj;
324             } else {
325                 return Lists.newArrayList(obj);
326             }
327         };
328     }

330     private static final DataFetcher descriptionDataFetcher = environment -> {
331         String lang = environment.getArgument(Ontology.LANGUAGE_OF_DESCRIPTION);
332         String code = environment.getArgument(Ontology.IDENTIFIER_KEY);
333 
334         Entity source = (Entity) environment.getSource();
335         Iterable<Description> descriptions = source.as(Described.class).getDescriptions();
336 
337         if (lang == null && code == null) {
338             int at = environment.getArgument(SLICE_PARAM);
339             List<Description> descList = Lists.newArrayList(descriptions);
340             return at >= 1 && descList.size() >= at ? descList.get(at - 1) : null;
341         } else {
342             for (Description next : descriptions) {
343                 String langCode = next.getLanguageOfDescription();
344                 if (langCode.equalsIgnoreCase(lang)) {
345                     if (code != null && !code.isEmpty()) {
346                         String ident = next.getDescriptionCode();
347                         if (ident.equals(code)) {
348                             return next;
349                         }
350                     } else {
351                         return next;
352                     }
353                 }
354             }
355             return null;
356         }
357     };

370     private DataFetcher manyToOneRelationshipFetcher(Function<Entity, Entity> f) {
371         return environment -> {
372             Entity elem = f.apply((Entity) environment.getSource());
373             if (elem != null &&
374                     AclManager.getAclFilterFunction(api().accessor())
375                             .compute(elem.asVertex())) {
376                 return elem;
377             }
378 
379             return null;
380         };
381     }

602                         .description("Pagination information")
603                         .type(newObject()
604                                 .name(PAGE_INFO)
605                                 .field(newFieldDefinition()
606                                         .name(HAS_PREVIOUS_PAGE)
607                                         .description("If a previous page of data is available")

610                                 )
611                                 .field(newFieldDefinition()
612                                         .name(PREVIOUS_PAGE)
613                                         .description("A cursor pointing to the previous page of items")
614                                         .type(GraphQLString)
615                                         .build()
616                                 )
617                                 .field(newFieldDefinition()
618                                         .name(HAS_NEXT_PAGE)
619                                         .description("If another page of data is available")

621                                         .build()
622                                 )
623                                 .field(newFieldDefinition()
624                                         .name(NEXT_PAGE)
