12   *
13   * @author Mike Bryant (http://github.com/mikesname)

15  public class GraphManagerFactory {

17      /**
18       * Obtain an instance of a graph manager.
19       *
20       * @param graph An indexable and transactional Blueprints graph.
21       * @return A graph manager instance.
22       */
23      // NB: Because Java doesn't support multiple wildcard bounds we do some checking
24      // of the bounds manually ourselves, which is ugly but should ensure it's safe
25      // to do an unchecked cast here.
26      @SuppressWarnings("unchecked")
27      public static GraphManager getInstance(FramedGraph<?> graph) {
28          Graph baseGraph = graph.getBaseGraph();
29  
30          if (!IndexableGraph.class.isAssignableFrom(baseGraph.getClass())) {
31              throw new RuntimeException("Graph instance must be transactional and indexable");
32          }
33  
34          if (Neo4jGraph.class.isAssignableFrom(baseGraph.getClass())) {
35              return new Neo4jGraphManager(graph);
36          } else {
37              return new BlueprintsGraphManager(graph);
38          }
39      }
