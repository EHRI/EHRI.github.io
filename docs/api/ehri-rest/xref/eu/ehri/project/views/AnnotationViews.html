52  /**
53   * View class for handling annotation-related operations.
54   *
55   * @author Mike Bryant (http://github.com/mikesname)
56   */
57  public final class AnnotationViews implements Scoped<AnnotationViews> {
58  
59      private final FramedGraph<?> graph;
60      private final AclManager acl;
61      private final ViewHelper helper;
62      private final GraphManager manager;
63  
64      /**
65       * Scoped constructor.
66       *
67       * @param graph The graph
68       * @param scope The view scope
69       */
70      public AnnotationViews(FramedGraph<?> graph, PermissionScope scope) {
71          this.graph = graph;
72          helper = new ViewHelper(graph, scope);
73          acl = helper.getAclManager();
74          manager = GraphManagerFactory.getInstance(graph);
75      }
76  
77      /**
78       * Constructor with system scope.
79       *
80       * @param graph The graph
81       */
82      public AnnotationViews(FramedGraph<?> graph) {
83          this(graph, SystemScope.getInstance());
84      }
85  
86      /**
87       * Create an annotation for a dependent node of an item. The entity and the
88       * dependent item can be the same.
89       *
90       * @param id           the identifier of the AccessibleEntity this annotation is attached to, as a target
91       * @param did          the identifier of the dependent item
92       * @param bundle       the annotation itself
93       * @param user         the user creating the annotation
94       * @param accessibleTo Users or groups who can access this annotation
95       * @return the created annotation
96       * @throws PermissionDenied
97       * @throws AccessDenied
98       * @throws ValidationError
99       * @throws ItemNotFound
100      */
101     public Annotation createFor(String id, String did, Bundle bundle, Accessor user, Collection<Accessor> accessibleTo)
102             throws PermissionDenied, AccessDenied, ValidationError, ItemNotFound {
103         AccessibleEntity entity = manager.getFrame(id, AccessibleEntity.class);
104         AnnotatableEntity dep = manager.getFrame(did, AnnotatableEntity.class);
105         helper.checkEntityPermission(entity, user, PermissionType.ANNOTATE);
106         helper.checkReadAccess(entity, user);
107 
108         if (!(entity.equals(dep) || isInSubtree(entity, dep))) {
109             // FIXME: Better error message here...
110             throw new PermissionDenied("Item is not covered by parent item's permissions");
111         }
112 
113         Annotation annotation = new BundleDAO(graph).create(bundle, Annotation.class);
114         entity.addAnnotation(annotation);
115         if (!entity.equals(dep)) {
116             dep.addAnnotationPart(annotation);
117         }
118         annotation.setAnnotator(graph.frame(user.asVertex(), Annotator.class));
119         acl.setAccessors(annotation, accessibleTo);
120         acl.withScope(SystemScope.INSTANCE)
121                 .grantPermission(annotation, PermissionType.OWNER, user);
122 
123         new ActionManager(graph, entity)
124                 .newEventContext(annotation,
125                         graph.frame(user.asVertex(), Actioner.class),
126                         EventTypes.annotation, Optional.<String>absent())
127                 .commit();
128         return annotation;
129     }
130 
131     private boolean isInSubtree(Frame parent, Frame child) {
132         // Check dependent is within item's subtree!
133         final Set<String> deps = Sets.newHashSet();
134         new Serializer(graph).traverseSubtree(parent, new TraversalCallback() {
135             @Override
136             public void process(Frame vertexFrame, int depth,
137                     String relation, int relationIndex) {
138                 deps.add(vertexFrame.getId());
139             }
140         });
141         return deps.contains(child.getId());
142     }
143 
144     @Override
145     public AnnotationViews withScope(PermissionScope scope) {
146         return new AnnotationViews(graph, scope);
147     }
