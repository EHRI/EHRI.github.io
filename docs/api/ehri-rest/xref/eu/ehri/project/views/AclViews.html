69      /**
70       * Set the global permission matrix for a user.
71       *
72       * @param accessor The user
73       * @param permissionSet The new permissions
74       * @param grantee The user/group granting the new permissions
75       * @throws PermissionDenied
76       */
77      public InheritedGlobalPermissionSet setGlobalPermissionMatrix(Accessor accessor,
78                                         GlobalPermissionSet permissionSet,
79              Accessor grantee) throws PermissionDenied {
80          checkGrantPermission(grantee, permissionSet);
81          acl.setPermissionMatrix(accessor, permissionSet);
82          boolean scoped = !scope.equals(SystemScope.INSTANCE);
83          // Log the action...
84          ActionManager.EventContext context = actionManager.logEvent(
85                  manager.cast(accessor, AccessibleEntity.class),
86                  manager.cast(grantee, Actioner.class),
87                  EventTypes.setGlobalPermissions);
88          if (scoped) {
89              context.addSubjects(manager.cast(scope, AccessibleEntity.class));
90          }
91          return acl.getInheritedGlobalPermissions(accessor);
92      }

111     /**
112      * Check the accessor has GRANT permissions to update another user's
113      * permissions.
114      * 
115      * @param accessor The user
116      * @param permissionSet The user's permissions.
117      * @throws PermissionDenied
118      */
119     private void checkGrantPermission(Accessor accessor,
120             GlobalPermissionSet permissionSet)
121             throws PermissionDenied {
122         Map<ContentTypes, Collection<PermissionType>> permissionMap
123                 = permissionSet.asMap();
124         // Check we have grant permissions for the requested content types
125         if (!acl.belongsToAdmin(accessor)) {
126             try {
127                 Permission grantPerm = manager.getFrame(
128                         PermissionType.GRANT.getName(), Permission.class);
129                 for (ContentTypes ctype : permissionMap.keySet()) {
130                     if (!acl.hasPermission(ctype, PermissionType.GRANT, accessor)) {
131                         throw new PermissionDenied(accessor.getId(),
132                                 ctype.toString(), grantPerm.getId(),
133                                 scope.getId());
134                     }
135                 }
136             } catch (ItemNotFound e) {
137                 throw new RuntimeException(
138                         "Unable to get node for permission type '"
139                                 + PermissionType.GRANT + "'", e);
140             }
141         }
142     }

165     public void revokePermissionGrant(PermissionGrant grant, Accessor user)
166             throws PermissionDenied {
167         // TODO: This should be rather more complicated and account for the
168         // fact that individual grants can, in theory, have more than one
169         // target content type.
170         for (PermissionGrantTarget tg : grant.getTargets()) {
171             switch (manager.getEntityClass(tg)) {
172                 case CONTENT_TYPE:
173                     helper.checkContentPermission(user, ContentTypes.withName(tg.getId()), PermissionType.GRANT);
174                     break;
175                 default:
176                     helper.checkEntityPermission(
177                         manager.cast(tg, AccessibleEntity.class), user, PermissionType.GRANT);
178             }
179         }
180         acl.revokePermissionGrant(grant);
181     }

222     private void ensureCanModifyGroupMembership(Group group, Accessor user, Accessor grantee)
223             throws PermissionDenied {
224         // If a user is not admin they can only add someone else to a group if
225         // a) they belong to that group themselves, and
226         // b) they have the modify permission on that group, and
227         // c) they have grant permissions for the user
228         if (!acl.belongsToAdmin(grantee)) {
229             boolean found = false;
230             for (Accessor acc : grantee.getAllParents()) {
231                 if (group.equals(acc)) {
232                     found = true;
233                     break;
234                 }
235             }
236             if (!found) {
237                 throw new PermissionDenied(grantee.getId(), group.getId(),
238                         "Non-admin users cannot add other users to groups that they" +
239                                 " do not themselves belong to.");
240             }
241             helper.checkEntityPermission(manager.cast(user, AccessibleEntity.class),
242                     grantee, PermissionType.GRANT);
243             helper.checkEntityPermission(group, grantee, PermissionType.UPDATE);
244         }
245     }
