57      public VirtualUnitViews(FramedGraph<?> graph) {
58          this.graph = graph;
59          this.manager = GraphManagerFactory.getInstance(graph);
60          this.aclManager = new AclManager(graph);
61          this.viewHelper = new ViewHelper(graph);

65      public void moveIncludedUnits(VirtualUnit from, VirtualUnit to, Iterable<DocumentaryUnit> included,
66              Accessor accessor) throws PermissionDenied {
67          // Wrap this in a list as a precaution because it
68          // will be iterated twice!
69          List<DocumentaryUnit> items = Lists.newArrayList(included);
70          removeIncludedUnits(from, items, accessor);
71          addIncludedUnits(to, items, accessor);
72      }

74      /**
75       * Add documentary units to be included in a virtual unit as child items.
76       *
77       * @param parent   The parent VU
78       * @param included A set of child DUs
79       * @param accessor The current user
80       * @throws PermissionDenied
81       */
82      public void addIncludedUnits(VirtualUnit parent, Iterable<DocumentaryUnit> included, Accessor accessor)
83              throws PermissionDenied {
84          viewHelper.checkEntityPermission(parent, accessor, PermissionType.UPDATE);
85          for (DocumentaryUnit unit : included) {
86              parent.addIncludedUnit(unit);
87          }
88      }
89  
90      /**
91       * Remove documentary units from a virtual unit as child items.
92       *
93       * @param parent   The parent VU
94       * @param included A set of child DUs
95       * @param accessor The current user
96       * @throws PermissionDenied
97       */
98      public void removeIncludedUnits(VirtualUnit parent, Iterable<DocumentaryUnit> included, Accessor accessor)
99              throws PermissionDenied {
100         viewHelper.checkEntityPermission(parent, accessor, PermissionType.UPDATE);
101         for (DocumentaryUnit unit : included) {
102             parent.removeIncludedUnit(unit);
103         }
104     }

106     /**
107      * Find virtual collections to which this item belongs.
108      *
109      * @param item     An item (typically a documentary unit)
110      * @param accessor The current user
111      * @return A set of top-level virtual units
112      */
113     public Iterable<VirtualUnit> getVirtualCollections(Frame item, Accessor accessor) {
114 
115         // This is a relatively complicated traversal. We want to go from the item,
116         // then to any descriptions, from those descriptions to any virtual units
117         // that reference them, and then up to the top-level item. It is complicated
118         // by the fact that the first encountered unit might actually be the top-level
119         // item, in which case our loop traversal will miss it, so we have to combine
120         // the result
121 
122         GremlinPipeline<Vertex, Vertex> pipe = new GremlinPipeline<>();
123         Pipeline<Vertex, Vertex> otherPipe = pipe.start(item.asVertex())
124                 .in(Ontology.DESCRIPTION_FOR_ENTITY)
125                 .in(Ontology.VC_DESCRIBED_BY)
126                 .as("n").out(Ontology.VC_IS_PART_OF)
127                 .loop("n", JavaHandlerUtils.defaultMaxLoops, new PipeFunction<LoopPipe.LoopBundle<Vertex>, Boolean>() {
128                     @Override
129                     public Boolean compute(LoopPipe.LoopBundle<Vertex> vertexLoopBundle) {
130                         return (!vertexLoopBundle.getObject()
131                                 .getEdges(Direction.OUT, Ontology.VC_IS_PART_OF)
132                                 .iterator().hasNext())
133                                 && manager.getEntityClass(vertexLoopBundle.getObject())
134                                 .equals(EntityClass.VIRTUAL_UNIT);
135                     }
136                 });
137 
138         GremlinPipeline<Vertex, Vertex> out = new GremlinPipeline<Vertex, Vertex>(item.asVertex())
139                 .copySplit(new GremlinPipeline<Vertex, Object>(item.asVertex())
140                         .in(Ontology.DESCRIPTION_FOR_ENTITY)
141                         .in(Ontology.VC_DESCRIBED_BY), otherPipe)
142                 .exhaustMerge().cast(Vertex.class).filter(new PipeFunction<Vertex, Boolean>() {
143                     @Override
144                     public Boolean compute(Vertex vertex) {
145                         return !vertex.getEdges(Direction.OUT, Ontology.VC_IS_PART_OF)
146                                 .iterator().hasNext();
147                     }
148                 }).filter(AclManager.getAclFilterFunction(accessor));
149 
150         return graph.frameVertices(out, VirtualUnit.class);
151     }
152 
153     /**
154      * Find virtual collections to which this user belongs.
155      *
156      * @param user     An user (typically a documentary unit)
157      * @param accessor The current user
158      * @return A set of top-level virtual units
159      */
160     public Iterable<VirtualUnit> getVirtualCollectionsForUser(Frame user, Accessor accessor) {
161         GremlinPipeline<Vertex, Vertex> pipe = new GremlinPipeline<>();
162         Pipeline<Vertex, Vertex> filtered = pipe.start(user.asVertex())
163                 .in(Ontology.VC_HAS_AUTHOR)
164                 .filter(AclManager.getAclFilterFunction(accessor));
165 
166         return graph.frameVertices(filtered, VirtualUnit.class);
167     }
