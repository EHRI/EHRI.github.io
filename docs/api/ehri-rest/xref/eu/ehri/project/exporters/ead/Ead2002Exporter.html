123     public Document export(DocumentaryUnit unit, String langCode) throws IOException {
124         // Root
125         Document doc = documentBuilder.newDocument();

131                 "xs:schemaLocation", "urn:isbn:1-931666-22-9 http://www.loc.gov/ead/ead.xsd");
132         doc.appendChild(rootElem);
133 
134         Element eadHeaderElem = doc.createElement("eadheader");
135         eadHeaderElem.setAttribute("relatedencoding", "DC");
136         eadHeaderElem.setAttribute("scriptencoding", "iso15924");
137         eadHeaderElem.setAttribute("repositoryencoding", "iso15511");
138         eadHeaderElem.setAttribute("dateencoding", "iso8601");
139         eadHeaderElem.setAttribute("countryencoding", "iso3166-1");
140 
141         rootElem.appendChild(eadHeaderElem);
142         Element eadIdElem = doc.createElement("eadid");
143         eadIdElem.setTextContent(unit.getId());
144         eadHeaderElem.appendChild(eadIdElem);
145 
146         Repository repository = unit.getRepository();
147         Optional<Description> descOpt = LanguageHelpers.getBestDescription(
148                 unit, Optional.<Description>absent(), langCode);
149 
150         for (Description desc : descOpt.asSet()) {
151             addFileDesc(doc, eadHeaderElem, repository, desc, langCode);
152             addProfileDesc(doc, eadHeaderElem, unit, desc);
153         }
154 
155         addRevisionDesc(doc, eadHeaderElem, unit);
156 
157         //
158         // Archdesc section
159         //
160         Element archDescElem = doc.createElement("archdesc");
161         rootElem.appendChild(archDescElem);
162 
163         for (Description desc : descOpt.asSet()) {
164             String level = Optional
165                     .fromNullable(desc.<String>getProperty("levelOfDescription"))
166                     .or("otherlevel");
167             archDescElem.setAttribute("level", level);
168             Element didElem = addDataSection(doc, archDescElem, unit, desc);
169 
170             for (Description repoDesc : LanguageHelpers.getBestDescription(repository,
171                     Optional.<Description>absent(), langCode).asSet()) {
172                 Element repoElem = doc.createElement("repository");
173                 didElem.appendChild(repoElem);
174                 Element corpElem = doc.createElement("corpname");
175                 repoElem.appendChild(corpElem);
176                 corpElem.setTextContent(repoDesc.getName());
177             }
178 
179             addPropertyValues(doc, archDescElem, desc);
180         }
181 
182         Element dscElem = doc.createElement("dsc");
183         archDescElem.appendChild(dscElem);
184         for (DocumentaryUnit child : getOrderedChildren(unit)) {
185             addEadLevel(doc, dscElem, 1, child, descOpt, langCode);
186         }
187 
188         for (Description desc : descOpt.asSet()) {
189             addControlAccess(doc, archDescElem, desc);
190         }
191 
192         return doc;
193     }

206     private void addRevisionDesc(Document doc, Element eadHeaderElem, DocumentaryUnit unit) {
207         List<List<SystemEvent>> eventList = Lists.newArrayList(
208                 api.events().aggregateForItem(unit));
209         if (!eventList.isEmpty()) {
210             Element revDescElem = doc.createElement("revisiondesc");
211             eadHeaderElem.appendChild(revDescElem);
212 
213             for (int i = eventList.size() - 1; i >= 0; i--) {
214                 List<SystemEvent> agg = eventList.get(i);
215                 SystemEvent event = agg.get(0);
216                 Element changeElem = doc.createElement("change");
217                 revDescElem.appendChild(changeElem);
218                 Element dateElem = doc.createElement("date");
219                 changeElem.appendChild(dateElem);
220                 dateElem.setTextContent(new DateTime(event.getTimestamp()).toString());
221                 Element itemElem = doc.createElement("item");
222                 changeElem.appendChild(itemElem);
223                 if (event.getLogMessage() == null || event.getLogMessage().isEmpty()) {
224                     itemElem.setTextContent(event.getEventType().name());
225                 } else {
226                     itemElem.setTextContent(String.format("%s [%s]",
227                             event.getLogMessage(), event.getEventType()));
228                 }
229             }
230         }

312     public void addEadLevel(Document doc, Element base, int num, DocumentaryUnit subUnit,
313             Optional<Description> priorDescOpt, String langCode) throws IOException {
314         logger.trace("Adding EAD sublevel: c" + num + " -> " + base.getTagName());
315         Optional<Description> descOpt = LanguageHelpers.getBestDescription(subUnit, priorDescOpt, langCode);
316         String levelTag = String.format("c%02d", num);
317         Element levelElem = doc.createElement(levelTag);
318         base.appendChild(levelElem);
319 
320         for (Description desc : descOpt.asSet()) {
321             for (String level : Optional.fromNullable(
322                     desc.<String>getProperty("levelOfDescription")).asSet()) {
323                 levelElem.setAttribute("level", level);
324             }
325         }
326 
327         for (Description desc : descOpt.asSet()) {
328             addDataSection(doc, levelElem, subUnit, desc);
329             addPropertyValues(doc, levelElem, desc);
330             addControlAccess(doc, levelElem, desc);
331         }
332 
333         for (DocumentaryUnit child : getOrderedChildren(subUnit)) {
334             addEadLevel(doc, levelElem, num + 1, child, descOpt, langCode);
335         }

338     private Element addDataSection(Document doc, Element levelElem, DocumentaryUnit subUnit, Description desc)
339             throws IOException {
340         Element didElem = doc.createElement("did");
341         levelElem.appendChild(didElem);
342         Element unitIdElem = doc.createElement("unitid");
343         didElem.appendChild(unitIdElem);
344         unitIdElem.setTextContent(subUnit.getIdentifier());
345 
346         Set<String> propertyKeys = desc.getPropertyKeys();
347 
348         Element unitTitleElem = doc.createElement("unittitle");
349         didElem.appendChild(unitTitleElem);
350         unitTitleElem.setTextContent(desc.getName());
351 
352         for (DatePeriod datePeriod : desc.as(DocumentaryUnitDescription.class).getDatePeriods()) {
353             if (DatePeriod.DatePeriodType.creation.equals(datePeriod.getDateType())) {
354                 String start = datePeriod.getStartDate();
355                 String end = datePeriod.getEndDate();
356                 Element dateElem = doc.createElement("unitdate");
357                 if (start != null && end != null) {
358                     DateTime startDateTime = new DateTime(start);
359                     DateTime endDateTime = new DateTime(end);
360                     dateElem.setAttribute("normal", String.format("%s/%s",
361                             unitDateNormalFormat.print(startDateTime),
362                             unitDateNormalFormat.print(endDateTime)));
363                     dateElem.setTextContent(String.format("%s/%s",
364                             startDateTime.year().get(), endDateTime.year().get()));
365                     didElem.appendChild(dateElem);
366                 } else if (start != null) {
367                     DateTime startDateTime = new DateTime(start);
368                     dateElem.setAttribute("normal", String.format("%s",
369                             unitDateNormalFormat.print(startDateTime)));
370                     dateElem.setTextContent(String.format("%s", startDateTime.year().get()));
371                     didElem.appendChild(dateElem);
372                 }
373             }
374         }
375 
376         for (Map.Entry<String, String> pair : textDidMappings.entrySet()) {
377             if (propertyKeys.contains(pair.getKey())) {
378                 Object value = desc.getProperty(pair.getKey());
379                 List values = value instanceof List ? (List) value : Lists.newArrayList(value);
380                 for (Object v : values) {
381                     Element elem = doc.createElement(pair.getValue());
382                     didElem.appendChild(elem);
383                     elem.setTextContent(v.toString());
384                 }
385             }
386         }
387 
388         if (propertyKeys.contains("languageOfMaterial")) {
389             Element langMatElem = doc.createElement("langmaterial");
390             didElem.appendChild(langMatElem);
391             Object value = desc.getProperty("languageOfMaterial");
392             List values = value instanceof List
393                     ? (List) value : Lists.newArrayList(value);
394             for (Object v : values) {
395                 Element lang = doc.createElement("language");
396                 lang.setTextContent(LanguageHelpers.codeToName(v.toString()));
397                 // Only add the language if its a 3-letter string
398                 if (v.toString().length() == 3) {
399                     lang.setAttribute("langcode", v.toString());
400                 }
401                 langMatElem.appendChild(lang);
402             }
403         }
404         return didElem;
405     }

407     private void addControlAccess(Document doc, Element element, Description desc) {
408         Map<AccessPointType, List<AccessPoint>> byType = Maps.newHashMap();
409         for (AccessPoint accessPoint : desc.getAccessPoints()) {
410             AccessPointType type = accessPoint.getRelationshipType();
411             if (controlAccessMappings.containsKey(type)) {
412                 if (byType.containsKey(type)) {
413                     byType.get(type).add(accessPoint);
414                 } else {
415                     byType.put(type, Lists.newArrayList(accessPoint));
416                 }
417             }
418         }
419 
420         for (Map.Entry<AccessPointType, List<AccessPoint>> entry : byType.entrySet()) {
421             AccessPointType type = entry.getKey();
422             Element ctrlElem = doc.createElement("controlaccess");
423             element.appendChild(ctrlElem);
424             for (AccessPoint accessPoint : entry.getValue()) {
425                 Element apElem = doc.createElement(controlAccessMappings.get(type));
426                 apElem.setTextContent(accessPoint.getName());
427                 ctrlElem.appendChild(apElem);
428             }
429         }
430     }

432     private void addPropertyValues(Document doc, Element base, Entity item) throws IOException {
433         Set<String> available = item.getPropertyKeys();
434 
435         for (Map.Entry<String, String> pair : multiValueTextMappings.entrySet()) {
436             if (available.contains(pair.getKey())) {
437                 Object value = item.getProperty(pair.getKey());
438                 List values = value instanceof List
439                         ? (List) value
440                         : Lists.newArrayList(value);
441                 for (Object v : values) {
442                     Element elem = doc.createElement(pair.getValue());
443                     base.appendChild(elem);
444                     createCDataElement(doc, elem, "p", v.toString());
445                 }
446             }
447         }
