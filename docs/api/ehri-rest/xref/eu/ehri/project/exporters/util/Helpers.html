300     public static Optional<String> countryCodeToContinent(String countryCode) {
301         String continentCode = countryCodesToContinents.get(countryCode.toUpperCase());
302         if (continentCode != null) {
303             return Optional.fromNullable(continentCodes.get(continentCode));
304         }
305         return Optional.absent();
306     }

308     /**
309      * Get the best description for a given language code.
310      *
311      * @param item         a described item
312      * @param priorDescOpt if the object is hierarchical, the parent-level
313      *                     description
314      * @param langCode     a 3-letter language code.
315      * @return the best matching description found
316      */
317     public static Optional<Description> getBestDescription(Described item, Optional<Description> priorDescOpt, String langCode) {
318         List<Description> descriptions = Lists.newArrayList(item.getDescriptions());
319         Collections.sort(descriptions, new Comparator<Description>() {
320             @Override
321             public int compare(Description d1, Description d2) {
322                 return d1.getId().compareTo(d2.getId());
323             }
324         });
325         Description fallBack = null;
326         for (Description description : descriptions) {
327             if (fallBack == null) {
328                 fallBack = description;
329             }
330             // First of all, check the description code (usually set to the
331             // EAD file ID.) If this is the same as the parent, return the
332             // current description.
333             for (Description parent : priorDescOpt.asSet()) {
334                 for (String code : Optional.fromNullable(parent.getDescriptionCode()).asSet()) {
335                     if (code.equals(description.getDescriptionCode())) {
336                         return Optional.of(description);
337                     }
338                 }
339             }
340 
341             // Otherwise, fall back to the first one with the same language
342             if (description.getLanguageOfDescription().equalsIgnoreCase(langCode)) {
343                 return Optional.of(description);
344             }
345         }
346         return Optional.fromNullable(fallBack);
347     }

349     public static Optional<Description> getBestDescription(Described item, String langCode) {
350         return getBestDescription(item , Optional.<Description>absent(), langCode);
351     }
