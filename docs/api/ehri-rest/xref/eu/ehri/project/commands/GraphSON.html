40  /**
41   * Dump the complete graph as graphSON file, or import such a dump
42   * <p/>
43   * Example usage:
44   * - stop the server
45   * $NEO4J_HOME/bin/neo4j stop
46   * - save a dump
47   * ./scripts/cmd graphson -d out graph.json
48   * or
49   * ./scripts/cmd graphson -d out - > graph.json
50   * - edit it
51   * - remove the graph
52   * rm -rf $NEO4J_HOME/data/graph.db
53   * - load edited graph
54   * ./scripts/cmd graphson -d in graph.json
55   * - start server
56   * $NEO4J_HOME/bin/neo4j start
57   *
58   * @author Mike Bryant (https://github.com/mikesname)
59   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
60   */
61  public class GraphSON extends BaseCommand implements Command {
62  
63      final static String NAME = "graphson";
64  
65      public GraphSON() {
66      }
67  
68      @Override
69      public String getHelp() {
70          return "export or import a GraphSON file.\n" + getUsage();
71      }
72  
73      @Override
74      public String getUsage() {
75          return "Usage: graphson -d [out|in] <filename>";
76      }
77  
78      @Override
79      protected void setCustomOptions(Options options) {
80          options.addOption(new Option("d", true, "Output or input a dump"));
81      }
82  
83      @Override
84      @SuppressWarnings("unchecked")
85      public int execWithOptions(final FramedGraph<?> graph,
86              CommandLine cmdLine) throws Exception {
87  
88          if (cmdLine.getArgList().size() < 1) {
89              throw new MissingArgumentException("Graph file path missing");
90          }
91  
92          String dumpMode = "out"; //defaults might also be handled by the parser?
93  
94          if (cmdLine.hasOption("d")) {
95              dumpMode = cmdLine.getOptionValue("d");
96          }
97  
98          // check if option is useful, otherwise print the help and bail out
99          if (dumpMode.contentEquals("out")) {
100             saveDump(graph, cmdLine);
101         } else if (dumpMode.contentEquals("in")) {
102             loadDump(graph, cmdLine);
103         } else {
104             throw new UnrecognizedOptionException("Unrecognised dump mode: '" + dumpMode + "'");
105         }
106 
107         return 0;
108     }
109 
110     public void saveDump(final FramedGraph<?> graph,
111             CommandLine cmdLine) throws IOException {
112 
113         String filepath = (String) cmdLine.getArgList().get(0);
114 
115         // if the file is '-' that means we do standard out
116         if (filepath.contentEquals("-")) {
117             // to stdout
118             GraphSONWriter.outputGraph(graph, System.out, GraphSONMode.EXTENDED);
119         } else {
120             // try to open or create the file for writing
121             OutputStream out = new FileOutputStream(filepath);
122             GraphSONWriter.outputGraph(graph, out, GraphSONMode.EXTENDED);
123             out.close();
124         }
125     }
126 
127     public void loadDump(final FramedGraph<?> graph,
128             CommandLine cmdLine) throws IOException {
129         GraphSONReader reader = new GraphSONReader(graph);
130         String filepath = (String) cmdLine.getArgList().get(0);
131 
132         InputStream inputStream = new FileInputStream(filepath);
133         InputStream readStream = filepath.toLowerCase().endsWith(".gz")
134                 ? new GZIPInputStream(inputStream)
135                 : inputStream;
136         try {
137             reader.inputGraph(readStream, 1000);
138             GraphManagerFactory.getInstance(graph).rebuildIndex();
139         } finally {
140             inputStream.close();
141         }
142     }
