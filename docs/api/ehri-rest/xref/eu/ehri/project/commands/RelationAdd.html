20  package eu.ehri.project.commands;

22  import com.tinkerpop.blueprints.TransactionalGraph;
23  import com.tinkerpop.blueprints.Vertex;
24  import com.tinkerpop.frames.FramedGraph;
25  import eu.ehri.project.core.GraphManager;
26  import eu.ehri.project.core.GraphManagerFactory;
27  import eu.ehri.project.exceptions.ItemNotFound;
28  import eu.ehri.project.models.utils.JavaHandlerUtils;
29  import org.apache.commons.cli.CommandLine;
30  import org.apache.commons.cli.Option;
31  import org.apache.commons.cli.Options;
32  
33  /**
34   * Add an arbitrary edge between two nodes.
35   *
36   * @author Mike Bryant (https://github.com/mikesname)
37   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
38   */
39  public class RelationAdd extends BaseCommand implements Command {
40  
41      final static String NAME = "add-rel";
42  
43      public RelationAdd() {
44      }
45  
46      @Override
47      protected void setCustomOptions(Options options) {
48          options.addOption(new Option("s", "single", false,
49             "Ensure the out entity only has one relationship of this type by removing any others"));
50          options.addOption(new Option("d", "allow-duplicates", false,
51                  "Allow creating multiple edges with the same label between the same two nodes"));
52      }
53  
54      @Override
55      public String getHelp() {
56          return "Usage: add-rel [OPTIONS] <source> <rel-name> <target>";
57      }
58  
59      @Override
60      public String getUsage() {
61          return "Create a relationship between a source and a target";
62      }
63  
64      @Override
65      public int execWithOptions(final FramedGraph<? extends TransactionalGraph> graph,
66              CommandLine cmdLine) throws ItemNotFound {
67  
68          GraphManager manager = GraphManagerFactory.getInstance(graph);
69  
70          if (cmdLine.getArgList().size() < 3)
71              throw new RuntimeException(getHelp());
72  
73          String src = (String)cmdLine.getArgList().get(0);
74          String label = (String)cmdLine.getArgList().get(1);
75          String dst = (String)cmdLine.getArgList().get(2);
76  
77          Vertex source = manager.getVertex(src);
78          Vertex target = manager.getVertex(dst);
79  
80          try {
81              if (cmdLine.hasOption("allow-duplicates")) {
82                  source.addEdge(label, target);
83              } else if (cmdLine.hasOption("unique")) {
84                  if (!JavaHandlerUtils.addUniqueRelationship(source, target, label)) {
85                      System.err.println("Relationship already exists");
86                  }
87              } else  {
88                  if (!JavaHandlerUtils.addSingleRelationship(source, target, label)) {
89                      System.err.println("Relationship already exists");
90                  }
91              }
92              graph.getBaseGraph().commit();
93          } catch (Exception e) {
94              graph.getBaseGraph().rollback();
95              throw new RuntimeException(e);
96          }
97  
98          return 0;
99      }
