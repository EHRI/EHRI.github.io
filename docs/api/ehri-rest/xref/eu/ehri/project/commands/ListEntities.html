46  /**
47   * Command for listing entities of a given type.
48   *
49   * @author Mike Bryant (https://github.com/mikesname)
50   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
51   */
52  public class ListEntities extends BaseCommand implements Command {
53  
54      final static String NAME = "list";
55  
56      @Override
57      @SuppressWarnings("static-access")
58      protected void setCustomOptions(Options options) {
59          options.addOption(OptionBuilder
60             .withType(String.class)
61             .withLongOpt("format").isRequired(false)
62             .hasArg(true).withArgName("f")
63             .withDescription("Format for output data, which defaults to just the id. " +
64                     "If provided can be one of: xml, json").create("f"));
65          options.addOption(OptionBuilder
66                  .withType(String.class)
67                  .withLongOpt("root-node").isRequired(false)
68                  .hasArg(true).withArgName("r")
69                  .withDescription("Name of the root node (default: '" + NAME + "')").create("r"));
70      }
71  
72      @Override
73      public String getHelp() {
74          return "Usage: list [OPTIONS] <type>";
75      }
76  
77      @Override
78      public String getUsage() {
79          return "List entities of a given type.";
80      }
81  
82      @Override
83      public int execWithOptions(final FramedGraph<?> graph, CommandLine cmdLine) throws Exception {
84  
85          // the first argument is the entity type, and that must be specified
86          if (cmdLine.getArgList().size() < 1)
87              throw new RuntimeException(getHelp());
88          EntityClass type = EntityClass.withName(cmdLine.getArgs()[0]);
89          Class<?> cls = type.getEntityClass();
90  
91          if (!AccessibleEntity.class.isAssignableFrom(cls))
92              throw new RuntimeException("Unknown accessible entity: " + type);
93  
94          GraphManager manager = GraphManagerFactory.getInstance(graph);
95          Serializer serializer = new Serializer(graph);
96          String rootName = cmdLine.getOptionValue("r", NAME);
97  
98          if (!cmdLine.hasOption("f")) {
99              // default to only outputting the id's
100             printIds(manager, type);
101         } else {
102             // if there is a second argument, that might be 'json' or 'xml'
103             String format = cmdLine.getOptionValue("f");
104             if (format.equalsIgnoreCase("xml")) {
105                 printXml(manager, serializer, type, getTransformer(Optional.<InputStream>absent()),
106                         rootName);
107             } else if (format.equalsIgnoreCase("json")) {
108                 printJson(manager, serializer, type);
109             } else {
110                 // If there's an XSLT file in the resources that is named
111                 // EntityType_destinationFormat.xslt use that...
112                 String xsltName = String.format("%s_%s.xslt", type.getName(), format);
113                 InputStream ios = EntityClass.class.getClassLoader().getResourceAsStream(xsltName);
114                 if (ios != null) {
115                     try {
116                         Transformer transformer = getTransformer(Optional.fromNullable(ios));
117                         printXml(manager, serializer, type, transformer, rootName);
118                     } finally {
119                         ios.close();
120                     }
121                 } else {
122                     // unknown format
123                     throw new RuntimeException(
124                             String.format("Unknown format: '%s' and no '%s' xslt found", format, xsltName));
125                 }
126             }
127         }
128 
129         return 0;
130     }
131 
132     private void printIds(GraphManager manager, EntityClass type) {
133         for (AccessibleEntity acc : manager.getFrames(type, AccessibleEntity.class)) {
134             System.out.println(acc.getId());
135         }
136     }
137 
138     private void printJson(GraphManager manager, Serializer serializer, EntityClass type)
139             throws SerializationError {
140 
141         // NOTE no json root {}, but always a list []
142         System.out.print("[\n");
143 
144         int cnt = 0;
145         for (AccessibleEntity acc : manager.getFrames(type, AccessibleEntity.class)) {
146             String jsonString = serializer.vertexFrameToJson(acc);
147             if (cnt != 0) System.out.print(",\n");
148             System.out.println(jsonString);
149             cnt++;
150         }
151         System.out.print("]\n"); // end list
152 
153     }
154 
155     private void printXml(GraphManager manager, Serializer serializer, EntityClass type,
156             Transformer transformer, String rootName)
157                 throws UnsupportedEncodingException, SerializationError, TransformerException {
158         System.out.print("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
159         System.out.print("<" + rootName + ">\n"); // root element
160         for (AccessibleEntity acc : manager.getFrames(type, AccessibleEntity.class)) {
161             transformer.transform(new DOMSource(serializer.vertexFrameToXml(acc)),
162                     new StreamResult(new OutputStreamWriter(System.out, "UTF-8")));
163         }
164         System.out.print("</" + rootName + ">\n"); // root element
165     }
166 
167     private static Transformer getTransformer(Optional<InputStream> xsltOpt) throws IOException, TransformerException {
168         TransformerFactory tf = TransformerFactory.newInstance();
169         Transformer transformer = xsltOpt.isPresent()
170                 ? tf.newTransformer(new StreamSource(xsltOpt.get()))
171                 : tf.newTransformer();
172         transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
173         transformer.setOutputProperty(OutputKeys.METHOD, "xml");
174         transformer.setOutputProperty(OutputKeys.INDENT, "yes");
175         transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
176         transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
177         return transformer;
178     }
