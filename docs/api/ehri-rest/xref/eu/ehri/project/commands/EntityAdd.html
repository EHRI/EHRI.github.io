45  public class EntityAdd extends BaseCommand {

52      @Override
53      protected void setCustomOptions(Options options) {
54          options.addOption(Option.builder("P")
55                  .argName("property=value")
56                  .numberOfArgs(2)
57                  .valueSeparator()
58                  .desc("Add a property with the given value")
59                  .build());
60          options.addOption(Option.builder()
61                  .longOpt("user")
62                  .hasArg()
63                  .required()
64                  .type(String.class)
65                  .hasArg().desc("Identifier of user taking action")
66                  .build());
67          options.addOption(Option.builder()
68                  .longOpt("log")
69                  .hasArg()
70                  .type(String.class)
71                  .desc("Log message for delete action.")
72                  .build());
73          options.addOption(Option.builder()
74                  .longOpt("scope")
75                  .hasArg()
76                  .type(String.class)
77                  .desc("Identifier of scope to import into, i.e. repository")
78                  .build());
79          options.addOption(Option.builder()
80                  .longOpt("update")
81                  .desc("Update item if it already exists")
82                  .build());
83      }

85      @Override
86      public String getUsage() {
87          return String.format("%s <type> [OPTIONS] [-Pkey=value]", NAME);
88      }
89  
90      @Override
91      public String getHelp() {
92          return "Create a new entity with the given id and properties";
93      }
94  
95      @Override
96      public int execWithOptions(FramedGraph<?> graph,
97              CommandLine cmdLine) throws ItemNotFound, ValidationError, PermissionDenied, DeserializationError {
98  
99          GraphManager manager = GraphManagerFactory.getInstance(graph);
100 
101         if (cmdLine.getArgList().size() < 1)
102             throw new RuntimeException(getUsage());
103 
104         String logMessage = "Imported from command-line";
105         if (cmdLine.hasOption("log")) {
106             logMessage = cmdLine.getOptionValue("log");
107         }
108 
109         // Find the agent
110         PermissionScope scope = SystemScope.getInstance();
111         if (cmdLine.hasOption("scope")) {
112             scope = manager.getEntity(cmdLine.getOptionValue("scope"), PermissionScope.class);
113         }
114 
115         // Find the user
116         UserProfile user = manager.getEntity(cmdLine.getOptionValue("user"),
117                 UserProfile.class);
118 
119         String typeName = cmdLine.getArgList().get(0);
120         EntityClass entityClass = EntityClass.withName(typeName);
121         Properties properties = cmdLine.getOptionProperties("P");
122 
123         Bundle.Builder builder = Bundle.Builder.withClass(entityClass);
124         for (Object prop : properties.keySet()) {
125             builder.addDataValue((String) prop, properties.getProperty((String) prop));
126         }
127         Bundle bundle = builder.build();
128         String id = entityClass.getIdGen().generateId(scope.idPath(), bundle);
129 
130         try {
131             createItem(graph, cmdLine, id, bundle, scope, user, logMessage);
132         } catch (ValidationError e) {
133             System.err.printf("A user a id: '%s' already exists%n", id);
134             return CmdEntryPoint.RetCode.BAD_DATA.getCode();
135         } catch (PermissionDenied e) {
136             System.err.printf("User %s does not have permission to perform that action.%n", user.getId());
137             return CmdEntryPoint.RetCode.BAD_PERMS.getCode();
138         } catch (DeserializationError e) {
139             System.err.println(e.getMessage());
140             return CmdEntryPoint.RetCode.BAD_DATA.getCode();
141         }
142 
143         return 0;
144     }
145 
146     // Suppressing warnings here because we throw a RuntimeException if the
147     // item class is not of an acceptable type.
148     @SuppressWarnings("unchecked")
149     public void createItem(FramedGraph<?> graph,
150             CommandLine cmdLine, String id, Bundle bundle,
151             PermissionScope scope, UserProfile user, String logMessage) throws DeserializationError,
152             ValidationError, PermissionDenied {
153 
154         if (!Accessible.class.isAssignableFrom(bundle.getBundleJavaClass())) {
155             throw new DeserializationError("Item class: " + bundle.getBundleJavaClass().getSimpleName() +
156                     " is not a first-class database item");
157         }
158 
159         LoggingCrudViews<?> view = new LoggingCrudViews(graph, bundle.getBundleJavaClass(), scope);
160         if (cmdLine.hasOption("update")) {
161             view.createOrUpdate(bundle.withId(id), user, getLogMessage(logMessage));
162         } else {
163             view.create(bundle.withId(id), user, getLogMessage(logMessage));
164         }
165     }
