49  /**
50   * Sanity check various parts of the graph.
51   *
52   * @author Mike Bryant (http://github.com/mikesname)
53   */
54  public class Check extends BaseCommand implements Command {
55  
56      final static String NAME = "check";
57  
58      public Check() {
59      }
60      
61  	@Override
62  	public String getHelp() {
63          return "perform various checks on the graph structure" +
64          		"\n" + getUsage();
65  	}
66  	
67      @Override
68      public String getUsage() {
69  		return "Usage: check [OPTIONS]";
70      }
71  
72      @Override
73      protected void setCustomOptions(Options options) {
74          options.addOption(new Option("q", "quick", false, "Quick checks only"));
75      }
76      
77      @Override
78      @SuppressWarnings("unchecked")
79      public int execWithOptions(final FramedGraph<?> graph,
80              CommandLine cmdLine) throws Exception {
81  
82          final GraphManager manager = GraphManagerFactory.getInstance(graph);
83          checkPermissionScopes(graph, manager);
84          checkOwnerPermGrantsHaveNoScope(manager);
85  
86          return 0;
87      }
88  
89      /**
90       * The following types of item should ALL have a permission scope.
91       *
92       *  Doc unit - either a repository or another doc unit
93       *  Concept - a vocabulary
94       *  Repository - a country
95       *  Hist agent - an auth set
96       *
97       * @param graph The graph
98       * @param manager The graph manager
99       * @throws Exception
100      */
101     public void checkPermissionScopes(final FramedGraph<?> graph,
102             final GraphManager manager) throws Exception {
103 
104         List<EntityClass> types = Lists.newArrayList(DOCUMENTARY_UNIT, REPOSITORY, CVOC_CONCEPT, HISTORICAL_AGENT);
105 
106         for (EntityClass entityClass : types) {
107             try (CloseableIterable<? extends Frame> items = manager.getFrames(entityClass, entityClass.getEntityClass())) {
108                 for (Frame item : items) {
109                     AccessibleEntity entity = graph.frame(item.asVertex(), AccessibleEntity.class);
110                     PermissionScope scope = entity.getPermissionScope();
111                     if (scope == null) {
112                         System.err.println("Missing scope: " + entity.getId() + " (" + entity.asVertex().getId() + ")");
113                     } else {
114                         switch (manager.getEntityClass(item)) {
115                             case DOCUMENTARY_UNIT:
116                                 checkIdGeneration(graph.frame(item.asVertex(), DocumentaryUnit.class), scope);
117                                 break;
118                             case REPOSITORY:
119                                 checkIdGeneration(graph.frame(item.asVertex(), Repository.class), scope);
120                                 break;
121                             default:
122                         }
123                     }
124                 }
125             }
126         }
127     }
128 
129     private void checkIdGeneration(IdentifiableEntity doc, PermissionScope scope) {
130         if (scope != null) {
131             String ident = doc.getIdentifier();
132             List<String> path = Lists.newArrayList(Iterables.concat(scope.idPath(), Lists.newArrayList(ident)));
133             String finalId = IdGeneratorUtils.joinPath(path);
134             if (!finalId.equals(doc.getId())) {
135                 System.err.println(String.format("Generated ID does not match scopes: '%s' -> %s + %s",
136                         doc.getId(), path, ident));
137             }
138         }
139     }
140 
141     private void checkOwnerPermGrantsHaveNoScope(final GraphManager manager) throws Exception {
142         try (CloseableIterable<PermissionGrant> items = manager
143                 .getFrames(EntityClass.PERMISSION_GRANT, PermissionGrant.class)) {
144             for (PermissionGrant grant : items) {
145                 Frame scope = grant.getScope();
146                 Frame perm = grant.getPermission();
147                 if (scope != null && perm != null && perm.getId().equals(PermissionType.OWNER.getName())) {
148                     System.err.println(
149                             String.format("Owner permission grant with scope: " + grant.asVertex().getId()));
150                 }
151             }
152         }
153 
154     }
