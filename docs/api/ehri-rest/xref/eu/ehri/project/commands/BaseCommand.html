20  package eu.ehri.project.commands;

22  import com.google.common.base.Optional;
23  import com.tinkerpop.frames.FramedGraph;
24  import eu.ehri.project.models.base.Accessor;
25  import eu.ehri.project.api.Api;
26  import eu.ehri.project.api.ApiFactory;
27  import org.apache.commons.cli.CommandLine;
28  import org.apache.commons.cli.CommandLineParser;
29  import org.apache.commons.cli.DefaultParser;
30  import org.apache.commons.cli.HelpFormatter;
31  import org.apache.commons.cli.Options;
32  import org.apache.commons.cli.ParseException;
33  import org.slf4j.Logger;
34  
35  import java.io.PrintWriter;
36  import java.io.StringWriter;

38  /**
39   * Abstract base class for commands. Provides the main
40   * entry points for interaction:
41   * <p>
42   * <ul>
43   * <li>Set options</li>
44   * <li>Get help</li>
45   * <li>Get usage</li>
46   * <li>Execute with options</li>
47   * </ul>
48   */
49  public abstract class BaseCommand implements Command {
50  
51      protected static final Logger logger = org.slf4j.LoggerFactory.getLogger(Command.class);
52      protected final Options options = new Options();
53      private final CommandLineParser parser = new DefaultParser();
54  
55      protected void setCustomOptions(Options options) {
56      }
57  
58      public abstract String getHelp();
59  
60      public abstract String getUsage();
61  
62      private String getHelpFooter() {
63          return "";
64      }
65  
66      /**
67       * Execute this command with the given database and un-parsed
68       * command line arguments
69       *
70       * @param graph the graph database
71       * @param args  the raw command line arguments
72       * @return a command return code
73       * @throws Exception
74       */
75      public final int exec(FramedGraph<?> graph, String[] args) throws Exception {
76          setCustomOptions(options);
77          return execWithOptions(graph, parser.parse(options, args));
78      }
79  
80      @Override
81      public String getDetailedHelp() {
82          HelpFormatter formatter = new HelpFormatter();
83          setCustomOptions(options);
84          StringWriter out = new StringWriter();
85          formatter.printHelp(
86                  new PrintWriter(out), 80, getUsage(), "\n" + getHelp() + "\n\n",
87                  options, 5, 0, "\n" + getHelpFooter());
88          return out.toString();
89      }
90  
91  
92      /**
93       * Execute this command with the given database and parsed command
94       * line arguments.
95       *
96       * @param graph   the graph database
97       * @param cmdLine the parsed command line object
98       * @return a command return code
99       * @throws Exception
100      */
101     public abstract int execWithOptions(FramedGraph<?> graph,
102             CommandLine cmdLine) throws Exception;
103 
104     /**
105      * Utility to get a parsed command line from some args. This is
106      * mainly helpful for testing.
107      *
108      * @param args A list of arg strings
109      * @return The parsed command line
110      */
111     public CommandLine getCmdLine(String[] args) throws ParseException {
112         setCustomOptions(options);
113         return parser.parse(options, args);
114     }
115 
116     protected Api api(FramedGraph<?> graph, Accessor accessor) {
117         return ApiFactory.withLogging(graph, accessor);
118     }
119 
120     /**
121      * Get an optional log message given a string that may or may not
122      * be empty.
123      *
124      * @param msg a possibly null or empty string
125      * @return an optional message string
126      */
127     protected Optional<String> getLogMessage(String msg) {
128         return (msg == null || msg.trim().isEmpty())
129                 ? Optional.<String>absent()
130                 : Optional.of(msg);
131     }
132 }
