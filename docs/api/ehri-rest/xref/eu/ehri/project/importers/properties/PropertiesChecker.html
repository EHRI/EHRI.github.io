20  package eu.ehri.project.importers.properties;

22  import eu.ehri.project.importers.ead.EadImporter;
23  import eu.ehri.project.models.EntityClass;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import java.util.Set;
28  
29  
30  public class PropertiesChecker {
31  
32      private static final Logger logger = LoggerFactory.getLogger(PropertiesChecker.class);
33      private final NodeProperties allowed;
34  
35      public PropertiesChecker(NodeProperties allowedproperties) {
36          allowed = allowedproperties;
37      }
38  
39      /*
40       * checks whether these actualproperties adhere to the allowedproperties
41       * the actualproperties can contain less than the allowedproperties, but must at least contain the required ones 
42       * and it will log the superfluous ones, so 
43       * @param actualproperties
44       * @return returns true if it contains the required ones and no new ones
45       */
46      public boolean check(ImportProperties actualproperties, EntityClass nodeEntity) {
47          String node = nodeEntity.getName();
48          boolean testresult = true;
49          if (allowed.getHandlerProperties(node) == null) {
50              logger.error("no properties allowed for " + node);
51              return false;
52          }
53          if (allowed.getRequiredProperties(node) != null) {
54              for (String required : allowed.getRequiredProperties(node)) {
55                  if (!actualproperties.containsPropertyValue(required)) {
56                      logger.error(node + " should contain required " + required);
57                      testresult = false;
58                  }
59              }
60          }
61          Set<String> handler = allowed.getHandlerProperties(node);
62          for (String property : actualproperties.getAllNonAttributeValues()) {
63              if(property.isEmpty() || property.equals("")){
64                  logger.error("property file contains empty right-hand side");
65                  testresult=false;
66              }
67              if (!handler.contains(property)) {
68                  if(nodeEntity.equals(EntityClass.DOCUMENTARY_UNIT_DESCRIPTION) &&
69                      property.endsWith(EadImporter.ACCESS_POINT)){
70                      logger.debug(property.substring(0, property.indexOf(EadImporter.ACCESS_POINT)+6));
71                   
72                  }else{
73                      logger.warn(node + " should NOT contain " + property);
74                  }
75  //                testresult = false;
76              }
77          }
78          return testresult;
79      }
80  }
