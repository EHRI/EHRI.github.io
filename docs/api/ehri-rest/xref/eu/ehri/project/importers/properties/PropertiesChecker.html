17      public PropertiesChecker(NodeProperties allowedproperties) {
18          allowed = allowedproperties;

21      /*
22       * checks whether these actualproperties adhere to the allowedproperties
23       * the actualproperties can contain less than the allowedproperties, but must at least contain the required ones 
24       * and it will log the superfluous ones, so 
25       * @param actualproperties
26       * @return returns true if it contains the required ones and no new ones
27       */
28      public boolean check(ImportProperties actualproperties, EntityClass nodeEntity) {
29          String node = nodeEntity.getName();
30          boolean testresult = true;
31          if (allowed.getHandlerProperties(node) == null) {
32              logger.error("no properties allowed for " + node);
33              return false;
34          }
35          if (allowed.getRequiredProperties(node) != null) {
36              for (String required : allowed.getRequiredProperties(node)) {
37                  if (!actualproperties.containsPropertyValue(required)) {
38                      logger.error(node + " should contain required " + required);
39                      testresult = false;
40                  }
41              }
42          }
43          Set<String> handler = allowed.getHandlerProperties(node);
44          for (String property : actualproperties.getAllNonAttributeValues()) {
45              if(property.isEmpty() || property.equals("")){
46                  logger.error("property file contains empty right-hand side");
47                  testresult=false;
48              }
49              if (!handler.contains(property)) {
50                  if(nodeEntity.equals(EntityClass.DOCUMENT_DESCRIPTION) &&
51                      property.endsWith(EadImporter.ACCESS_POINT)){
52                      logger.debug(property.substring(0, property.indexOf(EadImporter.ACCESS_POINT)+6));
53                   
54                  }else{
55                      logger.warn(node + " should NOT contain " + property);
56                  }
57  //                testresult = false;
58              }
59          }
60          return testresult;
61      }
