48  /**
49   * Importer of historical agents described in WP2.
50   * before importing the file: delete the columns with the reordering of the first and last name
51   * add a column 'id' with a unique identifier, prefixed with EHRI-Personalities or some such.
52   */
53  public abstract class Wp2PersonalitiesImporter extends MapImporter {
54  
55      private final XmlImportProperties p = new XmlImportProperties("wp2personalities.properties");
56  
57      private static final Logger logger = LoggerFactory.getLogger(Wp2PersonalitiesImporter.class);
58  
59      public Wp2PersonalitiesImporter(FramedGraph<?> framedGraph, Actioner actioner, PermissionScope permissionScope,
60              ImportLog log) {
61          super(framedGraph, permissionScope, actioner, log);
62      }
63  
64      @Override
65      public Accessible importItem(Map<String, Object> itemData) throws ValidationError {
66          Bundle descBundle = new Bundle(EntityClass.HISTORICAL_AGENT_DESCRIPTION,
67                  extractUnitDescription(itemData, EntityClass.HISTORICAL_AGENT_DESCRIPTION));
68          for (Map<String, Object> dpb : extractDates(itemData)) {
69              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
70          }
71          Bundle unit = new Bundle(EntityClass.HISTORICAL_AGENT, extractUnit(itemData))
72                  .withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
73  
74          BundleManager persister = getPersister();
75          Mutation<HistoricalAgent> mutation = persister.createOrUpdate(unit, HistoricalAgent.class);
76          HistoricalAgent frame = mutation.getNode();
77  
78          if (!permissionScope.equals(SystemScope.getInstance()) && mutation.created()) {
79              permissionScope.as(AuthoritativeSet.class).addItem(frame);
80              frame.setPermissionScope(permissionScope);
81          }
82  
83          handleCallbacks(mutation);
84          return frame;
85      }
86  
87      public Accessible importItem(Map<String, Object> itemData, int depth) throws ValidationError {
88          throw new UnsupportedOperationException("Not supported ever.");
89      }
90  
91      private Map<String, Object> extractUnit(Map<String, Object> itemData) {
92          //unit needs at least IDENTIFIER_KEY
93          Map<String, Object> item = Maps.newHashMap();
94          if (itemData.containsKey("id")) {
95              item.put(Ontology.IDENTIFIER_KEY, itemData.get("id"));
96          } else {
97              logger.error("missing objectIdentifier");
98          }
99          return item;
100     }
101 
102     private Map<String, Object> extractUnitDescription(Map<String, Object> itemData, EntityClass entityClass) {
103         Map<String, Object> item = Maps.newHashMap();
104 
105         for (String key : itemData.keySet()) {
106             if (!key.equals("id")) {
107                 if (!p.containsProperty(key)) {
108                     Helpers.putPropertyInGraph(item, SaxXmlHandler.UNKNOWN + key, itemData.get(key).toString());
109                 } else {
110                     Helpers.putPropertyInGraph(item, p.getProperty(key), itemData.get(key).toString());
111                 }
112             }
113 
114         }
115         //create all otherFormsOfName
116         if (!item.containsKey("typeOfEntity")) {
117             Helpers.putPropertyInGraph(item, "typeOfEntity", "person");
118         }
119         if (!item.containsKey(Ontology.LANGUAGE_OF_DESCRIPTION)) {
120             Helpers.putPropertyInGraph(item, Ontology.LANGUAGE_OF_DESCRIPTION, "en");
121         }
122         return item;
123     }
124 
125     /**
126      * @param itemData the item data map
127      * @return returns a List with Maps with DatePeriod.DATE_PERIOD_START_DATE and DatePeriod.DATE_PERIOD_END_DATE values
128      */
129     @Override
130     public List<Map<String, Object>> extractDates(Map<String, Object> itemData) {
131 
132         List<Map<String, Object>> l = Lists.newArrayList();
133         Map<String, Object> items = Maps.newHashMap();
134 
135         String start = (String) itemData.get("dateOfBirth");
136 
137         if (start != null && start.endsWith("00-00")) {
138             start = start.substring(0, 4);
139         }
140         if (start != null) {
141             items.put(Ontology.DATE_PERIOD_START_DATE, start);
142             l.add(items);
143         }
144         return l;
145     }
