20  /**
21   * Import EAD describing a Virtual Collection. some rules governing virtual collections:
22   *
23   * the archdesc should describe the purpose of this vc. it can not in itself refer to a DU.
24   *
25   * every c level is either 1) a virtual level (=VirtualLevel), or 2) it points to an existing DocumentaryUnit
26   * (=VirtualReferrer) (and consequently to the entire subtree beneath it) 1) there is no repository-tag with a
27   * ehri-label
28   *
29   * 2) there is exactly one repository-tag with an ehri-label <repository
30   * label="ehri_repository_vc">il-002777</repository> (this will not be shown in the portal) and exactly one unitid with
31   * a ehri-main-identifier label, that is identical to the existing unitid within the graph for this repository
32   *
33   * all other tags will be ignored, since the DocumentsDescription of the referred DocumentaryUnit will be shown. there
34   * should not be any c-levels beneath such a c-level
35   *
36   *
37   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
38   *
39   */

71              throws ValidationError {
72  
73          BundleDAO persister = getPersister(idPath);
74  
75          boolean isVirtualLevel = isVirtualLevel(itemData);
76          Bundle unit = new Bundle(unitEntity, extractVirtualUnit(itemData));
77  
78          if (isVirtualLevel) { //VirtualLevel
79              // Check for missing identifier, throw an exception when there is no ID.
80              if (unit.getDataValue(Ontology.IDENTIFIER_KEY) == null) {
81                  throw new ValidationError(unit, Ontology.IDENTIFIER_KEY,
82                          "Missing identifier " + Ontology.IDENTIFIER_KEY);
83              }
84              logger.debug("Imported item: " + itemData.get("name"));
85              Bundle descBundle = new Bundle(EntityClass.DOCUMENT_DESCRIPTION, extractUnitDescription(itemData, EntityClass.DOCUMENT_DESCRIPTION));
86              // Add dates and descriptions to the bundle since they're @Dependent
87              // relations.
88              for (Map<String, Object> dpb : extractDates(itemData)) {
89                  descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
90              }
91              for (Map<String, Object> rel : extractRelations(itemData)) {//, (String) unit.getErrors().get(IdentifiableEntity.IDENTIFIER_KEY)
92                  logger.debug("relation found: " + rel.get(Ontology.NAME_KEY));
93                  descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT, new Bundle(EntityClass.UNDETERMINED_RELATIONSHIP, rel));
94              }
95              Map<String, Object> unknowns = extractUnknownProperties(itemData);
96              if (!unknowns.isEmpty()) {
97                  logger.debug("Unknown Properties found");
98                  descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, new Bundle(EntityClass.UNKNOWN_PROPERTY, unknowns));
99              }
100             unit = unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
101             Mutation<VirtualUnit> mutation =
102                     persister.createOrUpdate(unit, VirtualUnit.class);
103             VirtualUnit frame = mutation.getNode();
104             // Set the repository/item relationship
105             //TODO: figure out another way to determine we're at the root, so we can get rid of the depth param
106             if (idPath.isEmpty() && mutation.created()) {
107                 EntityClass scopeType = manager.getEntityClass(permissionScope);
108                 if (scopeType.equals(EntityClass.USER_PROFILE)) {
109                     UserProfile responsibleUser = framedGraph.frame(permissionScope.asVertex(), UserProfile.class);
110                     frame.setAuthor(responsibleUser);
111                     //the top Virtual Unit does not have a permissionScope. 
112                 } else if (scopeType.equals(unitEntity)) {
113                     VirtualUnit parent = framedGraph.frame(permissionScope.asVertex(), VirtualUnit.class);
114                     parent.addChild(frame);
115                     frame.setPermissionScope(parent);
116                 } else if (unitEntity.equals(EntityClass.VIRTUAL_UNIT)) {
117                     // no scope needed for top VirtualUnit
118                 } else {
119                     logger.error("Unknown scope type for virtual unit: {}", scopeType);
120                 }
121             }
122             handleCallbacks(mutation);
123 //        if (mutation.created()) {

125 //        }
126             return frame;
127         } //find the referred Documentary Unit and RETURN it
128         //if (!isVirtualLevel) {//VirtualReferrer

132                 DocumentaryUnit d = findReferredToDocumentaryUnit(itemData);

146     @SuppressWarnings("unchecked")
147     @Override
148     protected Iterable<Map<String, Object>> extractRelations(Map<String, Object> data) {
149         final String REL = "AccessPoint";
150         List<Map<String, Object>> list = new ArrayList<Map<String, Object>>();
151         for (String key : data.keySet()) {
152             if (key.endsWith(REL)) {
153                 logger.debug(key + " found in data");
154                 //type, targetUrl, targetName, notes
155                 for (Map<String, Object> origRelation : (List<Map<String, Object>>) data.get(key)) {
156                     Map<String, Object> relationNode = new HashMap<String, Object>();
157                     for (String eventkey : origRelation.keySet()) {
158                         logger.debug(eventkey);
159                         if (eventkey.endsWith(REL)) {
160                             relationNode.put(Ontology.UNDETERMINED_RELATIONSHIP_TYPE, eventkey);
161                             relationNode.put(Ontology.NAME_KEY, origRelation.get(eventkey));
162                         } else {
163                             relationNode.put(eventkey, origRelation.get(eventkey));
164                         }
165                     }
166                     if (!relationNode.containsKey(Ontology.UNDETERMINED_RELATIONSHIP_TYPE)) {
167                         relationNode.put(Ontology.UNDETERMINED_RELATIONSHIP_TYPE, "corporateBodyAccessPoint");
168                     }
169                     list.add(relationNode);
170                 }
171             }
172         }
173         return list;
174     }

176     /**
177      * Creates a Map containing properties of a Virtual Unit.
178      *
179      * These properties are the unit's identifiers.
180      *
181      * @param itemData Map of all extracted information
182      * @return a Map representing a Documentary Unit node
183      * @throws ValidationError
184      */
185     protected Map<String, Object> extractVirtualUnit(Map<String, Object> itemData) throws ValidationError {
186 
187         Map<String, Object> unit = new HashMap<String, Object>();
188         if (itemData.get(OBJECT_ID) != null) {
189             unit.put(Ontology.IDENTIFIER_KEY, itemData.get(OBJECT_ID));
190         }
191         if (itemData.get(Ontology.OTHER_IDENTIFIERS) != null) {
192             logger.debug("otherIdentifiers is not null");
193             unit.put(Ontology.OTHER_IDENTIFIERS, itemData.get(Ontology.OTHER_IDENTIFIERS));
194         }
195         return unit;
196     }

204      * Creates a Map containing properties of a Documentary Unit description. These properties are the unit

212     protected Map<String, Object> extractDocumentDescription(Map<String, Object> itemData, int depth) throws ValidationError {
213 
214         Map<String, Object> unit = new HashMap<String, Object>();
215         for (String key : itemData.keySet()) {
216             if (!(key.equals(OBJECT_ID)
217                     || key.equals(Ontology.OTHER_IDENTIFIERS)
218                     || key.startsWith(SaxXmlHandler.UNKNOWN))) {
219                 unit.put(key, itemData.get(key));
220             }
221         }
222         return unit;

241         return !(itemData.containsKey(REPOID) && itemData.containsKey(UNITID));

244     private DocumentaryUnit findReferredToDocumentaryUnit(Map<String, Object> itemData) throws ItemNotFound {
245         if (itemData.containsKey(REPOID) && itemData.containsKey(UNITID)) {
246             String repositoryid = itemData.get(REPOID).toString();
247             String unitid = itemData.get(UNITID).toString();
248             Repository repository = manager.getFrame(repositoryid, Repository.class);
249             for (DocumentaryUnit unit : repository.getAllCollections()) {
250                 logger.debug(unit.getIdentifier() + " " + unit.getId() + " " + unitid);
251                 if (unit.getIdentifier().equals(unitid)) {
252                     return unit;
253                 }
254             }
255             throw new ItemNotFound(String.format("No item %s found in repo %s", unitid, repositoryid));
256         }
257         throw new ItemNotFound(String.format("Apparently no repositoryid/unitid combo given"));
258 
259     }
