215     /**
216      * Import an input stream.
217      *
218      * @param ios        The SKOS file input stream
219      * @param logMessage A log message
220      * @return A log of imported nodes
221      */
222     @Override
223     public ImportLog importFile(InputStream ios, String logMessage) throws IOException, ValidationError {
224 
225         // Create a new action for this import
226         Optional<String> logMsg = getLogMessage(logMessage);
227         ActionManager.EventContext eventContext = new ActionManager(framedGraph, vocabulary).newEventContext(
228                 actioner, EventTypes.ingest, logMsg);
229         // Create a manifest to store the results of the import.
230         ImportLog log = new ImportLog(logMsg.orElse(null));
231 
232         // NB: We rely in inference here, despite a sketchy understanding of how it works ;)
233         // The RDFS.subPropertyOf lets us fetch SKOS-XL pref/alt/hidden labels using
234         // the same property name as the plain SKOS variants.
235         OntModel model = ModelFactory.createOntologyModel(OntModelSpec.OWL_MEM_RULE_INF);
236         try {
237             model.add(SKOSXL.prefLabel, RDFS.subPropertyOf, SKOS.prefLabel);
238             model.add(SKOSXL.altLabel, RDFS.subPropertyOf, SKOS.altLabel);
239             model.add(SKOSXL.hiddenLabel, RDFS.subPropertyOf, SKOS.hiddenLabel);
240             // Since we use the same graph label for narrower/broader, enabling inverse
241             // inference on the model means we get consistent behaviour when syncing
242             // these relationships, regardless of how they are defined in the RDF...
243             model.add(SKOS.broader, OWL.inverseOf, SKOS.narrower);
244 
245             model.read(ios, null, format);
246             OntClass conceptClass = model.getOntClass(SkosRDFVocabulary.CONCEPT.getURI().toString());
247             logger.debug("in import file: {}", SkosRDFVocabulary.CONCEPT.getURI());
248             Map<Resource, Concept> imported = Maps.newHashMap();
249 
250             ExtendedIterator<? extends OntResource> itemIterator = conceptClass.listInstances();
251             try {
252                 while (itemIterator.hasNext()) {
253                     Resource item = itemIterator.next();
254 
255                     try {
256                         Mutation<Concept> graphConcept = importConcept(item);
257                         imported.put(item, graphConcept.getNode());
258 
259                         switch (graphConcept.getState()) {
260                             case UNCHANGED:
261                                 log.addUnchanged();
262                                 break;
263                             case CREATED:
264                                 log.addCreated();
265                                 eventContext.addSubjects(graphConcept.getNode());
266                                 break;
267                             case UPDATED:
268                                 log.addUpdated();
269                                 eventContext.addSubjects(graphConcept.getNode());
270                                 break;
271                         }
272                     } catch (ValidationError validationError) {
273                         if (tolerant) {
274                             logger.error(validationError.getMessage());
275                             log.addError(item.toString(), validationError.getMessage());
276                         } else {
277                             throw validationError;
278                         }
279                     }
280                 }
281             } finally {
282                 itemIterator.close();
283             }
284 
285             for (Map.Entry<Resource, Concept> pair : imported.entrySet()) {
286                 hookupRelationships(pair.getKey(), pair.getValue(), imported);
287             }
288 
289             for (Concept concept : imported.values()) {
290                 vocabulary.addItem(concept);
291                 concept.setPermissionScope(vocabulary);
292             }
293 
294             if (log.hasDoneWork()) {
295                 eventContext.commit();
296             }
297 
298             return log;
299         } finally {
300             model.close();
301         }
302     }

304     private Mutation<Concept> importConcept(Resource item) throws ValidationError {
305         logger.debug("Importing: {}", item);
306         Bundle.Builder builder = Bundle.Builder.withClass(EntityClass.CVOC_CONCEPT)
307                 .addDataValue(Ontology.IDENTIFIER_KEY, getId(URI.create(item.getURI())))
308                 .addDataValue(Ontology.URI_KEY, item.getURI());
309 
310         for (Map.Entry<String, URI> prop : SkosRDFVocabulary.GENERAL_PROPS.entrySet()) {
311             for (RDFNode target : getObjectWithPredicate(item, prop.getValue())) {
312                 if (target.isLiteral()) {
313                     if (prop.getKey().equals("latitude/longitude")) {
314                         String[] latLon = target.asLiteral().getString().split(",");
315                         if (latLon.length > 1) {
316                             builder.addDataValue("latitude", latLon[0]);
317                             builder.addDataValue("longitude", latLon[1]);
318                         }
319                     } else {
320                         builder.addDataMultiValue(prop.getKey(), target.asLiteral().getString());
321                     }
322                 } else {
323                     builder.addDataMultiValue(prop.getKey(), target.toString());
324                 }
325             }
326         }
327 
328         Map<AuthoritativeItem, String> linkedConcepts = Maps.newHashMap();
329 
330         List<Bundle> unknown = getAdditionalRelations(item, linkedConcepts);
331         for (Bundle description : getDescriptions(item)) {
332             Bundle withRels = description
333                     .withRelations(Ontology.HAS_UNKNOWN_PROPERTY, unknown);
334             builder.addRelation(Ontology.DESCRIPTION_FOR_ENTITY, withRels);
335         }
336 
337         Mutation<Concept> mut = dao.createOrUpdate(builder.build(), Concept.class);
338         solveUndeterminedRelationships(mut.getNode(), linkedConcepts);
339         return mut;
340     }

342     private void solveUndeterminedRelationships(Concept unit, Map<AuthoritativeItem, String> linkedConcepts) {
343         Api api = ApiFactory.noLogging(framedGraph, actioner.as(UserProfile.class));
344 
345         for (AuthoritativeItem concept : linkedConcepts.keySet()) {
346             try {
347                 String relType = linkedConcepts.get(concept);
348                 Bundle linkBundle = Bundle.of(EntityClass.LINK)
349                         .withDataValue(Ontology.LINK_HAS_TYPE, "associative")
350                         .withDataValue(relType.substring(0, relType.indexOf(":")), relType.substring(relType.indexOf(":") + 1))
351                         .withDataValue(Ontology.LINK_HAS_DESCRIPTION, ImportHelpers.RESOLVED_LINK_DESC);
352                 Link link = api.create(linkBundle, Link.class);
353                 unit.addLink(link);
354                 concept.addLink(link);
355             } catch (ValidationError | PermissionDenied | DeserializationError ex) {
356                 logger.error(ex.getMessage());
357             }
358         }
359     }

361     private List<Bundle> getAdditionalRelations(Resource item, Map<AuthoritativeItem, String> linkedItems) {
362         List<Bundle> unknown = Lists.newArrayList();
363 
364         for (Map.Entry<String, URI> rel : SkosRDFVocabulary.RELATION_PROPS.entrySet()) {
365             for (RDFNode annotation : getObjectWithPredicate(item, rel.getValue())) {
366                 if (annotation.isLiteral()) {
367                     unknown.add(Bundle.Builder.withClass(EntityClass.UNKNOWN_PROPERTY)
368                             .addDataValue(rel.getKey(), annotation.toString())
369                             .build());
370                 } else {
371                     if (rel.getKey().startsWith("skos:") || rel.getKey().startsWith("sem:")) {
372                         Optional<AuthoritativeItem> found = findRelatedConcept(annotation.toString());
373                         if (found.isPresent()) {
374                             linkedItems.put(found.get(), rel.getKey());
375                         } else {
376                             unknown.add(Bundle.Builder.withClass(EntityClass.UNKNOWN_PROPERTY)
377                                     .addDataValue(rel.getKey(), annotation.toString())
378                                     .build());
379                         }
380                     }
381                 }
382             }
383         }
384         return unknown;
385     }

387     private Optional<AuthoritativeItem> findRelatedConcept(String name) {
388         if (name != null) {
389             String[] domains = name.split("/");
390             if (domains.length > 2) {
391                 String cvocId = domains[domains.length - 2];
392                 String conceptId = domains[domains.length - 1];
393                 AuthoritativeSet referredSet;
394                 try {
395                     GraphManager manager = GraphManagerFactory.getInstance(framedGraph);
396                     referredSet = manager.getEntity(cvocId, AuthoritativeSet.class);
397                     for (AuthoritativeItem authItem : referredSet.getAuthoritativeItems()) {
398                         if (authItem.getIdentifier().equals(conceptId)) {
399                             return Optional.of(authItem);
400                         }
401                     }
402                 } catch (ItemNotFound ex) {
403                     logger.error("AuthoritativeSet with id {} not found: {}", cvocId, ex.getMessage());
404                 }
405             }
406         }
407         return Optional.empty();
408     }

410     private List<RDFNode> getObjectWithPredicate(Resource item, final URI propUri) {
411         // NB: this should be possible with simply item.listProperties(propUri)
412         // but for some reason that doesn't work... I can't grok why.
413         return item.listProperties().filterKeep(statement ->
414                 statement.getPredicate()
415                         .hasURI(propUri.toString()))
416                 .mapWith(Statement::getObject).toList();
417     }

419     private void connectRelation(Concept current, Resource item, Map<Resource, Concept> others,
420             URI propUri, Function<Concept, Iterable<Concept>> getter,
421             BiConsumer<Concept, Concept> addFunc, BiConsumer<Concept, Concept> dropFunc) {
422         Set<Concept> existingRelations = Sets.newHashSet(getter.apply(current));
423         Set<Concept> newRelations = getObjectWithPredicate(item, propUri)
424                 .stream()
425                 .filter(RDFNode::isResource)
426                 .map(n -> others.get(n.asResource()))
427                 .filter(Objects::nonNull)
428                 .collect(Collectors.toSet());
429         if (!existingRelations.equals(newRelations)) {
430             logger.debug("Updating relations for concept: {}: {} -> {} => {}",
431                     propUri, current.getId(), existingRelations, newRelations);
432 
433             Sets.difference(existingRelations, newRelations)
434                     .forEach(e -> dropFunc.accept(current, e));
435             Sets.difference(newRelations, existingRelations)
436                     .forEach(n -> addFunc.accept(current, n));
437         }
438     }

440     private void hookupRelationships(Resource item, Concept current, Map<Resource, Concept> conceptMap) {
441         connectRelation(current, item, conceptMap, SkosRDFVocabulary.BROADER.getURI(),
442                 Concept::getBroaderConcepts, Concept::addBroaderConcept, Concept::removeBroaderConcept);
443         connectRelation(current, item, conceptMap, SkosRDFVocabulary.NARROWER.getURI(),
444                 Concept::getNarrowerConcepts, Concept::addNarrowerConcept, Concept::removeNarrowerConcept);
445         connectRelation(current, item, conceptMap, SkosRDFVocabulary.RELATED.getURI(),
446                 Concept::getRelatedConcepts, Concept::addRelatedConcept, Concept::removeRelatedConcept);
447     }

449     private List<StringValue> getReifiedObjectValue(Resource item, URI propUri) {
450         List<StringValue> values = Lists.newArrayList();
451         for (RDFNode node : getObjectWithPredicate(item, propUri)) {
452             if (node.isLiteral()) {
453                 values.add(new StringValue(node.asLiteral().getString(), node.asLiteral().getLanguage()));
454             } else {
455                 Stream.of(SKOSXL.literalForm, RDF.value)
456                         .map(prop -> node.asResource().getProperty(prop))
457                         .filter(Objects::nonNull)
458                         .map(Statement::getObject)
459                         .filter(Objects::nonNull)
460                         .findFirst()
461                         .ifPresent(object -> values.add(new StringValue(
462                                 object.asLiteral().getString(), object.asLiteral().getLanguage())));
463             }
464         }
465         return values;
466     }

468     private List<Bundle> getDescriptions(Resource item) {
469         List<Bundle> descriptions = Lists.newArrayList();
470 
471         for (StringValue property : getReifiedObjectValue(item, SkosRDFVocabulary.PREF_LABEL.getURI())) {
472 
473             Bundle.Builder builder = Bundle.Builder.withClass(EntityClass.CVOC_CONCEPT_DESCRIPTION);
474 
475             String langCode2Letter = property.getLang();
476             String langCode3Letter = getLanguageCode(langCode2Letter, defaultLang);
477             Optional<String> descCode = getScriptCode(langCode2Letter);
478 
479             builder.addDataValue(Ontology.NAME_KEY, property.getValue())
480                     .addDataValue(Ontology.LANGUAGE, langCode3Letter);
481             descCode.ifPresent(code -> builder.addDataValue(Ontology.IDENTIFIER_KEY, code));
482 
483             for (Map.Entry<String, List<URI>> prop : SkosRDFVocabulary.LANGUAGE_PROPS.entrySet()) {
484                 List<String> values = Lists.newArrayList();
485                 for (URI uri : prop.getValue()) {
486                     for (StringValue target : getReifiedObjectValue(item, uri)) {
487                         String propLang2Letter = target.getLang();
488                         String propLanguageCode = getLanguageCode(propLang2Letter, defaultLang);
489                         Optional<String> propDescCode = getScriptCode(propLang2Letter);
490                         if (propLanguageCode.equals(langCode3Letter) && propDescCode.equals(descCode)) {
491                             values.add(target.getValue());
492                         }
493                     }
494                 }
495                 if (!values.isEmpty()) {
496                     // Sorting the related literal values in
497                     // natural order gives consistency between
498                     // import/export round-trips. The data is
499                     // otherwise unsorted.
500                     Collections.sort(values);
501                     builder.addDataValue(prop.getKey(), values);
502                 }
503             }
504             Bundle bundle = builder.build();
505             logger.trace(bundle.toJson());
506             descriptions.add(bundle);
507         }
508 
509         // Hack: if there's only one description it means we've only
510         // got one prefLabel with a language tag. In this case get
511         // a list of altLabels in all languages :(
512         if (descriptions.size() == 1) {
513             List<String> all = getReifiedObjectValue(item, SkosRDFVocabulary.ALT_LABEL.getURI())
514                     .stream()
515                     .map(StringValue::getValue)
516                     .distinct()
517                     .sorted()
518                     .collect(Collectors.toList());
519             if (!all.isEmpty()) {
520                 descriptions.set(0, descriptions.get(0).withDataValue(Skos.altLabel.toString(), all));
521             }
522         }
523 
524         return descriptions;
525     }

527     private static String getLanguageCode(String langCode2Letter, String defaultLang) {
528         if (langCode2Letter == null || langCode2Letter.trim().isEmpty()) {
529             return defaultLang;
530         }
531         List<String> parts = codeSplitter.splitToList(langCode2Letter);
532         if (parts.isEmpty()) {
533             return defaultLang;
534         } else {
535             return LanguageHelpers.iso639DashTwoCode(parts.get(0));
536         }
537     }

539     private static Optional<String> getScriptCode(String langCode) {
540         List<String> parts = codeSplitter.splitToList(langCode);
541         return parts.size() > 1 ? Optional.of(parts.get(1)) : Optional.empty();
542     }

544     private String getId(URI uri) {
545         if (baseURI != null && suffix != null && uri.toString().startsWith(baseURI)) {
546             String sub = uri.toString().substring(baseURI.length());
547             return sub.substring(0, sub.lastIndexOf(suffix));
548         } else if (baseURI != null && uri.toString().startsWith(baseURI)) {
549             return uri.toString().substring(baseURI.length());
550         } else if (uri.getFragment() != null) {
551             return uri.getFragment();
552         } else {
553             return uri.getPath().substring(uri.getPath().lastIndexOf("/") + 1)
554                     + (uri.getQuery() != null ? uri.getQuery() : "")
555                     + (uri.getFragment() != null ? uri.getFragment() : "");
556         }
557     }
