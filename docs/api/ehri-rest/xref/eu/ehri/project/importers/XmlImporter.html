147             if (eventkey.equals("maintenanceEvent/type")) {
148                 me.put(MaintenanceEvent.EVENTTYPE, event.get(eventkey));
149             } else if (eventkey.equals("maintenanceEvent/agentType")) {
150                 me.put(MaintenanceEvent.AGENTTYPE, event.get(eventkey));
151             } else {
152                 me.put(eventkey, event.get(eventkey));
153             }
154         }
155         if (!me.containsKey(MaintenanceEvent.EVENTTYPE)) {
156             me.put(MaintenanceEvent.EVENTTYPE, "unknown event type");
157         }
158         return me;

193     private Map<String, Object> matchDate(String date) {
194         Map<String, Object> data = new HashMap<String, Object>();
195         for (Pattern re : datePatterns) {
196             Matcher matcher = re.matcher(date);
197             if (matcher.matches()) {
198                 logger.debug("matched "+ date);
199                 data.put(Ontology.DATE_PERIOD_START_DATE, normaliseDate(matcher.group(1)));
200                 data.put(Ontology.DATE_PERIOD_END_DATE, normaliseDate(matcher.group(matcher
201                         .groupCount() > 1 ? 2 : 1), Ontology.DATE_PERIOD_END_DATE));
202                 break;
203             }
204         }
205         return data;
206     }

212     /**
213      * Normalise a date in a string.
214      * 
215      * @param date a String date that needs formatting
216      * @param beginOrEnd a string signifying whether this date is the begin of
217      * a period or the end of a period
218      * @return a String containing the formatted date.
219      */
220     public static String normaliseDate(String date, String beginOrEnd) {
221         DateTimeFormatter fmt = ISODateTimeFormat.date();
222         String returndate = fmt.print(DateTime.parse(date));
223         if (returndate.startsWith("00")) {
224 //            logger.debug("strange date: " + returndate);
225             returndate = "19" + returndate.substring(2);
226             date = "19" + date;
227 //            logger.debug("strange date: " + returndate);
228         }
229         if (Ontology.DATE_PERIOD_END_DATE.equals(beginOrEnd)) {
230             if (!date.equals(returndate)) {
231                 ParsePosition p = new ParsePosition(0);
232                 new SimpleDateFormat("yyyy-MM").parse(date, p);
233                 if (p.getIndex() > 0) {
234                     returndate = fmt.print(DateTime.parse(date).plusMonths(1).minusDays(1));
235                 } else {
236                     p = new ParsePosition(0);
237                     new SimpleDateFormat("yyyy").parse(date, p);
238                     if (p.getIndex() > 0) {
239                         returndate = fmt.print(DateTime.parse(date).plusYears(1).minusDays(1));
240                     }
241                 }
242             }
243         }
244         return returndate;
245     }
