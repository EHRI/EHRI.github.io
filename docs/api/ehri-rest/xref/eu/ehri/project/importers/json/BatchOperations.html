52       * @param tolerant whether to allow individual validation errors

55      public BatchOperations(FramedGraph<?> graph, PermissionScope scopeOpt, boolean version, boolean tolerant) {
56          this.graph = graph;

59          this.tolerant = tolerant;

108     /**
109      * Create or update a batch of items.
110      *
111      * @param inputStream an input stream containing a JSON list of
112      *                    bundles corresponding to the items to update
113      *                    or create.
114      * @param actioner    the current user
115      * @param logMessage  a log message
116      * @return an import log
117      */
118     public ImportLog batchImport(InputStream inputStream, Actioner actioner, Optional<String> logMessage)
119             throws DeserializationError, ItemNotFound, ValidationError {
120         ActionManager.EventContext ctx = actionManager.newEventContext(actioner,
121                 EventTypes.modification, logMessage);
122         ImportLog log = new ImportLog(logMessage);
123         try (CloseableIterable<Bundle> bundleIter = Bundle.bundleStream(inputStream)) {
124             for (Bundle bundle : bundleIter) {
125                 try {
126                     Mutation<Accessible> update = dao.createOrUpdate(bundle, Accessible.class);
127                     switch (update.getState()) {
128                         case UPDATED:
129                             log.addUpdated();
130                             ctx.addSubjects(update.getNode());
131                             if (version) {
132                                 ctx.createVersion(update.getNode(), update.getPrior().get());
133                             }
134                             break;
135                         case CREATED:
136                             log.addCreated();
137                             ctx.addSubjects(update.getNode());
138                         default:
139                             log.addUnchanged();
140                     }
141                 } catch (ValidationError e) {
142                     if (!tolerant) {
143                         throw e;
144                     } else {
145                         log.addError(bundle.getId(), e.getMessage());
146                         logger.warn("Validation error patching {}: {}", bundle.getId(), e);
147                     }
148                 }
149             }
150             if (log.hasDoneWork()) {
151                 ctx.commit();
152             }
153             return log;
154         } catch (RuntimeJsonMappingException e) {
155             throw new DeserializationError("Error reading JSON stream:", e);
156         }
157     }

159     /**
160      * Update a batch of items.
161      *
162      * @param inputStream an input stream containing a JSON list of
163      *                    bundles corresponding to the items to update.
164      * @param actioner    the current user
165      * @param logMessage  a log message
166      * @return an import log
167      */
168     public ImportLog batchUpdate(InputStream inputStream, Actioner actioner, Optional<String> logMessage)
169             throws DeserializationError, ItemNotFound, ValidationError {
170         ActionManager.EventContext ctx = actionManager.newEventContext(actioner,
171                 EventTypes.modification, logMessage);
172         ImportLog log = new ImportLog(logMessage);
173         try (CloseableIterable<Bundle> bundleIter = Bundle.bundleStream(inputStream)) {
174             for (Bundle bundle : bundleIter) {
175                 Entity entity = manager.getEntity(bundle.getId(), bundle.getType().getJavaClass());
176                 Bundle oldBundle = serializer.entityToBundle(entity);
177                 Bundle newBundle = oldBundle.mergeDataWith(bundle);
178                 try {
179                     Mutation<Accessible> update = dao.update(newBundle, Accessible.class);
180                     switch (update.getState()) {
181                         case UPDATED:
182                             log.addUpdated();
183                             ctx.addSubjects(update.getNode());
184                             if (version) {
185                                 ctx.createVersion(entity, oldBundle);
186                             }
187                             break;
188                         case UNCHANGED:
189                             log.addUnchanged();
190                             break;
191                         default:
192                             throw new RuntimeException("Unexpected status in batch update: " + update.getState());
193                     }
194                 } catch (ValidationError e) {
195                     if (!tolerant) {
196                         throw e;
197                     } else {
198                         log.addError(entity.getId(), e.getMessage());
199                         logger.warn("Validation error patching {}: {}", entity.getId(), e);
200                     }
201                 }
202             }
203             if (log.hasDoneWork()) {
204                 ctx.commit();
205             }
206             return log;
207         } catch (SerializationError serializationError) {
208             throw new RuntimeException(serializationError);
209         } catch (RuntimeJsonMappingException e) {
210             throw new DeserializationError("Error reading JSON stream:", e);
211         }
212     }

214     /**
215      * Delete a batch of items by ID.
216      *
217      * @param ids        a list of item IDs
218      * @param actioner   the current user
219      * @param logMessage an optional log message
220      * @return the number of items deleted
221      */
222     public int batchDelete(List<String> ids, Actioner actioner, Optional<String> logMessage)
223             throws ItemNotFound {
224         if (!ids.isEmpty()) {
225             int done = 0;
226             try {
227                 ActionManager.EventContext ctx = actionManager.newEventContext(actioner,
228                         EventTypes.deletion, logMessage);
229                 for (String id : ids) {
230                     Entity entity = manager.getEntity(id, Entity.class);
231                     ctx = ctx.addSubjects(entity.as(Accessible.class));
232                     if (version) {
233                         ctx = ctx.createVersion(entity);
234                     }
235                 }
236                 ctx.commit();
237                 for (String id : ids) {
238                     dao.delete(serializer.entityToBundle(manager.getEntity(id, Entity.class)));
239                     done++;
240                 }
241                 return done;
242             } catch (SerializationError serializationError) {
243                 throw new RuntimeException(serializationError);
244             }
245         } else {
246             return 0;
247         }
