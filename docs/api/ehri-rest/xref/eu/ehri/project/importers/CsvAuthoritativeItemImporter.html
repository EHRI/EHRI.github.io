49  /**
50   * Importer of authoritative items such as historical agents and concepts, loaded
51   * from a CSV file.
52   *
53   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)

55  public class CsvAuthoritativeItemImporter extends MapImporter {

59      private static final Logger logger = LoggerFactory.getLogger(CsvAuthoritativeItemImporter.class);
60  
61      public CsvAuthoritativeItemImporter(FramedGraph<?> framedGraph, PermissionScope permissionScope, ImportLog log) {
62          super(framedGraph, permissionScope, log);
63      }
64  
65      @Override
66      public AccessibleEntity importItem(Map<String, Object> itemData) throws ValidationError {
67  
68          BundleDAO persister = getPersister();
69  
70          Bundle unit = new Bundle(EntityClass.HISTORICAL_AGENT, extractUnit(itemData));
71  
72          Bundle descBundle = new Bundle(EntityClass.HISTORICAL_AGENT_DESCRIPTION, extractUnitDescription(itemData, EntityClass.HISTORICAL_AGENT_DESCRIPTION));
73  
74          unit = unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
75  
76          Mutation<AuthoritativeItem> mutation = persister.createOrUpdate(unit, AuthoritativeItem.class);
77          AuthoritativeItem frame = mutation.getNode();
78  
79          if (!permissionScope.equals(SystemScope.getInstance())
80                  && mutation.created()) {
81              manager.cast(permissionScope, AuthoritativeSet.class).addItem(frame);
82              frame.setPermissionScope(permissionScope);
83          }
84  
85          handleCallbacks(mutation);
86          return frame;
87  
88      }
89  
90      @Override
91      public AccessibleEntity importItem(Map<String, Object> itemData, List<String> idPath) throws
92              ValidationError {
93          throw new UnsupportedOperationException("Not supported ever.");
94      }

96      protected Map<String, Object> extractUnit(Map<String, Object> itemData) {
97          //unit needs at least IDENTIFIER_KEY
98          Map<String, Object> item = new HashMap<String, Object>();
99          if (itemData.containsKey("id")) {
100             item.put(Ontology.IDENTIFIER_KEY, itemData.get("id"));
101         } else {
102             logger.error("missing objectIdentifier");
103         }
104         return item;
105     }

109     protected Map<String, Object> extractUnitDescription(Map<String, Object> itemData, EntityClass entityClass) {
110         Map<String, Object> item = new HashMap<String, Object>();
111         item.put(Ontology.CREATION_PROCESS, Description.CreationProcess.IMPORT.toString());
112 
113 
114         SaxXmlHandler.putPropertyInGraph(item, Ontology.NAME_KEY, itemData.get("name").toString());
115         for (String key : itemData.keySet()) {
116             if (!key.equals("id") && !key.equals("name")) {
117                     SaxXmlHandler.putPropertyInGraph(item, key, itemData.get(key).toString());
118             }
119 
120         }
121         if (!item.containsKey("typeOfEntity")) {
122             SaxXmlHandler.putPropertyInGraph(item, "typeOfEntity", "subject");
123         }
124         if (!item.containsKey(Ontology.LANGUAGE_OF_DESCRIPTION)) {
125             SaxXmlHandler.putPropertyInGraph(item, Ontology.LANGUAGE_OF_DESCRIPTION, "en");
126         }
127         return item;
128     }

131     /**
132      * @param itemData
133      * @return returns a List with Maps with DatePeriod.DATE_PERIOD_START_DATE and DatePeriod.DATE_PERIOD_END_DATE values
134      */
135     @Override
136     public List<Map<String, Object>> extractDates(Map<String, Object> itemData) {
137 
138         List<Map<String, Object>> l = Lists.newArrayList();
139         Map<String, Object> items = Maps.newHashMap();
140 
141         String end = (String) itemData.get("DateofdeathYYYY-MM-DD");
142         String start = (String) itemData.get("DateofbirthYYYY-MM-DD");
143 
144         if (start != null && start.endsWith("00-00")) {
145             start = start.substring(0, 4);
146         }
147         if (end != null && end.endsWith("00-00")) {
148             end = end.substring(0, 4);
149         }
150         if (end != null || start != null) {
151             if (start != null)
152                 items.put(Ontology.DATE_PERIOD_START_DATE, start);
153             if (end != null)
154                 items.put(Ontology.DATE_PERIOD_END_DATE, end);
155             l.add(items);
156         }
157         return l;
158     }
