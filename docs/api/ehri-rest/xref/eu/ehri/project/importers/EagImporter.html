49      @Override
50      public Map<String, Object> extractUnit(Map<String, Object> itemData) throws ValidationError {
51          Map<String, Object> data = super.extractUnit(itemData);
52  
53          // MB: Hack hack hack - extract EHRI-specific 'priority' field out of the
54          // pattern "Priority: <digit>" in the maintenanceNotes field.
55          Object notes = itemData.get(MAINTENANCE_NOTES);
56          if (notes != null) {
57              if (notes instanceof ArrayList<?>) {
58                  for (Object n : (ArrayList<?>) notes) {
59                      if (n instanceof String) {
60                          Matcher m = priorityPattern.matcher((String) n);
61                          if (m.find()) {
62                              data.put(PRIORITY, Integer.parseInt(m.group(1)));
63                          }
64                      }
65                  }
66              }
67          }
68  
69          return data;
70      }

76      @Override
77      public Repository importItem(Map<String, Object> itemData) throws ValidationError {
78  
79          BundleDAO persister = new BundleDAO(framedGraph, permissionScope.idPath());
80  
81          Bundle unit = new Bundle(EntityClass.REPOSITORY, extractUnit(itemData));
82  
83          Map<String, Object> descmap = extractUnitDescription(itemData, EntityClass.REPOSITORY_DESCRIPTION);
84          descmap.put(Ontology.IDENTIFIER_KEY, descmap.get(Ontology.IDENTIFIER_KEY) + "#desc");
85          Bundle descBundle = new Bundle(EntityClass.REPOSITORY_DESCRIPTION, descmap);
86  
87          // Add dates and descriptions to the bundle since they're @Dependent
88          // relations.
89          for (Map<String, Object> dpb : extractDates(itemData)) {
90              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
91          }
92  
93          //add the address to the description bundle
94          Map<String, Object> address = extractAddress(itemData);
95          if (!address.isEmpty()) {
96              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_ADDRESS, new Bundle(EntityClass.ADDRESS, address));
97          }
98          Map<String, Object> unknowns = extractUnknownProperties(itemData);
99          if (!unknowns.isEmpty()) {
100             logger.debug("Unknown Properties found");
101             descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, new Bundle(EntityClass.UNKNOWN_PROPERTY, unknowns));
102         }
103         for (Map<String, Object> dpb : extractMaintenanceEvent(itemData)) {
104             logger.debug("maintenance event found");
105             //dates in maintenanceEvents are no DatePeriods, they are not something to search on
106             descBundle = descBundle.withRelation(Ontology.HAS_MAINTENANCE_EVENT, new Bundle(EntityClass.MAINTENANCE_EVENT, dpb));
107         }
108 
109         unit = unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
110 
111         Mutation<Repository> mutation = persister.createOrUpdate(unit, Repository.class);
112         handleCallbacks(mutation);
113 
114         if (mutation.created()) {
115             mutation.getNode().setCountry(framedGraph.frame(permissionScope.asVertex(), Country.class));
116             mutation.getNode().setPermissionScope(permissionScope);
117         }
118         return mutation.getNode();
119     }
