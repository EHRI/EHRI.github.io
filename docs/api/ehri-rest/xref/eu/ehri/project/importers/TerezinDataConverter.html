24      public static void readFile(InputStream stream) {
25          Scanner scanner = new Scanner(stream);
26          while (scanner.hasNextLine()) {
27              String line = scanner.nextLine();
28              String[] quotes = line.split("\"");
29              String[] values;
30              if (quotes.length == 1) {
31                  values = line.split(";");
32              } else {
33                  if (quotes[0].length() <= 1) {
34                      logger.error(quotes[0] + " has a problem: " + line);
35                      break;
36                  }
37                  values = new String[2];
38                  values[0] = quotes[0].substring(0, quotes[0].length() - 2);
39                  values[1] = quotes[1];
40              }
41              if (values.length == 2) {
42  //            logger.info(values[0] + " -> " + values[1]);
43                  String[] datevalues = values[1].split(";");
44                  for (String datevalue : datevalues) {
45                      List<Calendar> list = parseDate(datevalue);
46                      if (list == null) {
47                          //parse failed, print the date and the identifier
48                          logger.error(values[0] + " := " + datevalue);
49                      }
50                  }
51              } else {
52                  logger.error(values[0] + " has a problem: " + line);
53  
54              }
55  
56          }
57      }
