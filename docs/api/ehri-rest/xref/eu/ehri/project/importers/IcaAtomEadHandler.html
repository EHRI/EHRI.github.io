32  /**
33   * Handler of EADs exported from ICA-AtoM.
34   * Only titles have to be handled in a special way.
35   * makes use of icaatom.properties with format: part/of/path/=attribute
36   *
37   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
38   */
39  public class IcaAtomEadHandler extends EadHandler {
40  
41      private static final Logger logger = LoggerFactory
42              .getLogger(IcaAtomEadHandler.class);
43      private final List<DocumentaryUnit>[] children = new ArrayList[12];
44  
45      /**
46       * Set a custom resolver so EAD DTDs are never looked up online.
47       * @param publicId
48       * @param systemId
49       * @return returns essentially an empty dtd file
50       * @throws org.xml.sax.SAXException
51       * @throws java.io.IOException
52       */
53      public org.xml.sax.InputSource resolveEntity(String publicId, String systemId)
54              throws org.xml.sax.SAXException, java.io.IOException {
55          // This is the equivalent of returning a null dtd.
56          return new org.xml.sax.InputSource(new java.io.StringReader(""));
57      }
58      public IcaAtomEadHandler(AbstractImporter<Map<String, Object>> importer, XmlImportProperties properties) {
59          super(importer, properties);
60          children[depth] = new ArrayList<DocumentaryUnit>();
61      }
62      public IcaAtomEadHandler(AbstractImporter<Map<String, Object>> importer) {
63          this(importer, new XmlImportProperties("icaatom.properties"));
64      }
65  
66      @Override
67      protected void extractTitle(Map<String, Object> currentGraph) {
68      	if (!currentGraph.containsKey(Ontology.NAME_KEY)) {
69              //finding some name for this unit:
70              if (currentGraph.containsKey("title")) {
71                  currentGraph.put(Ontology.NAME_KEY, currentGraph.get("title"));
72              } else {
73                  logger.warn("DocumentaryUnit node without name field: ");
74                  currentGraph.put(Ontology.NAME_KEY, "UNKNOWN title");
75              }
76          }
77  
78      }

80  }
