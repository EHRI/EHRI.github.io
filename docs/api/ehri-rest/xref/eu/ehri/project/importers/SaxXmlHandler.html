41  /**
42   * Reader of XML files, creator of {@link Map}-based representations of EA* files.
43   * Makes use of properties file with format: 
44   * 
45   * path/within/xml/=node/property
46   *
47   * if no &lt;node&gt; is given, it is the default logical-unit or unit-description of this property file.
48   * with eac.properties this would be an HistoricalAgent with an HistoricalAgentDescription
49   * if there is a &lt;node&gt; given, it will translate to another graph node, like Address.
50   *
51   * lines starting with '{@literal @}' give the attributes:
52   * <code>{@literal @}attribute=tmpname
53   * path/within/xml/@tmpname=node/property</code>
54   *
55   * all tags not included in the properties file that have a  nodevalue will be put in a unknownproperties node, 
56   * with an edge to the unit-description.
57   *
58   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
59   */
60  public abstract class SaxXmlHandler extends DefaultHandler implements LexicalHandler {

134     /**
135      * Receive an opening tag. Initialise the current text to store the characters,
136      * create a language map to hold descriptions in different languages,
137      * push the level if this element warrants a new sub-node and store the attributes 
138      * that need to be stored.
139      */
140     @Override
141     public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
142         // initialise the text holding space
143     	currentText.push(new StringBuilder());
144 
145     	// retrieve the language from the attributes and
146     	// create a language map
147         Optional<String> lang = languageAttribute(attributes);
148         if (lang.isPresent()) {
149             languagePrefix = lang.get();
150             if (!languageMap.containsKey(languagePrefix)) {
151                 if (languageMap.isEmpty()) {
152                     currentGraphPath.peek().put(LANGUAGE_OF_DESCRIPTION, languageMap);
153                 }
154                 Map<String, Object> m = Maps.newHashMap();
155                 m.put(LANGUAGE_OF_DESCRIPTION, languagePrefix);
156                 languageMap.put(languagePrefix, m);
157             }
158         }
159 
160         // Update the path with the new element name
161         currentPath.push(withoutNamespace(qName));
162         if (needToCreateSubNode(qName)) { //a new subgraph should be created
163             depth++;
164             logger.debug("Pushing depth... " + depth + " -> " + qName);
165             currentGraphPath.push(Maps.<String, Object>newHashMap());
166         }
167 
168         // Store attributes that are listed in the .properties file
169         for (int attr = 0; attr < attributes.getLength(); attr++) { // only certain attributes get stored
170             String attributeName = withoutNamespace(attributes.getQName(attr));
171             if (properties.hasAttributeProperty(attributeName)
172                     && !properties.getAttributeProperty(attributeName).equals(LANGUAGE_OF_DESCRIPTION)) {
173               
174                 if (isKeyInPropertyFile(currentPath, "@" + properties.getAttributeProperty(attributeName), "")) {
175                     String path = getImportantPath(currentPath, "@" + properties.getAttributeProperty(attributeName), "");
176                     putPropertyInCurrentGraph(path, attributes.getValue(attr));
177                 } else if (isKeyInPropertyFile(currentPath, "@" + properties.getAttributeProperty(attributeName), "$" + attributes.getValue(attr))) {
178                     this.attribute = getImportantPath(currentPath, "@" + properties.getAttributeProperty(attributeName), "$" + attributes.getValue(attr));
179                 } else{
180                     logger.debug("attribute " + attributeName + " not found in properties");
181                 }
182             }
183         }
184     }

186     /**
187      * Receive an end element. Put the contents of the text holding space in the current graph
188      * as a property using the .properties file as a mapping.
189      */
190     @Override
191     public void endElement(String uri, String localName, String qName) throws SAXException {
192         if (languagePrefix == null) {
193             if(attribute == null){
194                 putPropertyInCurrentGraph(getImportantPath(currentPath), currentText.pop().toString());
195             }else{
196                 putPropertyInCurrentGraph(attribute, currentText.pop().toString());
197                 attribute= null;
198             }
199         } else {
200             putPropertyInGraph(languageMap.get(languagePrefix), getImportantPath(currentPath), currentText.pop().toString());
201         }

208     /**
209      * Insert a graph representation in the current graph (the one on top of 
210      * the currentGraphPath stack) as a list item at the given key.
211      * The value stored at the key is always a list - if the key did not exist
212      * yet, a new list is created to which the sub-graph is added.
213      * 
214      * @param key name of the edge to connect the sub-graph to the current graph
215      * @param subgraph Map graph representation to insert into the current graph
216      */
217     @SuppressWarnings("unchecked")
218     protected void putSubGraphInCurrentGraph(String key, Map<String, Object> subgraph) {
219         Map<String, Object> c = currentGraphPath.peek();
220         if (c.containsKey(key)) {
221             ((List<Map<String, Object>>) c.get(key)).add(subgraph);
222         } else {
223             List<Map<String, Object>> l = Lists.newArrayList();
224             l.add(subgraph);
225             c.put(key, l);
226         }
227     }

229     /**
230      * Get the language from the XML attributes, if it is there.
231      * Else return 'absent'.
232      * @param attributes SAX-parsed XML attributes
233      * @return an Optional containing the language, or 'absent'
234      */
235     private Optional<String> languageAttribute(Attributes attributes) {
236         for (int attr = 0; attr < attributes.getLength(); attr++) { // only certain attributes get stored
237             String isLangAttribute = withoutNamespace(attributes.getQName(attr));
238             String prop = properties.getAttributeProperty(isLangAttribute);
239             if (prop != null && prop.equals(LANGUAGE_OF_DESCRIPTION)) {
240                 logger.debug("Language detected!");
241                 return Optional.of(attributes.getValue(attr));
242             }
243         }
244         return Optional.absent();
245     }

248      * Get the element name without namespace prefix.
249      * 
250      * @param qName an element QName that may have a namespace prefix
251      * @return the element name without namespace prefix
252      */
253     protected String withoutNamespace(String qName) {
254         String name = qName;
255         int colon = qName.indexOf(":");
256         if (colon > -1) {
257 //            assert(qName.substring(0, colon).equals(prefixStack.peek()));
258             name = qName.substring(colon + 1);
259         }
260         return name;
261     }

264      * Receives character data in SAX events, converts multiple space to a single space
265      * and puts the characters in the current node, unless the input contains only 
266      * whitespace.
267      */
268     @Override
269     public void characters(char ch[], int start, int length) throws SAXException {
270         // NB: 'Blank' (whitespace) only strings are significant here, because
271         // otherwise a sequence of character, line-break, and an entity will
272         // end up being concatenated with the line-break removed. We therefore
273         // preserve all line breaks and other whitespace here and normalize
274         // it when the text gets added to the graph.
275         currentText.peek().append(ch, start, length);
276     }

289     /**
290      * Stores this property value pair in the given graph node representation.
291      * If the value is effectively empty, nothing happens.
292      * If the property already exists, it is added to the value list.
293      *
294      * @param c a Map representation of a graph node
295      * @param property the key to store the value for
296      * @param value the value to store
297      */
298     protected static void putPropertyInGraph(Map<String, Object> c, String property, String value) {
299         if (value == null)
300             return;
301         String valuetrimmed = value.trim();
302         if (valuetrimmed.isEmpty()) {
303             return;
304         }
305 
306         // Language properties 
307 
308         if (property.startsWith(LANGUAGE_KEY_PREFIX)) {
309             valuetrimmed = Helpers.iso639DashTwoCode(valuetrimmed);
310         }
311 
312         valuetrimmed = StringUtils.normalizeSpace(valuetrimmed);
313 
314         logger.debug("putProp: " + property + " " + valuetrimmed);
315 
316         Object propertyList;
317         if (c.containsKey(property)) {
318             propertyList = c.get(property);
319             if (propertyList instanceof List) {
320                 ((List<Object>) propertyList).add(valuetrimmed);
321             } else {
322                 List<Object> o = Lists.newArrayList();
323                 o.add(c.get(property));
324                 o.add(valuetrimmed);
325                 c.put(property, o);
326             }
327         } else {
328             c.put(property, valuetrimmed);
329         }
330     }

361     /**
362      * did/unitid/@ehrilabel$ehri_main_identifier=objectIdentifier
363      * 
364      * @param path did/unitid/
365      * @param attribute @ehrilabel
366      * @param attributevalue $ehri_main_identifier
367      * 
368      * @return the corresponding value to this path from the properties file. The search is inside out, so if
369      * both eadheader/ and ead/eadheader/ are specified, it will return the value for the first.
370      *
371      * If this path has no corresponding value in the properties file, it will return the entire path name, with _
372      * replacing the /
373      */
374     private String getImportantPath(Stack<String> path, String attribute, String attributevalue) {
375         
376         String all = "";
377         for (int i = path.size(); i > 0; i--) {
378             all = path.get(i - 1) + "/" + all;
379             if (properties.getProperty(all+attribute+attributevalue) != null) {
380                 return properties.getProperty(all+attribute+attributevalue);
381             }
382         }
383         return UNKNOWN + all.replace("/", "_");
384     }
385     /**
386      * did/unitid/@ehrilabel$ehri_main_identifier=objectIdentifier
387      * 
388      * @param path did/unitid/
389      * @param attribute {@literal @}ehrilabel
390      * @param attributevalue $ehri_main_identifier
391      * 
392      * @return returns true if this path is a key in the properties file. 
393      *
394      * If this path has no corresponding value in the properties file, it will return false
395      */
396     private boolean isKeyInPropertyFile(Stack<String> path, String attribute, String attributevalue) {
397         
398         String all = "";
399         for (int i = path.size(); i > 0; i--) {
400             all = path.get(i - 1) + "/" + all;
401             if (properties.getProperty(all+attribute+attributevalue) != null) {
402                 return true;
403             }
404         }
405         return false;
406     }
