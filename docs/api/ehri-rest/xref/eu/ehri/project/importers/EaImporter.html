39   * Import Map based representations of EAD or EAC for a given repository into the database.

49       * Construct an EadImporter object.

59      /**
60       * Extract properties from the itemData Map that belong to a generic unit and
61       * returns them as a new Map. Calls extractDocumentaryUnit.
62       *
63       * @param itemData a Map representation of a unit
64       * @return a new Map containing those properties that are specific to a unit
65       * @throws ValidationError when extractDocumentaryUnit throws it
66       */
67      protected Map<String, Object> extractUnit(Map<String, Object> itemData) throws ValidationError {
68          Map<String, Object> unit = extractDocumentaryUnit(itemData);
69          unit.put("typeOfEntity", itemData.get("typeOfEntity"));
70          return unit;
71      }
72  
73      /**
74       * Extract DocumentaryUnit properties from the itemData and return them as a new Map.
75       * This implementation only extracts the objectIdentifier.
76       * <p/>
77       * This implementation does not throw ValidationErrors.
78       *
79       * @param itemData a Map containing raw properties of a DocumentaryUnit
80       * @return a new Map containing the objectIdentifier property
81       * @throws ValidationError never
82       */
83      protected Map<String, Object> extractDocumentaryUnit(Map<String, Object> itemData) throws ValidationError {
84          Map<String, Object> unit = Maps.newHashMap();
85          unit.put(Ontology.IDENTIFIER_KEY, itemData.get("objectIdentifier"));
86          if (itemData.get(Ontology.OTHER_IDENTIFIERS) != null) {
87              logger.debug("otherIdentifiers is not null");
88              unit.put(Ontology.OTHER_IDENTIFIERS, itemData.get(Ontology.OTHER_IDENTIFIERS));
89          }
90          return unit;
91      }

93      /**
94       * Extract properties from the itemData Map that are marked as unknown, and return them in a new Map.
95       *
96       * @param itemData a Map containing raw properties of a unit
97       * @return returns a Map with all keys from itemData that start with SaxXmlHandler.UNKNOWN
98       * @throws ValidationError never
99       */
100     protected Map<String, Object> extractUnknownProperties(Map<String, Object> itemData) throws ValidationError {
101         Map<String, Object> unknowns = Maps.newHashMap();
102         for (Entry<String, Object> key : itemData.entrySet()) {
103             if (key.getKey().startsWith(SaxXmlHandler.UNKNOWN)) {
104                 unknowns.put(key.getKey().substring(SaxXmlHandler.UNKNOWN.length()), key.getValue());
105             }
106         }
107         return unknowns;
108     }

110     /**
111      * Extract node representations for related nodes based on the 'relation' property in the supplied data Map.
112      *
113      * @param itemData a Map containing raw properties of a unit
114      * @return an Iterable of new Maps representing related nodes and their types
115      */
116     protected Iterable<Map<String, Object>> extractRelations(Map<String, Object> itemData) {
117         String relName = "relation";
118         List<Map<String, Object>> listOfRelations = Lists.newArrayList();
119         for (Entry<String, Object> itemProperty : itemData.entrySet()) {
120             if (itemProperty.getKey().equals(relName)) {
121                 //type, targetUrl, targetName, notes
122                 for (Map<String, Object> origRelation : (List<Map<String, Object>>) itemProperty.getValue()) {
123                     Map<String, Object> relationNode = Maps.newHashMap();
124                     for (Entry<String, Object> relationProperty : origRelation.entrySet()) {
125                         if (relationProperty.getKey().equals(relName + "/type")) {
126                             relationNode.put(Ontology.ACCESS_POINT_TYPE, relationProperty.getValue());
127                         } else if (relationProperty.getKey().equals(relName + "/url")) {
128                             //try to find the original identifier
129                             relationNode.put(LINK_TARGET, relationProperty.getValue());
130                         } else if (relationProperty.getKey().equals(relName + "/" + Ontology.NAME_KEY)) {
131                             //try to find the original identifier
132                             relationNode.put(Ontology.NAME_KEY, relationProperty.getValue());
133                         } else if (relationProperty.getKey().equals(relName + "/notes")) {
134                             relationNode.put(Ontology.LINK_HAS_DESCRIPTION, relationProperty.getValue());
135                         } else {
136                             relationNode.put(relationProperty.getKey(), relationProperty.getValue());
137                         }
138                     }
139                     // Set a default relationship type if no type was found in the relationship
140                     if (!relationNode.containsKey(Ontology.ACCESS_POINT_TYPE)) {
141                         relationNode.put(Ontology.ACCESS_POINT_TYPE, "corporateBodyAccess");
142                     }
143                     listOfRelations.add(relationNode);
144                 }
145             }
146         }
147         return listOfRelations;
148     }

150     /**
151      * Extract a Map containing the properties of a documentary unit's description.
152      * Excludes unknown properties, object identifier(s), maintenance events, relations,
153      * addresses and *Access relations.
154      *
155      * @param itemData a Map containing raw properties of a unit
156      * @param entity   an EntityClass to get the multi-valuedness of properties for
157      * @return a Map representation of a DocumentDescription
158      */
159     protected Map<String, Object> extractUnitDescription(Map<String, Object> itemData, EntityClass entity) {
160         Map<String, Object> description = Maps.newHashMap();
161 
162         description.put(Ontology.CREATION_PROCESS, Description.CreationProcess.IMPORT.toString());
163 
164         for (Entry<String, Object> itemProperty : itemData.entrySet()) {
165             if (itemProperty.getKey().equals("descriptionIdentifier")) {
166                 description.put(Ontology.IDENTIFIER_KEY, itemProperty.getValue());
167             } else if (itemProperty.getKey().equals("conditionsOfAccess")) {
168                 description.put(itemProperty.getKey(), flattenNonMultivaluedProperties(
169                         itemProperty.getKey(), itemProperty.getValue(), entity));
170             } else if (!itemProperty.getKey().startsWith(SaxXmlHandler.UNKNOWN)
171                     && !itemProperty.getKey().equals("objectIdentifier")
172                     && !itemProperty.getKey().equals(Ontology.IDENTIFIER_KEY)
173                     && !itemProperty.getKey().equals(Ontology.OTHER_IDENTIFIERS)
174                     && !itemProperty.getKey().startsWith("maintenanceEvent")
175                     && !itemProperty.getKey().startsWith("relation")
176                     && !itemProperty.getKey().startsWith("IGNORE")
177                     && !itemProperty.getKey().startsWith("address/")
178                     && !itemProperty.getKey().endsWith("Access")
179                     && !itemProperty.getKey().endsWith("AccessPoint")) {
180                 description.put(itemProperty.getKey(), flattenNonMultivaluedProperties(
181                         itemProperty.getKey(), itemProperty.getValue(), entity));
182             }
183         }
184 
185         return description;
186     }

188     /**
189      * Extract an address node representation from the itemData.
190      *
191      * @param itemData a Map containing raw properties of a unit
192      * @return returns a Map with all address/ keys (may be empty)
193      */
194     protected Map<String, Object> extractAddress(Map<String, Object> itemData) {
195         Map<String, Object> address = Maps.newHashMap();
196         for (Entry<String, Object> itemProperty : itemData.entrySet()) {
197             if (itemProperty.getKey().startsWith("address/")) {
198                 address.put(itemProperty.getKey().substring(8), itemProperty.getValue());
199             }
200         }
201         return address;
202     }
