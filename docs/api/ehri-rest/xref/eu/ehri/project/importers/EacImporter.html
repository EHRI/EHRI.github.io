80      /**
81       * @param itemData the item data map
82       * @throws ValidationError
83       */
84      @Override
85      public HistoricalAgent importItem(Map<String, Object> itemData) throws ValidationError {
86  
87          BundleDAO persister = new BundleDAO(framedGraph, permissionScope.idPath());
88          Bundle descBundle = new Bundle(EntityClass.HISTORICAL_AGENT_DESCRIPTION,
89                  extractUnitDescription(itemData, EntityClass.HISTORICAL_AGENT_DESCRIPTION));
90  
91          // Add dates and descriptions to the bundle since they're @Dependent
92          // relations.
93          for (Map<String, Object> dpb : extractDates(itemData)) {
94              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
95          }
96  
97          // add the address to the description bundle
98          Map<String, Object> address = extractAddress(itemData);
99          if (!address.isEmpty()) {
100             descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_ADDRESS, new Bundle(EntityClass.ADDRESS, address));
101         }
102 
103         Map<String, Object> unknowns = extractUnknownProperties(itemData);
104         if (!unknowns.isEmpty()) {
105             logger.debug("Unknown Properties found");
106             descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, new Bundle(EntityClass.UNKNOWN_PROPERTY, unknowns));
107         }
108 
109         for (Map<String, Object> dpb : extractMaintenanceEvent(itemData)) {
110             logger.debug("maintenance event found");
111             //dates in maintenanceEvents are no DatePeriods, they are not something to search on
112             descBundle = descBundle.withRelation(Ontology.HAS_MAINTENANCE_EVENT, new Bundle(EntityClass.MAINTENANCE_EVENT, dpb));
113         }
114 
115         for (Map<String, Object> rel : extractRelations(itemData)) {
116             if (rel.containsKey(REL_TYPE) && rel.get(REL_TYPE).equals("subjectAccess")) {
117                 logger.debug("relation found");
118                 descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT, new Bundle(EntityClass.ACCESS_POINT, rel));
119             }
120         }
121 
122         Bundle unit = new Bundle(EntityClass.HISTORICAL_AGENT, extractUnit(itemData))
123                 .withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
124 
125         Mutation<HistoricalAgent> mutation = persister.createOrUpdate(unit, HistoricalAgent.class);
126         HistoricalAgent frame = mutation.getNode();
127         solveUndeterminedRelationships(frame, descBundle);
128 
129         // There may or may not be a specific scope here...
130         if (!permissionScope.equals(SystemScope.getInstance())
131                 && mutation.created()) {
132             manager.cast(permissionScope, AuthoritativeSet.class).addItem(frame);
133             frame.setPermissionScope(permissionScope);
134         }
135 
136         handleCallbacks(mutation);
137         return frame;
138 
139     }

141     @Override
142     protected Iterable<Map<String, Object>> extractRelations(Map<String, Object> itemData) {
143         String relKey = "relation";
144         List<Map<String, Object>> list = Lists.newArrayList();
145         for (String key : itemData.keySet()) {
146             if (key.equals(relKey)) {
147                 //name identifier
148                 for (Map<String, Object> origRelation : (List<Map<String, Object>>) itemData.get(key)) {
149                     Map<String, Object> relationNode = Maps.newHashMap();
150                     for (String eventkey : origRelation.keySet()) {
151                         if (eventkey.equals(REL_TYPE)) {
152                             relationNode.put(Ontology.ACCESS_POINT_TYPE, origRelation.get(eventkey) + "Access");
153                         } else if (eventkey.equals(REL_NAME) && origRelation.get(REL_TYPE).equals("subject")) {
154                             Map<String, Object> m = (Map) ((List) origRelation.get(eventkey)).get(0);
155                             //try to find the original identifier
156                             relationNode.put(LINK_TARGET, m.get("concept"));
157                             //try to find the original name
158                             relationNode.put(Ontology.NAME_KEY, m.get(REL_NAME));
159                             relationNode.put("cvoc", m.get("cvoc"));
160                         } else {
161                             relationNode.put(eventkey, origRelation.get(eventkey));
162                         }
163                     }
164                     if (!relationNode.containsKey(Ontology.ACCESS_POINT_TYPE)) {
165                         // Corporate bodies are the default type
166                         relationNode.put(Ontology.ACCESS_POINT_TYPE, "corporateBodyAccess");
167                     }
168                     list.add(relationNode);
169                 }
170             }
171         }
172         return list;
173     }

175     @Override
176     protected Map<String, Object> extractUnitDescription(Map<String, Object> itemData, EntityClass entity) {
177         Map<String, Object> description = Maps.newHashMap();
178         description.put(Ontology.CREATION_PROCESS, Description.CreationProcess.IMPORT.toString());
179 
180         for (String key : itemData.keySet()) {
181             if (key.equals("descriptionIdentifier")) {
182                 description.put(Ontology.IDENTIFIER_KEY, itemData.get(key));
183                 //resolved in EacHandler
184             } else if (key.equals("book")) {
185                 extractBooks(itemData.get(key), description);
186             } else if (!key.startsWith(SaxXmlHandler.UNKNOWN)
187                     && !key.equals("objectIdentifier")
188                     && !key.equals(Ontology.OTHER_IDENTIFIERS)
189                     && !key.equals(Ontology.IDENTIFIER_KEY)
190                     && !key.startsWith("maintenanceEvent")
191                     && !key.startsWith("IGNORE")
192                     && !key.startsWith("relation")
193                     && !key.startsWith("address/")) {
194                 description.put(key, flattenNonMultivaluedProperties(key, itemData.get(key), entity));
195             }
196         }
197 
198         return description;
199     }

201     protected void solveUndeterminedRelationships(HistoricalAgent unit, Bundle descBundle) throws ValidationError {
202 
203         //Try to resolve the undetermined relationships
204         //we can only create the annotations after the DocumentaryUnit and its Description have been added to the graph,
205         //so they have id's. 
206         for (Description unitdesc : unit.getDescriptions()) {
207             // Put the set of relationships into a HashSet to remove duplicates.
208             for (AccessPoint rel : Sets.newHashSet(unitdesc.getAccessPoints())) {
209                 Vertex relationVertex = rel.asVertex();
210                 for (String key : relationVertex.getPropertyKeys()) {
211                     logger.debug("solving undetermindRels: {} {} ({})",
212                             key, relationVertex.getProperty(key), descBundle.getId());
213                 }
214                 /*
215                  * the wp2 undetermined relationship that can be resolved have a 'cvoc' and a 'concept' attribute.
216                  * they need to be found in the vocabularies that are in the graph
217                  */
218                 if (relationVertex.getPropertyKeys().contains("cvoc")) {
219                     String cvocId = relationVertex.getProperty("cvoc");
220                     String conceptId = relationVertex.getProperty(LINK_TARGET);
221                     logger.debug("{} -> {}", cvocId, conceptId);
222                     try {
223                         Vocabulary vocabulary = manager.getFrame(cvocId, Vocabulary.class);
224                         for (Concept concept : vocabulary.getConcepts()) {
225                             logger.debug("********************* {} {}", concept.getId(), concept.getIdentifier());
226                             if (concept.getIdentifier().equals(conceptId)) {
227                                 try {
228                                     String linkType = relationVertex.getProperty(REL_TYPE);
229                                     Bundle linkBundle = new Bundle(EntityClass.LINK)
230                                             .withDataValue(Ontology.LINK_HAS_TYPE, linkType)
231                                             .withDataValue(Ontology.LINK_HAS_DESCRIPTION, RESOLVED_LINK_DESC);
232                                     UserProfile user = manager.cast(log.getActioner(), UserProfile.class);
233                                     Link link = new CrudViews<>(framedGraph, Link.class).create(linkBundle, user);
234                                     unit.addLink(link);
235                                     concept.addLink(link);
236                                     link.addLinkBody(rel);
237                                 } catch (PermissionDenied ex) {
238                                     logger.error("Unexpected permission error on EAC relationship creation: {}", ex);
239                                     throw new RuntimeException(ex);
240                                 }
241                             }
242                         }
243                     } catch (ItemNotFound ex) {
244                         logger.error("Vocabulary with id {} not found: {}", cvocId, ex);
245                     }
246                 }
247             }
248         }
249     }

251     private void extractBooks(Object books, Map<String, Object> description) {
252         List<String> createdBy = Lists.newArrayList();
253         List<String> subjectOf = Lists.newArrayList();
254 
255         if (books instanceof List) {
256             for (Object book : (List) books) {
257                 if (book instanceof Map) {
258                     Map bookEntry = (Map) book;
259                     boolean created = false;
260                     String publication = "";
261                     for (Object entryKey : bookEntry.keySet()) {
262                         switch ((String) entryKey) {
263                             case "bookentry":
264                                 if (bookEntry.get(entryKey) instanceof List) {
265                                     for (Object entry : (List) bookEntry.get(entryKey)) {
266                                         if (entry instanceof Map
267                                                 && ((Map) entry).containsKey(REL_TYPE)
268                                                 && ((Map) entry).containsKey("bookentry")) {
269                                             String type = ((Map) entry).get(REL_TYPE).toString();
270                                             String value = ((Map) entry).get("bookentry").toString();
271                                             String join = type.equals("isbn") || type.equals("creator")
272                                                     ? " " + type + ":"
273                                                     : "";
274                                             publication = publication.concat(join + value);
275                                         }
276                                     }
277                                 }
278                                 break;
279                             case REL_TYPE:
280                                 created = (bookEntry.get(entryKey).equals("creatorOf"));
281                                 break;
282                             default:
283                                 publication = publication.concat(entryKey + ":" + bookEntry.get(entryKey));
284                                 break;
285                         }
286                     }
287                     if (created) {
288                         createdBy.add(publication);
289                     } else {
290                         subjectOf.add(publication);
291                     }
292                 }
293             }
294         }
295         if (!createdBy.isEmpty()) {
296             description.put("createdBy", createdBy);
297         }
298 
299         if (!subjectOf.isEmpty()) {
300             description.put("subjectOf", subjectOf);
301         }
302     }
