64      /**
65       * @param itemData
66       * @throws ValidationError
67       */
68      @Override
69      public HistoricalAgent importItem(Map<String, Object> itemData) throws ValidationError {
70  
71          BundleDAO persister = new BundleDAO(framedGraph, permissionScope.idPath());
72  
73          Bundle unit = new Bundle(EntityClass.HISTORICAL_AGENT, extractUnit(itemData));
74  
75          Bundle descBundle = new Bundle(EntityClass.HISTORICAL_AGENT_DESCRIPTION, extractUnitDescription(itemData, EntityClass.HISTORICAL_AGENT_DESCRIPTION));
76  
77          // Add dates and descriptions to the bundle since they're @Dependent
78          // relations.
79          for (Map<String, Object> dpb : extractDates(itemData)) {
80              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
81          }
82  
83          //add the address to the description bundle
84          Map<String, Object> address = extractAddress(itemData);
85          if (!address.isEmpty()) {
86              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_ADDRESS, new Bundle(EntityClass.ADDRESS, address));
87          }
88  
89          Map<String, Object> unknowns = extractUnknownProperties(itemData);
90          if (!unknowns.isEmpty()) {
91              logger.debug("Unknown Properties found");
92              descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, new Bundle(EntityClass.UNKNOWN_PROPERTY, unknowns));
93          }
94  
95          for (Map<String, Object> dpb : extractMaintenanceEvent(itemData)) {
96              logger.debug("maintenance event found");
97              //dates in maintenanceEvents are no DatePeriods, they are not something to search on
98              descBundle = descBundle.withRelation(Ontology.HAS_MAINTENANCE_EVENT, new Bundle(EntityClass.MAINTENANCE_EVENT, dpb));
99          }
100 
101         for (Map<String, Object> rel : extractRelations(itemData)) {
102             logger.debug("relation found");
103             descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT, new Bundle(EntityClass.UNDETERMINED_RELATIONSHIP, rel));
104         }
105 
106         unit = unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
107 
108         Mutation<HistoricalAgent> mutation = persister.createOrUpdate(unit, HistoricalAgent.class);
109         HistoricalAgent frame = mutation.getNode();
110 
111         if (mutation.created()) {
112             solveUndeterminedRelationships(frame, descBundle);
113         }
114 
115         // There may or may not be a specific scope here...
116         if (!permissionScope.equals(SystemScope.getInstance())
117                 && mutation.created()) {
118             manager.cast(permissionScope, AuthoritativeSet.class).addItem(frame);
119             frame.setPermissionScope(permissionScope);
120         }
121 
122         handleCallbacks(mutation);
123         return frame;
124 
125     }

127     @Override
128     protected Map<String, Object> extractUnitDescription(Map<String, Object> itemData, EntityClass entity) {
129         Map<String, Object> description = new HashMap<String, Object>();
130         description.put(Ontology.CREATION_PROCESS, Description.CreationProcess.IMPORT.toString());
131 
132         for (String key : itemData.keySet()) {
133             if (key.equals("descriptionIdentifier")) {
134                 description.put(Ontology.IDENTIFIER_KEY, itemData.get(key));
135             } else if (key.startsWith("name")) {
136                 Object name = itemData.get(key);
137                 if (name instanceof List) {
138                     for (Object nameentry : (List) name) {
139                         if (nameentry instanceof Map) {
140                             String nameType = (String) ((Map<String, Object>) nameentry).get("name/nameType");
141                             String namePart = (String) ((Map<String, Object>) nameentry).get("name/namePart");
142                             if (namePart != null && nameType != null) {
143                                 if (nameType.equals("authorized"))
144                                     description.put(Ontology.NAME_KEY, namePart);
145                                 else if (nameType.equals("parallel"))
146                                     description.put("parallelFormsOfName", namePart);
147                                 else
148                                     description.put("otherFormsOfName", namePart);
149                             }
150                         }
151                     }
152                 }
153             } else if (!key.startsWith(SaxXmlHandler.UNKNOWN)
154                     && !key.equals("objectIdentifier")
155                     && !key.equals(Ontology.IDENTIFIER_KEY)
156                     && !key.startsWith("maintenanceEvent")
157                     && !key.startsWith("relation")
158                     && !key.startsWith("address/")) {
159                 description.put(key, changeForbiddenMultivaluedProperties(key, itemData.get(key), entity));
160             }
161 
162         }
163 //        assert(description.containsKey(IdentifiableEntity.IDENTIFIER_KEY));
164         return description;
165     }

167     /**
168      * Tries to resolve the undetermined relationships for IcaAtoM eac files by iterating through all UndeterminedRelationships,
169      * finding the DescribedEntity meant by the 'targetUrl' in the Relationship and creating an Annotation for it.
170      *
171      *
172      * @param frame
173      * @param descBundle
174      * @throws ValidationError
175      */
176     private void solveUndeterminedRelationships(HistoricalAgent frame, Bundle descBundle)
177             throws ValidationError {
178         //Try to resolve the undetermined relationships
179         //we can only create the annotations after the HistoricalAgent and it Description have been added to the graph,
180         //so they have id's. 
181         Description histdesc = null;
182         //we need the id (not the identifier) of the description, this requires some checking
183         for (Description thisAgentDescription : frame.getDescriptions()) {
184             //is thisAgentDescription the one we just created?
185             if (thisAgentDescription.asVertex().getProperty(Ontology.IDENTIFIER_KEY).equals(descBundle.getData().get(Ontology.IDENTIFIER_KEY))) {
186                 histdesc = thisAgentDescription;
187                 break;
188             }
189         }
190         if (histdesc == null) {
191             logger.warn("newly created description not found");
192         } else {
193             // Put the set of relationships into a HashSet to remove duplicates.
194             for (UndeterminedRelationship rel : Sets.newHashSet(histdesc.getUndeterminedRelationships())) {
195                 //our own ica-atom generated eac files have as target of a relation the url of the ica-atom
196                 //this must be matched back to descriptionUrl property in a previously created HistoricalAgentDescription
197                 String targetUrl = (String)rel.asVertex().getProperty(LINK_TARGET);
198                 Iterable<Vertex> docs = framedGraph.getVertices("descriptionUrl", targetUrl);
199                 if (docs.iterator().hasNext()) {
200                     String annotationType = rel.asVertex().getProperty(Ontology.LINK_HAS_TYPE).toString();
201                     DescribedEntity targetEntity = framedGraph.frame(docs.iterator().next(), Description.class).getEntity();
202                     try {
203                         Bundle linkBundle = new Bundle(EntityClass.LINK)
204                                 .withDataValue(Ontology.LINK_HAS_TYPE, annotationType)
205                                 .withDataValue(Ontology.LINK_HAS_DESCRIPTION, rel.asVertex().getProperty(Ontology.LINK_HAS_DESCRIPTION));
206                         Link link = new CrudViews<Link>(framedGraph, Link.class).create(linkBundle, userProfile);
207                         frame.addLink(link);
208                         targetEntity.addLink(link);
209                         link.addLinkBody(rel);
210 
211                         //attach the mirror Undetermined Relationship as a body to this Annotation
212                         String thisUrl = descBundle.getData().get("descriptionUrl").toString();
213                         for (Description targetEntityDescription : targetEntity.getDescriptions()) {
214                             for (UndeterminedRelationship remoteRel : Sets.newHashSet(targetEntityDescription.getUndeterminedRelationships())) {
215                                 //check that both the body targeturl and the type are the same
216                                 if (thisUrl.equals(remoteRel.asVertex().getProperty(LINK_TARGET))
217                                         && annotationType.equals(remoteRel.asVertex().getProperty(Ontology.LINK_HAS_TYPE))
218                                         ) {
219                                     link.addLinkBody(remoteRel);
220                                 }
221                             }
222                         }
223                     } catch (IntegrityError e) {
224                         logger.error(e.getMessage());
225                         throw new RuntimeException(e);
226                     } catch (PermissionDenied ex) {
227                         logger.error(ex.getMessage());
228                         throw new RuntimeException(ex);
229                     }
230                 } else {
231                     logger.info("relation found, but target " + rel.asVertex().getProperty(LINK_TARGET) + " not in graph");
232                 }
233             }
234         }
235     }
