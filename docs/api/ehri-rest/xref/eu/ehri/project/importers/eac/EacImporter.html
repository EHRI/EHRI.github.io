86      /**
87       * @param itemData the item data map
88       */
89      @Override
90      public HistoricalAgent importItem(Map<String, Object> itemData) throws ValidationError {
91  
92          BundleManager persister = new BundleManager(framedGraph, permissionScope.idPath());
93          Bundle descBundle = Bundle.of(EntityClass.HISTORICAL_AGENT_DESCRIPTION,
94                  extractUnitDescription(itemData, EntityClass.HISTORICAL_AGENT_DESCRIPTION));
95  
96          // Add dates and descriptions to the bundle since they're @Dependent
97          // relations.
98          for (Map<String, Object> dpb : ImportHelpers.extractDates(itemData)) {
99              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, Bundle.of(EntityClass.DATE_PERIOD, dpb));
100         }
101 
102         // add the address to the description bundle
103         Map<String, Object> address = ImportHelpers.extractAddress(itemData);
104         if (!address.isEmpty()) {
105             descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_ADDRESS, Bundle.of(EntityClass.ADDRESS, address));
106         }
107 
108         Map<String, Object> unknowns = ImportHelpers.extractUnknownProperties(itemData);
109         if (!unknowns.isEmpty()) {
110             logger.debug("Unknown Properties found");
111             descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, Bundle.of(EntityClass.UNKNOWN_PROPERTY, unknowns));
112         }
113 
114         for (Map<String, Object> dpb : ImportHelpers.extractSubNodes(Entities.MAINTENANCE_EVENT, itemData)) {
115             logger.debug("maintenance event found");
116             //dates in maintenanceEvents are no DatePeriods, they are not something to search on
117             descBundle = descBundle.withRelation(Ontology.HAS_MAINTENANCE_EVENT, Bundle.of(EntityClass.MAINTENANCE_EVENT, dpb));
118         }
119 
120         for (Map<String, Object> rel : extractRelations(itemData)) {
121             if (rel.containsKey(REL_TYPE) && rel.get(REL_TYPE).equals(AccessPointType.subject.name())) {
122                 logger.debug("relation found");
123                 descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT,
124                         Bundle.of(EntityClass.ACCESS_POINT, rel));
125             }
126         }
127 
128         Bundle unit = Bundle.of(EntityClass.HISTORICAL_AGENT, extractUnit(itemData))
129                 .withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
130 
131         Mutation<HistoricalAgent> mutation = persister.createOrUpdate(unit, HistoricalAgent.class);
132         HistoricalAgent frame = mutation.getNode();
133         solveUndeterminedRelationships(frame, descBundle);
134 
135         // There may or may not be a specific scope here...
136         if (!permissionScope.equals(SystemScope.getInstance())
137                 && mutation.created()) {
138             permissionScope.as(AuthoritativeSet.class).addItem(frame);
139             frame.setPermissionScope(permissionScope);
140         }
141 
142         handleCallbacks(mutation);
143         return frame;
144 
145     }

147     private Iterable<Map<String, Object>> extractRelations(Map<String, Object> itemData) {
148         List<Map<String, Object>> list = Lists.newArrayList();
149         for (String key : itemData.keySet()) {
150             if (key.equals(Entities.ACCESS_POINT)) {
151                 //name identifier
152                 for (Map<String, Object> origRelation : (List<Map<String, Object>>) itemData.get(key)) {
153                     Map<String, Object> relationNode = Maps.newHashMap();
154                     for (String eventkey : origRelation.keySet()) {
155                         if (eventkey.equals(REL_TYPE)) {
156                             relationNode.put(Ontology.ACCESS_POINT_TYPE, origRelation.get(eventkey));
157                         } else if (eventkey.equals(REL_NAME)) {
158                             //try to find the original identifier
159                             relationNode.put(ImportHelpers.LINK_TARGET, origRelation.get("concept"));
160                             //try to find the original name
161                             relationNode.put(Ontology.NAME_KEY, origRelation.get(REL_NAME));
162                             relationNode.put("cvoc", origRelation.get("cvoc"));
163                         } else {
164                             relationNode.put(eventkey, origRelation.get(eventkey));
165                         }
166                     }
167                     if (!relationNode.containsKey(Ontology.ACCESS_POINT_TYPE)) {
168                         // Corporate bodies are the default type
169                         relationNode.put(Ontology.ACCESS_POINT_TYPE, AccessPointType.corporateBody);
170                     }
171                     list.add(relationNode);
172                 }
173             }
174         }
175         return list;
176     }

178     protected Map<String, Object> extractUnitDescription(Map<String, Object> itemData, EntityClass entity) {
179         Map<String, Object> description = Maps.newHashMap();
180         description.put(Ontology.CREATION_PROCESS, Description.CreationProcess.IMPORT.toString());
181 
182         for (String key : itemData.keySet()) {
183             if (key.equals("descriptionIdentifier")) {
184                 description.put(Ontology.IDENTIFIER_KEY, itemData.get(key));
185             } else if (!key.startsWith(ImportHelpers.UNKNOWN_PREFIX)
186                     && !key.equals(ImportHelpers.OBJECT_IDENTIFIER)
187                     && !key.equals(Ontology.OTHER_IDENTIFIERS)
188                     && !key.equals(Ontology.IDENTIFIER_KEY)
189                     && !key.startsWith(Entities.MAINTENANCE_EVENT)
190                     && !key.startsWith(Entities.ACCESS_POINT)
191                     && !key.startsWith("IGNORE")
192                     && !key.startsWith("address/")) {
193                 description.put(key, ImportHelpers.flattenNonMultivaluedProperties(key, itemData.get(key), entity));
194             }
195         }
196 
197         return description;
198     }

200     private void solveUndeterminedRelationships(HistoricalAgent unit, Bundle descBundle) throws ValidationError {
201 
202         //Try to resolve the undetermined relationships
203         //we can only create the annotations after the DocumentaryUnit and its Description have been added to the graph,
204         //so they have id's.
205         Api api = ApiFactory.noLogging(framedGraph, actioner.as(UserProfile.class));
206         Bundle linkBundle = Bundle.of(EntityClass.LINK)
207                 .withDataValue(Ontology.LINK_HAS_DESCRIPTION, ImportHelpers.RESOLVED_LINK_DESC);
208 
209         for (Description unitdesc : unit.getDescriptions()) {
210             // Put the set of relationships into a HashSet to remove duplicates.
211             for (AccessPoint rel : Sets.newHashSet(unitdesc.getAccessPoints())) {
212                 Vertex relationVertex = rel.asVertex();
213                 for (String key : relationVertex.getPropertyKeys()) {
214                     logger.debug("solving undetermindRels: {} {} ({})",
215                             key, relationVertex.getProperty(key), descBundle.getId());
216                 }
217                 // the wp2 undetermined relationship that can be resolved have a 'cvoc' and a 'concept' attribute.
218                 // they need to be found in the vocabularies that are in the graph
219                 if (relationVertex.getPropertyKeys().contains("cvoc")) {
220                     String cvocId = relationVertex.getProperty("cvoc");
221                     String conceptId = relationVertex.getProperty(ImportHelpers.LINK_TARGET);
222                     logger.debug("{} -> {}", cvocId, conceptId);
223                     try {
224                         Vocabulary vocabulary = manager.getEntity(cvocId, Vocabulary.class);
225                         for (Concept concept : vocabulary.getConcepts()) {
226                             logger.debug("********************* {} {}", concept.getId(), concept.getIdentifier());
227                             if (concept.getIdentifier().equals(conceptId)) {
228                                 try {
229                                     String linkType = relationVertex.getProperty(REL_TYPE);
230                                     Bundle data = linkBundle.withDataValue(Ontology.LINK_HAS_TYPE, linkType);
231                                     Link link = api.create(data, Link.class);
232                                     unit.addLink(link);
233                                     concept.addLink(link);
234                                     link.addLinkBody(rel);
235                                 } catch (PermissionDenied | DeserializationError ex) {
236                                     logger.error("Unexpected error on EAC relationship creation: {}", ex);
237                                     throw new RuntimeException(ex);
238                                 }
239                             }
240                         }
241                     } catch (ItemNotFound ex) {
242                         logger.error("Vocabulary with id {} not found: {}", cvocId, ex);
243                     }
244                 }
245             }
246         }
247     }

249     private Map<String, Object> extractUnit(Map<String, Object> itemData) throws ValidationError {
250         Map<String, Object> data = ImportHelpers.extractIdentifiers(itemData);
251         data.put("typeOfEntity", itemData.get("typeOfEntity"));
252         return data;
