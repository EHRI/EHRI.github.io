34      @Override

36          return possibleSubnodes.containsKey(getImportantPath(currentPath));

39      @Override
40      public void endElement(String uri, String localName, String qName) throws SAXException {
41          //if a subnode is ended, add it to the super-supergraph
42          super.endElement(uri, localName, qName);
43  
44          if (needToCreateSubNode(qName)) {
45              logger.debug("endElement: " + qName);
46  
47              logger.debug("just before popping: " + depth + "-" + getImportantPath(currentPath) + "-" + qName);
48              Map<String, Object> currentGraph = currentGraphPath.pop();
49              putSubGraphInCurrentGraph(getImportantPath(currentPath), currentGraph);
50  //            currentGraphPath.pop();
51              depth--;
52          }
53          
54          currentPath.pop();
55          //an EAG file consists of only 1 element, so if we're back at the root, we're done
56          if (currentPath.isEmpty()) {
57              try {
58                  logger.debug("depth close " + depth + " " + qName);
59                  //TODO: add any mandatory fields not yet there:
60                  if (!currentGraphPath.peek().containsKey("objectIdentifier")) {
61                      logger.warn("no objectIdentifier found");
62                      putPropertyInCurrentGraph("objectIdentifier", "id");
63                  }
64                  if (!currentGraphPath.peek().containsKey("typeOfEntity")) {
65                      putPropertyInCurrentGraph("typeOfEntity", "organisation");
66                  }
67                  if (!currentGraphPath.peek().containsKey(Ontology.NAME_KEY)) {
68                      logger.debug("no " + Ontology.NAME_KEY + " found");
69                      putPropertyInCurrentGraph(Ontology.NAME_KEY, "title");
70                  }
71                  if (!currentGraphPath.peek().containsKey(Ontology.LANGUAGE_OF_DESCRIPTION)) {
72                      logger.debug("no " + Ontology.LANGUAGE_OF_DESCRIPTION + " found");
73                      putPropertyInCurrentGraph(Ontology.LANGUAGE_OF_DESCRIPTION, "en");
74                  }
75                  if (!currentGraphPath.peek().containsKey("rulesAndConventions")) {
76                      logger.debug("no rulesAndConventions found");
77                      putPropertyInCurrentGraph("rulesAndConventions", "ISDIAH");
78                  }
79                  importer.importItem(currentGraphPath.pop(), Lists.<String>newArrayList());
80  
81              } catch (ValidationError ex) {
82                  logger.error(ex.getMessage());
83              }
84          }
