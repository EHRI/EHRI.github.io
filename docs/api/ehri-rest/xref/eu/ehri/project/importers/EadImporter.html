37   * Import EAD for a given repository into the database. Due to the laxness of the EAD standard this is a fairly complex
38   * procedure. An EAD a single entity at the highest level of description or multiple top-level entities, with or without
39   * a hierarchical structure describing their child items. This means that we need to recursively descend through the
40   * archdesc and c,c01-12 levels.

42   * TODO: Extensive cleanups, optimisation, and rationalisation.

44   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
45   *
46   */

66      /**
67       * Import a single archdesc or c01-12 item, keeping a reference to the hierarchical depth.
68       *
69       * @param itemData The data map
70       * @param idPath The identifiers of parent documents,
71       *               not including those of the overall permission scope
72       * @throws ValidationError when the itemData does not contain an identifier for the unit or...
73       */
74      @Override
75      public AbstractUnit importItem(Map<String, Object> itemData, List<String> idPath)
76              throws ValidationError {
77  
78          BundleDAO persister = getPersister(idPath);
79  
80          // extractDocumentaryUnit does not throw ValidationError on missing ID
81          Bundle unit = new Bundle(unitEntity, extractDocumentaryUnit(itemData));
82          
83          // Check for missing identifier, throw an exception when there is no ID.
84          if (unit.getDataValue(Ontology.IDENTIFIER_KEY) == null) {
85              throw new ValidationError(unit, Ontology.IDENTIFIER_KEY,
86                      "Missing identifier " + Ontology.IDENTIFIER_KEY);
87          }
88          logger.debug("Imported item: " + itemData.get("name"));
89          Bundle descBundle = new Bundle(EntityClass.DOCUMENT_DESCRIPTION, extractUnitDescription(itemData, EntityClass.DOCUMENT_DESCRIPTION));
90          // Add dates and descriptions to the bundle since they're @Dependent
91          // relations.
92          for (Map<String, Object> dpb : extractDates(itemData)) {
93              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
94          }
95          for (Map<String, Object> rel : extractRelations(itemData)) {//, (String) unit.getErrors().get(IdentifiableEntity.IDENTIFIER_KEY)
96              logger.debug("relation found: " + rel.get(Ontology.NAME_KEY));
97              for(String s : rel.keySet()){
98                  logger.debug(s);
99              }
100             descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT, new Bundle(EntityClass.UNDETERMINED_RELATIONSHIP, rel));
101         }
102         Map<String, Object> unknowns = extractUnknownProperties(itemData);
103         if (!unknowns.isEmpty()) {
104             StringBuilder unknownProperties = new StringBuilder();
105             for(String u : unknowns.keySet()){
106                 unknownProperties.append(u);
107             }
108             logger.debug("Unknown Properties found: " + unknownProperties.toString());
109             descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, new Bundle(EntityClass.UNKNOWN_PROPERTY, unknowns));
110         }
111 
112         Mutation<DocumentaryUnit> mutation =
113                 persister.createOrUpdate(mergeWithPreviousAndSave(unit, descBundle, idPath), DocumentaryUnit.class);
114         DocumentaryUnit frame = mutation.getNode();
115 
116         // Set the repository/item relationship
117         if (idPath.isEmpty() && mutation.created()) {
118             EntityClass scopeType = manager.getEntityClass(permissionScope);
119             if (scopeType.equals(EntityClass.REPOSITORY)) {
120                 Repository repository = framedGraph.frame(permissionScope.asVertex(), Repository.class);
121                 frame.setRepository(repository);
122                 frame.setPermissionScope(repository);
123             } else if (scopeType.equals(unitEntity)) {
124                 DocumentaryUnit parent = framedGraph.frame(permissionScope.asVertex(), DocumentaryUnit.class);
125                 parent.addChild(frame);
126                 frame.setPermissionScope(parent);
127             } else {
128                 logger.error("Unknown scope type for documentary unit: {}", scopeType);
129             }
130         }
131         handleCallbacks(mutation);
132         logger.debug("============== "+frame.getIdentifier()+" created:" + mutation.created());
133         if (mutation.created()) {
134             solveUndeterminedRelationships(frame, descBundle);
135         }
136         return frame;

150     protected Bundle mergeWithPreviousAndSave(Bundle unit, Bundle descBundle, List<String> idPath) throws ValidationError {
151         final String languageOfDesc = descBundle.getDataValue(Ontology.LANGUAGE_OF_DESCRIPTION);
152         final String thisSourceFileId = descBundle.getDataValue(Ontology.SOURCEFILE_KEY);
153         /*
154          * for some reason, the idpath from the permissionscope does not contain the parent documentary unit.
155          * TODO: so for now, it is added manually
156          */
157         List<String> lpath = new ArrayList<String>();
158         for(String p : getPermissionScope().idPath()){
159             lpath.add(p);
160         }
161         for(String p : idPath){
162             lpath.add(p);
163         }
164         Bundle withIds = unit.generateIds(lpath);                
165         
166         if (manager.exists(withIds.getId())) {
167             try {
168                 //read the current item’s bundle
169                 Bundle oldBundle = mergeSerializer
170                         .vertexFrameToBundle(manager.getVertex(withIds.getId()));
171                 //filter out dependents that a) are descriptions, b) have the same language/code
172                 Bundle.Filter filter = new Bundle.Filter() {
173                     @Override
174                     public boolean remove(String relationLabel, Bundle bundle) {
175                         String lang = bundle.getDataValue(Ontology.LANGUAGE);
176                         String oldSourceFileId = bundle.getDataValue(Ontology.SOURCEFILE_KEY);
177                         return bundle.getType().equals(EntityClass.DOCUMENT_DESCRIPTION)
178                                 && (lang != null
179                                 && lang.equals(languageOfDesc)
180                                 && (oldSourceFileId != null && oldSourceFileId.equals(thisSourceFileId))
181                                 );
182                     }
183                 };
184                 Bundle filtered = oldBundle.filterRelations(filter);
185                 
186                 //if this desc-id already exists, but with a different sourceFileId, 
187                 //change the desc-id
188                 String defaultDescIdentifier= withIds.getId()+"-"+languageOfDesc.toLowerCase();
189                 String newDescIdentifier=withIds.getId()+"-"+thisSourceFileId.toLowerCase().replace("#", "-");
190                 if(manager.exists(newDescIdentifier)){
191                         descBundle=descBundle.withDataValue(Ontology.IDENTIFIER_KEY, newDescIdentifier);
192                 } else if(manager.exists(defaultDescIdentifier)){
193                     Bundle oldDescBundle = mergeSerializer
194                         .vertexFrameToBundle(manager.getVertex(defaultDescIdentifier));
195                     //if the previous had NO sourcefile_key OR it was different:
196                     if(oldDescBundle.getDataValue(Ontology.SOURCEFILE_KEY) == null
197                             || ! thisSourceFileId.equals(oldDescBundle.getDataValue(Ontology.SOURCEFILE_KEY).toString())){
198                         descBundle=descBundle.withDataValue(Ontology.IDENTIFIER_KEY, newDescIdentifier);
199                         logger.info("other description found ("+defaultDescIdentifier+"), creating new description id: " + descBundle.getDataValue(Ontology.IDENTIFIER_KEY).toString());
200                     }
201                 }
202 
203                 return withIds.withRelations(filtered.getRelations())
204                         .withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
205 
206             } catch (SerializationError ex) {
207                 throw new ValidationError(unit, "serialization error", ex.getMessage());
208             } catch (ItemNotFound ex) {
209                 throw new ValidationError(unit, "item not found exception", ex.getMessage());
210             }
211         } else {
212             return unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
213         }
214     }

216     /**
217      * subclasses can override this method to cater to their special needs for UndeterminedRelationships
218      * by default, it expects something like this in the original EAD:
219      * 
220      * <persname source="terezin-victims" authfilenumber="PERSON.ITI.1514982">Kien,
221                         Leonhard (* 11.5.1886)</persname>
222      *
223      * it works in unison with the extractRelations() method. 
224      * 
225                         * 
226      * @param unit
227      * @param descBundle - not used
228      * @throws ValidationError 
229      */
230     protected void solveUndeterminedRelationships(DocumentaryUnit unit, Bundle descBundle) throws ValidationError {
231         //Try to resolve the undetermined relationships
232         //we can only create the annotations after the DocumentaryUnit and its Description have been added to the graph,
233         //so they have id's. 
234         for (Description unitdesc : unit.getDescriptions()) {
235             // Put the set of relationships into a HashSet to remove duplicates.
236             for (UndeterminedRelationship rel : Sets.newHashSet(unitdesc.getUndeterminedRelationships())) {
237                 /*
238                  * the wp2 undetermined relationship that can be resolved have a 'cvoc' and a 'concept' attribute.
239                  * they need to be found in the vocabularies that are in the graph
240                  */
241                 if (rel.asVertex().getPropertyKeys().contains("cvoc")) {
242                     String cvoc_id = (String) rel.asVertex().getProperty("cvoc");
243                     String concept_id = (String) rel.asVertex().getProperty("concept");
244                     if(concept_id == null){
245                       concept_id = (String) rel.asVertex().getProperty("target");
246                     }
247                     logger.debug("cvoc:"+cvoc_id + "  concept:" + concept_id);
248                     Vocabulary vocabulary;
249                     try {
250                         vocabulary = manager.getFrame(cvoc_id, Vocabulary.class);
251                         for (Concept concept : vocabulary.getConcepts()) {
252                         logger.debug("*********************" + concept.getId() + " " + concept.getIdentifier());
253                         if (concept.getIdentifier().equals(concept_id)) {
254                             try {
255                                 Bundle linkBundle = new Bundle(EntityClass.LINK)
256                                         .withDataValue(Ontology.LINK_HAS_TYPE, rel.asVertex().getProperty("type").toString())
257                                         .withDataValue(Ontology.LINK_HAS_DESCRIPTION, RESOLVED_LINK_DESC);
258                                 UserProfile user = manager.getFrame(this.log.getActioner().getId(), UserProfile.class);
259                                 Link link = new CrudViews<Link>(framedGraph, Link.class).create(linkBundle, user);
260                                 unit.addLink(link);
261                                 concept.addLink(link);
262                                 link.addLinkBody(rel);
263                             } catch (PermissionDenied ex) {
264                                 logger.error(ex.getMessage());
265                             }
266 
267                         }
268 
269                     }
270                     } catch (ItemNotFound ex) {
271                         logger.error("Vocabulary with id " + cvoc_id +" not found. "+ex.getMessage());
272                     }
273                     
274                 }else{
275                     logger.debug("no cvoc found");
276                 }
277                 
278             }
279         }

283    @Override
284     protected Iterable<Map<String, Object>> extractRelations(Map<String, Object> data) {
285         final String REL = "relation";
286         List<Map<String, Object>> list = new ArrayList<Map<String, Object>>();
287         for (String key : data.keySet()) {
288             if (key.equals(REL)) {
289                 //name identifier
290                 for (Map<String, Object> origRelation : (List<Map<String, Object>>) data.get(key)) {
291                     Map<String, Object> relationNode = new HashMap<String, Object>();
292                     if(origRelation.containsKey("type")){
293                             //try to find the original identifier
294                             relationNode.put(LINK_TARGET, origRelation.get("concept"));
295                             //try to find the original name
296                             relationNode.put(Ontology.NAME_KEY, origRelation.get("name"));
297                             relationNode.put("cvoc", origRelation.get("cvoc"));
298                             relationNode.put(Ontology.UNDETERMINED_RELATIONSHIP_TYPE, origRelation.get("type"));
299                     } else {
300                         relationNode.put(Ontology.NAME_KEY, origRelation.get(REL));
301                     }
302                     if (!relationNode.containsKey(Ontology.UNDETERMINED_RELATIONSHIP_TYPE)) {
303                         logger.debug("relationNode without type: "+relationNode.get(Ontology.NAME_KEY));
304                         relationNode.put(Ontology.UNDETERMINED_RELATIONSHIP_TYPE, "corporateBodyAccess");
305                     }
306                     list.add(relationNode);
307                 }
308             }
309             else 
310     if (key.endsWith(ACCESS_POINT)) {
311 
312                 logger.debug(key + data.get(key).getClass());
313                 if (data.get(key) instanceof List) {
314                     for (Object o : (List) data.get(key)) {
315                         logger.debug("" + o.getClass());
316                     }
317                     //type, targetUrl, targetName, notes
318                     for (Map<String, Object> origRelation : (List<Map<String, Object>>) data.get(key)) {
319                         if (origRelation.isEmpty()) {
320                             break;
321                         }
322                         Map<String, Object> relationNode = new HashMap<String, Object>();
323                         for (String eventkey : origRelation.keySet()) {
324                             if (eventkey.endsWith(ACCESS_POINT)) {
325                                 relationNode.put(Ontology.UNDETERMINED_RELATIONSHIP_TYPE, eventkey.substring(0, eventkey.indexOf("Point")));
326                                 relationNode.put(Ontology.NAME_KEY, origRelation.get(eventkey));
327 //logger.debug("------------------" + eventkey.substring(0, eventkey.indexOf("Point")) + ": "+ origRelation.get(eventkey));                            
328                             } else {
329                                 relationNode.put(eventkey, origRelation.get(eventkey));
330                             }
331                         }
332                         if (!relationNode.containsKey(Ontology.UNDETERMINED_RELATIONSHIP_TYPE)) {
333                             relationNode.put(Ontology.UNDETERMINED_RELATIONSHIP_TYPE, "corporateBodyAccess");
334                         }
335                         //if no name is given, it was apparently an empty <controlaccess> tag?
336                         if (relationNode.containsKey(Ontology.NAME_KEY)) {
337                             list.add(relationNode);
338                         }
339                     }
340                 } else {
341                         Map<String, Object> relationNode = new HashMap<String, Object>();
342                         relationNode.put(Ontology.UNDETERMINED_RELATIONSHIP_TYPE, key.substring(0, key.indexOf("Point")));
343                         relationNode.put(Ontology.NAME_KEY, data.get(key));
344                         list.add(relationNode);
345                     
346                 }
347             }
348         }
349         return list;
350     }
