43      // Splitter to breaking up
44      private static final Splitter codeSplitter = Splitter.on("-").omitEmptyStrings().limit(2);

46      /**
47       * Keys in the graph that encode a language code must start with the LANGUAGE_KEY_PREFIX.
48       */
49      public static final String LANGUAGE_KEY_PREFIX = "language";
50      /**
51       * Limited selection of bibliographical 3-letter codes for the languages
52       * we're most likely to run into, and their mappings to ISO639-2 Term codes.
53       */
54      private static final ImmutableMap<String, String> iso639BibTermLookup = ImmutableMap.<String, String>builder()
55              .put("alb", "sqi") // albanian
56              .put("arm", "hye") // armenian
57              .put("baq", "eus") // basque
58              .put("ger", "deu") // german
59              .put("dut", "nld") // dutch
60              .put("rum", "ron") // romanian
61              .put("mac", "mkd") // macedonian
62              .put("slo", "slk") // slovak
63              .put("fre", "fra") // french
64              .put("cze", "ces") // czech
65              .build();

67      private static final Map<String, Locale> locale2Map;
68      private static final Map<String, String> locale3Map;
69      private static final Map<String, Locale> localeNameMap;

71      /**
72       * Stores this property value pair in the given graph node representation.
73       * If the value is effectively empty, nothing happens.
74       * If the property already exists, it is added to the value list.
75       *
76       * @param c a Map representation of a graph node
77       * @param property the key to store the value for
78       * @param value the value to store
79       */
80      public static void putPropertyInGraph(Map<String, Object> c, String property, String value) {
81          if (value == null)
82              return;
83          String valuetrimmed = value.trim();
84          if (valuetrimmed.isEmpty()) {
85              return;
86          }
87  
88          // Language properties
89  
90          if (property.startsWith(LANGUAGE_KEY_PREFIX)) {
91              valuetrimmed = iso639DashTwoCode(valuetrimmed);
92          }
93  
94          valuetrimmed = StringUtils.normalizeSpace(valuetrimmed);
95  
96          logger.debug("putProp: " + property + " " + valuetrimmed);
97  
98          Object propertyList;
99          if (c.containsKey(property)) {
100             propertyList = c.get(property);
101             if (propertyList instanceof List) {
102                 ((List<Object>) propertyList).add(valuetrimmed);
103             } else {
104                 List<Object> o = Lists.newArrayList();
105                 o.add(c.get(property));
106                 o.add(valuetrimmed);
107                 c.put(property, o);
108             }
109         } else {
110             c.put(property, valuetrimmed);
111         }
112     }

114     static {
115         String[] languages = Locale.getISOLanguages();
116         locale2Map = Maps.newHashMapWithExpectedSize(languages.length);
117         locale3Map = Maps.newHashMapWithExpectedSize(languages.length);
118         localeNameMap = Maps.newHashMapWithExpectedSize(languages.length);
119         for (String language : languages) {
120             Locale locale = new Locale(language);
121             locale2Map.put(language, locale);
122             locale3Map.put(locale.getISO3Language(), language);
123             localeNameMap.put(locale.getDisplayLanguage(Locale.ENGLISH).toLowerCase(), locale);
124         }
125 
126     }

128     /**
129      * Take an ISO-639-1 code or a language name and try and map to a valid ISO639-2 code.
130      * 
131      * @param nameOrCode a language code or name to convert
132      * @return the ISO 639-2 language code for that code or name, or the input string if
133      *         no conversion was found
134      */
135     public static String iso639DashTwoCode(String nameOrCode) {
136         if (nameOrCode.length() == 2 && locale2Map.containsKey(nameOrCode)) {
137             return locale2Map.get(nameOrCode).getISO3Language();
138         } else if (nameOrCode.length() == 3 && iso639BibTermLookup.containsKey(nameOrCode)) {
139             return iso639BibTermLookup.get(nameOrCode);
140         } else if (nameOrCode.length() > 3 && localeNameMap.containsKey(nameOrCode.toLowerCase())) {
141             return localeNameMap.get(nameOrCode.toLowerCase()).getISO3Language();
142             /* FIXME the localeNameMap depends on locale and translating an 
143              * English name to a code fails when executed on 
144              * e.g. a server with non-English locale
145              */
146         }
147         return nameOrCode;
148     }
149 
150     /**
151      * Take an ISO-639-2 code or a language name and try and map to a valid ISO639-1 code.
152      *
153      * @param nameOrCode a language code or name to convert
154      * @return the ISO 639-1 language code for that code or name, or the input string if
155      *         no conversion was found
156      */
157     public static String iso639DashOneCode(String nameOrCode) {
158         if (nameOrCode.length() == 3 && locale3Map.containsKey(nameOrCode)) {
159             return locale3Map.get(nameOrCode);
160         } else if (nameOrCode.length() == 3 && iso639BibTermLookup.containsKey(nameOrCode)) {
161             return locale3Map.get(iso639BibTermLookup.get(nameOrCode));
162         } else if (nameOrCode.length() > 3 && localeNameMap.containsKey(nameOrCode.toLowerCase())) {
163             return localeNameMap.get(nameOrCode.toLowerCase()).getLanguage();
164         } else if (nameOrCode.length() > 2 && nameOrCode.contains("-")) {
165             // Attempt to handle codes like 'heb-Hebr' and 'eng-Latn'
166             List<String> parts = Lists.newArrayList(codeSplitter.split(nameOrCode));
167             if (parts.size() == 1) {
168                 return iso639DashOneCode(parts.get(0));
169             } else if (parts.size() == 2) {
170                 return iso639DashOneCode(parts.get(0)) + "-" + parts.get(1);
171             }
172         }
173         return nameOrCode;
174     }
