33      /**
34       * Set the accessors who are able to view an item. If no accessors
35       * are set, the item is globally readable.
36       *
37       * @param id          The ID of the item
38       * @param accessorIds The IDs of the users who can access this item.
39       * @return the updated object
40       * @throws PermissionDenied
41       * @throws ItemNotFound
42       * @throws BadRequester
43       */
44      @POST
45      @Path("/{id:[^/]+}")
46      public Response setVisibility(@PathParam("id") String id,
47              @QueryParam(ACCESSOR_PARAM) List<String> accessorIds)
48              throws PermissionDenied, ItemNotFound, BadRequester, SerializationError {
49          graph.getBaseGraph().checkNotInTransaction();
50          try {
51              AccessibleEntity item = manager.getFrame(id, AccessibleEntity.class);
52              Accessor current = getRequesterUserProfile();
53              Set<Accessor> accessors = getAccessors(accessorIds, current);
54              aclViews.setAccessors(item, accessors, current);
55              graph.getBaseGraph().commit();
56              return single(item);
57          } finally {
58              cleanupTransaction();
59          }
60      }
