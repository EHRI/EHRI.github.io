83      @POST
84      @Consumes(MediaType.APPLICATION_JSON)
85      @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
86      public Response createGroup(Bundle bundle,
87              @QueryParam(ACCESSOR_PARAM) List<String> accessors,
88              @QueryParam(MEMBER_PARAM) List<String> members)
89              throws PermissionDenied, ValidationError, IntegrityError,
90              DeserializationError, ItemNotFound, BadRequester {
91          final UserProfile currentUser = getCurrentUser();
92          try {
93              final Set<Accessor> groupMembers = Sets.newHashSet();
94              for (String member : members) {
95                  groupMembers.add(manager.getFrame(member, Accessor.class));
96              }
97              return createItem(bundle, accessors, new Handler<Group>() {
98                  @Override
99                  public void process(Group group) throws PermissionDenied {
100                     for (Accessor member : groupMembers) {
101                         aclViews.addAccessorToGroup(group, member, currentUser);
102                     }
103                 }
104             });
105         } catch (ItemNotFound e) {
106             graph.getBaseGraph().rollback();
107             throw new DeserializationError("User or group not found: " + e.getValue());
108         }
109     }

132     /**
133      * Add an accessor to a group.
134      */
135     @POST
136     @Path("/{id:[^/]+}/{aid:.+}")
137     public Response addMember(@PathParam("id") String id,
138             @PathParam("aid") String aid)
139             throws PermissionDenied, ItemNotFound, BadRequester {
140         graph.getBaseGraph().checkNotInTransaction();
141         Group group = manager.getFrame(id, EntityClass.GROUP, Group.class);
142         Accessor accessor = manager.getFrame(aid, Accessor.class);
143         try {
144             aclViews.addAccessorToGroup(group, accessor, getRequesterUserProfile());
145             graph.getBaseGraph().commit();
146             return Response.status(Status.OK).location(getItemUri(accessor)).build();
147         } finally {
148             cleanupTransaction();
149         }
150     }
