106     /**
107      * Fetch a list of items by their ID.
108      *
109      * @param ids  a list of string IDs
110      * @param gids a list of graph number IDs
111      * @return A serialized list of items
112      * @throws ItemNotFound
113      */
114     @GET
115     @Produces(MediaType.APPLICATION_JSON)
116     public Response list(@QueryParam("id") List<String> ids,
117             @QueryParam("gid") List<Long> gids) throws ItemNotFound {
118 
119         try (Tx tx = beginTx()) {
120             for (Long id : gids) {
121                 if (graph.getVertex(id) == null) {
122                     throw new ItemNotFound(String.valueOf(id));
123                 }
124             }
125 
126             Stream<Vertex> verticesByGids = gids.stream().map(graph::getVertex);
127 
128             // Object a lazily-computed view of the ids->vertices...
129             Iterable<Vertex> verticesByIds = manager.getVertices(ids);
130             Iterable<Vertex> allVertices = Iterables.concat(verticesByIds, verticesByGids.collect(Collectors.toList()));
131 
132             PipeFunction<Vertex, Boolean> filter = AclManager
133                     .getAclFilterFunction(getRequesterUserProfile());
134             GremlinPipeline<Vertex, Vertex> filtered = new GremlinPipeline<Vertex, Vertex>(allVertices)
135                     .filter(aclManager.getContentTypeFilterFunction()).filter(filter);
136             Response response = streamingVertexList(filtered);
137             tx.success();
138             return response;
139         }
140     }

160     /**
161      * Fetch an item of any type by ID.
162      *
163      * @param id the item's ID
164      * @return A serialized representation.
165      * @throws ItemNotFound
166      */
167     @GET
168     @Produces(MediaType.APPLICATION_JSON)
169     @Path("{id:[^/]+}")
170     public Response get(@PathParam("id") String id) throws ItemNotFound, AccessDenied {
171         try (final Tx tx = beginTx()) {
172             Vertex item = manager.getVertex(id);
173 
174             // If the item doesn't exist or isn't a content type throw 404
175             Accessor currentUser = getRequesterUserProfile();
176             if (item == null || !aclManager.getContentTypeFilterFunction().compute(item)) {
177                 throw new ItemNotFound(id);
178             } else if (!AclManager.getAclFilterFunction(currentUser).compute(item)) {
179                 throw new AccessDenied(currentUser.getId(), id);
180             }
181 
182             Response response = single(graph.frame(item, Accessible.class));
183             tx.success();
184             return response;
185         }
186     }

647     private static class IdSet {
648         final List<String> ids;
649         final List<Long> gids;
650 
651         IdSet(List<String> ids, List<Long> gids) {
652             this.ids = ids;
653             this.gids = gids;
654         }
655     }
656 
657     private IdSet parseGraphIds(String json) throws IOException, DeserializationError {
658         try {
659             TypeReference<List<Object>> typeRef = new TypeReference<List<Object>>() {
660             };
661             List<Object> jsonValues = jsonMapper.readValue(json, typeRef);
662             List<String> ids = Lists.newArrayList();
663             List<Long> gids = Lists.newArrayList();
664 
665             for (Object js : jsonValues) {
666                 if (js instanceof Integer) {
667                     gids.add(Long.valueOf((Integer) js));
668                 } else if (js instanceof Long) {
669                     gids.add((Long) js);
670                 } else if (js instanceof String) {
671                     ids.add((String) js);
672                 }
673             }
674             return new IdSet(ids, gids);
675         } catch (JsonMappingException e) {
