39  /**
40   * Provides a web service interface for the HistoricalAgent model.
41   *
42   * @author Mike Bryant (http://github.com/mikesname)
43   */
44  @Path(Entities.HISTORICAL_AGENT)
45  public class HistoricalAgentResource extends AbstractAccessibleEntityResource<HistoricalAgent>
46          implements GetResource, ListResource, CreateResource, UpdateResource, DeleteResource {
47  
48      public HistoricalAgentResource(@Context GraphDatabaseService database) {
49          super(database, HistoricalAgent.class);
50      }
51  
52      @GET
53      @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
54      @Path("{id:.+}")
55      @Override
56      public Response get(@PathParam("id") String id) throws ItemNotFound {
57          return getItem(id);
58  
59      }
60  
61      @GET
62      @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
63      @Override
64      public Response list() {
65          return listItems();
66      }
67  
68      @POST
69      @Consumes(MediaType.APPLICATION_JSON)
70      @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
71      @Override
72      public Response create(Bundle bundle,
73                             @QueryParam(ACCESSOR_PARAM) List<String> accessors)
74              throws PermissionDenied, ValidationError, DeserializationError {
75          try (final Tx tx = graph.getBaseGraph().beginTx()) {
76              Response item = createItem(bundle, accessors);
77              tx.success();
78              return item;
79          }
80      }
81  
82      @PUT
83      @Consumes(MediaType.APPLICATION_JSON)
84      @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
85      @Path("{id:.+}")
86      @Override
87      public Response update(@PathParam("id") String id, Bundle bundle)
88              throws PermissionDenied, ValidationError,
89              DeserializationError, ItemNotFound {
90          try (final Tx tx = graph.getBaseGraph().beginTx()) {
91              Response response = updateItem(id, bundle);
92              tx.success();
93              return response;
94          }
95      }
96  
97      @DELETE
98      @Path("{id:.+}")
99      @Override
100     public Response delete(@PathParam("id") String id)
101             throws PermissionDenied, ItemNotFound, ValidationError {
102         try (final Tx tx = graph.getBaseGraph().beginTx()) {
103             Response response = deleteItem(id);
104             tx.success();
105             return response;
106         }
107     }
