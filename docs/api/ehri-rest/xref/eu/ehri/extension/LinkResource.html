107     /**
108      * Create a link between two items.
109      *
110      * @param targetId  The link target
111      * @param sourceId  The link source
112      * @param bundle      The link body data
113      * @param bodies    optional list of entities to provide the body
114      * @param accessors The IDs of accessors who can see this link
115      * @return The created link item
116      * @throws PermissionDenied
117      * @throws ValidationError
118      * @throws DeserializationError
119      * @throws ItemNotFound
120      * @throws BadRequester
121      * @throws SerializationError
122      */
123     @POST
124     @Consumes(MediaType.APPLICATION_JSON)
125     @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
126     @Path("{targetId:.+}/{sourceId:.+}")
127     public Response createLinkFor(
128             @PathParam("targetId") String targetId,
129             @PathParam("sourceId") String sourceId, Bundle bundle,
130             @QueryParam(BODY_PARAM) List<String> bodies,
131             @QueryParam(ACCESSOR_PARAM) List<String> accessors)
132             throws PermissionDenied, ValidationError, DeserializationError,
133             ItemNotFound, BadRequester, SerializationError {
134         Accessor user = getRequesterUserProfile();
135         try {
136             Link link = linkViews.createLink(targetId,
137                     sourceId, bodies, bundle, user);
138             aclManager.setAccessors(link,
139                     getAccessors(accessors, user));
140             graph.getBaseGraph().commit();
141             return creationResponse(link);
142         } catch (SerializationError e) {
143             graph.getBaseGraph().rollback();
144             throw new RuntimeException(e);
145         } finally {
146             cleanupTransaction();
147         }
148 
149     }

151     /**
152      * Delete an access point.
153      */
154     @DELETE
155     @Path("/accessPoint/{id:.+}")
156     public Response deleteAccessPoint(@PathParam("id") String id)
157             throws AccessDenied, PermissionDenied, ItemNotFound, ValidationError,
158             BadRequester, SerializationError {
159         graph.getBaseGraph().checkNotInTransaction();
160         try {
161             Accessor userProfile = getRequesterUserProfile();
162             UndeterminedRelationship rel = manager.getFrame(id, UndeterminedRelationship.class);
163             Description description = rel.getDescription();
164             if (description == null) {
165                 throw new ItemNotFound(id);
166             }
167             DescribedEntity item = description.getEntity();
168             if (item == null) {
169                 throw new ItemNotFound(id);
170             }
171             descriptionViews.delete(item.getId(), id, userProfile, getLogMessage());
172             graph.getBaseGraph().commit();
173             return Response.status(Status.OK).build();
174         } finally {
175             cleanupTransaction();
176         }
177     }

194     /**
195      * Delete a link. If the optional ?accessPoint=[ID] parameter is also given
196      * the access point associated with the link will also be deleted.
197      */
198     @DELETE
199     @Path("/for/{id:.+}/{linkId:.+}")
200     public Response deleteLinkForItem(@PathParam("id") String id, @PathParam("linkId") String linkId)
201             throws AccessDenied, PermissionDenied, ItemNotFound, ValidationError,
202             BadRequester {
203         graph.getBaseGraph().checkNotInTransaction();
204         try {
205             helper.checkEntityPermission(manager.getFrame(id, AccessibleEntity.class),
206                     getRequesterUserProfile(), PermissionType.ANNOTATE);
207             Actioner actioner = manager.cast(getRequesterUserProfile(), Actioner.class);
208             Link link = manager.getFrame(linkId, EntityClass.LINK, Link.class);
209             actionManager.logEvent(link, actioner, EventTypes.deletion);
210             manager.deleteVertex(link.asVertex());
211             graph.getBaseGraph().commit();
212             return Response.ok().build();
213         } finally {
214             cleanupTransaction();
215         }
216     }
