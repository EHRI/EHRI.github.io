113     @POST
114     @Consumes(MediaType.APPLICATION_JSON)
115     @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
116     @Path("/{id:.+}/" + Entities.CVOC_CONCEPT)
117     @Override
118     public Response createChild(@PathParam("id") String id,
119                                 Bundle bundle, @QueryParam(ACCESSOR_PARAM) List<String> accessors)
120             throws PermissionDenied, ValidationError,
121             DeserializationError, ItemNotFound {
122         try (final Tx tx = graph.getBaseGraph().beginTx()) {
123             Accessor user = getRequesterUserProfile();
124             final Concept parent = views.detail(id, user);
125             Response item = createItem(bundle, accessors, new Handler<Concept>() {
126                 @Override
127                 public void process(Concept concept) {
128                     parent.addNarrowerConcept(concept);
129                     concept.setVocabulary(parent.getVocabulary());
130                 }
131             }, views.setScope(parent.getVocabulary()));
132             tx.success();
133             return item;
134         }
135     }

138      * Add an existing concept to the list of 'narrower' relations.
139      *
140      * @param id         The item ID
141      * @param idNarrower The narrower item ID
142      * @throws AccessDenied
143      * @throws PermissionDenied

146     @POST

148     public Response addNarrowerCvocConcept(
149             @PathParam("id") String id,
150             @PathParam("idNarrower") String idNarrower)
151             throws AccessDenied, PermissionDenied, ItemNotFound {
152         try (final Tx tx = graph.getBaseGraph().beginTx()) {
153             Accessor accessor = getRequesterUserProfile();
154             Concept concept = views.detail(id, accessor);
155             Concept relatedConcept = views.detail(idNarrower, accessor);
156             helper.checkEntityPermission(concept, accessor, PermissionType.UPDATE);
157             helper.checkEntityPermission(relatedConcept, accessor, PermissionType.UPDATE);
158             concept.addNarrowerConcept(relatedConcept);
159             tx.success();
160             return Response.status(Status.OK).build();
161         }
