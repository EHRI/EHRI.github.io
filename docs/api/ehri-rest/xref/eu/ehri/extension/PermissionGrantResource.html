44      @GET

46      @Path("/{id:.+}")

51                  EntityClass.PERMISSION_GRANT, PermissionGrant.class);

60       * @throws BadRequester

62      @DELETE
63      @Path("/{id:.+}")
64      public Response revokePermissionGrant(@PathParam("id") String id)
65              throws ItemNotFound, PermissionDenied, BadRequester {
66          try {
67              new AclViews(graph).revokePermissionGrant(manager.getFrame(id,
68                      EntityClass.PERMISSION_GRANT, PermissionGrant.class),
69                      getRequesterUserProfile());
70              graph.getBaseGraph().commit();
71              return Response.status(Status.OK).build();
72          } finally {
