42  /**
43   * Provides a web service interface for the PermissionGrant model.
44   * <p>
45   * FIXME: PermissionGrant is not currently an AccessibleEntity so
46   * handling it is complicated. We need to re-architect the REST views
47   * to handle more than just the initially-envisaged scenarios.
48   *
49   * @author Mike Bryant (http://github.com/mikesname)
50   */
51  @Path(Entities.PERMISSION_GRANT)
52  public class PermissionGrantResource extends AbstractRestResource {
53  
54      public PermissionGrantResource(@Context GraphDatabaseService database) {
55          super(database);
56      }
57  
58      /**
59       * Fetch a given permission grant.
60       *
61       * @param id The ID of the permission grant
62       * @return The permission grant
63       * @throws ItemNotFound
64       * @throws PermissionDenied
65       */
66      @GET
67      @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
68      @Path("{id:.+}")
69      public Response getPermissionGrant(@PathParam("id") String id)
70              throws ItemNotFound, PermissionDenied, SerializationError {
71          try (final Tx tx = graph.getBaseGraph().beginTx()) {
72              PermissionGrant grant = manager.getFrame(id,
73                      EntityClass.PERMISSION_GRANT, PermissionGrant.class);
74              Response response = single(grant);
75              tx.success();
76              return response;
77          }
78      }
79  
80      /**
81       * Revoke a particular permission grant.
82       *
83       * @param id The ID of the permission grant
84       * @throws ItemNotFound
85       * @throws PermissionDenied
86       */
87      @DELETE
88      @Path("{id:.+}")
89      public Response revokePermissionGrant(@PathParam("id") String id)
90              throws ItemNotFound, PermissionDenied {
91          try (final Tx tx = graph.getBaseGraph().beginTx()) {
92              new AclViews(graph).revokePermissionGrant(manager.getFrame(id,
93                              EntityClass.PERMISSION_GRANT, PermissionGrant.class),
94                      getRequesterUserProfile());
95              tx.success();
96              return Response.status(Status.OK).build();
97          }
98      }
99  }
