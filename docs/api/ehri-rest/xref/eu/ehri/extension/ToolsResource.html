97       * @param languageCode     the language code of created concepts
98       * @param accessPointTypes the access point types to process
99       * @param tolerant         proceed even if there are integrity errors due

101      * @param excludeSingle    don't create concepts/links for access points that

103      * @return the number of links created
104      * @throws ItemNotFound
105      * @throws BadRequester
106      * @throws ValidationError
107      * @throws PermissionDenied

110     @POST
111     @Produces(MediaType.APPLICATION_JSON)
112     @Path("/generateConcepts/{repositoryId:.+}/{vocabularyId:.+}")
113     public long autoLinkRepositoryDocs(
114             @PathParam("repositoryId") String repositoryId,
115             @PathParam("vocabularyId") String vocabularyId,
116             @QueryParam(ACCESS_POINT_TYPE_PARAM) List<String> accessPointTypes,
117             @QueryParam(LANG_PARAM) @DefaultValue(DEFAULT_LANG) String languageCode,
118             @QueryParam(SINGLE_PARAM) @DefaultValue("true") boolean excludeSingle,
119             @QueryParam(TOLERANT_PARAM) @DefaultValue("false") boolean tolerant)
120             throws ItemNotFound, BadRequester, ValidationError,
121             PermissionDenied, DeserializationError {
122         try (final Tx tx = graph.getBaseGraph().beginTx()) {
123             UserProfile user = getCurrentUser();
124             Repository repository = manager.getFrame(repositoryId, Repository.class);
125             Vocabulary vocabulary = manager.getFrame(vocabularyId, Vocabulary.class);
126 
127             long linkCount = linker
128                     .withAccessPointTypes(accessPointTypes)
129                     .withTolerant(tolerant)
130                     .withExcludeSingles(excludeSingle)
131                     .withDefaultLanguage(languageCode)
132                     .withLogMessage(getLogMessage())
133                     .createAndLinkRepositoryVocabulary(repository, vocabulary, user);
134 
135             tx.success();
136             return linkCount;

230     /**
231      * Regenerate the hierarchical graph ID for a given item, optionally
232      * renaming it.
233      *
234      * The default mode is to output items whose IDs would change, without
235      * actually changing them. The {@code collisions} parameter will <b>only</b>
236      * output items that would cause collisions if renamed, whereas {@code tolerant}
237      * mode will skip them altogether.
238      *
239      * The {@code commit} flag will cause renaming to take place.
240      *
241      * @param id         the item's existing ID
242      * @param collisions only output items that if renamed to the
243      *                   regenerated ID would cause collisions
244      * @param tolerant   skip items that could cause collisions rather
245      *                   then throwing an error
246      * @param commit     whether or not to rename the item
247      * @return a tab old-to-new mapping, or an empty
248      * body if nothing was changed
249      * @throws ItemNotFound
250      * @throws IOException
251      */
252     @POST
253     @Produces("text/csv")
254     @Path("/_regenerateId/{id:.+}")
255     public String regenerateId(
256             @PathParam("id") String id,
257             @QueryParam("collisions") @DefaultValue("false") boolean collisions,
258             @QueryParam("tolerant") @DefaultValue("false") boolean tolerant,
259             @QueryParam("commit") @DefaultValue("false") boolean commit)
260             throws ItemNotFound, IOException, IdRegenerator.IdCollisionError {
261         try (final Tx tx = graph.getBaseGraph().beginTx()) {
262             AccessibleEntity item = manager.getFrame(id, AccessibleEntity.class);
263             Optional<List<String>> remap = new IdRegenerator(graph)
264                     .withActualRename(commit)
265                     .collisionMode(collisions)
266                     .skippingCollisions(tolerant)
267                     .reGenerateId(item);
268             tx.success();
269             return makeCsv(Lists.newArrayList(remap.asSet()));
270         }
271     }

361     /**
362      * Regenerate description IDs.
363      *
364      * @throws ItemNotFound
365      * @throws IOException
366      */
367     @POST
368     @Produces("text/plain")
369     @Path("/_regenerateDescriptionIds")
370     public String regenerateDescriptionIds(
371             @QueryParam("buffer") @DefaultValue("-1") int bufferSize,
372             @QueryParam("commit") @DefaultValue("false") boolean commit)
373             throws IOException, ItemNotFound, IdRegenerator.IdCollisionError {
374         EntityClass[] types = {EntityClass.DOCUMENT_DESCRIPTION, EntityClass
375                 .CVOC_CONCEPT_DESCRIPTION, EntityClass.HISTORICAL_AGENT_DESCRIPTION, EntityClass
376                 .REPOSITORY_DESCRIPTION};
377         Serializer depSerializer = new Serializer.Builder(graph).dependentOnly().build();
378         int done = 0;
379         try (final Tx tx = graph.getBaseGraph().beginTx()) {
380             for (EntityClass entityClass : types) {
381                 try (CloseableIterable<Description> descriptions = manager.getFrames(entityClass, Description.class)) {
382                     for (Description desc : descriptions) {
383                         DescribedEntity entity = desc.getEntity();
384                         if (entity != null) {
385                             PermissionScope scope = entity.getPermissionScope();
386                             List<String> idPath = scope != null
387                                     ? Lists.newArrayList(scope.idPath())
388                                     : Lists.<String>newArrayList();
389                             idPath.add(entity.getIdentifier());
390                             Bundle descBundle = depSerializer.vertexFrameToBundle(desc);
391                             String newId = entityClass.getIdgen().generateId(idPath, descBundle);
392                             if (!newId.equals(desc.getId()) && commit) {
393                                 manager.renameVertex(desc.asVertex(), desc.getId(), newId);
394                                 done++;
395 
396                                 if (bufferSize > 0 && done % bufferSize == 0) {
397                                     tx.success();
398                                 }
399                             }
400                         }
401                     }
402                 }
403             }
404             if (commit && done > 0) {
405                 tx.success();
406             }
407             return String.valueOf(done);
408         } catch (SerializationError e) {
409             throw new RuntimeException(e);
410         }
411     }

413     @POST
414     @Produces("text/csv")
415     @Path("/_setIdsOnEventLinks")
416     public String setIdsOnEventLinks()
417             throws IOException, ItemNotFound, IdRegenerator.IdCollisionError {
418         try (final Tx tx = graph.getBaseGraph().beginTx()) {
419             long done = 0;
420             for (Vertex v : graph.getVertices()) {
421                 if ("eventLink".equals(v.getProperty("_debugType"))) {
422                     String id = v.getProperty(EntityType.ID_KEY);
423                     if (id == null) {
424                         UUID timeBasedUUID = GenericIdGenerator.getTimeBasedUUID();
425                         v.setProperty(EntityType.ID_KEY, timeBasedUUID.toString());
426                         v.setProperty(EntityType.TYPE_KEY, Entities.EVENT_LINK);
427                         done++;
428 
429                         if (done % 10000 == 0) {
430                             graph.getBaseGraph().commit();
431                         }
432                     }
433                 }
434             }
435             tx.success();
436             return String.valueOf(done);
437         }

442     private String makeCsv(List<List<String>> rows) throws IOException {
443         StringWriter writer = new StringWriter();
444         CSVWriter csvWriter = new CSVWriter(writer, '\t', CSVWriter.NO_QUOTE_CHARACTER);
445         for (List<String> remap : rows) {
446             String[] strings = remap.toArray(new String[2]);
447             csvWriter.writeNext(strings);
448         }
449         csvWriter.close();
450         return writer.toString();
451     }
