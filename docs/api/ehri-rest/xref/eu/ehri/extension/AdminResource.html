60   * Provides additional Admin methods needed by client systems
61   * and general graph-maintenance functionality.

65  @Path(AdminResource.ENDPOINT)
66  public class AdminResource extends AbstractRestResource {

68      public static final String ENDPOINT = "admin";
69      public static final String DEFAULT_USER_ID_PREFIX = "user";
70      public static final String DEFAULT_USER_ID_FORMAT = "%s%06d";
71  
72      public AdminResource(@Context GraphDatabaseService database) {
73          super(database);
74      }

76      /**
77       * Export the DB as a stream of JSON in
78       * <a href="https://github.com/tinkerpop/blueprints/wiki/GraphSON-Reader-and-Writer-Library">GraphSON</a> format.
79       * <p/>
80       * The mode used is EXTENDED.
81       */
82      @GET
83      @Produces(MediaType.APPLICATION_JSON)
84      @Path("/_exportGraphSON")
85      public Response getGraphSON() throws Exception {
86          return Response.ok(new StreamingOutput() {
87              @Override
88              public void write(OutputStream stream) throws IOException, WebApplicationException {
89                  try (final Tx tx = graph.getBaseGraph().beginTx()) {
90                      GraphSONWriter.outputGraph(graph, stream, GraphSONMode.EXTENDED);
91                      tx.success();
92                  }
93              }
94          }).build();
95      }

97      /**
98       * Re-build the graph's internal lucene index.
99       * <p/>
100      * NB: This takes a lot of memory for large graphs. Do
101      * not use willy-nilly and increase the heap size as
102      * necessary. TODO: Add incremental buffered commit.
103      *
104      * @throws java.lang.Exception
105      */
106     @POST
107     @Produces(MediaType.APPLICATION_JSON)
108     @Path("/_reindexInternal")
109     public Response reindexInternal() throws Exception {
110         try (final Tx tx = graph.getBaseGraph().beginTx()) {
111             manager.rebuildIndex();
112             tx.success();
113             return Response.ok().build();
114         }
115     }

117     /**
118      * Create a new user with a default name and identifier.
119      *
120      * @param jsonData Additional key/value data for the created object
121      * @param groups   IDs for groups to which the user should belong
122      * @return A new user
123      * @throws Exception
124      */
125     @POST
126     @Produces({MediaType.APPLICATION_JSON, MediaType.TEXT_XML})
127     @Consumes(MediaType.APPLICATION_JSON)
128     @Path("/createDefaultUserProfile")
129     public Response createDefaultUserProfile(String jsonData,
130             @QueryParam(GROUP_PARAM) List<String> groups) throws Exception {
131         try (final Tx tx = graph.getBaseGraph().beginTx()) {
132             String ident = getNextDefaultUserId();
133             Bundle bundle = Bundle.Builder.withClass(EntityClass.USER_PROFILE)
134                     .addDataValue(Ontology.IDENTIFIER_KEY, ident)
135                     .addDataValue(Ontology.NAME_KEY, ident)
136                     .addData(parseUserData(jsonData))
137                     .build();
138 
139             // NB: This assumes that admin's ID is the same as its identifier.
140             Accessor accessor = manager.getFrame(Group.ADMIN_GROUP_IDENTIFIER,
141                     Accessor.class);
142             Crud<UserProfile> view = ViewFactory.getCrudWithLogging(graph, UserProfile.class);
143             UserProfile user = view.create(bundle, accessor);
144 
145             // add to the groups
146             for (String groupId : groups) {
147                 Group group = manager.getFrame(groupId, EntityClass.GROUP, Group.class);
148                 group.addMember(user);
149             }
150 
151             // Grant them owner permissions on their own account.
152             new AclManager(graph).grantPermission(user, PermissionType.OWNER, user
153             );
154             Response response = creationResponse(user);
155             tx.success();
156             return response;
157         }
158     }

162     private String getNextDefaultUserId() {
163         try (CloseableIterable<Vertex> query = manager.getVertices(EntityClass.USER_PROFILE)) {
164             long start = Iterables.size(query) + 1;
165             while (manager.exists(String.format(DEFAULT_USER_ID_FORMAT,
166                     DEFAULT_USER_ID_PREFIX, start))) start++;
167             return String.format(DEFAULT_USER_ID_FORMAT, DEFAULT_USER_ID_PREFIX, start);
168         }
169     }
170 
171     private Map<String, Object> parseUserData(String json) throws IOException {
172         if (json == null || json.trim().equals("")) {
173             return Maps.newHashMap();
174         } else {
175             TypeReference<HashMap<String, Object>> typeRef = new TypeReference<
176                     HashMap<String, Object>
177                     >() {
178             };
179             return jsonMapper.readValue(json, typeRef);
180         }
181     }
